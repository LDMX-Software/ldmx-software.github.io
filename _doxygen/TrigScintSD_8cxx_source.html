25G4bool TrigScintSD::ProcessHits(G4Step* step, G4TouchableHistory* history) {
26 // Get the energy deposited by the particle during the step
27 auto energy{step->GetTotalEnergyDeposit()};
28
29 // If a non-Geantino particle doesn't deposit energy during the step,
30 // skip processing it.
31 if (auto particleDef{step->GetTrack()->GetDefinition()};
32 (energy == 0) && ((particleDef != G4Geantino::Definition()) ||
33 (particleDef != G4ChargedGeantino::Definition())))
34 return false;
35
36 // Create a new instance of a calorimeter hit
37 auto hit{new G4CalorimeterHit()};
38
39 // Set the energy deposition
40 hit->setEdep(energy);
41
42 // Set the hit position
43 auto position{0.5 * (step->GetPreStepPoint()->GetPosition() +
44 step->GetPostStepPoint()->GetPosition())};
45 auto volumePosition{step->GetPreStepPoint()
46 ->GetTouchableHandle()
47 ->GetHistory()
48 ->GetTopTransform()
49 .Inverse()
50 .TransformPoint(G4ThreeVector())};
51 hit->setPosition(position[0], position[1], volumePosition.z());
52
53 // Get the track associated with this step
54 auto track{step->GetTrack()};
55
56 // Set the global time.
57 hit->setTime(track->GetGlobalTime());
58
59 // Set the ID on the hit.
60 auto bar{track->GetVolume()->GetCopyNo()};
61 ldmx::TrigScintID id(moduleId_, bar);
62 hit->setID(id.raw());
63
64 // Set the track ID on the hit.
65 hit->setTrackID(track->GetTrackID());
66
67 // Set the PDG code from the track.
68 hit->setPdgCode(track->GetParticleDefinition()->GetPDGEncoding());
69
70 hitsCollection_->insert(hit);
71
72 return true;
73}
