27G4bool TrigScintSD::ProcessHits(G4Step* step, G4TouchableHistory* history) {
28 // Get the energy deposited by the particle during the step
29 auto energy{step->GetTotalEnergyDeposit()};
30
31 // If a non-Geantino particle doesn't deposit energy during the step,
32 // skip processing it.
33 if (energy == 0 and not isGeantino(step)) return false;
34
35 // Create a new instance of a calorimeter hit
36 // emplace_back returns a *reference* to the hit that was constructed
37 // and we should keep that reference so that we are editing the correct hit
38 ldmx::SimCalorimeterHit& hit = hits_.emplace_back();
39
40 G4StepPoint* prePoint = step->GetPreStepPoint();
41 G4StepPoint* postPoint = step->GetPostStepPoint();
42
43 // A Geant4 "touchable" is a way to uniquely identify a particular volume,
44 // short for touchable detector element. See the detector definition and
45 // response section of the Geant4 application developers manual for details.
46 //
47 // The TouchableHandle is just a reference counted pointer to a
48 // G4TouchableHistory object, which is a concrete implementation of a
49 // G4Touchable interface.
50 //
51 auto touchableHistory{prePoint->GetTouchableHandle()->GetHistory()};
52 // Affine transform for converting between local and global coordinates
53 auto topTransform{touchableHistory->GetTopTransform()};
54 // Set the hit position
55 auto position{0.5 * (prePoint->GetPosition() +
56 postPoint->GetPosition())};
57
58 // Convert the center of the bar to its corresponding global position
59 auto volumePosition{topTransform.Inverse().TransformPoint(G4ThreeVector())};
60 hit.setPosition(position[0], position[1], volumePosition.z());
61
62 // Get the track associated with this step
63 auto track{step->GetTrack()};
64
65 // Set the ID on the hit.
66 auto bar{track->GetVolume()->GetCopyNo()};
67 ldmx::TrigScintID id(module_id_, bar);
68 hit.setID(id.raw());
69
70 // add single contrib to this calorimeter hit
71 // IncidentID - this track's ID
72 // Track ID
73 // PDG ID
74 // energy deposited
75 // global time of this hit
76 hit.addContrib(track->GetTrackID(), track->GetTrackID(),
77 track->GetParticleDefinition()->GetPDGEncoding(), energy,
78 track->GetGlobalTime());
79
80
81
82 // Step details
83 hit.setPathLength(step->GetStepLength());
84 hit.setVelocity(track->GetVelocity());
85 // Convert pre/post step position from global coordinates to coordinates
86 // within the scintillator bar
87 const auto localPreStepPoint{
88 topTransform.TransformPoint(prePoint->GetPosition())};
89 const auto localPostStepPoint{
90 topTransform.TransformPoint(postPoint->GetPosition())};
91 hit.setPreStepPosition(localPreStepPoint[0], localPreStepPoint[1],
92 localPreStepPoint[2]);
93
94 hit.setPostStepPosition(localPostStepPoint[0], localPostStepPoint[1],
95 localPostStepPoint[2]);
96
97 hit.setPreStepTime(prePoint->GetGlobalTime());
98 hit.setPostStepTime(postPoint->GetGlobalTime());
99
100
101 return true;
102}
