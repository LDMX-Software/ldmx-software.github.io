27G4bool TrigScintSD::ProcessHits(G4Step* step, G4TouchableHistory* history) {
28 // Get the energy deposited by the particle during the step
29 auto energy{step->GetTotalEnergyDeposit()};
30
31 // If a non-Geantino particle doesn't deposit energy during the step,
32 // skip processing it.
33 if (energy == 0 and not isGeantino(step)) return false;
34
35 // Create a new instance of a calorimeter hit
36 // emplace_back returns a *reference* to the hit that was constructed
37 // and we should keep that reference so that we are editing the correct hit
38 ldmx::SimCalorimeterHit& hit = hits_.emplace_back();
39
40 // Set the hit position
41 auto position{0.5 * (step->GetPreStepPoint()->GetPosition() +
42 step->GetPostStepPoint()->GetPosition())};
43 auto volumePosition{step->GetPreStepPoint()
44 ->GetTouchableHandle()
45 ->GetHistory()
46 ->GetTopTransform()
47 .Inverse()
48 .TransformPoint(G4ThreeVector())};
49 hit.setPosition(position[0], position[1], volumePosition.z());
50
51 // Get the track associated with this step
52 auto track{step->GetTrack()};
53
54 // Set the ID on the hit.
55 auto bar{track->GetVolume()->GetCopyNo()};
56 ldmx::TrigScintID id(module_id_, bar);
57 hit.setID(id.raw());
58
59 // add single contrib to this calorimeter hit
60 // IncidentID - this track's ID
61 // Track ID
62 // PDG ID
63 // energy deposited
64 // global time of this hit
65 hit.addContrib(track->GetTrackID(), track->GetTrackID(),
66 track->GetParticleDefinition()->GetPDGEncoding(), energy,
67 track->GetGlobalTime());
68
69 return true;
70}
