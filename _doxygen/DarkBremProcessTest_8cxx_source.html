51void secondary_generation(simcore::darkbrem::G4eDarkBremsstrahlung& process,
52 const G4ThreeVector& incident_momentum) {
53 // maximum momentum allowed to be leftover in units of MeV
54 static const double max_leftover_momentum{1e-6};
55
56 static const G4ThreeVector origin(0., 0., 0.);
57 static const G4double Mel{G4Electron::Electron()->GetPDGMass()};
58
59 // owned and cleaned by track
60 G4DynamicParticle* incident_electron =
61 new G4DynamicParticle(G4Electron::Electron(), incident_momentum);
62
63 G4Track incident_track(incident_electron, 0., origin);
64
65 G4StepPoint* post_step_point = new G4StepPoint; // owned and clenaed by step
66 post_step_point->SetPosition(origin);
67 post_step_point->SetMomentumDirection(incident_momentum.unit());
68 post_step_point->SetMass(Mel);
69 post_step_point->SetKineticEnergy(sqrt(incident_momentum.mag2() + Mel * Mel) -
70 Mel);
71
72 G4StepPoint* pre_step_point = new G4StepPoint; // owned and cleaned by step
73
74 G4Step incident_step;
75 incident_step.SetTrack(&incident_track);
76 incident_track.SetStep(&incident_step);
77 incident_step.SetPostStepPoint(post_step_point);
78 incident_step.SetPreStepPoint(pre_step_point);
79
80 G4VParticleChange* particle_change =
81 process.PostStepDoIt(incident_track, incident_step);
82
83 G4ThreeVector total_momentum = incident_momentum;
84 for (unsigned int iSecondary{0};
85 iSecondary < particle_change->GetNumberOfSecondaries(); iSecondary++) {
86 G4ThreeVector secondary_momentum =
87 particle_change->GetSecondary(iSecondary)->GetMomentum();
88 total_momentum -= secondary_momentum;
89 }
90 particle_change->Clear(); // reset particle change for next time
91
92 // consider the total momentum 0 if it is less than an eV
93 CHECK(total_momentum.mag() <= 1e-6);
94}
