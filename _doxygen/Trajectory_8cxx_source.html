15 Trajectory::Trajectory(const G4Track* aTrack) : genStatus_(0) {
16  // Copy basic info from the track.
17  particleDef_ = aTrack->GetDefinition();
18  mass_ = aTrack->GetDynamicParticle()->GetMass();
19  trackID_ = aTrack->GetTrackID();
20  parentID_ = aTrack->GetParentID();
21  globalTime_ = aTrack->GetGlobalTime();
22  vertexPosition_ = aTrack->GetVertexPosition();
23  energy_ = aTrack->GetVertexKineticEnergy() + mass_;
24 
25  // Set the process type.
26  const G4VProcess* process = aTrack->GetCreatorProcess();
27  if (process) {
28  const G4String& processName = process->GetProcessName();
29  ldmx::SimParticle::ProcessType processType =
30  ldmx::SimParticle::findProcessType(processName);
31  processType_ = processType;
32  // Uncomment this to see what process types are being saved. --JM
33  // std::cout << "Trajectory - set process type " << processType
34  // << " from <" << processName << ">" << std::endl;
35  } else {
36  processType_ = ldmx::SimParticle::ProcessType::unknown;
37  }
38 
39  // Get the track information. This is used to set the track vertex
40  // volume and initial momentum.
41  auto trackInfo{
42  dynamic_cast<UserTrackInformation*>(aTrack->GetUserInformation())};
43 
44  // Set the vertex volume
45  vertexVolume_ = trackInfo->getVertexVolume();
46  vertexRegion_ = aTrack->GetLogicalVolumeAtVertex()->GetRegion()->GetName();
47 
48  // Set initial momentum from track information.
49  const G4ThreeVector& p = trackInfo->getInitialMomentum();
50  initialMomentum_.set(p.x(), p.y(), p.z());
51 
52  // If the track has not been stepped, then only the first point is added.
53  // Otherwise, the track has already been stepped so we add also its last
54  // location which should be its endpoint.
55  trajPoints_ = new TrajectoryPointContainer();
56  trajPoints_->push_back(new G4TrajectoryPoint(aTrack->GetVertexPosition()));
57  if (aTrack->GetTrackStatus() == G4TrackStatus::fStopAndKill) {
58  trajPoints_->push_back(new G4TrajectoryPoint(aTrack->GetPosition()));
59  }
60 }

101 void Trajectory::MergeTrajectory(G4VTrajectory* secondTrajectory) {
102  if (secondTrajectory == NULL) {
103  return;
104  }
105 
106  Trajectory* seco = (Trajectory*)secondTrajectory;
107  G4int ent = seco->GetPointEntries();
108  for (int i = 1; i < ent; i++) {
109  trajPoints_->push_back((*(seco->trajPoints_))[i]);
110  }
111  delete (*seco->trajPoints_)[0];
112  seco->trajPoints_->clear();
113 }
