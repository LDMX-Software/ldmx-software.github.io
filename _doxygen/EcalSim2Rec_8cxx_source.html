116 
117  EcalHit recHit;
118

120  double energy = simHit.getEdep() + hitNoise;
121  recHit.setAmplitude(energy);
122  if (energy > readoutThreshold_) {
123  recHit.setEnergy(((energy/MIP_SI_RESPONSE)*layerWeights_.at(hit_pair.first)+energy)*secondOrderEnergyCorrection_);

143  // Construct a hit in the ith position

148

160

163
