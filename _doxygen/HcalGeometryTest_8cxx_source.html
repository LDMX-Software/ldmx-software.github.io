38  void analyze(const framework::Event &event) final override {
39  const auto simHits =
40  event.getCollection<ldmx::SimCalorimeterHit>("HcalSimHits");
41 
42  CHECK(simHits.size() > 0);
43 
44  for (int ihit = 0; ihit < simHits.size(); ihit++) {
45  auto hit = simHits.at(ihit);
46  double x = hit.getPosition()[0];
47  double y = hit.getPosition()[1];
48  double z = hit.getPosition()[2];
49 
50  unsigned int hitID = hit.getID();
51  ldmx::HcalID detID(hitID);
52  int section = detID.section();
53  int layer = detID.layer();
54  int strip = detID.strip();
55 
56  // get the Hcal geometry
57  const auto &hcalGeometry = getCondition<ldmx::HcalGeometry>(
58  ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
59 
60  // get position
61  auto positionMap = hcalGeometry.getStripCenterPosition(detID);
62 
63  auto target_z =
64  Approx(z).epsilon(MAX_ENERGY_PERCENT_ERROR_ZPOSITION / 100);
65  CHECK(positionMap.Z() == target_z);
66  if (section == ldmx::HcalID::HcalSection::BACK) {
67  if (layer % 2 == 1) {
68  auto target_y =
69  Approx(y).epsilon(MAX_ENERGY_PERCENT_ERROR_YPOSITION / 100);
70  CHECK(positionMap.Y() == target_y);
71  } else {
72  auto target_x =
73  Approx(x).epsilon(MAX_ENERGY_PERCENT_ERROR_YPOSITION / 100);
74  CHECK(positionMap.X() == target_x);
75  }
76  } else {
77  if ((section == ldmx::HcalID::HcalSection::TOP) ||
78  (section == ldmx::HcalID::HcalSection::BOTTOM)) {
79  auto target_y =
80  Approx(y).epsilon(MAX_ENERGY_PERCENT_ERROR_YPOSITION / 100);
81  CHECK(positionMap.Y() == target_y);
82  } else if ((section == ldmx::HcalID::HcalSection::LEFT) ||
83  (section == ldmx::HcalID::HcalSection::RIGHT)) {
84  auto target_x =
85  Approx(x).epsilon(MAX_ENERGY_PERCENT_ERROR_YPOSITION / 100);
86  CHECK(positionMap.X() == target_x);
87  }
88  }
89  }
90  return;
91  }
