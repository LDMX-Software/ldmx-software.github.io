23void GammaPhysics::ConstructProcess() {
24 aParticleIterator->reset();
25
26 // Loop through all of the particles and find the "gamma".
27 while ((*aParticleIterator)()) {
28 G4ParticleDefinition* particle = aParticleIterator->value();
29 G4ProcessManager* pmanager = particle->GetProcessManager();
30 G4String particleName = particle->GetParticleName();
31
32 if (particleName == "gamma") {
33 // Get the process list associated with the gamma.
34 G4ProcessVector* vProcess = pmanager->GetProcessList();
35
36 // Search the list for the process "photoNuclear". When found,
37 // change the calling order so photonNuclear is called before
38 // EM processes. The biasing operator needs the photonNuclear
39 // process to be called first because the cross-section is
40 // needed to bias down other process.
41 for (int iProcess = 0; iProcess < vProcess->size(); ++iProcess) {
42 G4String processName = (*vProcess)[iProcess]->GetProcessName();
43 if (processName == "photonNuclear") {
44 pmanager->SetProcessOrderingToFirst((*vProcess)[iProcess],
45 G4ProcessVectorDoItIndex::idxAll);
46 }
47 }
48
49 // Add the gamma -> mumu to the physics list.
50 pmanager->AddDiscreteProcess(&gammaConvProcess);
51 }
52 }
