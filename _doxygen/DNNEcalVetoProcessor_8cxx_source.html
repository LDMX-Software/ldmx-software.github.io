36 void DNNEcalVetoProcessor::produce(framework::Event& event) {
37  ldmx::EcalVetoResult result;
38 
39  // Get the Ecal Geometry
40  const ldmx::EcalHexReadout& hexReadout = getCondition<ldmx::EcalHexReadout>(
41  ldmx::EcalHexReadout::CONDITIONS_OBJECT_NAME);
42 
43  // Get the collection of digitized Ecal hits from the event.
44  const auto ecalRecHits = event.getCollection<ldmx::EcalHit>("EcalRecHits");
45  auto nhits = std::count_if(
46  ecalRecHits.begin(), ecalRecHits.end(),
47  [](const ldmx::EcalHit& hit) { return hit.getEnergy() > 0; });
48 
49  if (nhits < max_num_hits_) {
50  // make inputs
51  make_inputs(hexReadout, ecalRecHits);
52  // run the DNN
53  auto outputs = rt_->run(input_names_, data_)[0];
54  result.setDiscValue(outputs.at(1));
55  } else {
56  result.setDiscValue(-99);
57  }
58 
59  if (debug_) {
60  std::cout << "... disc_val = " << result.getDisc() << std::endl;
61  }
62 
63  result.setVetoResult(result.getDisc() > disc_cut_);
64 
65  // If the event passes the veto, keep it. Otherwise, drop the event.
66  if (result.passesVeto()) {
67  setStorageHint(framework::hint_shouldKeep);
68  } else {
69  setStorageHint(framework::hint_shouldDrop);
70  }
71 
72  event.add(collectionName_, result);
73 }

75 void DNNEcalVetoProcessor::make_inputs(
76  const ldmx::EcalHexReadout& geom,
77  const std::vector<ldmx::EcalHit>& ecalRecHits) {
78  // clear data
79  for (auto& v : data_) {
80  std::fill(v.begin(), v.end(), 0);
81  }
82 
83  unsigned idx = 0;
84  for (const auto& hit : ecalRecHits) {
85  if (hit.getEnergy() <= 0) continue;
86  double x = 0, y = 0, z = 0;
87  geom.getCellAbsolutePosition(hit.getID(), x, y, z);
88 
89  data_[0].at(coordinate_x_offset_ + idx) = x;
90  data_[0].at(coordinate_y_offset_ + idx) = y;
91  data_[0].at(coordinate_z_offset_ + idx) = z;
92 
93  ldmx::EcalID id(hit.getID());
94  data_[1].at(feature_x_offset_ + idx) = x;
95  data_[1].at(feature_y_offset_ + idx) = y;
96  data_[1].at(feature_z_offset_ + idx) = z;
97  data_[1].at(feature_layerid_offset_ + idx) = id.layer();
98  data_[1].at(feature_energy_offset_ + idx) = std::log(hit.getEnergy());
99 
100  ++idx;
101  }
102 
103  if (debug_) {
104  for (unsigned iname = 0; iname < input_names_.size(); ++iname) {
105  std::cout << "=== " << input_names_[iname] << " ===" << std::endl;
106  for (unsigned i = 0; i < input_sizes_[iname]; ++i) {
107  std::cout << data_[iname].at(i) << ", ";
108  if ((i + 1) % max_num_hits_ == 0) {
109  std::cout << std::endl;
110  }
111  }
112  }
113  } // debug
