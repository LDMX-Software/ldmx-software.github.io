100  void DNNEcalVetoProcessor::produce(Event &event) {
101  EcalVetoResult result;
102 
103  // Get the collection of digitized Ecal hits from the event.
104  const auto ecalRecHits = event.getCollection<EcalHit>("EcalRecHits");
105  auto nhits = std::count_if(ecalRecHits.begin(), ecalRecHits.end(), [](const EcalHit& hit){ return hit.getEnergy()>0; });
106 
107  if (nhits < max_num_hits_) {
108  // make inputs
109  make_inputs(ecalRecHits);
110  // run the DNN
111 #ifdef LDMX_USE_ONNXRUNTIME
112  auto outputs = rt_->run(input_names_, data_)[0];
113  result.setDiscValue(outputs.at(1));
114 #endif
115  } else {
116  result.setDiscValue(-99);
117  }
118 
119  if (debug_) {
120  std::cout << "... disc_val = " << result.getDisc() << std::endl;
121  }
122 
123  result.setVetoResult(result.getDisc() > disc_cut_);
124 
125  // If the event passes the veto, keep it. Otherwise, drop the event.
126  if (result.passesVeto()) {
127  setStorageHint(hint_shouldKeep);
128  } else {
129  setStorageHint(hint_shouldDrop);
130  }
131 
132  event.add(collectionName_, result);
133  }

135  void DNNEcalVetoProcessor::make_inputs(const std::vector<EcalHit>& ecalRecHits) {
136  // clear data

141  unsigned idx = 0;

144  double x = 0, y = 0, z = 0;

146 
147  data_[0].at(coordinate_x_offset_ + idx) = x;
148  data_[0].at(coordinate_y_offset_ + idx) = y;
149  data_[0].at(coordinate_z_offset_ + idx) = z;
150 
151  data_[1].at(feature_x_offset_ + idx) = x;
152  data_[1].at(feature_y_offset_ + idx) = y;
153  data_[1].at(feature_z_offset_ + idx) = z;
154  data_[1].at(feature_layerid_offset_ + idx) = hit.getLayer();
155  data_[1].at(feature_energy_offset_ + idx) = std::log(hit.getEnergy());
156 
157  ++idx;
158  }
159
