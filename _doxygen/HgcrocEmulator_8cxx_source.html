6HgcrocEmulator::HgcrocEmulator(const framework::config::Parameters &ps) {
7 // settings of readout chip that are the same for all chips
8 // used in actual digitization
9 noise_ = ps.getParameter<bool>("noise");
10 timingJitter_ = ps.getParameter<double>("timingJitter");
11 rateUpSlope_ = ps.getParameter<double>("rateUpSlope");
12 timeUpSlope_ = ps.getParameter<double>("timeUpSlope");
13 rateDnSlope_ = ps.getParameter<double>("rateDnSlope");
14 timeDnSlope_ = ps.getParameter<double>("timeDnSlope");
15 timePeak_ = ps.getParameter<double>("timePeak");
16 clockCycle_ = ps.getParameter<double>("clockCycle");
17 nADCs_ = ps.getParameter<int>("nADCs");
18 iSOI_ = ps.getParameter<int>("iSOI");
19
20 // Time -> clock counts conversion
21 // time [ns] * ( 2^10 / max time in ns ) = clock counts
22 ns_ = 1024. / clockCycle_;
23
24 hit_merge_ns_ = 0.05; // combine at 50 ps level
25
26 // Configure the pulse shape function
27 pulseFunc_ =
28 TF1("pulseFunc",
29 "[0]*((1.0+exp([1]*(-[2]+[3])))*(1.0+exp([5]*(-[6]+[3]))))/"
30 "((1.0+exp([1]*(x-[2]+[3]-[4])))*(1.0+exp([5]*(x-[6]+[3]-[4]))))",
31 0.0, (double)nADCs_ * clockCycle_);
32 pulseFunc_.FixParameter(0, 1.0); // amplitude is set externally
33 pulseFunc_.FixParameter(1, rateUpSlope_);
34 pulseFunc_.FixParameter(2, timeUpSlope_);
35 pulseFunc_.FixParameter(3, timePeak_);
36 pulseFunc_.FixParameter(4, 0); // not using time offset in this way
37 pulseFunc_.FixParameter(5, rateDnSlope_);
38 pulseFunc_.FixParameter(6, timeDnSlope_);
39}

45bool HgcrocEmulator::digitize(
46 const int &channelID,
47 std::vector<std::pair<double, double>> &arriving_pulses,
48 std::vector<ldmx::HgcrocDigiCollection::Sample> &digiToAdd) const {
49 // step 0: prepare ourselves for emulation
50
51 digiToAdd.clear(); // make sure it is clean
52
53 // Configure chip settings based off of table (that may have been passed)
54 double totMax = getCondition(channelID, "TOT_MAX");
55 double padCapacitance = getCondition(channelID, "PAD_CAPACITANCE");
56 double gain = this->gain(channelID);
57 double pedestal = this->pedestal(channelID);
58 double toaThreshold = getCondition(channelID, "TOA_THRESHOLD");
59 double totThreshold = getCondition(channelID, "TOT_THRESHOLD");
60 // measTime defines the point in the BX where an in-time
61 // (time=0 in times vector) hit would arrive.
62 // Used to determine BX boundaries and TOA behavior.
63 double measTime = getCondition(channelID, "MEAS_TIME");
64 double drainRate = getCondition(channelID, "DRAIN_RATE");
65 double readoutThresholdFloat = this->readoutThreshold(channelID);
66 int readoutThreshold = int(readoutThresholdFloat);
67
68 // sort by amplitude
69 // ==> makes sure that puleses are merged towards higher ones
70 std::sort(
71 arriving_pulses.begin(), arriving_pulses.end(),
72 [](const std::pair<double, double> &a,
73 const std::pair<double, double> &b) { return a.first > b.first; });
74
75 // step 1: gather voltages into groups separated by (programmable) ns, single
76 // pass
77 CompositePulse pulse(pulseFunc_, gain, pedestal);
78
79 for (auto hit : arriving_pulses) pulse.addOrMerge(hit, hit_merge_ns_);
80
81 // TODO step 2: add timing jitter
82 // if (noise_) pulse.jitter();
83
85
86 // step 3: go through each BX sample one by one
87 bool doReadout = false;
88 bool wasTOA = false;
89 for (int iADC = 0; iADC < nADCs_; iADC++) {
90 double startBX = (iADC - iSOI_) * clockCycle_ - measTime;
91
92 // step 3b: check each merged hit to see if it peaks in this BX. If so,
93 // check its peak time to see if it's over TOT or TOA.
94 bool startTOT = false;
95 bool overTOA = false;
96 double toverTOA = -1;
97 double toverTOT = -1;
98 for (auto hit : pulse.hits()) {
99 int hitBX = int((hit.second + measTime) / clockCycle_ + iSOI_);
100 if (hitBX != iADC)
101 continue; // if this hit wasn't in the current BX, continue...
102
103 double vpeak = pulse(hit.second);
104
105 if (vpeak > totThreshold) {
106 startTOT = true;
107 if (toverTOT < hit.second)
108 toverTOT = hit.second; // use the latest time in the window
109 }
110
111 if (vpeak > toaThreshold) {
112 if (!overTOA || hit.second < toverTOA) toverTOA = hit.second;
113 overTOA = true;
114 }
115
116 } // loop over sim hits
117
118 // check for the case of a TOA even though the peak is in the next BX
119 if (!overTOA && pulse(startBX + clockCycle_) > toaThreshold) {
120 if (pulse(startBX) < toaThreshold) {
121 // pulse crossed TOA threshold somewhere between the start of this
122 // basket and the end
123 overTOA = true;
124 toverTOA = startBX + clockCycle_;
125 }
126 }
127
128 if (startTOT) {
129 // above TOT threshold -> do TOT readout mode
130
131 // @TODO NO NOISE
132 // CompositePulse includes pedestal, we need to remove it
133 // when calculating the charge deposited.
134 double charge_deposited =
135 (pulse(toverTOT) - gain * pedestal) * padCapacitance;
136
137 // Measure Time Over Threshold (TOT) by using the drain rate.
138 // 1. Use drain rate to see how long it takes for the charge to drain off
139 // 2. Translate this into DIGI samples
140
141 // Assume linear drain with slope drain rate:
142 // y-intercept = pulse amplitude
143 // slope = drain rate
144 // ==> x-intercept = amplitude / rate
145 // actual time over threshold using the real signal voltage amplitude
146 double tot = charge_deposited / drainRate;
147
148 // calculate the index that tot will complete on
149 int num_whole_clocks = int(tot / clockCycle_);
150
151 // calculate the TDC counts for this tot measurement
152 // internally, the chip uses 12 bits (2^12 = 4096)
153 // to measure a maximum of tot Max [ns]
154 int tdc_counts = int(tot * 4096 / totMax) + pedestal;
155
156 // were we already over TOA? TOT is reported in BX where TOA went over
157 // threshold...
158 int toa{0};
159 if (wasTOA) {
160 // TOA was in the past
161 toa = digiToAdd.back().toa();
162 } else {
163 // TOA is here and we need to find it
164 double timecross = pulse.findCrossing(startBX, toverTOT, toaThreshold);
165 toa = int((timecross - startBX) * ns_);
166 // keep inside valid limits
167 if (toa == 0) toa = 1;
168 if (toa > 1023) toa = 1023;
169 }
170
171 digiToAdd.emplace_back(
172 false, true, // mark as a TOT measurement
173 (iADC > 0) ? digiToAdd.at(iADC - 1).adc_t()
174 : pedestal, // ADC t-1 is first measurement
175 tdc_counts, // TOT
176 toa // TOA is third measurement
177 );
178
179 // TODO: properly handle saturation and recovery, eventually.
180 // Now just kill everything...
181 while (digiToAdd.size() < nADCs_) {
182 digiToAdd.emplace_back(true, false, // flags to mark type of sample
183 0x3FF, 0x3FF, 0);
184 }
185
186 return true; // always readout
187 } else {
188 // determine the voltage at the sampling time
189 double bxvolts = pulse((iADC - iSOI_) * clockCycle_);
190 // add noise if requested
191 if (noise_) bxvolts += noise(channelID);
192 // convert to integer and keep in range (handle low and high saturation)
193 int adc = bxvolts / gain;
194 if (adc < 0) adc = 0;
195 if (adc > 1023) adc = 1023;
196
197 // check for TOA
198 int toa(0);
199 if (pulse(startBX) < toaThreshold && overTOA) {
200 double timecross = pulse.findCrossing(startBX, toverTOA, toaThreshold);
201 toa = int((timecross - startBX) * ns_);
202 // keep inside valid limits
203 if (toa == 0) toa = 1;
204 if (toa > 1023) toa = 1023;
205 wasTOA = true;
206 } else {
207 wasTOA = false;
208 }
209
210 digiToAdd.emplace_back(
211 false, false, // use flags to mark this sample as an ADC measurement
212 (iADC > 0) ? digiToAdd.at(iADC - 1).adc_t()
213 : pedestal, // ADC t-1 is first measurement
214 adc, // ADC[t] is the second field
215 toa // TOA is third measurement
216 );
217 } // TOT or ADC Mode
218 } // sampling baskets
219
220 // we only get here if we never went into TOT mode
221 // check the SOI to see if we should read out
222 return digiToAdd.at(iSOI_).adc_t() >= readoutThreshold;
223} // HgcrocEmulator::digitize

226std::vector<ldmx::HgcrocDigiCollection::Sample> HgcrocEmulator::noiseDigi(
227 const int& channel, const double& soi_amplitude) const {
228 // get chip conditions from emulator
229 double pedestal{this->pedestal(channel)};
230 double gain{this->gain(channel)};
231 // fill a digi with noise samples
232 std::vector<ldmx::HgcrocDigiCollection::Sample> noise_digi;
233 for (int iADC{0}; iADC<nADCs_; iADC++) {
234 // gen noise for ADC samples
235 int adc_tm1{pedestal};
236 if (iADC > 0)
237 adc_tm1 = noise_digi.at(iADC-1).adc_t();
238 else
239 adc_tm1 += noise(channel)/gain;
240 int adc_t{pedestal + noise(channel)/gain};
241
242 if (iADC == iSOI_)
243 adc_t += soi_amplitude/gain;
244
245 // set toa to 0 (not determined)
246 // put new sample into noise digi
247 noise_digi.emplace_back(false,false,adc_tm1,adc_t,0);
248 } // samples in noise digi
249 return noise_digi;
250}
