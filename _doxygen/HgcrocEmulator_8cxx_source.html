6  HgcrocEmulator::HgcrocEmulator(const Parameters& ps) {
7 
8  //settings of readout chip that are the same for all chips
9  // used in actual digitization
10  noiseRMS_ = ps.getParameter<double>("noiseRMS");
11  timingJitter_ = ps.getParameter<double>("timingJitter");
12  clockCycle_ = ps.getParameter<double>("clockCycle");
13  totMax_ = ps.getParameter<double>("totMax");
14  rateUpSlope_ = ps.getParameter<double>("rateUpSlope");
15  timeUpSlope_ = ps.getParameter<double>("timeUpSlope");
16  rateDnSlope_ = ps.getParameter<double>("rateDnSlope");
17  timeDnSlope_ = ps.getParameter<double>("timeDnSlope");
18  timePeak_ = ps.getParameter<double>("timePeak");
19  nADCs_ = ps.getParameter<int>("nADCs");
20  iSOI_ = ps.getParameter<int>("iSOI");
21  noise_ = ps.getParameter<bool>("noise");
22  readoutPadCapacitance_ = ps.getParameter<double>("readoutPadCapacitance");
23 
24  //conditions/settings of chip that may change between chips
25  // the ones passed here are the "defaults", i.e. if
26  // no extra conditions information is passed, then the emulator
27  // uses these parameters
28  gain_ = ps.getParameter<double>("gain");
29  pedestal_ = ps.getParameter<double>("pedestal");
30  readoutThreshold_ = ps.getParameter<double>("readoutThreshold");
31  toaThreshold_ = ps.getParameter<double>("toaThreshold");
32  totThreshold_ = ps.getParameter<double>("totThreshold");
33  measTime_ = ps.getParameter<double>("measTime");
34  drainRate_ = ps.getParameter<double>("drainRate");
35 
36  //Time -> clock counts conversion
37  // time [ns] * ( 2^10 / max time in ns ) = clock counts
38  ns_ = 1024./clockCycle_;
39 
40  // Configure the pulse shape function
41  pulseFunc_ = TF1(
42  "pulseFunc",
43  "[0]*((1.0+exp([1]*(-[2]+[3])))*(1.0+exp([5]*(-[6]+[3]))))/((1.0+exp([1]*(x-[2]+[3]-[4])))*(1.0+exp([5]*(x-[6]+[3]-[4]))))",
44  0.0,(double) nADCs_*clockCycle_
45  );
46  pulseFunc_.FixParameter( 1 , rateUpSlope_ );
47  pulseFunc_.FixParameter( 2 , timeUpSlope_ );
48  pulseFunc_.FixParameter( 3 , timePeak_ );
49  pulseFunc_.FixParameter( 5 , rateDnSlope_ );
50  pulseFunc_.FixParameter( 6 , timeDnSlope_ );
51

58  bool HgcrocEmulator::digitize(
59  const int &channelID,
60  const std::vector<double> &voltages,
61  const std::vector<double> &times,
62  std::vector<HgcrocDigiCollection::Sample> &digiToAdd
63  ) const {
64 
65  digiToAdd.clear(); //make sure it is clean
66 
67  //sum all voltages and do a voltage-weighted average to get the hit time
68  // exclude any hits with times outside the sampling region
69  double signalAmplitude = 0.0;
70  double timeInWindow = 0.0;
71  for ( int iContrib = 0; iContrib < voltages.size(); iContrib++ ) {
72 
73  if ( times.at(iContrib) < 0 or times.at(iContrib) > clockCycle_*nADCs_ ) {
74  //invalid contribution - outside time range or time is unset
75  continue;
76  }
77 
78  signalAmplitude += voltages.at(iContrib);
79  timeInWindow += voltages.at(iContrib) * times.at(iContrib);
80  }
81  if ( signalAmplitude > 0. ) timeInWindow /= signalAmplitude; //voltage weighted average
82 
83  // put noise onto timing
84  //TODO more physical way of simulating the timing jitter
85  if ( noise_ ) timeInWindow += noiseInjector_->Gaus( 0. , timingJitter_ );
86 
87  //set time in the window to zero if noise pushed it below zero
88  //TODO better (more physical) method for handling this case?
89  if ( timeInWindow < 0. ) timeInWindow = 0.;
90 
91  //setup up pulse by changing the amplitude and timing parameters
92  configurePulse( signalAmplitude , timeInWindow );
93 
94  //Configure chip settings based off of table (that may have been passed)
95  double gain = getCondition( channelID , "gain" , gain_ );
96  double pedestal = getCondition( channelID , "pedestal" , pedestal_ );
97  double toaThreshold = getCondition( channelID , "toaThreshold" , toaThreshold_ );
98  double totThreshold = getCondition( channelID , "totThreshold" , totThreshold_ );
99  double measTime = getCondition( channelID , "measTime" , measTime_ );
100  double drainRate = getCondition( channelID , "drainRate" , drainRate_ );
101 
102  double readoutThresholdFloat = getCondition( channelID , "readoutThreshold" , readoutThreshold_ );
103  int readoutThreshold = int(readoutThresholdFloat);
104  /* debug printout
105  std::cout << "Configuration: {"
106  << "gain: " << gain << " mV/ADC, "
107  << "pedestal: " << pedestal << ", "
108  << "readout: " << readoutThreshold << " mV, "
109  << "toa: " << toaThreshold << " mV, "
110  << "tot: " << totThreshold << " mV, "
111  << "measTime: " << measTime << " ns, "
112  << "drainRate: " << drainRate << " mV/ns }"
113  << std::endl;
114  */
115 
116  auto measurePulse = [this,&gain,&pedestal](double time, bool withNoise) {
117  auto signal = gain*pedestal + pulseFunc_.Eval(time);
118  if(withNoise) signal += noiseInjector_->Gaus(0.,noiseRMS_);
119  return signal;
120  };
121 
122  // choose readout mode
123  double pulsePeak = measurePulse( timeInWindow , false );
124  if (verbose_) {
125  std::cout << "Pulse: { "
126  << "Amplitude: " << pulsePeak << "mV, "
127  << "Beginning: " << measurePulse(0.,false) << "mV, "
128  << "Time: " << timeInWindow << "ns } -> ";
129  }
130  if ( pulsePeak < totThreshold ) {
144  //below TOT threshold -> do ADC readout mode
145  if (verbose_) std::cout << "ADC Mode { ";
146 
147  //measure time of arrival (TOA) using TOA threshold
148  double toa(0.);
149  // make sure pulse crosses TOA threshold
150  if ( measurePulse(0.,false) < toaThreshold and pulsePeak > toaThreshold ) {
151  toa = pulseFunc_.GetX(toaThreshold-gain*pedestal, -nADCs_*clockCycle_, timeInWindow);
152  }
153  if (verbose_) std::cout << "TOA: " << toa << "ns, ";
154 
155  //measure ADCs
156  for ( unsigned int iADC = 0; iADC < nADCs_; iADC++ ) {
157  double fullMeasTime = iADC*clockCycle_ + measTime;
158  digiToAdd.emplace_back(
159  false, false, //use flags to mark this sample as an ADC measurement
160  iADC > 0 ? digiToAdd.at(iADC-1).adc_t() : pedestal, //ADC t-1 is first measurement
161  measurePulse( fullMeasTime, noise_ )/gain, //ADC t is second measurement
162  toa * ns_ //TOA is third measurement
163  );
164  if (verbose_) std::cout << " ADC " << iADC << ": " << digiToAdd[iADC].adc_t() << ", ";
165  }
166  if (verbose_) std::cout << "}" << std::endl;
167 
172  return (digiToAdd.at(iSOI_).adc_t() >= readoutThreshold);
173 
174  } else {
184  // above TOT threshold -> do TOT readout mode
185 
186  double charge_deposited = signalAmplitude * readoutPadCapacitance_;
187 
188  // Measure Time Over Threshold (TOT) by using the drain rate.
189  // 1. Use drain rate to see how long it takes for the charge to drain off
190  // 2. Translate this into DIGI samples
191 
192  // Assume linear drain with slope drain rate:
193  // y-intercept = pulse amplitude
194  // slope = drain rate
195  // ==> x-intercept = amplitude / rate
196  //actual time over threshold using the real signal voltage amplitude
197  double tot = charge_deposited / drainRate;
198 
199  double toa(0.); //default is earliest possible time
200  // check if first half is just always above readout
201  if ( measurePulse(0.,false) < totThreshold )
202  toa = pulseFunc_.GetX(totThreshold-gain*pedestal, 0., timeInWindow);
203 
204  // calculate the index that tot will complete on
205  int num_whole_clocks = int( tot / clockCycle_ );
206 
207  // calculate the TDC counts for this tot measurement
208  // internally, the chip uses 12 bits (2^12 = 4096)
209  // to measure a maximum of tot Max [ns]
210  int tdc_counts = int( tot * 4096 / totMax_ ) + pedestal;
211 
212  if (verbose_) {
213  std::cout << "TOT Mode { "
214  << "TOA: " << toa << " ns, "
215  << "TOT: " << tot << " ns, "
216  << "TDC: " << tdc_counts << "}"
217  << std::endl;
218  }
219 
220  //Notice that if tot > max time = digiToAdd.size()*clockCycle_,
221  // this will return just the max time
222  for ( unsigned int iADC = 0; iADC < nADCs_; iADC++ ) {
223  bool tot_progress, tot_complete;
224  int secon_measurement;
225  if ( iADC == iSOI_ ) {
226  //for in-time hits, the TOT is reported in the SOI
227  secon_measurement = tdc_counts;
228  tot_progress = false;
229  tot_complete = true;
230  } else {
231  //TOT still in progress or already completed
232  double fullMeasTime = iADC*clockCycle_ + measTime;
233  //TODO what does the pulse ADC measurement look like during TOT
234  secon_measurement = measurePulse( fullMeasTime, noise_ )/gain;
235  tot_progress = (iADC < num_whole_clocks);
236  tot_complete = false;
237  }
238  digiToAdd.emplace_back(
239  tot_progress, tot_complete, //flags to mark type of sample
240  iADC > 0 ? digiToAdd.at(iADC-1).adc_t() : pedestal, //first measurement is ADC t-1
241  secon_measurement,
242  toa*ns_ //last measurement is TOA
243  );
244  }
245 
249  return true;
250  } //where is the amplitude of the hit w.r.t. readout and TOT thresholds
251 
252  } //HgcrocEmulator::digitize
