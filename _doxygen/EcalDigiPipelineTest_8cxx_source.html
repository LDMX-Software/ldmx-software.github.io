99class isCloseEnough : public Catch::MatcherBase<double> {
100 private:
102 double truth_;
103
105 const double max_absolute_diff_;
106
108 const double max_relative_diff_;
109
110 public:
116 isCloseEnough(double const &truth, double const &abs_diff,
117 double const &rel_diff)
118 : truth_{truth},
119 max_absolute_diff_{abs_diff},
120 max_relative_diff_{rel_diff} {}
121
129 bool match(const double& daq_energy) const override {
130 return (
131 daq_energy == Approx(truth_).epsilon(max_relative_diff_)
132 or
133 daq_energy == Approx(truth_).margin(max_absolute_diff_)
134 );
135 }
136
140 virtual std::string describe() const override {
141 std::ostringstream ss;
142 ss << "is within an absolute difference of "
143 << max_absolute_diff_ << "MeV OR a relative difference of "
144 << max_relative_diff_ << " with " << truth_ << " MeV.";
145 return ss.str();
146 }
147};

233class EcalCheckEnergyReconstruction : public framework::Analyzer {
234 public:
235 EcalCheckEnergyReconstruction(const std::string &name, framework::Process &p)
236 : framework::Analyzer(name, p) {}
237 ~EcalCheckEnergyReconstruction() {}
238
239 void onProcessStart() final override {
240 getHistoDirectory();
241 ntuple_.create("EcalDigiTest");
242 ntuple_.addVar<float>("EcalDigiTest", "SimEnergy");
243 ntuple_.addVar<float>("EcalDigiTest", "RecEnergy");
244 ntuple_.addVar<float>("EcalDigiTest", "TrigPrimEnergy");
245
246 ntuple_.addVar<int>("EcalDigiTest", "DaqDigi");
247 ntuple_.addVar<int>("EcalDigiTest", "DaqDigiIsADC");
248 ntuple_.addVar<int>("EcalDigiTest", "DaqDigiADC");
249 ntuple_.addVar<int>("EcalDigiTest", "DaqDigiTOT");
250 ntuple_.addVar<int>("EcalDigiTest", "TrigPrimDigiEncoded");
251 ntuple_.addVar<int>("EcalDigiTest", "TrigPrimDigiLinear");
252 }
253
254 void analyze(const framework::Event &event) final override {
255 const auto simHits =
256 event.getCollection<ldmx::SimCalorimeterHit>("EcalSimHits");
257
258 REQUIRE(simHits.size() == 1);
259
260 float truth_energy = simHits.at(0).getEdep();
261 ntuple_.setVar<float>("SimEnergy", truth_energy);
262
263 const auto daqDigis{
264 event.getObject<ldmx::HgcrocDigiCollection>("EcalDigis")};
265
266 CHECK(daqDigis.getNumDigis() == 1);
267 auto daqDigi = daqDigis.getDigi(0);
268 ntuple_.setVar<int>("DaqDigi", daqDigi.soi().raw());
269 bool is_in_adc_mode = daqDigi.isADC();
270 ntuple_.setVar<int>("DaqDigiIsADC", is_in_adc_mode);
271 ntuple_.setVar<int>("DaqDigiADC", daqDigi.soi().adc_t());
272 ntuple_.setVar<int>("DaqDigiTOT", daqDigi.tot());
273
274 const auto recHits = event.getCollection<ldmx::EcalHit>("EcalRecHits");
275 CHECK(recHits.size() == 1);
276
277 auto hit = recHits.at(0);
278 ldmx::EcalID id(hit.getID());
279 CHECK_FALSE(hit.isNoise());
280 CHECK(id.raw() == simHits.at(0).getID());
281
282 double daq_energy{hit.getAmplitude()};
283 CHECK_THAT(daq_energy,
284 isCloseEnough(truth_energy,MAX_ENERGY_ERROR_DAQ,MAX_ENERGY_PERCENT_ERROR_DAQ));
285 ntuple_.setVar<float>("RecEnergy", hit.getAmplitude());
286
287 const auto trigDigis{
288 event.getObject<ldmx::HgcrocTrigDigiCollection>("ecalTrigDigis")};
289 CHECK(trigDigis.size() == 1);
290
291 auto trigDigi = trigDigis.at(0);
292 float tp_energy = 8 * trigDigi.linearPrimitive() * 320. / 1024 * MeV_per_fC;
293
294 CHECK_THAT(tp_energy ,
295 isCloseEnough(truth_energy,MAX_ENERGY_ERROR_TP,MAX_ENERGY_PERCENT_ERROR_TP));
296 ntuple_.setVar<float>("TrigPrimEnergy", tp_energy);
297 ntuple_.setVar<int>("TrigPrimDigiEncoded", trigDigi.getPrimitive());
298 ntuple_.setVar<int>("TrigPrimDigiLinear", trigDigi.linearPrimitive());
299
300 return;
301 }
302}; // EcalCheckEnergyReconstruction
