102class isCloseEnough : public Catch::Matchers::MatcherBase<double> {
103 private:
105 double truth_;
106
108 const double max_absolute_diff_;
109
111 const double max_relative_diff_;
112
113 public:
119 isCloseEnough(double const &truth, double const &abs_diff,
120 double const &rel_diff)
121 : truth_{truth},
122 max_absolute_diff_{abs_diff},
123 max_relative_diff_{rel_diff} {}
124
132 bool match(const double& daq_energy) const override {
133 return (
134 daq_energy == Approx(truth_).epsilon(max_relative_diff_)
135 or
136 daq_energy == Approx(truth_).margin(max_absolute_diff_)
137 );
138 }
139
143 virtual std::string describe() const override {
144 std::ostringstream ss;
145 ss << "is within an absolute difference of "
146 << max_absolute_diff_ << "MeV OR a relative difference of "
147 << max_relative_diff_ << " with " << truth_ << " MeV.";
148 return ss.str();
149 }
150};

236class EcalCheckEnergyReconstruction : public framework::Analyzer {
237 public:
238 EcalCheckEnergyReconstruction(const std::string &name, framework::Process &p)
239 : framework::Analyzer(name, p) {}
240 ~EcalCheckEnergyReconstruction() {}
241
242 void onProcessStart() final override {
243 getHistoDirectory();
244 ntuple_.create("EcalDigiTest");
245 ntuple_.addVar<float>("EcalDigiTest", "SimEnergy");
246 ntuple_.addVar<float>("EcalDigiTest", "RecEnergy");
247 ntuple_.addVar<float>("EcalDigiTest", "TrigPrimEnergy");
248
249 ntuple_.addVar<int>("EcalDigiTest", "DaqDigi");
250 ntuple_.addVar<int>("EcalDigiTest", "DaqDigiIsADC");
251 ntuple_.addVar<int>("EcalDigiTest", "DaqDigiADC");
252 ntuple_.addVar<int>("EcalDigiTest", "DaqDigiTOT");
253 ntuple_.addVar<int>("EcalDigiTest", "TrigPrimDigiEncoded");
254 ntuple_.addVar<int>("EcalDigiTest", "TrigPrimDigiLinear");
255 }
256
257 void analyze(const framework::Event &event) final override {
258 const auto simHits =
259 event.getCollection<ldmx::SimCalorimeterHit>("EcalSimHits");
260
261 REQUIRE(simHits.size() == 1);
262
263 float truth_energy = simHits.at(0).getEdep();
264 ntuple_.setVar<float>("SimEnergy", truth_energy);
265
266 const auto daqDigis{
267 event.getObject<ldmx::HgcrocDigiCollection>("EcalDigis")};
268
269 CHECK(daqDigis.getNumDigis() == 1);
270 auto daqDigi = daqDigis.getDigi(0);
271 ntuple_.setVar<int>("DaqDigi", daqDigi.soi().raw());
272 bool is_in_adc_mode = daqDigi.isADC();
273 ntuple_.setVar<int>("DaqDigiIsADC", is_in_adc_mode);
274 ntuple_.setVar<int>("DaqDigiADC", daqDigi.soi().adc_t());
275 ntuple_.setVar<int>("DaqDigiTOT", daqDigi.tot());
276
277 const auto recHits = event.getCollection<ldmx::EcalHit>("EcalRecHits");
278 CHECK(recHits.size() == 1);
279
280 auto hit = recHits.at(0);
281 ldmx::EcalID id(hit.getID());
282 CHECK_FALSE(hit.isNoise());
283 CHECK(id.raw() == simHits.at(0).getID());
284
285 double daq_energy{hit.getAmplitude()};
286 CHECK_THAT(daq_energy,
287 isCloseEnough(truth_energy,MAX_ENERGY_ERROR_DAQ,MAX_ENERGY_PERCENT_ERROR_DAQ));
288 ntuple_.setVar<float>("RecEnergy", hit.getAmplitude());
289
290 const auto trigDigis{
291 event.getObject<ldmx::HgcrocTrigDigiCollection>("ecalTrigDigis")};
292 CHECK(trigDigis.size() == 1);
293
294 auto trigDigi = trigDigis.at(0);
295 float tp_energy = 8 * trigDigi.linearPrimitive() * 320. / 1024 * MeV_per_fC;
296
297 CHECK_THAT(tp_energy ,
298 isCloseEnough(truth_energy,MAX_ENERGY_ERROR_TP,MAX_ENERGY_PERCENT_ERROR_TP));
299 ntuple_.setVar<float>("TrigPrimEnergy", tp_energy);
300 ntuple_.setVar<int>("TrigPrimDigiEncoded", trigDigi.getPrimitive());
301 ntuple_.setVar<int>("TrigPrimDigiLinear", trigDigi.linearPrimitive());
302
303 return;
304 }
305}; // EcalCheckEnergyReconstruction
