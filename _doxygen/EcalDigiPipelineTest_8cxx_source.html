197  void analyze(const ldmx::Event &event) final override {
198 
199  const auto simHits = event.getCollection<ldmx::SimCalorimeterHit>("EcalSimHits");
200 
201  REQUIRE(simHits.size() == 1);
202 
203  float truth_energy = simHits.at(0).getEdep();
204  ntuple_.setVar<float>("SimEnergy", truth_energy);
205 
206  const auto daqDigis{event.getObject<ldmx::HgcrocDigiCollection>("EcalDigis")};
207 
208  CHECK(daqDigis.getNumDigis() == 1);
209  auto daqDigi = daqDigis.getDigi(0);
210  ntuple_.setVar<int>("DaqDigi", daqDigi.soi().raw());
211  bool is_in_adc_mode = daqDigi.isADC();
212  ntuple_.setVar<int>("DaqDigiIsADC", is_in_adc_mode);
213  ntuple_.setVar<int>("DaqDigiADC", daqDigi.soi().adc_t());
214  ntuple_.setVar<int>("DaqDigiTOT", daqDigi.tot());
215 
216  const auto recHits = event.getCollection<ldmx::EcalHit>("EcalRecHits");
217  CHECK(recHits.size() == 1);
218 
219  auto hit = recHits.at(0);
220  ldmx::EcalID id(hit.getID());
221  CHECK_FALSE(hit.isNoise());
222  CHECK(id.raw() == simHits.at(0).getID());
223 
224  // define target energy by using the settings at the top
225  auto target_daq_energy =
226  Approx(truth_energy)
227  .epsilon(MAX_ENERGY_PERCENT_ERROR_DAQ_TOT_MODE / 100);
228  if (is_in_adc_mode)
229  target_daq_energy =
230  Approx(truth_energy).margin(MAX_ENERGY_ERROR_DAQ_ADC_MODE);
231 
232  CHECK(hit.getAmplitude() == target_daq_energy);
233  ntuple_.setVar<float>("RecEnergy", hit.getAmplitude());
234 
235  const auto trigDigis{
236  event.getObject<ldmx::HgcrocTrigDigiCollection>("ecalTrigDigis")};
237  CHECK(trigDigis.size() == 1);
238 
239  auto trigDigi = trigDigis.at(0);
240  float tp_energy = 8 * trigDigi.linearPrimitive() * 320. / 1024 * MeV_per_fC;
241 
242  auto target_tp_energy =
243  Approx(truth_energy)
244  .epsilon(MAX_ENERGY_PERCENT_ERROR_TP_TOT_MODE / 100);
245  if (is_in_adc_mode)
246  target_tp_energy =
247  Approx(truth_energy).margin(MAX_ENERGY_ERROR_TP_ADC_MODE);
248 
249  CHECK(tp_energy == target_tp_energy);
250  ntuple_.setVar<float>("TrigPrimEnergy", tp_energy);
251  ntuple_.setVar<int>("TrigPrimDigiEncoded", trigDigi.getPrimitive());
252  ntuple_.setVar<int>("TrigPrimDigiLinear", trigDigi.linearPrimitive());
253 
254  return;
255  }
