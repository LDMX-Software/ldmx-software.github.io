60bool StorageControl::keepEvent(bool event_completed) const {
61 if (not event_completed) return false;
62
63 int votesKeep(0), votesDrop(0);
64 // loop over all rules and then over all hints
65 for (auto rule : rules_) {
66 for (auto hint : hints_) {
67 if (!rule.matches(hint)) continue;
68 if (hint.hint_ == hint_shouldKeep || hint.hint_ == hint_mustKeep)
69 votesKeep++;
70 else if (hint.hint_ == hint_shouldDrop || hint.hint_ == hint_mustDrop)
71 votesDrop++;
72 }
73 }
74
75 // easy case
76 if (!votesKeep && !votesDrop) return defaultIsKeep_;
77
78 // harder cases
79 if (votesKeep > votesDrop) return true;
80 if (votesDrop > votesKeep) return false;
81
82 // at the end, go with the default
83 return defaultIsKeep_;
84}
