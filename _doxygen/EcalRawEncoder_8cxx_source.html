192 word |= (link_id & packing::utility::mask<16>) << 16;

205 link_crc << word;
206
207 // special "header" word from ROC
208 word = 0;
209 word |= 0b0101 << 28;
210 word |= (bunch_id & packing::utility::mask<12>) << 16;
211 word |= (rreq & packing::utility::mask<6> ) << 10;
212 word |= (orbit & packing::utility::mask<3>) << 7;
213 // skipping hamming error bits because we will set them all to false here
214 word |= 0b0101;
215 buffer.push_back(word);
216 fpga_crc << word;
217 link_crc << word;
218
227 // put samples into buffer
228 for (auto const& [channel, sample] : channels) {
229 buffer.push_back(sample);
230 fpga_crc << sample;
231 link_crc << sample;
232 }
233 buffer.push_back(link_crc.get());
234 fpga_crc << link_crc.get();
235 }
236 }
237 i_bx++;
238 }
239
240 event.add(output_name_, buffer);

242 return;
