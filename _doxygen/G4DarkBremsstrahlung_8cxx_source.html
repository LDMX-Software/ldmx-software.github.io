161G4double G4DarkBremsstrahlung::GetMeanFreePath(const G4Track& track, G4double,
162 G4ForceCondition*) {
163 // won't happen if it isn't applicable
164 if (not IsApplicable(*track.GetParticleDefinition())) return DBL_MAX;
165
166 G4double energy = track.GetDynamicParticle()->GetKineticEnergy();
167 G4double SIGMA = 0;
168 G4Material* materialWeAreIn = track.GetMaterial();
169 const G4ElementVector* theElementVector = materialWeAreIn->GetElementVector();
170 const G4double* NbOfAtomsPerVolume = materialWeAreIn->GetVecNbOfAtomsPerVolume();
171
172 for (size_t i = 0; i < materialWeAreIn->GetNumberOfElements(); i++) {
173 G4double AtomicZ = (*theElementVector)[i]->GetZ();
174 G4double AtomicA = (*theElementVector)[i]->GetA() / (g / mole);
175
176 G4double element_xsec;
177
178 if (cache_xsec_)
179 element_xsec = element_xsec_cache_.get(energy, AtomicA, AtomicZ);
180 else
181 element_xsec =
182 model_->ComputeCrossSectionPerAtom(energy, AtomicA, AtomicZ);
183
184 SIGMA += NbOfAtomsPerVolume[i] * element_xsec;
185 }
186 SIGMA *= global_bias_;
187 if (GetVerboseLevel() > 3) {
188 G4cout << "G4DBrem : sigma = " << SIGMA
189 << " initIntLenLeft = " << theInitialNumberOfInteractionLength
190 << " nIntLenLeft = " << theNumberOfInteractionLengthLeft << G4endl;
191 }
192 return SIGMA > DBL_MIN ? 1. / SIGMA : DBL_MAX;
