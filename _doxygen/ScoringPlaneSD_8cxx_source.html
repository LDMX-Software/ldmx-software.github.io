34G4bool ScoringPlaneSD::ProcessHits(G4Step* step, G4TouchableHistory* history) {
35 // Get the edep from the step.
36 G4double edep = step->GetTotalEnergyDeposit();
37
38 // Create a new hit object.
39 G4TrackerHit* hit = new G4TrackerHit();
40
41 // Assign track ID for finding the SimParticle in post event processing.
42 hit->setTrackID(step->GetTrack()->GetTrackID());
43 hit->setPdgID(step->GetTrack()->GetDynamicParticle()->GetPDGcode());
44
45 // Set the edep.
46 hit->setEdep(edep);
47
48 // Set the start position.
49 G4StepPoint* prePoint = step->GetPreStepPoint();
50
51 // Set the end position.
52 G4StepPoint* postPoint = step->GetPostStepPoint();
53
54 G4ThreeVector start = prePoint->GetPosition();
55 G4ThreeVector end = postPoint->GetPosition();
56
57 // Set the mid position.
58 G4ThreeVector mid = 0.5 * (start + end);
59 hit->setPosition(mid.x(), mid.y(), mid.z());
60
61 // Compute path length.
62 G4double pathLength =
63 sqrt(pow(start.x() - end.x(), 2) + pow(start.y() - end.y(), 2) +
64 pow(start.z() - end.z(), 2));
65 hit->setPathLength(pathLength);
66
67 // Set the global time.
68 hit->setTime(step->GetTrack()->GetGlobalTime());
69
70 // Set the momentum
71 G4ThreeVector p = postPoint->GetMomentum();
72 hit->setMomentum(p.x(), p.y(), p.z());
73 hit->setEnergy(postPoint->GetTotalEnergy());
74
75 /*
76 * Set the 32-bit ID on the hit.
77 */
78 int cpNumber = prePoint->GetTouchableHandle()->GetCopyNumber();
79 ldmx::SimSpecialID id = ldmx::SimSpecialID::ScoringPlaneID(cpNumber);
80 hit->setID(id.raw());
81
82 /*
83 * Debug print.
84 */
85 if (this->verboseLevel > 2) {
86 hit->Print();
87 std::cout << std::endl;
88 }
89
90 // Insert hit into current hits collection.
91 hitsCollection_->insert(hit);
92
93 return true;
94}
