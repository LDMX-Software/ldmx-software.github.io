32G4bool ScoringPlaneSD::ProcessHits(G4Step* step, G4TouchableHistory* history) {
33 // Get the edep from the step.
34 G4double edep = step->GetTotalEnergyDeposit();
35
36 // Create a new hit object.
37 ldmx::SimTrackerHit& hit{hits_.emplace_back()};
38
39 // Assign track ID for finding the SimParticle in post event processing.
40 hit.setTrackID(step->GetTrack()->GetTrackID());
41 hit.setPdgID(step->GetTrack()->GetDynamicParticle()->GetPDGcode());
42
43 // Set the edep.
44 hit.setEdep(edep);
45
46 // Set the start position.
47 G4StepPoint* prePoint = step->GetPreStepPoint();
48
49 // Set the end position.
50 G4StepPoint* postPoint = step->GetPostStepPoint();
51
52 G4ThreeVector start = prePoint->GetPosition();
53 G4ThreeVector end = postPoint->GetPosition();
54
55 // Set the mid position.
56 G4ThreeVector mid = 0.5 * (start + end);
57 hit.setPosition(mid.x(), mid.y(), mid.z());
58
59 // Compute path length.
60 G4double pathLength =
61 sqrt(pow(start.x() - end.x(), 2) + pow(start.y() - end.y(), 2) +
62 pow(start.z() - end.z(), 2));
63 hit.setPathLength(pathLength);
64
65 // Set the global time.
66 hit.setTime(step->GetTrack()->GetGlobalTime());
67
68 // Set the momentum
69 G4ThreeVector p = postPoint->GetMomentum();
70 hit.setMomentum(p.x(), p.y(), p.z());
71 hit.setEnergy(postPoint->GetTotalEnergy());
72
73 /*
74 * Set the 32-bit ID on the hit.
75 */
76 int cpNumber = prePoint->GetTouchableHandle()->GetCopyNumber();
77 ldmx::SimSpecialID id = ldmx::SimSpecialID::ScoringPlaneID(cpNumber);
78 hit.setID(id.raw());
79
80 return true;
81}
