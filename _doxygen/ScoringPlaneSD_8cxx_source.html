36  G4bool ScoringPlaneSD::ProcessHits(G4Step* step, G4TouchableHistory* history) {
37 
38  // Get the edep from the step.
39  G4double edep = step->GetTotalEnergyDeposit();
40 
41  // Create a new hit object.
42  G4TrackerHit* hit = new G4TrackerHit();
43 
44  // Assign track ID for finding the SimParticle in post event processing.
45  hit->setTrackID(step->GetTrack()->GetTrackID());
46  hit->setPdgID(step->GetTrack()->GetDynamicParticle()->GetPDGcode());
47 
48  // Set the edep.
49  hit->setEdep(edep);
50 
51  // Set the start position.
52  G4StepPoint* prePoint = step->GetPreStepPoint();
53 
54  // Set the end position.
55  G4StepPoint* postPoint = step->GetPostStepPoint();
56 
57  G4ThreeVector start = prePoint->GetPosition();
58  G4ThreeVector end = postPoint->GetPosition();
59 
60  // Set the mid position.
61  G4ThreeVector mid = 0.5 * (start + end);
62  hit->setPosition(mid.x(), mid.y(), mid.z());
63 
64  // Compute path length.
65  G4double pathLength = sqrt(pow(start.x() - end.x(), 2)
66  + pow(start.y() - end.y(), 2)
67  + pow(start.z() - end.z(), 2));
68  hit->setPathLength(pathLength);
69 
70  // Set the global time.
71  hit->setTime(step->GetTrack()->GetGlobalTime());
72 
73  // Set the momentum
74  G4ThreeVector p = postPoint->GetMomentum();
75  hit->setMomentum(p.x(), p.y(), p.z());
76  hit->setEnergy(postPoint->GetTotalEnergy());
77 
78  /*
79  * Set the 32-bit ID on the hit.
80  */
81  int cpNumber = prePoint->GetTouchableHandle()->GetCopyNumber();
82  SimSpecialID id=SimSpecialID::ScoringPlaneID(cpNumber);
83  hit->setID(id.raw());
84 
85  /*
86  * Debug print.
87  */
88  if (this->verboseLevel > 2) {
89  hit->Print();
90  std::cout << std::endl;
91  }
92 
93  // Insert hit into current hits collection.
94  hitsCollection_->insert(hit);
95 
96  return true;
97  }
