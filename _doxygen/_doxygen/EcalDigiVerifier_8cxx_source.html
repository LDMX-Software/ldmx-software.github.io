16  void EcalDigiVerifier::analyze(const ldmx::Event& event) {
17 
18  //get truth information sorted into an ID based map
19  std::vector<SimCalorimeterHit> ecalSimHits = event.getCollection<SimCalorimeterHit>( ecalSimHitColl_ , ecalSimHitPass_ );
20 
21  //sort sim hits by ID
22  std::sort( ecalSimHits.begin() , ecalSimHits.end() ,
23  []( const SimCalorimeterHit &lhs , const SimCalorimeterHit &rhs ) {
24  return lhs.getID() < rhs.getID();
25  }
26  );
27 
28  std::vector<EcalHit> ecalRecHits = event.getCollection<EcalHit>( ecalRecHitColl_ , ecalRecHitPass_ );
29 
30  //sort rec hits by ID
31  std::sort( ecalRecHits.begin() , ecalRecHits.end() ,
32  []( const EcalHit &lhs , const EcalHit &rhs ) {
33  return lhs.getID() < rhs.getID();
34  }
35  );
36 
37  double totalRecEnergy = 0.;
38  for ( const EcalHit &recHit : ecalRecHits ) {
39 
40  //skip anything that digi flagged as noise
41  if ( recHit.isNoise() ) continue;
42 
43  int rawID = recHit.getID();
44 
45  //get information for this hit
46  int numSimHits = 0;
47  double totalSimEDep = 0.;
48  for ( const SimCalorimeterHit &simHit : ecalSimHits ) {
49  if ( rawID == simHit.getID() ) {
50  numSimHits += simHit.getNumberOfContribs();
51  totalSimEDep += simHit.getEdep();
52  } else if ( rawID < simHit.getID() ) {
53  //later sim hits - all done
54  break;
55  }
56  }
57 
58  histograms_.fill( "num_sim_hits_per_cell" , numSimHits );
59  histograms_.fill( "sim_edep__rec_amplitude" , totalSimEDep , recHit.getAmplitude() );
60 
61  totalRecEnergy += recHit.getEnergy();
62  }
63 
64  histograms_.fill( "total_rec_energy" , totalRecEnergy );
65 
66  if ( totalRecEnergy > 6000. ) {
67  setStorageHint( hint_shouldKeep );
68  } else {
69  setStorageHint( hint_shouldDrop );
70  }
71 
72  return;
73  }
74 
75 }
