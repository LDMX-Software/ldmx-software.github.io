33  G4bool HcalSD::ProcessHits(G4Step* aStep, G4TouchableHistory* ROhist) {
34 
35  // Determine if current particle of this step is a Geantino.
36  G4ParticleDefinition* pdef = aStep->GetTrack()->GetDefinition();
37  bool isGeantino = false;
38  if (pdef == G4Geantino::Definition() || pdef == G4ChargedGeantino::Definition()) {
39  isGeantino = true;
40  }
41 
42  // Get the edep from the step.
43  G4double edep = aStep->GetTotalEnergyDeposit();
44 
45  // Skip steps with no energy dep which come from non-Geantino particles.
46  if (edep == 0.0 && !isGeantino) {
47  if (verboseLevel > 2) {
48  std::cout << "CalorimeterSD skipping step with zero edep." << std::endl << std::endl;
49  }
50  return false;
51  }
52 
53  //---------------------------------------------------------------------------------------------------
54  // Birks' Law
55  // ===========
56  //
57  // In the case of Scintillator as active medium, we can
58  // describe the quenching effects with the Birks' law,
59  // using the expression and the coefficients taken from
60  // the paper NIM 80 (1970) 239-244 for the organic
61  // scintillator NE-102:
62  // S*dE/dr
63  // dL/dr = -----------------------------------
64  // 1 + C1*(dE/dr)
65  // with:
66  // S=1
67  // C1 = 1.29 x 10^-2 g*cm^-2*MeV^-1
68  // C2 = 9.59 x 10^-6 g^2*cm^-4*MeV^-2
69  // These are the same values used by ATLAS TileCal
70  // and CMS HCAL (and also the default in Geant3).
71  // You can try different values for these parameters,
72  // to have an idea on the uncertainties due to them,
73  // by uncommenting one of the lines below.
74  // To get the "dE/dr" that appears in the formula,
75  // which has the dimensions
76  // [ dE/dr ] = MeV * cm^2 / g
77  // we have to divide the energy deposit in MeV by the
78  // product of the step length (in cm) and the density
79  // of the scintillator:
80 
81  G4double birksFactor(1.0);
82  G4double stepLength = aStep->GetStepLength() / CLHEP::cm;
83  //Do not apply Birks for gamma deposits!
84  if (stepLength > 1.0e-6) //Check, cut if necessary.
85  {
86  G4double rho = aStep->GetPreStepPoint()->GetMaterial()->GetDensity() / (CLHEP::g / CLHEP::cm3);
87  G4double dedx = edep / (rho * stepLength); //[MeV*cm^2/g]
88  birksFactor = 1.0 / (1.0 + birksc1_ * dedx + birksc2_ * dedx * dedx);
89  if (aStep->GetTrack()->GetDefinition() == G4Gamma::GammaDefinition()) birksFactor = 1.0;
90  if (aStep->GetTrack()->GetDefinition() == G4Neutron::NeutronDefinition()) birksFactor = 1.0;
91  }
92 
93 
94  // Create a new cal hit.
95  G4CalorimeterHit* hit = new G4CalorimeterHit();
96 
97  // Set the edep.
98  hit->setEdep(edep*birksFactor);
99 
100  // Get the scintillator solid box
101  G4Box* scint = static_cast<G4Box*>(aStep->GetPreStepPoint()->GetTouchableHandle()->GetVolume()->GetLogicalVolume()->GetSolid());
102 
103  // Set the step mid-point as the hit position.
104  G4StepPoint* prePoint = aStep->GetPreStepPoint();
105  G4StepPoint* postPoint = aStep->GetPostStepPoint();
106  G4ThreeVector position = 0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
107  G4ThreeVector localPosition = aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()->GetTopTransform().TransformPoint(position);
108  hit->setPosition(position[0], position[1], position[2]);
109 
110  // Set the global time.
111  hit->setTime(aStep->GetTrack()->GetGlobalTime());
112 
113  // Create the ID for the hit.
114  int copyNum = aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()->GetVolume(layerDepth_)->GetCopyNo();
115  int section = copyNum / 1000;
116  int layer = copyNum % 1000;
117 
118  //stripID: back Hcal, segmented along y direction for now every 10 cm -- alternate x-y in the future?
119  // left/right side hcal: segmented along x direction every 10 cm
120  // top/bottom side hcal: segmented along y direction every 10 cm
121 
122  int stripID = -1;
123  // Odd layers have bars horizontal
124  // Even layers have bars vertical
125  // 5cm wide bars are HARD-CODED
126  if (section==HcalID::BACK && layer % 2 == 1) stripID = int( (localPosition.y()+scint->GetYHalfLength())/50.0);
127  else if (section==HcalID::BACK && layer % 2 == 0) stripID = int( (localPosition.x()+scint->GetXHalfLength())/50.0);
128  else stripID = int( (localPosition.z()+scint->GetZHalfLength())/50.0);
129 
130  // std::cout << "---" << std::endl;
131  // std::cout << "GetXHalfLength = " << scint->GetXHalfLength() << "\t GetYHalfLength = " << scint->GetYHalfLength() << "\t GetZHalfLength = " << scint->GetZHalfLength() << std::endl;
132  // std::cout << "xpos = " << localPosition.x() << "\t ypos = " << localPosition.y() << "\t zpos = " << localPosition.z() << std::endl;
133  // std::cout << "xpos_g = " << position.x() << "\t ypos_g = " << position.y() << "\t zpos_g = " << position.z() << std::endl;
134  // std::cout << "Layer = " << layer << "\t section = " << section << "\t strip = " << stripID << std::endl;
135 
136  HcalID id(section,layer,stripID);
137  hit->setID(id.raw());
138 
139  // Set the track ID on the hit.
140  hit->setTrackID(aStep->GetTrack()->GetTrackID());
141 
142  // Set the PDG code from the track.
143  hit->setPdgCode(aStep->GetTrack()->GetParticleDefinition()->GetPDGEncoding());
144 
145  //do we want to set the hit coordinate in the middle of the absorber?
146  // G4ThreeVector volumePosition = aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()->GetTopTransform().Inverse().TransformPoint(G4ThreeVector());
147  // if (section==HcalID::BACK) hit->setPosition(position[0], position[1], volumePosition.z());
148  // elseif (section==HcalID::TOP || section==HcalID::BOTTOM) hit->setPosition(position[0], volumePosition.y(), position[2]);
149  // elseif (section==HcalID::LEFT || section==HcalID::RIGHT) hit->setPosition(volumePosition.x(),position[1] , position[2]);
150 
151 
152  if (this->verboseLevel > 2) {
153  std::cout << "Created new SimCalorimeterHit in detector " << this->GetName()
154  << " subdet ID <" << subdet_ << ">, layer <" << layer << "> and section <" << section << ">, copynum <" << copyNum << ">"
155  << std::endl;
156  hit->Print();
157  std::cout << std::endl;
158  }

161  hitsCollection_->insert(hit);
162

164  }
