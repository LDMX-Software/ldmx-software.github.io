15  Trajectory::Trajectory(const G4Track* aTrack) :
16  genStatus_(0) {
17 
18  // Copy basic info from the track.
19  particleDef_ = aTrack->GetDefinition();
20  mass_ = aTrack->GetDynamicParticle()->GetMass();
21  trackID_ = aTrack->GetTrackID();
22  parentID_ = aTrack->GetParentID();
23  globalTime_ = aTrack->GetGlobalTime();
24  vertexPosition_ = aTrack->GetVertexPosition();
25  energy_ = aTrack->GetVertexKineticEnergy() + mass_;
26 
27  // Set the process type.
28  const G4VProcess* process = aTrack->GetCreatorProcess();
29  if (process) {
30  const G4String& processName = process->GetProcessName();
31  SimParticle::ProcessType processType = SimParticle::findProcessType(processName);
32  processType_ = processType;
33  // Uncomment this to see what process types are being saved. --JM
34  //std::cout << "Trajectory - set process type " << processType
35  // << " from <" << processName << ">" << std::endl;
36  } else {
37  processType_ = SimParticle::ProcessType::unknown;
38  }
39 
40  // Get the track information. This is used to set the track vertex
41  // volume and initial momentum.
42  auto trackInfo{dynamic_cast<UserTrackInformation*>(aTrack->GetUserInformation())};
43 
44  // Set the vertex volume
45  vertexVolume_ = trackInfo->getVertexVolume();
46  vertexRegion_ = aTrack->GetLogicalVolumeAtVertex()->GetRegion()->GetName();
47 
48  // Set initial momentum from track information.
49  const G4ThreeVector& p = trackInfo->getInitialMomentum();
50  initialMomentum_.set(p.x(), p.y(), p.z());
51 
52  // If the track has not been stepped, then only the first point is added.
53  // Otherwise, the track has already been stepped so we add also its last location
54  // which should be its endpoint.
55  trajPoints_ = new TrajectoryPointContainer();
56  trajPoints_->push_back(new G4TrajectoryPoint(aTrack->GetVertexPosition()));
57  if (aTrack->GetTrackStatus() == G4TrackStatus::fStopAndKill) {
58  trajPoints_->push_back(new G4TrajectoryPoint(aTrack->GetPosition()));
59  }
60  }
