52 uint64_t RandomNumberSeedService::getSeed(const std::string& name) const {
53  uint64_t seed(0);
54  std::map<std::string,uint64_t>::const_iterator i=seeds_.find(name);
55  if (i==seeds_.end()) {
56  // hash is sum of characters shifted by position, mod 8
57  for (size_t j=0; j<name.size(); j++)
58  seed+=(uint64_t(name[j])<<(j%8));
59  seed+=masterSeed_;
60  // break const here only to cache the seed
61  seeds_[name]=seed;
62  } else seed=i->second;
63  return seed;
64 }
