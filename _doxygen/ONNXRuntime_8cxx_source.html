17 ONNXRuntime::ONNXRuntime(const std::string& model_path,
18  const SessionOptions* session_options) {
19  // create session
20  if (session_options) {
21  session_.reset(new Session(env_, model_path.c_str(), *session_options));
22  } else {
23  SessionOptions sess_opts;
24  sess_opts.SetIntraOpNumThreads(1);
25  session_.reset(new Session(env_, model_path.c_str(), sess_opts));
26  }
27  AllocatorWithDefaultOptions allocator;
28 
29  // get input names and shapes
30  size_t num_input_nodes = session_->GetInputCount();
31  input_node_strings_.resize(num_input_nodes);
32  input_node_names_.resize(num_input_nodes);
33  input_node_dims_.clear();
34 
35  for (size_t i = 0; i < num_input_nodes; i++) {
36  // get input node names
37  std::string input_name(session_->GetInputName(i, allocator));
38  input_node_strings_[i] = input_name;
39  input_node_names_[i] = input_node_strings_[i].c_str();
40 
41  // get input shapes
42  auto type_info = session_->GetInputTypeInfo(i);
43  auto tensor_info = type_info.GetTensorTypeAndShapeInfo();
44  size_t num_dims = tensor_info.GetDimensionsCount();
45  input_node_dims_[input_name].resize(num_dims);
46  tensor_info.GetDimensions(input_node_dims_[input_name].data(), num_dims);
47 
48  // set the batch size to 1 by default
49  input_node_dims_[input_name].at(0) = 1;
50  }
51 
52  size_t num_output_nodes = session_->GetOutputCount();
53  output_node_strings_.resize(num_output_nodes);
54  output_node_names_.resize(num_output_nodes);
55  output_node_dims_.clear();
56 
57  for (size_t i = 0; i < num_output_nodes; i++) {
58  // get output node names
59  std::string output_name(session_->GetOutputName(i, allocator));
60  output_node_strings_[i] = output_name;
61  output_node_names_[i] = output_node_strings_[i].c_str();
62 
63  // get output node types
64  auto type_info = session_->GetOutputTypeInfo(i);
65  auto tensor_info = type_info.GetTensorTypeAndShapeInfo();
66  size_t num_dims = tensor_info.GetDimensionsCount();
67  output_node_dims_[output_name].resize(num_dims);
68  tensor_info.GetDimensions(output_node_dims_[output_name].data(), num_dims);
69 
70  // the 0th dim depends on the batch size
71  output_node_dims_[output_name].at(0) = -1;
72  }
73 }

77 FloatArrays ONNXRuntime::run(const std::vector<std::string>& input_names,
78  FloatArrays& input_values,
79  const std::vector<std::string>& output_names,
80  int64_t batch_size) const {
81  assert(input_names.size() == input_values.size());
82  assert(batch_size > 0);
83 
84  // create input tensor objects from data values
85  std::vector<Value> input_tensors;
86  auto memory_info =
87  MemoryInfo::CreateCpu(OrtArenaAllocator, OrtMemTypeDefault);
88  for (const auto& name : input_node_strings_) {
89  auto iter = std::find(input_names.begin(), input_names.end(), name);
90  if (iter == input_names.end()) {
91  throw std::runtime_error("Input " + name + " is not provided!");
92  }
93  auto value = input_values.begin() + (iter - input_names.begin());
94  auto input_dims = input_node_dims_.at(name);
95  input_dims[0] = batch_size;
96  auto expected_len = std::accumulate(input_dims.begin(), input_dims.end(), 1,
97  std::multiplies<int64_t>());
98  if (expected_len != (int64_t)value->size()) {
99  throw std::runtime_error("Input array " + name + " has a wrong size of " +
100  std::to_string(value->size()) + ", expected " +
101  std::to_string(expected_len));
102  }
103  auto input_tensor =
104  Value::CreateTensor<float>(memory_info, value->data(), value->size(),
105  input_dims.data(), input_dims.size());
106  assert(input_tensor.IsTensor());
107  input_tensors.emplace_back(std::move(input_tensor));
108  }
109 
110  // set output node names; will get all outputs if `output_names` is not
111  // provided
112  std::vector<const char*> run_output_node_names;
113  if (output_names.empty()) {
114  run_output_node_names = output_node_names_;
115  } else {
116  for (const auto& name : output_names) {
117  run_output_node_names.push_back(name.c_str());
118  }
119  }
120 
121  // run
122  auto output_tensors =
123  session_->Run(RunOptions{nullptr}, input_node_names_.data(),
124  input_tensors.data(), input_tensors.size(),
125  run_output_node_names.data(), run_output_node_names.size());
126 
127  // convert output to floats
128  FloatArrays outputs;
129  for (auto& output_tensor : output_tensors) {
130  assert(output_tensor.IsTensor());
131 
132  // get output shape
133  auto tensor_info = output_tensor.GetTensorTypeAndShapeInfo();
134  auto length = tensor_info.GetElementCount();
135 
136  auto floatarr = output_tensor.GetTensorMutableData<float>();
137  outputs.emplace_back(floatarr, floatarr + length);
138  }
139  assert(outputs.size() == run_output_node_names.size());
140 
141  return outputs;
142 }
