44void NtuplizeTrigScintQIEDigis::analyze(const framework::Event& event) {
45 const auto& digis{event.getCollection<trigscint::TrigScintQIEDigis>(input_name_,input_pass_)};
46 ldmxsw_event_ = event.getEventNumber();
47 for (const auto& digi : digis) {

49 chan_id_ = digi.getChanID();
50 elec_id_ = digi.getElecID();
