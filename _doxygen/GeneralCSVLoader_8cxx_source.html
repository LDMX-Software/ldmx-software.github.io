27bool GeneralCSVLoader::nextRow() {
28 do {
29 std::string line=getNextLine();
30 if (line.empty()) return false;
31 // it's a comment!
32 if (line[0]=='#') continue;
33 // split into pieces
34 std::vector<std::string> line_split;
35 // explicitly erase trailing white space
36 line.erase(std::find_if(line.rbegin(), line.rend(),
37 [](unsigned char c) { return !std::isspace(c); }).base(), line.end());
38 boost::tokenizer<boost::escaped_list_separator<char>> tok(line);
39 for (auto chunk : tok){
40 line_split.push_back(chunk);
41 }
42 if (colNames_.empty()) {
43 colNames_.swap(line_split);
44 continue;
45 }
46 if (line_split.size()==colNames_.size()) rowData_.swap(line_split);
47 else {
48 EXCEPTION_RAISE("CSVLineMismatch","Reading CSV found line with "+std::to_string(line_split.size())+" in CSV with "+std::to_string(colNames_.size()) +" columns");
49 }
50
51 return true;
52 } while (true);
53}

59std::string StringCSVLoader::getNextLine() {
60
61 std::string retval;
62 if (rowBegin_!=rowEnd_)
63 if (rowEnd_==std::string::npos) retval=source_.substr(rowBegin_,rowEnd_);
64 else retval=source_.substr(rowBegin_,rowEnd_-rowBegin_);
65
66 // now we look for the follow on.
67 // find the first non-end-of-line character
68 rowBegin_=source_.find_first_not_of(linesep_,rowEnd_);
69 if (rowBegin_!=std::string::npos) {
70 rowEnd_=source_.find_first_of(linesep_,rowBegin_);
71 } else rowEnd_=std::string::npos;
72
73 return retval;
74}
