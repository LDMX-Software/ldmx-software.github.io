41void TrackingRecoDQM::TrackMonitoring(const std::vector<ldmx::Track>& tracks,
42 const std::string title) {
43
44 //If I have truth information, sort the tracks vector according to their trackID and truthProb
45 std::vector<ldmx::Track> uniqueTracks; // real tracks (truth_prob > cut), unique
46 std::vector<ldmx::Track> duplicateTracks; // real tracks (truth_prob > cut), duplicated
47 std::vector<ldmx::Track> fakeTracks; // fake tracks (truth_prob < cut)
48
49 if (doTruthComparison) {
50 sortTracks(tracks,uniqueTracks, duplicateTracks,fakeTracks);
51 }
52 else {
53 uniqueTracks = tracks;
54 }
55
56 for (auto &track : uniqueTracks) {
57
58 //Perigee track parameters
59
60 double trk_d0 = track.getD0();
61 double trk_z0 = track.getZ0();
62 double trk_qop = track.getQoP();
63 double trk_theta = track.getTheta();
64 double trk_phi = track.getPhi();
65 double trk_p = 1./abs(trk_qop);
66
67 std::vector<double> trk_mom = track.getMomentum();
68
69 //The transverse momentum in the bending plane
70 double pt_bending = std::sqrt(trk_mom[0]*trk_mom[0] + trk_mom[1]*trk_mom[1]);
71
72 //The momentum in the plane transverse wrt the beam axis
73 double pt_beam = std::sqrt(trk_mom[1]*trk_mom[1] + trk_mom[2]*trk_mom[2]);
74
75 histograms_.fill(title+"d0",trk_d0);
76 histograms_.fill(title+"z0",trk_z0);
77 histograms_.fill(title+"qop",trk_qop);
78 histograms_.fill(title+"phi",trk_phi);
79 histograms_.fill(title+"theta",trk_theta);
80
81 histograms_.fill(title+"p", std::abs(1./trk_qop));
82 histograms_.fill(title+"px", trk_mom[0]);
83 histograms_.fill(title+"py", trk_mom[1]);
84 histograms_.fill(title+"pz", trk_mom[2]);
85
86 histograms_.fill(title+"pt_bending", pt_bending);
87 histograms_.fill(title+"pt_beam", pt_beam);
88
89 histograms_.fill(title+"nHits", track.getNhits());
90 histograms_.fill(title+"Chi2", track.getChi2());
91 histograms_.fill(title+"ndf", track.getNdf());
92 histograms_.fill(title+"Chi2/ndf", track.getChi2()/track.getNdf());
93 histograms_.fill(title+"nShared", track.getNsharedHits());
94
95
96 //Covariance matrix
97 Acts::BoundSymMatrix cov = tracking::sim::utils::unpackCov(track.getPerigeeCov());
98
99 double sigmad0 = sqrt(cov(Acts::BoundIndices::eBoundLoc0,Acts::BoundIndices::eBoundLoc0));
100 double sigmaz0 = sqrt(cov(Acts::BoundIndices::eBoundLoc1,Acts::BoundIndices::eBoundLoc1));
101 double sigmaphi = sqrt(cov(Acts::BoundIndices::eBoundPhi ,Acts::BoundIndices::eBoundPhi));
102 double sigmatheta = sqrt(cov(Acts::BoundIndices::eBoundTheta,Acts::BoundIndices::eBoundTheta));
103 double sigmaqop = sqrt(cov(Acts::BoundIndices::eBoundQOverP,Acts::BoundIndices::eBoundQOverP));
104
105 histograms_.fill(title+"d0_err", sigmad0);
106 histograms_.fill(title+"z0_err", sigmaz0);
107 histograms_.fill(title+"phi_err", sigmaphi);
108 histograms_.fill(title+"theta_err",sigmatheta);
109 histograms_.fill(title+"qop_err", sigmaqop);
110
111 double sigmap = (1./trk_qop)*(1./trk_qop)*sigmaqop;
112 histograms_.fill(title+"p_err", sigmap);
113
114 if (doTruthComparison) {
115
116 //Truth Comparison
117 ldmx::TruthTrack* truth_trk = nullptr;
118
119 auto it = std::find_if(truthTrackCollection_->begin(),
120 truthTrackCollection_->end(),[&](const ldmx::TruthTrack& tt) {
121 return tt.getTrackID() == track.getTrackID();
122 });
123
124 double trackTruthProb = track.getTruthProb();
125
126 if (it != truthTrackCollection_->end() && trackTruthProb >= trackProb_cut_)
127 truth_trk = &(*it);
128
129
130 //Found matched track
131 if (truth_trk) {
132
133 double truth_d0 = truth_trk->getD0();
134 double truth_z0 = truth_trk->getZ0();
135 double truth_phi = truth_trk->getPhi();
136 double truth_theta = truth_trk->getTheta();
137 double truth_qop = truth_trk->getQoP();
138 double truth_p = 1. / abs(truth_trk->getQoP());
139
140 histograms_.fill(title+"truth_d0", truth_d0);
141 histograms_.fill(title+"truth_z0", truth_z0);
142 histograms_.fill(title+"truth_phi", truth_phi);
143 histograms_.fill(title+"truth_theta",truth_theta);
144 histograms_.fill(title+"truth_qop", truth_qop);
145 histograms_.fill(title+"truth_p", truth_p);
146
147 double res_d0 = trk_d0 - truth_d0;
148 double res_z0 = trk_z0 - truth_z0;
149 double res_phi = trk_phi - truth_phi;
150 double res_theta = trk_theta - truth_theta;
151 double res_qop = trk_qop - truth_qop;
152 double res_p = trk_p - truth_p;
153
154 histograms_.fill(title+"res_d0", res_d0);
155 histograms_.fill(title+"res_z0", res_z0);
156 histograms_.fill(title+"res_phi", res_phi);
157 histograms_.fill(title+"res_theta",res_theta);
158 histograms_.fill(title+"res_qop", res_qop);
159 histograms_.fill(title+"res_p", res_p);
160
161 double pull_d0 = res_d0 / sigmad0;
162 double pull_z0 = res_z0 / sigmaz0;
163 double pull_phi = res_phi / sigmaphi;
164 double pull_theta = res_theta / sigmatheta;
165 double pull_qop = res_qop / sigmaqop;
166 double pull_p = res_p / sigmap;
167
168 histograms_.fill(title+"pull_d0", pull_d0);
169 histograms_.fill(title+"pull_z0", pull_z0);
170 histograms_.fill(title+"pull_phi", pull_phi);
171 histograms_.fill(title+"pull_theta",pull_theta);
172 histograms_.fill(title+"pull_qop", pull_qop);
173 histograms_.fill(title+"pull_p", pull_p);
174
175
176 //Fill reco plots for efficiencies - numerator
177
178 histograms_.fill(title+"match_d0", trk_d0);
179 histograms_.fill(title+"match_z0", trk_z0);
180 histograms_.fill(title+"match_phi", trk_phi);
181 histograms_.fill(title+"match_theta",trk_theta);
182 histograms_.fill(title+"match_p", trk_p);
183
184 } //found matched track
185 }//do TruthComparison
186 }//loop on unique tracks (or not split)
187
188
189 for (auto& ftrack : fakeTracks) {
190 histograms_.fill(title+"fake_d0", ftrack.getD0());
191 histograms_.fill(title+"fake_z0", ftrack.getZ0());
192 histograms_.fill(title+"fake_phi", ftrack.getPhi());
193 histograms_.fill(title+"fake_theta",ftrack.getTheta());
194 histograms_.fill(title+"fake_p", 1. / abs(ftrack.getQoP()));
195 histograms_.fill(title+"fake_nHits", ftrack.getNhits());
196 histograms_.fill(title+"fake_Chi2", ftrack.getChi2());
197 histograms_.fill(title+"fake_Chi2/ndf", ftrack.getChi2()/ftrack.getNdf());
198 histograms_.fill(title+"fake_nShared", ftrack.getNsharedHits());
199 }
200
201 for (auto& dtrack : duplicateTracks) {
202 histograms_.fill(title+"dup_d0", dtrack.getD0());
203 histograms_.fill(title+"dup_z0", dtrack.getZ0());
204 histograms_.fill(title+"dup_phi", dtrack.getPhi());
205 histograms_.fill(title+"dup_theta",dtrack.getTheta());
206 histograms_.fill(title+"dup_p", 1. / abs(dtrack.getQoP()));
207 histograms_.fill(title+"dup_nHits", dtrack.getNhits());
208 histograms_.fill(title+"dup_Chi2", dtrack.getChi2());
209 histograms_.fill(title+"dup_Chi2/ndf", dtrack.getChi2()/dtrack.getNdf());
210 histograms_.fill(title+"dup_nShared", dtrack.getNsharedHits());
211 }

216void TrackingRecoDQM::sortTracks(const std::vector<ldmx::Track>& tracks,
217 std::vector<ldmx::Track>& uniqueTracks,
218 std::vector<ldmx::Track>& duplicateTracks,
219 std::vector<ldmx::Track>& fakeTracks) {
220
221 // Create a copy of the const vector so we can sort it
222 std::vector<ldmx::Track> sortedTracks = tracks;
223
224 // Sort the vector of Track objects based on their trackID member
225 std::sort(sortedTracks.begin(),
226 sortedTracks.end(),
227 [](ldmx::Track& t1,ldmx::Track& t2){
228 return t1.getTrackID() < t2.getTrackID();
229 }
230 );
231
232 // Loop over the sorted vector of Track objects
233 for (size_t i = 0; i < sortedTracks.size(); i++) {
234 if (sortedTracks[i].getTruthProb() < trackProb_cut_)
235 fakeTracks.push_back(sortedTracks[i]);
236 else { //not a fake track
237 // If this is the first Track object with this trackID, add it to the uniqueTracks vector directly
238 if (uniqueTracks.size() == 0 || sortedTracks[i].getTrackID() != sortedTracks[i-1].getTrackID()) {
239 uniqueTracks.push_back(sortedTracks[i]);
240 }
241 // Otherwise, add it to the duplicateTracks vector if its truthProb is lower than the existing Track object
242 // Otherwise, if the truthProbability is higher than the track stored in uniqueTracks, put it in uniqueTracks and move the uniqueTracks.back to duplicateTracks.
243 else if (sortedTracks[i].getTruthProb() > uniqueTracks.back().getTruthProb()) {
244 duplicateTracks.push_back(uniqueTracks.back());
245 uniqueTracks.back() = sortedTracks[i];
246 }
247 // Otherwise, add it to the duplicateTracks vector
248 else {
249 duplicateTracks.push_back(sortedTracks[i]);
250 }
251 } //a real track
252 } //loop on sorted tracks
253 // The total number of elements in the uniqueTracks and duplicateTracks vectors should be equal to the number of elements in the original tracks vector
254 if (uniqueTracks.size() + duplicateTracks.size() + fakeTracks.size() != tracks.size()) {
255 std::cerr << "Error: unique and duplicate tracks vectors do not add up to original tracks vector" << std::endl;
256 return;
257 }
258
259 if (debug_) {
260 // Iterate through the uniqueTracks vector and duplicateTracks vector
261 std::cout << "Unique tracks:" << std::endl;
262 for (const ldmx::Track& track : uniqueTracks) {
263 std::cout << "Track ID: " << track.getTrackID() << ", Truth Prob: " << track.getTruthProb() << std::endl;
264 }
265 std::cout << "Duplicate tracks:" << std::endl;
266 for (const ldmx::Track& track : duplicateTracks) {
267 std::cout << "Track ID: " << track.getTrackID() << ", Truth Prob: " << track.getTruthProb() << std::endl;
268 }
269 std::cout << "Fake tracks:" << std::endl;
270 for (const ldmx::Track& track : fakeTracks) {
271 std::cout << "Track ID: " << track.getTrackID() << ", Truth Prob: " << track.getTruthProb() << std::endl;
272 }
273 }
274}
