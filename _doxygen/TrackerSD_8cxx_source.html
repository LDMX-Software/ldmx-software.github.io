32G4bool TrackerSD::ProcessHits(G4Step* aStep, G4TouchableHistory*) {
33 // Determine if current particle of this step is a Geantino.
34 G4ParticleDefinition* pdef = aStep->GetTrack()->GetDefinition();
35 bool isGeantino = false;
36 if (pdef == G4Geantino::Definition() ||
37 pdef == G4ChargedGeantino::Definition()) {
38 isGeantino = true;
39 }
40
41 // Get the edep from the step.
42 G4double edep = aStep->GetTotalEnergyDeposit();
43
44 // Skip steps with no energy dep which come from non-Geantino particles.
45 if (edep == 0.0 && !isGeantino) {
46 if (verboseLevel > 2) {
47 std::cout << "TrackerSD skipping step with zero edep" << std::endl
48 << std::endl;
49 }
50 return false;
51 }
52
53 // Create a new hit object.
54 G4TrackerHit* hit = new G4TrackerHit();
55
56 // Assign track ID for finding the SimParticle in post event processing.
57 hit->setTrackID(aStep->GetTrack()->GetTrackID());
58
59 // Set the edep.
60 hit->setEdep(edep);
61
62 // Set the start position.
63 G4StepPoint* prePoint = aStep->GetPreStepPoint();
64 // hit->setStartPosition(prePoint->GetPosition());
65
66 // Set the end position.
67 G4StepPoint* postPoint = aStep->GetPostStepPoint();
68 // hit->setEndPosition(postPoint->GetPosition());
69
70 G4ThreeVector start = prePoint->GetPosition();
71 G4ThreeVector end = postPoint->GetPosition();
72
73 // Set the mid position.
74 G4ThreeVector mid = 0.5 * (start + end);
75 hit->setPosition(mid.x(), mid.y(), mid.z());
76
77 // Compute path length.
78 G4double pathLength =
79 sqrt(pow(start.x() - end.x(), 2) + pow(start.y() - end.y(), 2) +
80 pow(start.z() - end.z(), 2));
81 hit->setPathLength(pathLength);
82
83 // Set the global time.
84 hit->setTime(aStep->GetTrack()->GetGlobalTime());
85
86 /*
87 * Compute and set the momentum.
88 */
89 /*
90 double mag = (prePoint->GetMomentum().mag() + postPoint->GetMomentum().mag())
91 / 2; G4ThreeVector p = (postPoint->GetPosition() - prePoint->GetPosition());
92 if (mag > 0) {
93 p.setMag(mag);
94 }
95 */
96 G4ThreeVector p = postPoint->GetMomentum();
97 hit->setMomentum(p.x(), p.y(), p.z());
98
99 /*
100 * Set the 32-bit ID on the hit.
101 */
102 int copyNum =
103 prePoint->GetTouchableHandle()->GetHistory()->GetVolume(2)->GetCopyNo();
104 int layer = copyNum / 10;
105 int module = copyNum % 10;
106 ldmx::TrackerID id(subDetID_, layer, module);
107 hit->setID(id.raw());
108 hit->setLayerID(layer);
109 hit->setModuleID(module);
110
111 // Set energy and pdg code of SimParticle (common things requested)
112 hit->setEnergy(postPoint->GetTotalEnergy());
113 hit->setPdgID(aStep->GetTrack()->GetDynamicParticle()->GetPDGcode());
114
115 /*
116 * Debug print.
117 */
118 if (this->verboseLevel > 2) {
119 hit->Print();
120 }
121
122 // Insert hit into current hits collection.
123 hitsCollection_->insert(hit);
124
125 return true;
126}

136void TrackerSD::EndOfEvent(G4HCofThisEvent*) {
137 // Print number of hits.
138 if (this->verboseLevel > 0) {
139 std::cout << GetName() << " had " << hitsCollection_->entries()
140 << " hits in event" << std::endl;
141 }
142
143 // Print each hit in hits collection.
144 if (this->verboseLevel > 1) {

146 (*hitsCollection_)[iHit]->Print();
