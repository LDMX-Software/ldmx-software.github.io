26G4bool TrackerSD::ProcessHits(G4Step* aStep, G4TouchableHistory*) {
27
28 // Get the edep from the step.
29 G4double edep = aStep->GetTotalEnergyDeposit();
30
31 // Skip steps with no energy dep which come from non-Geantino particles.
32 if (edep == 0.0 and not isGeantino(aStep)) {
33 if (verboseLevel > 2) {
34 std::cout << "TrackerSD skipping step with zero edep" << std::endl
35 << std::endl;
36 }
37 return false;
38 }
39
40 // Create a new hit object.
41 ldmx::SimTrackerHit& hit{hits_.emplace_back()};
42
43 // Assign track ID for finding the SimParticle in post event processing.
44 hit.setTrackID(aStep->GetTrack()->GetTrackID());
45
46 // Set the edep.
47 hit.setEdep(edep);
48
49 // Set the start position.
50 G4StepPoint* prePoint = aStep->GetPreStepPoint();
51 // hit->setStartPosition(prePoint->GetPosition());
52
53 // Set the end position.
54 G4StepPoint* postPoint = aStep->GetPostStepPoint();
55 // hit->setEndPosition(postPoint->GetPosition());
56
57 G4ThreeVector start = prePoint->GetPosition();
58 G4ThreeVector end = postPoint->GetPosition();
59
60 // Set the mid position.
61 G4ThreeVector mid = 0.5 * (start + end);
62 hit.setPosition(mid.x(), mid.y(), mid.z());
63
64 // Compute path length.
65 G4double pathLength =
66 sqrt(pow(start.x() - end.x(), 2) + pow(start.y() - end.y(), 2) +
67 pow(start.z() - end.z(), 2));
68 hit.setPathLength(pathLength);
69
70 // Set the global time.
71 hit.setTime(aStep->GetTrack()->GetGlobalTime());
72
73 /*
74 * Compute and set the momentum.
75 */
76 G4ThreeVector p = postPoint->GetMomentum();
77 hit.setMomentum(p.x(), p.y(), p.z());
78
79 /*
80 * Set the 32-bit ID on the hit.
81 */
82 int copyNum =
83 prePoint->GetTouchableHandle()->GetHistory()->GetVolume(2)->GetCopyNo();
84 int layer = copyNum / 10;
85 int module = copyNum % 10;
86 ldmx::TrackerID id(subDetID_, layer, module);
87 hit.setID(id.raw());
88 hit.setLayerID(layer);
89 hit.setModuleID(module);
90
91 // Set energy and pdg code of SimParticle (common things requested)
92 hit.setEnergy(postPoint->GetTotalEnergy());
93 hit.setPdgID(aStep->GetTrack()->GetDynamicParticle()->GetPDGcode());
94
95 return true;
96}
