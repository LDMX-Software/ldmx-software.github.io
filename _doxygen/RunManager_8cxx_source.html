52 void RunManager::setupPhysics() {
53  auto pList{physicsListFactory_.GetReferencePhysList("FTFP_BERT")};
54 
55  parallelWorldPath_ = parameters_.getParameter<std::string>("scoringPlanes");
56  isPWEnabled_ = !parallelWorldPath_.empty();
57  if (isPWEnabled_) {
58  std::cout
59  << "[ RunManager ]: Parallel worlds physics list has been registered."
60  << std::endl;
61  pList->RegisterPhysics(new G4ParallelWorldPhysics("ldmxParallelWorld"));
62  }
63 
64  pList->RegisterPhysics(new GammaPhysics);
65  pList->RegisterPhysics(new darkbrem::APrimePhysics(
66  parameters_.getParameter<framework::config::Parameters>("dark_brem")));
67 
68  auto biasing_operators{
69  parameters_.getParameter<std::vector<framework::config::Parameters>>(
70  "biasing_operators", {})};
71  if (!biasing_operators.empty()) {
72  std::cout << "[ RunManager ]: Biasing enabled with "
73  << biasing_operators.size() << " operator(s)." << std::endl;
74 
75  // create all the biasing operators that will be used
76  for (framework::config::Parameters& bop : biasing_operators) {
77  simcore::PluginFactory::getInstance().createBiasingOperator(
78  bop.getParameter<std::string>("class_name"),
79  bop.getParameter<std::string>("instance_name"), bop);
80  }
81 
82  // Instantiate the constructor used when biasing
83  G4GenericBiasingPhysics* biasingPhysics = new G4GenericBiasingPhysics();
84 
85  // specify which particles are going to be biased
86  // this will put a biasing interface wrapper around *all* processes
87  // associated with these particles
88  for (const simcore::XsecBiasingOperator* bop :
89  simcore::PluginFactory::getInstance().getBiasingOperators()) {
90  std::cout << "[ RunManager ]: Biasing operator '" << bop->GetName()
91  << "' set to bias " << bop->getParticleToBias() << std::endl;
92  biasingPhysics->Bias(bop->getParticleToBias());
93  }
94 
95  // Register the physics constructor to the physics list:
96  pList->RegisterPhysics(biasingPhysics);
97  }

102 void RunManager::Initialize() {
103  setupPhysics();
104 
105  // The parallel world needs to be registered before the mass world is
106  // constructed i.e. before G4RunManager::Initialize() is called.
107  if (isPWEnabled_) {
108  std::cout << "[ RunManager ]: Parallel worlds have been enabled."
109  << std::endl;
110 
111  auto validateGeometry_{parameters_.getParameter<bool>("validate_detector")};
112  G4GDMLParser* pwParser = new G4GDMLParser();
113  pwParser->Read(parallelWorldPath_, validateGeometry_);
114  this->getDetectorConstruction()->RegisterParallelWorld(
115  new ParallelWorld(pwParser, "ldmxParallelWorld", conditionsIntf_));
116  }
117 
118  // This is where the physics lists are told to construct their particles and
119  // their processes
120  // They are constructed in order, so it is important to register the biasing
121  // physics *after* any other processes that need to be able to be biased
122  G4RunManager::Initialize();
123 
124  // Instantiate the primary generator action
125  auto primaryGeneratorAction{new PrimaryGeneratorAction(parameters_)};
126  SetUserAction(primaryGeneratorAction);
127 
128  // Get instances of all G4 actions
129  // also create them in the factory
130  auto actions{PluginFactory::getInstance().getActions()};
131 
132  // Create all user actions
133  auto userActions{
134  parameters_.getParameter<std::vector<framework::config::Parameters>>(
135  "actions", {})};
136  for (auto& userAction : userActions) {
137  PluginFactory::getInstance().createAction(
138  userAction.getParameter<std::string>("class_name"),
139  userAction.getParameter<std::string>("instance_name"), userAction);
140  }
141 
142  // Register all actions with the G4 engine
143  for (const auto& [key, act] : actions) {
144  std::visit([this](auto&& arg) { this->SetUserAction(arg); }, act);
145  }
146 }

148 void RunManager::TerminateOneEvent() {
149  // have geant4 do its own thing
150  G4RunManager::TerminateOneEvent();
151 
152  // reset dark brem process (if needed)
153  G4ProcessTable* ptable = G4ProcessTable::GetProcessTable();
154  G4int verbosity = ptable->GetVerboseLevel();
155 
156  // Only one of these processes should be in the table
157  // (i.e. either the Dark Brem is biased or its not)
158  // BUT we want to be able to cover both options without
159  // the user having to configure it, so we set both
160  // of these processes to active (by passing 'true')
161  // while the table is silenced. If the table isn't silenced,
162  // the process that isn't in the table will cause the table
163  // to throw a "not found" warning.
164  std::vector<G4String> dark_brem_processes = {
165  darkbrem::G4eDarkBremsstrahlung::PROCESS_NAME,
166  "biasWrapper(" + darkbrem::G4eDarkBremsstrahlung::PROCESS_NAME + ")"};
167  ptable->SetVerboseLevel(
168  0); // silent ptable while searching for process that may/may not exist
169  for (auto const& name : dark_brem_processes)
170  ptable->SetProcessActivation(name, true);
171  ptable->SetVerboseLevel(verbosity);
172 
173  if (this->GetVerboseLevel() > 1) {
174  std::cout << "[ RunManager ] : "
175  << "Reset the dark brem process (if it was activated)."
176  << std::endl;
177  }
178  ptable->SetVerboseLevel(verbosity);
179 }
