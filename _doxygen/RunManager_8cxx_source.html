53void RunManager::setupPhysics() {
54 auto pList{physicsListFactory_.GetReferencePhysList("FTFP_BERT")};
55
56 parallelWorldPath_ = parameters_.getParameter<std::string>("scoringPlanes");
57 isPWEnabled_ = !parallelWorldPath_.empty();
58 if (isPWEnabled_) {
59 std::cout
60 << "[ RunManager ]: Parallel worlds physics list has been registered."
61 << std::endl;
62 pList->RegisterPhysics(new G4ParallelWorldPhysics("ldmxParallelWorld"));
63 }
64
65 pList->RegisterPhysics(new GammaPhysics);
66 pList->RegisterPhysics(new darkbrem::APrimePhysics(
67 parameters_.getParameter<framework::config::Parameters>("dark_brem")));
68
69 auto biasing_operators{
70 parameters_.getParameter<std::vector<framework::config::Parameters>>(
71 "biasing_operators", {})};
72 if (!biasing_operators.empty()) {
73 std::cout << "[ RunManager ]: Biasing enabled with "
74 << biasing_operators.size() << " operator(s)." << std::endl;
75
76 // create all the biasing operators that will be used
77 for (framework::config::Parameters& bop : biasing_operators) {
78 simcore::XsecBiasingOperator::Factory::get().make(
79 bop.getParameter<std::string>("class_name"),
80 bop.getParameter<std::string>("instance_name"), bop);
81 }
82
83 // Instantiate the constructor used when biasing
84 G4GenericBiasingPhysics* biasingPhysics = new G4GenericBiasingPhysics();
85
86 // specify which particles are going to be biased
87 // this will put a biasing interface wrapper around *all* processes
88 // associated with these particles
89 simcore::XsecBiasingOperator::Factory::get().apply([biasingPhysics](auto bop) {
90 std::cout << "[ RunManager ]: Biasing operator '" << bop->GetName()
91 << "' set to bias " << bop->getParticleToBias() << std::endl;
92 biasingPhysics->Bias(bop->getParticleToBias());
93 });
94
95 // Register the physics constructor to the physics list:
96 pList->RegisterPhysics(biasingPhysics);
97 }
98
99 this->SetUserInitialization(pList);
100}

102void RunManager::Initialize() {
103 setupPhysics();
104
105 // The parallel world needs to be registered before the mass world is
106 // constructed i.e. before G4RunManager::Initialize() is called.
107 if (isPWEnabled_) {
108 std::cout << "[ RunManager ]: Parallel worlds have been enabled."
109 << std::endl;
110
111 auto validateGeometry_{parameters_.getParameter<bool>("validate_detector")};
112 G4GDMLParser* pwParser = new G4GDMLParser();
113 pwParser->Read(parallelWorldPath_, validateGeometry_);
114 this->getDetectorConstruction()->RegisterParallelWorld(
115 new ParallelWorld(pwParser, "ldmxParallelWorld"));
116 }
117
118 // This is where the physics lists are told to construct their particles and
119 // their processes
120 // They are constructed in order, so it is important to register the biasing
121 // physics *after* any other processes that need to be able to be biased
122 G4RunManager::Initialize();
123
124 // create our G4User actions
125 auto primary_action{new g4user::PrimaryGeneratorAction(parameters_)};
126 auto run_action{new g4user::RunAction};
127 auto event_action{new g4user::EventAction};
128 auto tracking_action{new g4user::TrackingAction};
129 auto stepping_action{new g4user::SteppingAction};
130 auto stacking_action{new g4user::StackingAction};
131 // ...and register them with G4
132 SetUserAction(primary_action);
133 SetUserAction(run_action);
134 SetUserAction(event_action);
135 SetUserAction(tracking_action);
136 SetUserAction(stepping_action);
137 SetUserAction(stacking_action);
138
139 // Create all user actions and attch them to the corresponding G4 actions
140 auto user_actions{
141 parameters_.getParameter<std::vector<framework::config::Parameters>>(
142 "actions", {})};
143 for (auto& user_action : user_actions) {
144 auto ua = UserAction::Factory::get().make(
145 user_action.getParameter<std::string>("class_name"),
146 user_action.getParameter<std::string>("instance_name"), user_action);
147 for (auto& type : ua->getTypes()) {
148 if (type == simcore::TYPE::RUN) {
149 run_action->registerAction(ua.get());
150 } else if (type == simcore::TYPE::EVENT) {
151 event_action->registerAction(ua.get());
152 } else if (type == simcore::TYPE::TRACKING) {
153 tracking_action->registerAction(ua.get());
154 } else if (type == simcore::TYPE::STEPPING) {
155 stepping_action->registerAction(ua.get());
156 } else if (type == simcore::TYPE::STACKING) {
157 stacking_action->registerAction(ua.get());
158 } else {
159 EXCEPTION_RAISE("ActionType","Action type does not exist.");
160 }
161 }
162 }

165void RunManager::TerminateOneEvent() {
166 // have geant4 do its own thing
167 G4RunManager::TerminateOneEvent();
168
169 // reset dark brem process (if needed)
170 G4ProcessTable* ptable = G4ProcessTable::GetProcessTable();
171 G4int verbosity = ptable->GetVerboseLevel();
172
173 // Only one of these processes should be in the table
174 // (i.e. either the Dark Brem is biased or its not)
175 // BUT we want to be able to cover both options without
176 // the user having to configure it, so we set both
177 // of these processes to active (by passing 'true')
178 // while the table is silenced. If the table isn't silenced,
179 // the process that isn't in the table will cause the table
180 // to throw a "not found" warning.
181 std::vector<G4String> dark_brem_processes = {
182 darkbrem::G4eDarkBremsstrahlung::PROCESS_NAME,
183 "biasWrapper(" + darkbrem::G4eDarkBremsstrahlung::PROCESS_NAME + ")"};
184 ptable->SetVerboseLevel(
185 0); // silent ptable while searching for process that may/may not exist
186 for (auto const& name : dark_brem_processes)
187 ptable->SetProcessActivation(name, true);
188 ptable->SetVerboseLevel(verbosity);
189
190 if (this->GetVerboseLevel() > 1) {
191 std::cout << "[ RunManager ] : "
192 << "Reset the dark brem process (if it was activated)."
193 << std::endl;
194 }
195 ptable->SetVerboseLevel(verbosity);
196}
