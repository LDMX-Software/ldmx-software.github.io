50void RunManager::setupPhysics() {
51 auto pList{physicsListFactory_.GetReferencePhysList("FTFP_BERT")};
52
53 parallelWorldPath_ = parameters_.getParameter<std::string>("scoringPlanes");
54 isPWEnabled_ = !parallelWorldPath_.empty();
55 if (isPWEnabled_) {
56 std::cout
57 << "[ RunManager ]: Parallel worlds physics list has been registered."
58 << std::endl;
59 pList->RegisterPhysics(new G4ParallelWorldPhysics("ldmxParallelWorld"));
60 }
61
62 pList->RegisterPhysics(new GammaPhysics{"GammaPhysics", parameters_});
63 pList->RegisterPhysics(new APrimePhysics(
64 parameters_.getParameter<framework::config::Parameters>("dark_brem")));
65
66 auto biasing_operators{
67 parameters_.getParameter<std::vector<framework::config::Parameters>>(
68 "biasing_operators", {})};
69 if (!biasing_operators.empty()) {
70 std::cout << "[ RunManager ]: Biasing enabled with "
71 << biasing_operators.size() << " operator(s)." << std::endl;
72
73 // create all the biasing operators that will be used
74 for (framework::config::Parameters& bop : biasing_operators) {
75 simcore::XsecBiasingOperator::Factory::get().make(
76 bop.getParameter<std::string>("class_name"),
77 bop.getParameter<std::string>("instance_name"), bop);
78 }
79
80 // Instantiate the constructor used when biasing
81 G4GenericBiasingPhysics* biasingPhysics = new G4GenericBiasingPhysics();
82
83 // specify which particles are going to be biased
84 // this will put a biasing interface wrapper around *all* processes
85 // associated with these particles
86 simcore::XsecBiasingOperator::Factory::get().apply([biasingPhysics](
87 auto bop) {
88 std::cout << "[ RunManager ]: Biasing operator '" << bop->GetName()
89 << "' set to bias " << bop->getParticleToBias() << std::endl;
90 biasingPhysics->Bias(bop->getParticleToBias());
91 });
92
93 // Register the physics constructor to the physics list:
94 pList->RegisterPhysics(biasingPhysics);
95 }
96
97 this->SetUserInitialization(pList);
98}

100void RunManager::Initialize() {
101 setupPhysics();
102
103 // The parallel world needs to be registered before the mass world is
104 // constructed i.e. before G4RunManager::Initialize() is called.
105 if (isPWEnabled_) {
106 std::cout << "[ RunManager ]: Parallel worlds have been enabled."
107 << std::endl;
108
109 auto validateGeometry_{parameters_.getParameter<bool>("validate_detector")};
110 G4GDMLParser* pwParser = new G4GDMLParser();
111 pwParser->Read(parallelWorldPath_, validateGeometry_);
112 this->getDetectorConstruction()->RegisterParallelWorld(
113 new ParallelWorld(pwParser, "ldmxParallelWorld"));
114 }
115
116 // This is where the physics lists are told to construct their particles and
117 // their processes
118 // They are constructed in order, so it is important to register the biasing
119 // physics *after* any other processes that need to be able to be biased
120 G4RunManager::Initialize();
121
122 // create our G4User actions
123 auto primary_action{new g4user::PrimaryGeneratorAction(parameters_)};
124 auto run_action{new g4user::RunAction};
125 auto event_action{new g4user::EventAction};
126 auto tracking_action{new g4user::TrackingAction};
127 auto stepping_action{new g4user::SteppingAction};
128 auto stacking_action{new g4user::StackingAction};
129 // ...and register them with G4
130 SetUserAction(primary_action);
131 SetUserAction(run_action);
132 SetUserAction(event_action);
133 SetUserAction(tracking_action);
134 SetUserAction(stepping_action);
135 SetUserAction(stacking_action);
136
137 // Create all user actions and attch them to the corresponding G4 actions
138 auto user_actions{
139 parameters_.getParameter<std::vector<framework::config::Parameters>>(
140 "actions", {})};
141 for (auto& user_action : user_actions) {
142 auto ua = UserAction::Factory::get().make(
143 user_action.getParameter<std::string>("class_name"),
144 user_action.getParameter<std::string>("instance_name"), user_action);
145 for (auto& type : ua->getTypes()) {
146 if (type == simcore::TYPE::RUN) {
147 run_action->registerAction(ua.get());
148 } else if (type == simcore::TYPE::EVENT) {
149 event_action->registerAction(ua.get());
150 } else if (type == simcore::TYPE::TRACKING) {
151 tracking_action->registerAction(ua.get());
152 } else if (type == simcore::TYPE::STEPPING) {
153 stepping_action->registerAction(ua.get());
154 } else if (type == simcore::TYPE::STACKING) {
155 stacking_action->registerAction(ua.get());
156 } else {
157 EXCEPTION_RAISE("ActionType", "Action type does not exist.");
158 }
159 }
160 }

163void RunManager::TerminateOneEvent() {
164 // have geant4 do its own thing
165 G4RunManager::TerminateOneEvent();
166
167 // go through the processes attached to the electron and
168 // reactivate any process that contains the G4DarkBremmstrahlung name
169 // this covers both cases where the process is biased and not
170 static auto reactivate_dark_brem = [](G4ProcessManager* pman) {
171 for (int i_proc{0}; i_proc < pman->GetProcessList()->size(); i_proc++) {
172 G4VProcess* p{(*(pman->GetProcessList()))[i_proc]};
173 if (p->GetProcessName().contains(G4DarkBremsstrahlung::PROCESS_NAME)) {
174 pman->SetProcessActivation(p, true);
175 break;
176 }
177 }
178 };
179
180 reactivate_dark_brem(G4Electron::Definition()->GetProcessManager());
181
182 if (this->GetVerboseLevel() > 1) {
183 std::cout << "[ RunManager ] : "
184 << "Reset the dark brem process (if it was activated)."
185 << std::endl;
186 }
187}
