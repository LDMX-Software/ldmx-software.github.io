24 void TargetDarkBremFilter::stepping(const G4Step* step) {
25  // don't process if event is already aborted
26  if (G4EventManager::GetEventManager()->GetConstCurrentEvent()->IsAborted())
27  return;
28 
29  // Get the track associated with this step.
30  auto track{step->GetTrack()};
31 
32  // Leave if track is not an electron
33  auto particle_def{track->GetParticleDefinition()};
34  if (particle_def != G4Electron::Electron()) {
35  if (particle_def == simcore::darkbrem::G4APrime::APrime() and
36  track->GetCurrentStepNumber() == 1) {
49  if (isOutsideTargetRegion(track->GetLogicalVolumeAtVertex()))
50  AbortEvent("A' was not created within target.");
51  } // first step of A'
52  return;
53  }
54 
55  // Leave if track is not primary
56  if (track->GetParentID() != 0) return;
57 
58  // Leave if track is not in target region
59  if (isOutsideTargetRegion(track->GetVolume())) return;
60 
61  if (isOutsideTargetRegion(track->GetNextVolume()) // leaving target region
62  or
63  track->GetTrackStatus() == fStopAndKill // stopping within target region
64  or track->GetKineticEnergy() == 0. // stopping within target region
65  ) {
66  // Get the electron secondries
67  const std::vector<G4Track*>* secondaries{step->GetSecondary()};
68  if (not secondaries or secondaries->size() == 0) {
69  AbortEvent("Primary electron did not create secondaries.");
70  return;
71  } else {
72  // check secondaries to see if we made a dark brem
73  for (auto& secondary_track : *secondaries) {
74  if (secondary_track->GetParticleDefinition() ==
75  simcore::darkbrem::G4APrime::APrime()) {
76  // we found an A', woo-hoo!
77 
78  if (secondary_track->GetTotalEnergy() < threshold_) {
79  AbortEvent(
80  "A' was not created with total energy above input threshold.");
81  return;
82  }
83 
84  if (isOutsideTargetRegion(secondary_track->GetVolume())) {
85  AbortEvent("A' was not created within target.");
86  return;
87  }
88 
89  /* debug message
90  std::cout << "[ TargetDarkBremFilter ] : "
91  << "(" <<
92  G4EventManager::GetEventManager()->GetConstCurrentEvent()->GetEventID()
93  << ") "
94  << "Found an A' weighted " << secondary_track->GetWeight()
95  << " originating in " <<
96  secondary_track->GetVolume()->GetLogicalVolume()->GetName()
97  << " with energy " << secondary_track->GetTotalEnergy() << " MeV."
98  << std::endl;
99  */
100 
101  // we found a good A', so we can leave
102  return;
103  } // this secondary is an A'
104  } // loop through secondaries
105  } // are there secondaries to loop through
106 
107  // got here without finding A'
108  AbortEvent("Primary electron did not create A'.");
109  return;
110  } // should check secondaries of primary
111 
112  // get here if we are the primary in the target region,
113  // but shouldn't check the secondaries yet
114  return;
115 }
