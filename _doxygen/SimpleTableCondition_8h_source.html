101class HomogenousTableCondition : public BaseTableCondition {
102 public:
103 HomogenousTableCondition(const std::string& name,
104 const std::vector<std::string>& columns)
105 : BaseTableCondition(name, columns) {}
106
107 virtual ~HomogenousTableCondition() {}
108
112 void clear() {
113 keys_.clear();
114 values_.clear();
115 }
116
118 void add(unsigned int id, const std::vector<T>& values) {
119 if (values.size() != columnCount_) {
120 EXCEPTION_RAISE("ConditionsException",
121 getName() + ": Attempted to insert a row with " +
122 std::to_string(values.size()) +
123 " columns into a table with " +
124 std::to_string(columnCount_) + " columns");
125 }
126 std::size_t loc = findKey(id);
127 if (loc != getRowCount()) {
128 EXCEPTION_RAISE("ConditionsException",
129 "Attempted to add condition in " + getName() +
130 " for existing id " + std::to_string(id));
131 }
132 loc = findKeyInsert(id); // where to put it
133
134 // insert into the keys
135 keys_.insert(keys_.begin() + loc, id);
136 // insert into the values
137 values_.insert(values_.begin() + loc * columnCount_, values.begin(),
138 values.end());
139 }
140
145 T get(unsigned int id, unsigned int col) const {
146 std::size_t irow = findKey(id);
147 if (col >= columnCount_ || irow == getRowCount()) { // raise exception
148 EXCEPTION_RAISE("ConditionsException",
149 "No such column " + std::to_string(col) + " or id " +
150 std::to_string(id));
151 }
152 return values_[irow * columnCount_ + col];
153 }
154
159 std::pair<unsigned int, std::vector<T> > getRow(unsigned int irow) const {
160 if (irow >= getRowCount()) { // raise exception
161 EXCEPTION_RAISE("ConditionsException",
162 "Row out of range: " + std::to_string(irow));
163 }
164 std::vector<T> rv(&(values_[irow * columnCount_]),
165 &(values_[(irow + 1) * columnCount_]));
166 return std::pair<unsigned int, std::vector<T> >(keys_[irow], rv);
167 }
168
175 T getByName(unsigned int id, const std::string& colname) const {
176 return get(id, getColumnNumber(colname));
177 }
178
179 private:
180 std::vector<T> values_; // unrolled array
181};
