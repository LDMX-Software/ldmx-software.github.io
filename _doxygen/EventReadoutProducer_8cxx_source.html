27 // initialize QIE object for linearizing ADCs
28 SimQIE qie;
29
30 const auto digis{event.getCollection<trigscint::TrigScintQIEDigis>(
31 inputCollection_, inputPassName_)};
32
33 std::vector<trigscint::EventReadout> channelReadoutEvents;
34 for (const auto &digi : digis) {
35 trigscint::EventReadout outEvent;
36 auto adc{digi.getADC()};
37 auto tdc{digi.getTDC()};
38
39 //copy over from qie digi for convenience
40 outEvent.setChanID(digi.getChanID());
41 outEvent.setElecID(digi.getElecID());
42 outEvent.setTimeSinceSpill(digi.getTimeSinceSpill());
43 // elecID increases monotonically with 8 channels per fiber
44 outEvent.setFiberNb( (int)(digi.getElecID()/8) );
45 if ( outEvent.getFiberNb() == fiberToShift_ )
46 outEvent.setTimeOffset( timeShift_);
47
48 outEvent.setADC(adc);
49 outEvent.setTDC(tdc);
50 std::vector <float> charge;
51 std::vector <float> chargeErr;
52
53 float avgQ = 0;
54 float totPosQ = 0;
55 int iS=0;
56 int nPos=0;
57 float earlyPed=0;
58 for (auto& val: adc) {
59 float Q=qie.ADC2Q(val);
60 charge.push_back( Q );
61 chargeErr.push_back( qie.QErr(Q) );
62 avgQ+=Q; //charge.back();
63 if ( Q > 0 ) {
64 totPosQ+=Q;
65 nPos++;
66 }
67 if (verbose_)
68 ldmx_log(debug) << "got adc value " << val << " and charge " << Q; //qie.ADC2Q(val);
69 if (iS < nPedSamples_)
70 earlyPed+=Q;
71 iS++;
72 }
73 outEvent.setQ(charge); //set in proper order before sorting
74 outEvent.setQError(chargeErr); //set in proper order before sorting
75 earlyPed/=nPedSamples_;
76 outEvent.setEarlyPedestal(earlyPed);
77 //now calculate the pedestal as the average of the middle half of the sorted vector
78 float ped = 0;
79 std::sort(charge.begin(), charge.end());
80 int quartLength=(int)charge.size()/4;
81 for (int i = quartLength; i < 3*quartLength ; i++) {
82 ped+=charge[i];
83 }
84 ped/=2*quartLength;
85 //median: technically only true for odd number of elements but good enough
86 float medQ=charge[ (int)charge.size()/2 ];
87 float minQ=charge[0];
88 float maxQ=charge[charge.size() -1];
89
90 outEvent.setTotQ(totPosQ-nPos*ped); //store (event) ped subtracted total charge, before dividing by N
91 // outEvent.setTotQ(totPosQ-adc.size()*ped); //store (event) ped subtracted total charge, before dividing by N
92 // outEvent.setTotQ(avgQ-adc.size()*ped); //store (event) ped subtracted total charge, before dividing by N
93 avgQ/=adc.size();
94 outEvent.setPedestal(ped);
95 outEvent.setAvgQ(avgQ);
96 outEvent.setMedQ(medQ);
97 outEvent.setMinQ(minQ);
98 outEvent.setMaxQ(maxQ);
99
100
101 //and the noise as the RMSE of that
102 float diffSq = 0;
103 for (int i = quartLength; i < 3*quartLength ; i++) {
104 // for (auto& val: charge) {
105 diffSq+=(charge[i]-ped)*(charge[i]-ped);
106 }
107 diffSq/=2*quartLength; //adc.size();
108 outEvent.setNoise( sqrt(diffSq) );
109
110
111 uint flagSpike = (maxQ/outEvent.getTotQ() > 0.8 ); // skip "unnaturally" narrow hits
112 uint flagPlateau = ( fabs(ped) > 15 ); //threshold // //skip events that have strange plateaus
113 uint flagLongPulse = 0; //might be easier to deal with in hit reconstruction directly. could copy channel flags to hit flags and add this one there
114 uint flagOscillation = ( fabs(avgQ/ped) > 3./4 );
115 /*
116if ( fabs(chan.getPedestal()) < 15 //threshold // //skip events that have strange plateaus
117 // && (chan.getAvgQ()/chan.getPedestal()<0.8) //skip events that have strange oscillations
118 && 1 < nSampAboveThr && nSampAboveThr < 10 ) // skip one-time sample flips and long weird pulses
119 */
120 uint flag=flagSpike+2*flagPlateau+4*flagLongPulse+8*flagOscillation;
121 ldmx_log(debug) << "Got quality flag " << flag << " made up of (spike/plateau/long pulse/oscillation) " << flagSpike << "+" << flagPlateau << "+" << flagLongPulse << "+" << flagOscillation ;
122 outEvent.setQualityFlag( flag );
123
124
125 // if (ped > 15 )
126 // continue;
127 ldmx_log(debug) << "In event " << event.getEventHeader().getEventNumber() <<
128 ", set pedestal = " << outEvent.getPedestal() << //ped <<
129 " fC, noise = " << outEvent.getNoise() << " fC for channel "<<
130 outEvent.getChanID();
131
132 channelReadoutEvents.push_back( outEvent );
133 }
134 // Create the container to hold the
135 // digitized trigger scintillator hits.
136
137 event.add(outputCollection_, channelReadoutEvents);
138 ldmx_log(debug) << "\n";
139}
