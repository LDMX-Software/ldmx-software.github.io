37void EventReadoutProducer::produce(framework::Event &event) {
38 // initialize QIE object for linearizing ADCs
39 SimQIE qie;
40
41 const auto digis{event.getCollection<trigscint::TrigScintQIEDigis>(
42 inputCollection_, inputPassName_)};
43
44 std::vector<trigscint::EventReadout> channelReadoutEvents;
45 for (const auto &digi : digis) {
46 trigscint::EventReadout outEvent;
47 auto adc{digi.getADC()};
48 auto tdc{digi.getTDC()};
49
50 //copy over from qie digi for convenience
51 outEvent.setChanID(digi.getChanID());
52 outEvent.setElecID(digi.getElecID());
53 outEvent.setTimeSinceSpill(digi.getTimeSinceSpill());
54 // elecID increases monotonically with 8 channels per fiber
55 outEvent.setFiberNb( (int)(digi.getElecID()/8) );
56 if ( outEvent.getFiberNb() == fiberToShift_ )
57 outEvent.setTimeOffset( timeShift_);
58
59 outEvent.setADC(adc);
60 outEvent.setTDC(tdc);
61 std::vector <float> charge;
62 std::vector <float> chargeErr;
63
64 float avgQ = 0;
65 float totPosQ = 0;
66 int iS=0;
67 int nPos=0;
68 float earlyPed=0;
69 for (auto& val: adc) {
70 float Q=qie.ADC2Q(val);
71 charge.push_back( Q );
72 chargeErr.push_back( qie.QErr(Q) );
73 avgQ+=Q; //charge.back();
74 if ( Q > 0 ) {
75 totPosQ+=Q;
76 nPos++;
77 }
78 if (verbose_)
79 ldmx_log(debug) << "got adc value " << val << " and charge " << Q; //qie.ADC2Q(val);
80 if (iS < nPedSamples_)
81 earlyPed+=Q;
82 iS++;
83 }
84 outEvent.setQ(charge); //set in proper order before sorting
85 outEvent.setQError(chargeErr); //set in proper order before sorting
86 earlyPed/=nPedSamples_;
87 outEvent.setEarlyPedestal(earlyPed);
88
89 // oscillation check. for this the pulse charge needs to be in order, so set this up now
90 // the period is 4.
91 // ansatz: an oscillation is a repeated shape. normalize to maxq=1, in every interval of 4 samples
92 // if after normalization the same numbers are repeated, it's an oscillation
93 // edge case: multiple single PE peaks with that repetition. we probably don't need to keep those anyway
94 if (verbose_)
95 ldmx_log(debug) << "going into oscillations check ";
96 std::vector <float> chargeCheck = {NULL};
97 float minCharge=10; //no point in looking at oscillations just around the pedestal. nearest edge is 10.35 fC //ADC=0 corresponds to -16 fC.
98 for (int i=3; i<charge.size()-4; i++) {
99 float maxSamp=minCharge;
100 for (int iQ=0; iQ<4; iQ++) { //find the local max in the 4 samples
101 // ldmx_log(debug) << "got charge " << charge[i+iQ];
102 if (charge[i+iQ] > maxSamp)
103 maxSamp=charge[i+iQ];
104 }
105 if (verbose_)
106 ldmx_log(debug) << "got max charge " << maxSamp;
107 for (int iQ=0; iQ<4; iQ++) //store the locally normalized numbers. even if the period is 5 this should work for a while
108 chargeCheck.push_back(charge[i+iQ]/maxSamp);
109 i+=3; //to increment by 4, do 3 here and 1 in the loop
110 }
111
112
113 //now calculate the pedestal as the average of the middle half of the sorted vector
114 float ped = 0;
115 std::sort(charge.begin(), charge.end());
116 int pedLength=(int)charge.size()/5; //actually pulse can be up to 12 samples = 2/5*30
117 int pedOffset=2; //but still skip the lowest (and highest) few
118 // for (int i = pedLength; i < 3*pedLength ; i++) { //use 2nd and third 5th
119 for (int i = pedOffset; i < 2*pedLength+pedOffset ; i++) { //use 1st and 2nd 5th
120 ped+=charge[i];
121 }
122 ped/=2*pedLength;
123 //median: technically only true for odd number of elements but good enough
124 float medQ=charge[ (int)charge.size()/2 ];
125 float minQ=charge[0];
126 float maxQ=charge[charge.size() -1];
127
128 outEvent.setTotQ(totPosQ);//-nPos*ped); //store (event) ped subtracted total charge, before dividing by N --> actually, ped subtraction makes it confusing
129 // outEvent.setTotQ(totPosQ-adc.size()*ped); //store (event) ped subtracted total charge, before dividing by N
130 // outEvent.setTotQ(avgQ-adc.size()*ped); //store (event) ped subtracted total charge, before dividing by N
131 avgQ/=adc.size();
132 outEvent.setPedestal(ped);
133 outEvent.setAvgQ(avgQ);
134 outEvent.setMedQ(medQ);
135 outEvent.setMinQ(minQ);
136 outEvent.setMaxQ(maxQ);
137
138 //and the noise as the RMSE of that, same interval as pedestal
139 float diffSq = 0;
140 // for (int i = pedLength; i < 3*pedLength ; i++) {
141 for (int i = pedOffset; i < 2*pedLength+pedOffset ; i++) {
142 diffSq+=(charge[i]-ped)*(charge[i]-ped);
143 }
144 diffSq/=2*pedLength; //adc.size();
145 outEvent.setNoise( sqrt(diffSq) );
146
147 //oscillation check
148 uint flagOscillation = 0;
149 //no need to run tedious oscillation check for all-neg channels
150 if (maxQ > minCharge) {
151 int maxID=0;
152 //find the first occurence of a local max
153 for (int i=0; i<chargeCheck.size()-4; i++) {
154 if (chargeCheck[i]==1) { //an actual local max has been normalised by its own value
155 maxID=i;
156 // ldmx_log(debug) << "storing max index " <<maxID <<" and size of vector is " << chargeCheck.size()-4;
157 break;
158 }
159 }
160 int lastMatchSample=0;
161 //start from local max
162 bool doBreak=false;
163 for (int i=maxID; i<chargeCheck.size()-4; i++) {
164 if (verbose_)
165 ldmx_log(debug) << "Checking how many matching groups of four we can find, starting at index " << i;
166
167 for (int iQ=0; iQ<4; iQ++) { //check if they are consistently close
168 if (verbose_)
169 ldmx_log(debug) << "Comparing " << chargeCheck[i+iQ] << " (sample "<< i+iQ << ") to " << chargeCheck[i+4+iQ] << " (sample "<< i+4+iQ << "), ratio is " << chargeCheck[i+iQ]/chargeCheck[i+4+iQ];
170 //we can be generous in these crietira since we will require an unbroken suite of 8 matches to call it oscillation
171 if ( fabs(chargeCheck[i+iQ]/chargeCheck[i+4+iQ]-1) <0.5 || //need this tolerance to be kind of large, most actual peaks won't pass it by far anyway.
172 (chargeCheck[i+4+iQ]<0.01 && fabs(chargeCheck[i+iQ]/chargeCheck[i+4+iQ]) <5 ) ) // for very small numbers, one ADC difference can be a factor 3 so add some margin
173 lastMatchSample=i+iQ;
174 else {
175 if (verbose_)
176 ldmx_log(debug) << "Oscillation check for channel " << digi.getChanID() << " breaking at time sample " << i+iQ;
177 doBreak=true; //break outer loop
178 break; //break this loop
179 }
180 }
181 if (doBreak) {
182 break;
183 }
184 if (verbose_)
185 ldmx_log(debug) << "Current lastMatchSample " << lastMatchSample;
186 if (lastMatchSample-maxID>=2*4 ) { //we had at least a couple of oscillations (2nd period was fully matched by third)
187 flagOscillation = 1; //there is another check possible later too, commented for now
188 break; //we've seen what we need to see
189 }
190 i+=3;
191 }
192 } //if positive maxQ
193
194
195 // //use the top and bottom ends of the sorted q as another oscillation catcher: we don't expect that the top values will be high and basically identical unless they are from an oscillation
196 int nHigh=0;
197 int quartLength=(int)charge.size()/4;
198 for (int i = 4*quartLength-2; i >= 3*quartLength ; i--) { //maxQ is already at last index
199 if (charge[i]/maxQ > 0.66)
200 nHigh++;
201 }
202
203
204 uint flagSpike = (maxQ/outEvent.getTotQ() > 0.9 ) || (charge[charge.size()-2]/maxQ < 0.25) ; // skip "unnaturally" narrow hits (all charge in one sample or huge drop to second highest)
205 uint flagPlateau = ( ped>15 || nHigh >= 5); //( fabs(ped) > 15 ); //threshold // //skip events that have strange plateaus
206 uint flagLongPulse = 0; //easier to deal with in hit reconstruction directly. copy channel flags to hit flags and add this one there
207 uint flagNoise= (outEvent.getNoise() > 3.5 || outEvent.getNoise()==0); // =0 is typically from funky events but could be too harsh maybe
208 /* //let this wait for now
209 //if we have many high counts, a small event pedestal (where they weren't included), and an avgQ ~ maxQ/nHigh, then this is an oscillation
210 if ( (quartLength-nHigh<2 && ped<10) || fabs( maxQ/(avgQ*nHigh)-1 ) <0.1 )
211 flagOscillation=1;
212*/
213
214
215 /* //this seems to not catch what we want it to
216if ( fabs(chan.getPedestal()) < 15 //threshold // //skip events that have strange plateaus
217 // && (chan.getAvgQ()/chan.getPedestal()<0.8) //skip events that have strange oscillations
218 && 1 < nSampAboveThr && nSampAboveThr < 10 ) // skip one-time sample flips and long weird pulses
219 */
220 uint flag=flagSpike+2*flagPlateau+4*flagLongPulse+8*flagOscillation+16*flagNoise;
221 ldmx_log(debug) << "Got quality flag " << flag << " made up of (spike/plateau/long pulse/oscillation/noise) " << flagSpike << "+" << flagPlateau << "+" << flagLongPulse << "+" << flagOscillation << "+" << flagNoise ;
222 outEvent.setQualityFlag( flag );
223
224
225 // if (ped > 15 )
226 // continue;
227 ldmx_log(debug) << "In event " << event.getEventHeader().getEventNumber() <<
228 ", set pedestal = " << outEvent.getPedestal() << //ped <<
229 " fC, noise = " << outEvent.getNoise() << " fC for channel "<<
230 outEvent.getChanID();
231
232 channelReadoutEvents.push_back( outEvent );
233 }
234 // Create the container to hold the
235 // digitized trigger scintillator hits.
236
237 event.add(outputCollection_, channelReadoutEvents);
238 ldmx_log(debug) << "\n";
239}
