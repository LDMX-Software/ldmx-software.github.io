15LHEParticle::LHEParticle(std::string& line) {
16 std::istringstream iss(line);
17 std::vector<std::string> tokens;
18 do {
19 std::string elem;
20 iss >> elem;
21 if (elem.size() != 0) {
22 tokens.push_back(elem);
23 }
24 } while (iss);
25
26 if (tokens.size() != 13) {
27 EXCEPTION_RAISE("TokenNum",
28 "Wrong number of tokens in LHE particle record.");
29 }
30
31 idup_ = atof(tokens[0].c_str());
32 istup_ = atoi(tokens[1].c_str());
33 mothup_[0] = atoi(tokens[2].c_str());
34 mothup_[1] = atoi(tokens[3].c_str());
35 icolup_[0] = atoi(tokens[4].c_str());
36 icolup_[1] = atoi(tokens[5].c_str());
37 pup_[0] = atof(tokens[6].c_str());
38 pup_[1] = atof(tokens[7].c_str());
39 pup_[2] = atof(tokens[8].c_str());
40 pup_[3] = atof(tokens[9].c_str());
41 pup_[4] = atof(tokens[10].c_str());
42 vtimup_ = atof(tokens[11].c_str());
43 spinup_ = atof(tokens[12].c_str());
44
45 mothers_[0] = NULL;
46 mothers_[1] = NULL;
47}
