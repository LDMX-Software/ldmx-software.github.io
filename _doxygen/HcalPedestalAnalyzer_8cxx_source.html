63void HcalPedestalAnalyzer::analyze(const framework::Event& event) {
64 auto const& digis{
65 event.getObject<ldmx::HgcrocDigiCollection>(input_name_, input_pass_)};
66
67 for (std::size_t i_digi{0}; i_digi < digis.size(); i_digi++) {
68 auto d{digis.getDigi(i_digi)};
69 ldmx::HcalDigiID detid(d.id());
70
71 Channel& chan=pedestal_data_[detid];
72
73 bool has_tot=false;
74 bool has_toa=false;
75 bool has_under=false;
76 bool has_over=false;
77
78 for (int i = 0; i < digis.getNumSamplesPerDigi(); i++) {
79 if (d.at(i).tot()>0) has_tot=true;
80 if (d.at(i).toa()>0) has_toa=true;
81 if (d.at(i).adc_t()<low_cutoff_) has_under=true;
82 if (d.at(i).adc_t()>high_cutoff_) has_over=true;
83 }
84
85 if (has_tot && filter_noTOT) chan.rejects[0]++;
86 if (has_toa && filter_noTOA) chan.rejects[1]++;
87 if (has_under) chan.rejects[2]++;
88 if (has_over) chan.rejects[3]++;
89
90 if (has_tot && filter_noTOT) continue; // ignore this
91 if (has_toa && filter_noTOA) continue; // ignore this
92 if (has_under) continue; // ignore this, set threshold to zero to disable requirement
93 if (has_over) continue; // ignore this, set threshold larger than 1024 to disable requirement
94
95 for (int i = 0; i < digis.getNumSamplesPerDigi(); i++) {
96 int adc=d.at(i).adc_t();
97
98 chan.sum+=adc;
99 chan.sum_sq+=adc*adc;
100 chan.entries++;
101 if (chan.hist) chan.hist->Fill(adc);
102 else if (make_histos_)
103 chan.adcs.push_back(adc);
104 }
105
106 // histogram-related business
107 if (make_histos_ && !chan.hist && chan.entries>250)
108 create_and_fill(chan,detid);
109 }

112void HcalPedestalAnalyzer::create_and_fill(Channel& chan, ldmx::HcalDigiID detid) {
113 if (chan.entries==0) return;
114
115 TDirectory* hdir=getHistoDirectory();
116 hdir->cd();
117 char hname[120];
118 sprintf(hname,"pedestal_%d_%d_%d_%d",detid.section(),detid.layer(),detid.strip(),detid.end());
119 // logic: 100 bins to +/- 5 sigma based on first 250 events.
120 double mean=(chan.sum*1.0)/chan.entries;
121 double rms=sqrt(chan.sum_sq/chan.entries-mean*mean);
122 if (rms*5<50)
123 chan.hist=new TH1D(hname,hname,30,int(mean)-15,int(mean)+15);
124 else chan.hist=new TH1D(hname,hname,100,mean-5*rms,mean+5*rms);
125 for (auto x: chan.adcs)
126 chan.hist->Fill(x);
127 chan.adcs.clear();
128}

130void HcalPedestalAnalyzer::onProcessEnd() {
131 FILE* fout=fopen(output_file_.c_str(),"w");
132
133 time_t t = time(NULL);
134 struct tm *gmtm = gmtime(&t);
135 char times[1024];
136 strftime(times, sizeof(times), "%Y-%m-%d %H:%M:%S GMT", gmtm);
137 fprintf(fout,"# %s\n",comments_.c_str());
138 fprintf(fout,"# Produced %s\n",times);
139 fprintf(fout,"DetID,PEDESTAL_ADC,PEDESTAL_RMS_ADC\n");
140
141 for (auto ichan: pedestal_data_) {
142 if (ichan.second.entries==0) {
143 std::cout << "All entries filtered for " <<ichan.first << " for TOT " << ichan.second.rejects[0] << " for TOA " << ichan.second.rejects[1]
144 << " for underthreshold " << ichan.second.rejects[2]<< " for overthreshold " << ichan.second.rejects[3] << std::endl;
145 continue; // all entries were filtered out
146 }
147
148 double mean=ichan.second.sum*1.0/ichan.second.entries;
149 double rms=sqrt(ichan.second.sum_sq/ichan.second.entries-mean*mean);
150 fprintf(fout,"0x%08x,%9.3f,%9.3f\n",ichan.first.raw(),mean,rms);
151 }
152
153 fclose(fout);
154}
