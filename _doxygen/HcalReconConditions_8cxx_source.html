19bool HcalReconConditions::is_adc(const ldmx::HcalDigiID& id,
20 double sum_tot) const {
21 // check if the linearization has been done correctly
22 // a non-zero flag value is implicitly converted to true
23 if (totCalib(id, i_flagged) == 1) {
24 return true;
25 }
26
27 // if we are in ADC range (which was used as a reference in linearization),
28 // we use ADC
29 if (sum_tot < totCalib(id, i_lower_offset)) {
30 return true;
31 }
32
33 return false;
34}

36double HcalReconConditions::linearize(const ldmx::HcalDigiID& id,
37 double sum_tot) const {
38 // we know we have a linearization fit and are in TOT range,
39 // the lower side of TOT needs to be linearized with a specialized power law
40 if (sum_tot < totCalib(id, i_cut_point_tot)) {
41 return pow((sum_tot - totCalib(id, i_lower_offset)) /
42 totCalib(id, i_low_slope),
43 1 / totCalib(id, i_low_power)) +
44 totCalib(id, i_tot_not);
45 }
46
47 // we know sum_tot is >= lower offset and >= tot cut
48 // higher tot, linearized with adc using a simple linear mapping
49 return (sum_tot - totCalib(id, i_high_offset)) * totCalib(id, i_high_slope);
50}
