140  int getModuleID(double x, double y) const {
141  int bestID = -1;
142  double bestDist = 1E6;
143  for(auto const& module : modulePositionMap_) {
144  int mID = module.first;
145  double mX = module.second.first;
146  double mY = module.second.second;
147  double dist = sqrt( (x-mX)*(x-mX) + (y-mY)*(y-mY) );
148  if(dist < moduler_) return mID;
149  if(dist < bestDist) { bestID = mID; bestDist = dist; }
150  }
151  return bestID;
152  }
