109  int getModuleID(double x, double y) const {
110  int bestID = -1;
111  double bestDist = 1E6;
112  for(auto const& module : modulePositionMap_) {
113  int mID = module.first;
114  double mX = module.second.first;
115  double mY = module.second.second;
116  double dist = sqrt( (x-mX)*(x-mX) + (y-mY)*(y-mY) );
117  if(dist < moduler_) return mID;
118  if(dist < bestDist) { bestID = mID; bestDist = dist; }
119  }
120  return bestID;
121  }
