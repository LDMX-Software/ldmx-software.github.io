38  HcalID HcalDigiProducer::generateRandomID(HcalID::HcalSection sec){
39  int layer, strip;
40  HcalID::HcalSection section=sec;
41  if( sec == HcalID::BACK ){
42  layer=random_->Integer(NUM_BACK_HCAL_LAYERS_);
43  strip=random_->Integer(STRIPS_BACK_PER_LAYER_/SUPER_STRIP_SIZE_);
44  }else if( sec == HcalID::TOP || sec == HcalID::BOTTOM ){
45  layer=random_->Integer(NUM_SIDE_TB_HCAL_LAYERS_);
46  section=HcalID::HcalSection(random_->Integer(2)+1);
47  strip=random_->Integer(STRIPS_SIDE_TB_PER_LAYER_);
48  }else if( sec == HcalID::LEFT || sec == HcalID::RIGHT ){
49  layer=random_->Integer(NUM_SIDE_LR_HCAL_LAYERS_);
50  section=HcalID::HcalSection(random_->Integer(2)+3);
51  strip=random_->Integer(STRIPS_SIDE_LR_PER_LAYER_);
52  }else
53  std::cout << "WARNING [HcalDigiProducer::generateRandomID]: HcalSection is not known" << std::endl;
54 
55  return HcalID(section, layer, strip);
56  }

202  }
