24 // Function to convert charge to ADC count
25 // Working: The method checks in which QIE subrange does the charge lie,
26 // applies a corresponding gain to it and digitizes it.
27 int SimQIE::Q2ADC(float Charge) {
28  float qq = gain_ * Charge; // including QIE gain
29  if (isnoise_) qq += trg_->Gaus(mu_, sg_); // Adding gaussian random noise.
30 
31  if (qq <= edges_[0]) return 0;
32  if (qq >= edges_[16]) return 255;
33 
34  int ID = 8;
35  int a = 0;
36  int b = 16;
37 
38  // Binary search to find the subrange
39  while (b - a != 1) {
40  if (qq > edges_[(a + b) / 2]) {
41  a = (a + b) / 2;
42  } else
43  b = (a + b) / 2;
44  }
45  return 64 * (int)(a / 4) + nbins_[a % 4] +
46  floor((qq - edges_[a]) / sense_[a]);
47 }

49 // Function to convert ADCs back to charge
50 // The method checks to which QIE subrange does the ADC correspnd to
51 // and returns the mean charge of the correspnding bin in the subrange
52 float SimQIE::ADC2Q(int ADC) {
53  if (ADC <= 0) return -16;
54  if (ADC >= 255) return 350000;
55 
56  int rr = ADC / 64; // range
57  int v1 = ADC % 64; // temp. var
58  int ss = 0; // sub range
59 
60  for (int i = 1; i < 4; i++) { // to get the subrange
61  if (v1 > nbins_[i]) ss++;
62  }
63  int cc = 64 * rr + nbins_[ss];
64  float temp = edges_[4 * rr + ss] + (v1 - nbins_[ss]) * sense_[4 * rr + ss] +
65  sense_[4 * rr + ss] / 2;
66  return (temp / gain_);
67 }

70 float SimQIE::QErr(float Q) {
71  if (Q <= edges_[0]) return 0;
72  if (Q >= edges_[16]) return 0;
73 
74  int ID = 8;
75  int a = 0;
76  int b = 16;

78  if (Q > edges_[(a + b) / 2])
79  a = (a + b) / 2;
80  else
81  b = (a + b) / 2;
82  }
