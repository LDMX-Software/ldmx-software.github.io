54const ConditionsObject* Conditions::getConditionPtr(
55 const std::string& condition_name) {
56 const ldmx::EventHeader& context = *(process_.getEventHeader());
57 auto cacheptr = cache_.find(condition_name);
58
59 if (cacheptr == cache_.end()) {
60 auto copptr = providerMap_.find(condition_name);
61
62 if (copptr == providerMap_.end()) {
63 EXCEPTION_RAISE(
64 "ConditionUnavailable",
65 std::string("No provider is available for : " + condition_name));
66 }
67
68 std::pair<const ConditionsObject*, ConditionsIOV> cond =
69 copptr->second->getCondition(context);
70
71 if (!cond.first) {
72 EXCEPTION_RAISE(
73 "ConditionUnavailable",
74 std::string("Null condition returned for requested item : " +
75 condition_name));
76 }
77
78 // first request, create a cache entry
79 CacheEntry ce;
80 ce.iov = cond.second;
81 ce.obj = cond.first;
82 ce.provider = copptr->second;
83 cache_[condition_name] = ce;
84 return ce.obj;
85 } else {
87 if (cacheptr->second.iov.validForEvent(context))
88 return cacheptr->second.obj;
89 else {
90 // if not, we release the old object
91 cacheptr->second.provider->releaseConditionsObject(cacheptr->second.obj);
92 // now ask for a new one
93 std::pair<const ConditionsObject*, ConditionsIOV> cond =
94 cacheptr->second.provider->getCondition(context);
95
96 if (!cond.first) {
97 std::stringstream s;
98 s << "Unable to update condition '" << condition_name << "' for event "
99 << context.getEventNumber() << " run " << context.getRun();
100 if (context.isRealData())
101 s << " DATA";
102 else
103 s << " MC";
104 EXCEPTION_RAISE("ConditionUnavailable", s.str());
105 }
106 cacheptr->second.iov = cond.second;
107 cacheptr->second.obj = cond.first;
108 return cond.first;
109 }
110 }
111}
