41  auto cacheptr = cache_.find(condition_name);
42 
43  if (cacheptr == cache_.end()) {
44  auto copptr = providerMap_.find(condition_name);
45 
46  if (copptr==providerMap_.end()) {
47  EXCEPTION_RAISE("ConditionUnavailable",std::string("No provider is available for : "+condition_name));
48  }
49 
50  std::pair<const ConditionsObject*,ConditionsIOV> cond=copptr->second->getCondition(condition_name,context);
51 
52  if (!cond.first) {
53  EXCEPTION_RAISE("ConditionUnavailable",std::string("Null condition returned for requested item : "+condition_name));
54  }
55  // first request, create a cache entry
56  CacheEntry ce;
57  ce.iov=cond.second;
58  ce.obj=cond.first;
59  ce.provider=copptr->second;
60  cache_[condition_name]=ce;
61  } else {
63  if (cacheptr->second.iov.validForEvent(context)) return cacheptr->second.obj;
64  else {
65  // if not, we release the old object
66  cacheptr->second.provider->releaseConditionsObject(cacheptr->second.obj);
67  // now ask for a new one
68  std::pair<const ConditionsObject*,ConditionsIOV> cond=cacheptr->second.provider->getCondition(condition_name,context);
69 
70  if (!cond.first) {
71  std::stringstream s;

73  << condition_name << "' for event "
74  << context.getEventNumber() << " run " << context.getRun();
75  if (context.isRealData()) s << " DATA";

78  }
79  cacheptr->second.iov=cond.second;
