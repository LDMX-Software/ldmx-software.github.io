177  void EcalVetoProcessor::produce(Event& event) {
178 
179  EcalVetoResult result;
180 
181  clearProcessor();
182 
183  // Get the collection of Ecal scoring plane hits. If it doesn't exist,
184  // don't bother adding any truth tracking information.
185 
186  std::vector<double> recoilP;
187  std::vector<float> recoilPos;
188  std::vector<double> recoilPAtTarget;
189  std::vector<float> recoilPosAtTarget;
190 
191  if (event.exists("EcalScoringPlaneHits")) {
192 
193  //
194  // Loop through all of the sim particles and find the recoil electron.
195  //
196 
197  // Get the collection of simulated particles from the event
198  auto particleMap{event.getMap< int, SimParticle >("SimParticles")};
199 
200  // Search for the recoil electron
201  auto [recoilTrackID, recoilElectron] = Analysis::getRecoil(particleMap);
202 
203  // Find ECAL SP hit for recoil electron
204  auto ecalSpHits{event.getCollection< SimTrackerHit >( "EcalScoringPlaneHits" )};
205  float pmax = 0;
206  for ( SimTrackerHit &spHit : ecalSpHits ) {
207 
208  if (spHit.getLayerID() != 31 || spHit.getMomentum()[2] <= 0) continue;
209 
210  if (spHit.getTrackID() == recoilTrackID) {
211  if(sqrt(pow(spHit.getMomentum()[0],2) + pow(spHit.getMomentum()[1],2) + pow(spHit.getMomentum()[2],2)) > pmax) {
212  recoilP = spHit.getMomentum();
213  recoilPos = spHit.getPosition();
214  pmax = sqrt(pow(recoilP[0],2) + pow(recoilP[1],2) + pow(recoilP[2],2));
215  }
216  }
217  }
218 
219  // Find target SP hit for recoil electron
220  if (event.exists("TargetScoringPlaneHits")) {
221  std::vector< SimTrackerHit > targetSpHits = event.getCollection< SimTrackerHit >( "TargetScoringPlaneHits" );
222  pmax = 0;
223  for ( SimTrackerHit &spHit : targetSpHits ) {
224 
225  if (spHit.getLayerID() != 1 || spHit.getMomentum()[2] <= 0) continue;
226 
227  if (spHit.getTrackID() == recoilTrackID) {
228  if(sqrt(pow(spHit.getMomentum()[0],2) + pow(spHit.getMomentum()[1],2) + pow(spHit.getMomentum()[2],2)) > pmax) {
229  recoilPAtTarget = spHit.getMomentum();
230  recoilPosAtTarget = spHit.getPosition();
231  pmax = sqrt(pow(recoilPAtTarget[0],2) + pow(recoilPAtTarget[1],2) + pow(recoilPAtTarget[2],2));
232  }
233  }
234  }
235  }
236  }
237 
238 
239  // Get projected trajectories for electron and photon
240  std::vector<XYCoords> ele_trajectory, photon_trajectory;
241  if(recoilP.size() > 0) {
242  ele_trajectory = getTrajectory(recoilP, recoilPos);
243  std::vector<double> pvec = recoilPAtTarget.size() ? recoilPAtTarget : std::vector<double>{0.0, 0.0, 0.0};
244  std::vector<float> posvec = recoilPosAtTarget.size() ? recoilPosAtTarget : std::vector<float>{0.0, 0.0, 0.0};
245  photon_trajectory = getTrajectory({-pvec[0], -pvec[1], 4000.0 - pvec[2]}, posvec);
246  }
247 
248  float recoilPMag = recoilP.size() ? sqrt(pow(recoilP[0],2) + pow(recoilP[1],2) + pow(recoilP[2],2)) : -1.0;
249  float recoilTheta = recoilPMag > 0 ? recoilP[2]/recoilPMag : -1.0;
250 
251  std::vector<double> ele_radii = radius68_thetalt10_plt500;
252  if(recoilTheta<10 && recoilPMag >= 500) ele_radii = radius68_thetalt10_pgt500;
253  else if(recoilTheta>=10 && recoilTheta<20) ele_radii = radius68_theta10to20;
254  else if(recoilTheta>=20) ele_radii = radius68_thetagt20;
255  // Use default binning
256  std::vector<double> photon_radii = radius68_thetalt10_plt500;
257 
258 
259  // Get the collection of digitized Ecal hits from the event.
260  const std::vector< EcalHit > ecalRecHits = event.getCollection< EcalHit >( "EcalRecHits" );
261 
262  int globalCentroid = GetShowerCentroidIDAndRMS( ecalRecHits , showerRMS_);
263  /* ~~ Fill the hit map ~~ O(n) */
264  fillHitMap( ecalRecHits , cellMap_);
265  bool doTight = true;
266  /* ~~ Fill the isolated hit maps ~~ O(n) */
267  fillIsolatedHitMap( ecalRecHits , globalCentroid, cellMap_, cellMapTightIso_, doTight);
268 
269  //Loop over the hits from the event to calculate the rest of the important quantities
270 
271  float wavgLayerHit = 0;
272  float xMean = 0;
273  float yMean = 0;
274 
275  // Containment variables
276  unsigned int nregions = 5;
277  std::vector<float> electronContainmentEnergy (nregions, 0.0);
278  std::vector<float> photonContainmentEnergy (nregions, 0.0);
279  std::vector<float> outsideContainmentEnergy (nregions, 0.0);
280  std::vector<int> outsideContainmentNHits (nregions, 0);
281  std::vector<float> outsideContainmentXmean (nregions, 0.0);
282  std::vector<float> outsideContainmentYmean (nregions, 0.0);
283  std::vector<float> outsideContainmentXstd (nregions, 0.0);
284  std::vector<float> outsideContainmentYstd (nregions, 0.0);
285 
286  for ( const EcalHit &hit : ecalRecHits ) {
287  //Layer-wise quantities
288  LayerCellPair hit_pair = hitToPair(hit);
289  ecalLayerEdepRaw_[hit_pair.first] = ecalLayerEdepRaw_[hit_pair.first] + hit.getEnergy();
290  if(hit.getLayer() >= 20)
291  ecalBackEnergy_ += hit.getEnergy();
292  if (maxCellDep_ < hit.getEnergy())
293  maxCellDep_ = hit.getEnergy();
294  if (hit.getEnergy() > 0) {
295  nReadoutHits_++;
296  ecalLayerEdepReadout_[hit_pair.first] += hit.getEnergy();
297  ecalLayerTime_[hit_pair.first] += (hit.getEnergy()) * hit.getTime();
298  xMean += getCellCentroidXYPair(hit_pair.second).first * hit.getEnergy();
299  yMean += getCellCentroidXYPair(hit_pair.second).second * hit.getEnergy();
300  avgLayerHit_ += hit.getLayer();
301  wavgLayerHit += hit.getLayer() * hit.getEnergy();
302  if (deepestLayerHit_ < hit.getLayer()) {
303  deepestLayerHit_ = hit.getLayer();
304  }
305  XYCoords xy_pair = getCellCentroidXYPair(hit_pair.second);
306  float distance_ele_trajectory = ele_trajectory.size() ? sqrt( pow((xy_pair.first - ele_trajectory[hit.getLayer()].first),2) + pow((xy_pair.second - ele_trajectory[hit.getLayer()].second),2) ) : -1.0;
307  float distance_photon_trajectory = photon_trajectory.size() ? sqrt( pow((xy_pair.first - photon_trajectory[hit.getLayer()].first),2) + pow((xy_pair.second - photon_trajectory[hit.getLayer()].second),2) ) : -1.0;
308  // Decide which region a hit goes into and add to sums
309  for(unsigned int ireg = 0; ireg < nregions; ireg++) {
310  if(distance_ele_trajectory >= ireg*ele_radii[hit.getLayer()] && distance_ele_trajectory < (ireg+1)*ele_radii[hit.getLayer()])
311  electronContainmentEnergy[ireg] += hit.getEnergy();
312  if(distance_photon_trajectory >= ireg*photon_radii[hit.getLayer()] && distance_photon_trajectory < (ireg+1)*photon_radii[hit.getLayer()])
313  photonContainmentEnergy[ireg] += hit.getEnergy();
314  if(distance_ele_trajectory > (ireg+1)*ele_radii[hit.getLayer()] && distance_photon_trajectory > (ireg+1)*photon_radii[hit.getLayer()]) {
315  outsideContainmentEnergy[ireg] += hit.getEnergy();
316  outsideContainmentNHits[ireg] += 1;
317  outsideContainmentXmean[ireg] += xy_pair.first*hit.getEnergy();
318  outsideContainmentYmean[ireg] += xy_pair.second*hit.getEnergy();
319  }
320  }
321  }
322  }
323 
324  for (int iLayer = 0; iLayer < ecalLayerEdepReadout_.size(); iLayer++) {
325  for (auto cell : cellMapTightIso_[iLayer]) {
326  if (cell.second > 0) {
327  summedTightIso_ += cell.second;
328  }
329  }
330  ecalLayerTime_[iLayer] = ecalLayerTime_[iLayer] / ecalLayerEdepReadout_[iLayer];
331  summedDet_ += ecalLayerEdepReadout_[iLayer];
332  }
333 
334  if (nReadoutHits_ > 0) {
335  avgLayerHit_ /= nReadoutHits_;
336  wavgLayerHit /= summedDet_;
337  xMean /= summedDet_;
338  yMean /= summedDet_;
339  } else {
340  wavgLayerHit = 0;
341  avgLayerHit_ = 0;
342  xMean = 0;
343  yMean = 0;
344  }
345 
346  for(unsigned int ireg = 0; ireg < nregions; ireg++) {
347  if(outsideContainmentEnergy[ireg] > 0) {
348  outsideContainmentXmean[ireg] /= outsideContainmentEnergy[ireg];
349  outsideContainmentYmean[ireg] /= outsideContainmentEnergy[ireg];
350  }
351  }
352 
353  // Loop over hits a second time to find the standard deviations.
354  for ( const EcalHit &hit : ecalRecHits ) {
355  LayerCellPair hit_pair = hitToPair(hit);
356  if (hit.getEnergy() > 0) {
357  xStd_ += pow((getCellCentroidXYPair(hit_pair.second).first - xMean), 2) * hit.getEnergy();
358  yStd_ += pow((getCellCentroidXYPair(hit_pair.second).second - yMean), 2) * hit.getEnergy();
359  stdLayerHit_ += pow((hit.getLayer() - wavgLayerHit), 2) * hit.getEnergy();
360  }
361  XYCoords xy_pair = getCellCentroidXYPair(hit_pair.second);
362  float distance_ele_trajectory = ele_trajectory.size() ? sqrt( pow((xy_pair.first - ele_trajectory[hit.getLayer()].first),2) + pow((xy_pair.second - ele_trajectory[hit.getLayer()].second),2) ) : -1.0;
363  float distance_photon_trajectory = photon_trajectory.size() ? sqrt( pow((xy_pair.first - photon_trajectory[hit.getLayer()].first),2) + pow((xy_pair.second - photon_trajectory[hit.getLayer()].second),2) ) : -1.0;
364  for(unsigned int ireg = 0; ireg < nregions; ireg++) {
365  if(distance_ele_trajectory > (ireg+1)*ele_radii[hit.getLayer()] && distance_photon_trajectory > (ireg+1)*photon_radii[hit.getLayer()]) {
366  outsideContainmentXstd[ireg] += pow((xy_pair.first - outsideContainmentXmean[ireg]),2) * hit.getEnergy();
367  outsideContainmentYstd[ireg] += pow((xy_pair.second - outsideContainmentYmean[ireg]),2) * hit.getEnergy();
368  }
369  }
370  }
371 
372  if (nReadoutHits_ > 0) {
373  xStd_ = sqrt (xStd_ / summedDet_);
374  yStd_ = sqrt (yStd_ / summedDet_);
375  stdLayerHit_ = sqrt (stdLayerHit_ / summedDet_);
376  } else {
377  xStd_ = 0;
378  yStd_ = 0;
379  stdLayerHit_ = 0;
380  }
381 
382  for(unsigned int ireg = 0; ireg < nregions; ireg++) {
383  if(outsideContainmentEnergy[ireg] > 0) {
384  outsideContainmentXstd[ireg] = sqrt(outsideContainmentXstd[ireg]/outsideContainmentEnergy[ireg]);
385  outsideContainmentYstd[ireg] = sqrt(outsideContainmentYstd[ireg]/outsideContainmentEnergy[ireg]);
386  }
387  }
388 
389  // end loop over sim hits
390 
391 
392  /* Code for fiducial region below */
393 
394  std::vector<float> faceXY(2);
395 
396  if (!recoilP.empty() && recoilP[2] != 0) {
397  faceXY[0] = ((223.8 - 220.0) * (recoilP[0] / recoilP[2])) + recoilPos[0];
398  faceXY[1] = ((223.8 - 220.0) * (recoilP[1] / recoilP[2])) + recoilPos[1];
399  } else {
400  faceXY[0] = -9999.0;
401  faceXY[1] = -9999.0;
402  }
403 
404  int inside = 0;
405  int up = 0;
406  int step = 0;
407  int index;
408  float cell_radius = 5.0;
409 
410  std::vector<float>::iterator it;
411  it = std::lower_bound(mapsx.begin(), mapsx.end(), faceXY[0]);
412 
413  index = std::distance( mapsx.begin(), it);
414 
415  if (index == mapsx.size()) {
416  index += -1;
417  }
418 
419  if (!recoilP.empty() && faceXY[0] != -9999.0) {
420  while (true) {
421  std::vector<double> dis(2);
422 
423  dis[0] = faceXY[0] - mapsx[index + step];
424  dis[1] = faceXY[1] - mapsy[index + step];
425 
426  float celldis = sqrt (pow(dis[0],2) + pow(dis[1],2));
427 
428  if (celldis <= cell_radius) {
429  inside = 1;
430  break;
431  }
432 
433  if ((abs(dis[0]) > 5 && up == 0) || index + step == mapsx.size()-1) {
434  up = 1;
435  step = 0;
436  } else if ((abs(dis[0]) > 5 && up == 1) || (index + step == 0 && up == 1)) {
437  break;
438  }
439 
440  if (up == 0) {
441  step += 1;
442  } else {
443  step += -1;
444  }
445  }
446  }
447 
448  result.setVariables(nReadoutHits_, deepestLayerHit_, summedDet_, summedTightIso_, maxCellDep_,

489  /* Function to calculate the energy weighted shower centroid */
490  int EcalVetoProcessor::GetShowerCentroidIDAndRMS(const std::vector<EcalHit> &ecalRecHits, double& showerRMS) {
491  XYCoords wgtCentroidCoords = std::make_pair<float, float>(0., 0.);
492  float sumEdep = 0;
493  int returnCellId = 1e6;
494  //Calculate Energy Weighted Centroid
495  for (const EcalHit &hit : ecalRecHits ) {
496  LayerCellPair hit_pair = hitToPair(hit);
497  CellEnergyPair cell_energy_pair = std::make_pair(hit_pair.second, hit.getEnergy());
498  XYCoords centroidCoords = getCellCentroidXYPair(hit_pair.second);
499  wgtCentroidCoords.first = wgtCentroidCoords.first + centroidCoords.first * cell_energy_pair.second;
500  wgtCentroidCoords.second = wgtCentroidCoords.second + centroidCoords.second * cell_energy_pair.second;
501  sumEdep += cell_energy_pair.second;
502  }
503  wgtCentroidCoords.first = (sumEdep > 1E-6) ? wgtCentroidCoords.first / sumEdep : wgtCentroidCoords.first;
504  wgtCentroidCoords.second = (sumEdep > 1E-6) ? wgtCentroidCoords.second / sumEdep : wgtCentroidCoords.second;
505  //Find Nearest Cell to Centroid
506  float maxDist = 1e6;
507  for ( const EcalHit &hit : ecalRecHits ) {
508  LayerCellPair hit_pair = hitToPair(hit);
509  XYCoords centroidCoords = getCellCentroidXYPair(hit_pair.second);
510 
511  float deltaR = pow(pow((centroidCoords.first - wgtCentroidCoords.first), 2) + pow((centroidCoords.second - wgtCentroidCoords.second), 2), .5);
512  showerRMS += deltaR * hit.getEnergy();
513  if (deltaR < maxDist) {
514  maxDist = deltaR;
515  returnCellId = hit_pair.second;
516  }
517  }
518  if (sumEdep > 0)
519  showerRMS = showerRMS / sumEdep;
520  return returnCellId;
521  }
