18double to_double_ft(uint64_t i) {
19 static const unsigned int bits = 32;
20 static const unsigned int expbits = 8;
21 long double result;
22 long long shift;
23 unsigned bias;
24 unsigned significandbits = bits - expbits - 1; // -1 for sign bit
25
26 if (i == 0) return 0.0;
27
28 // pull the significand
29 result = (i&((1LL<<significandbits)-1)); // mask
30 result /= (1LL<<significandbits); // convert back to float
31 result += 1.0f; // add the one back on
32
33 // deal with the exponent
34 bias = (1<<(expbits-1)) - 1;
35 shift = ((i>>significandbits)&((1LL<<expbits)-1)) - bias;
36 while(shift > 0) { result *= 2.0; shift--; }
37 while(shift < 0) { result /= 2.0; shift++; }
38
39 // sign it
40 result *= (i>>(bits-1))&1? -1.0: 1.0;
41
42 return result;
43}

133
134 FiberTrackerEvent() = default;

136 FiberTrackerEvent(const std::vector<uint32_t>& spill_data, std::size_t i_word) {
137 trigger_timestamp_lsb = spill_data.at(i_word);
138 trigger_timestamp_msb = spill_data.at(i_word+1);
139 event_timestamp_lsb = spill_data.at(i_word+2);
140 event_timestamp_msb = spill_data.at(i_word+3);
141 channel_hits.clear();
142 channel_hits.reserve(6);
143 for (std::size_t i{i_word+4}; i < i_word+10 and i < spill_data.size(); i++)
144 channel_hits.push_back(spill_data.at(i));
145 }

392void FiberTrackerRawDecoder::produce(framework::Event& event) {
393 // only add and fill when file able to readout packet
394 if (not spill_packet_.next(ft_event_)) {
395 // no more events in this spill
396 if (file_reader_ >> spill_packet_) {
397 spill_tree_->Fill();
398 // next spill loaded, pop its first event
399 spill_packet_.next(ft_event_);
400 } else {
401#ifdef DEBUG
402 std::cout << "no more events" << std::endl;
403#endif
404 return;
405 }
406 }
407 tree_->Fill();
408
409 event.add(output_name_+"TriggerTSLSB", ft_event_.trigger_timestamp_lsb);
410 event.add(output_name_+"TriggerTSMSB", ft_event_.trigger_timestamp_msb);
411 event.add(output_name_+"EventTSLSB", ft_event_.event_timestamp_lsb);
412 event.add(output_name_+"EventTSMSB", ft_event_.event_timestamp_msb);
413 event.add(output_name_+"Hits", ft_event_.channel_hits);
414 return;
415} // produce
