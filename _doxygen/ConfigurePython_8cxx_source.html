33static std::string getPyString(PyObject* pyObj) {
34 std::string retval;
35 PyObject* pyStr = PyUnicode_AsEncodedString(pyObj, "utf-8", "Error ~");
36 retval = PyBytes_AS_STRING(pyStr);
37 Py_XDECREF(pyStr);

71static std::map<std::string, std::any> getMembers(PyObject* object) {
72 PyObject* dictionary{PyObject_GetAttrString(object, "__dict__")};
73
74 if (dictionary == 0) {
75 if (PyDict_Check(object))
76 dictionary = object;
77 else {
78 EXCEPTION_RAISE("ObjFail", "Python Object does not have __dict__ member");
79 }
80 }
81
82 PyObject *key(0), *value(0);
83 Py_ssize_t pos = 0;
84
85 std::map<std::string, std::any> params;
86
87 while (PyDict_Next(dictionary, &pos, &key, &value)) {
88 std::string skey{getPyString(key)};
89
90 if (PyLong_Check(value)) {
91 if (PyBool_Check(value)) {
92 params[skey] = bool(PyLong_AsLong(value));
93 } else {
94 params[skey] = int(PyLong_AsLong(value));
95 }
96 } else if (PyFloat_Check(value)) {
97 params[skey] = PyFloat_AsDouble(value);
98 } else if (PyUnicode_Check(value)) {
99 params[skey] = getPyString(value);
100 } else if (PyList_Check(
101 value)) { // assume everything is same value as first value
102 if (PyList_Size(value) > 0) {
103 auto vec0{PyList_GetItem(value, 0)};
104
105 if (PyLong_Check(vec0)) {
106 std::vector<int> vals;
107
108 for (auto j{0}; j < PyList_Size(value); j++)
109 vals.push_back(PyLong_AsLong(PyList_GetItem(value, j)));
110
111 params[skey] = vals;
112
113 } else if (PyFloat_Check(vec0)) {
114 std::vector<double> vals;
115
116 for (auto j{0}; j < PyList_Size(value); j++)
117 vals.push_back(PyFloat_AsDouble(PyList_GetItem(value, j)));
118
119 params[skey] = vals;
120
121 } else if (PyUnicode_Check(vec0)) {
122 std::vector<std::string> vals;
123 for (Py_ssize_t j = 0; j < PyList_Size(value); j++) {
124 PyObject* elem = PyList_GetItem(value, j);
125 vals.push_back(getPyString(elem));
126 }
127
128 params[skey] = vals;
129
130 } else {
131 // RECURSION zoinks!
132 // If the objects stored in the list doesn't
133 // satisfy any of the above conditions, just
134 // create a vector of parameters objects
135 std::vector<framework::config::Parameters> vals;
136 for (auto j{0}; j < PyList_Size(value); ++j) {
137 auto elem{PyList_GetItem(value, j)};
138
139 vals.emplace_back();
140 vals.back().setParameters(getMembers(elem));
141 }
142 params[skey] = vals;
143
144 } // type of object in python list
145 } // python list has non-zero size
146 } else {
147 // object got here, so we assume
148 // it is a higher level object
149 //(same logic as last option for a list)
150
151 // RECURSION zoinks!
152 framework::config::Parameters val;
153 val.setParameters(getMembers(value));
154
155 params[skey] = val;
156
157 } // python object type
158 } // loop through python dictionary
159
160 return std::move(params);
161}

163ConfigurePython::ConfigurePython(const std::string& pythonScript, char* args[],
164 int nargs) {
165 // assumes that nargs >= 0
166 // this is true always because we error out if no python script has been
167 // found
168
169 std::string cmd = pythonScript;
170 if (pythonScript.rfind("/") != std::string::npos) {
171 cmd = pythonScript.substr(pythonScript.rfind("/") + 1);
172 }
173 cmd = cmd.substr(0, cmd.find(".py"));
174
175 // python needs the argument list as if you are on the command line
176 // targs = [ script , arg0 , arg1 , ... ] ==> len(targs) = nargs+1
177 // PySys_SetArgvEx uses wchar_t instead of char in python3
178 wchar_t** targs = new wchar_t*[nargs + 1];
179 targs[0] = Py_DecodeLocale(pythonScript.c_str(), NULL);
180 for (int i = 0; i < nargs; i++) targs[i + 1] = Py_DecodeLocale(args[i], NULL);
181
182 // name our program after the script that is being run
183 Py_SetProgramName(targs[0]);
184
185 // start up python interpreter
186 Py_Initialize();
187
188 // The third argument to PySys_SetArgvEx tells python to import
189 // the args and add the directory of the first argument to
190 // the PYTHONPATH
191 // This way, the command to import the module just needs to be
192 // the name of the python script
193 PySys_SetArgvEx(nargs + 1, targs, 1);
194
195 // the following line is what actually runs the script
196 PyObject* script = PyImport_ImportModule(cmd.c_str());
197
198 // script has been run so we can
199 // free up arguments to python script
200 for (int i = 0; i < nargs + 1; i++) PyMem_RawFree(targs[i]);
201 delete[] targs;
202
203 if (script == 0) {
204 PyErr_Print();
205 EXCEPTION_RAISE("ConfigureError", "Problem loading python script");
206 }
207
208 PyObject* pCMod = PyObject_GetAttrString(script, root_module.c_str());
209 Py_DECREF(script); // don't need the script anymore
210 if (pCMod == 0) {
211 PyErr_Print();
212 EXCEPTION_RAISE("ConfigureError",
213 "Did not import root configuration module " + root_module);
214 }
215
216 PyObject* pProcessClass = PyObject_GetAttrString(pCMod, root_class.c_str());
217 Py_DECREF(pCMod); // don't need the config module anymore
218 if (pProcessClass == 0) {
219 PyErr_Print();
220 EXCEPTION_RAISE(
221 "ConfigureError",
222 "Did not import root configuration class " + root_class);
223 }
224
225 PyObject* pProcess = PyObject_GetAttrString(pProcessClass, root_object.c_str());
226 Py_DECREF(pProcessClass); // don't need the Process class anymore
227 if (pProcess == 0) {
228 // wasn't able to get lastProcess class member
229 PyErr_Print();
230 EXCEPTION_RAISE(
231 "ConfigureError",
232 "Process object not defined. This object is required to run.");
233 } else if (pProcess == Py_None) {
234 // lastProcess was left undefined
235 EXCEPTION_RAISE(
236 "ConfigureError",
237 "Did not create a configuration class instance");
238 }
239
240 // okay, now we have fully imported the script and gotten the handle
241 // to the last Process object defined in the script.
242 // We can now look at pProcess and get all of our parameters out of it.
243
244 configuration_.setParameters(getMembers(pProcess));
245
246 // all done with python nonsense
247 // delete one parent python object
248 // MEMORY still not sure if this is enough, but not super worried about it
249 // because this only happens once per run
250 Py_DECREF(pProcess);
251 // close up python interpreter
252 if (Py_FinalizeEx() < 0) {
253 PyErr_Print();
254 EXCEPTION_RAISE("PyError",
255 "I wasn't able to close up the python interpreter!");
256 }
257}
