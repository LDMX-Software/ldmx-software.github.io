66  static std::map< std::string, std::any > getMembers(PyObject* object) {
67 
68  PyObject* dictionary{PyObject_GetAttrString(object, "__dict__")};
69 
70  if ( dictionary == 0 ) {
71  if (PyDict_Check(object)) dictionary = object;
72  else {
73  EXCEPTION_RAISE(
74  "ObjFail",
75  "Python Object does not have __dict__ member"
76  );
77  }
78  }
79 
80  PyObject *key(0), *value(0);
81  Py_ssize_t pos = 0;
82 
83  std::map < std::string, std::any > params;
84 
85  while (PyDict_Next(dictionary, &pos, &key, &value)) {
86 
87  std::string skey{getPyString(key)};
88 
89  if (PyLong_Check(value)) {
90  if (PyBool_Check(value)) {
91  params[skey] = bool(PyLong_AsLong(value));
92  } else {
93  params[skey] = int(PyLong_AsLong(value));
94  }
95  } else if (PyFloat_Check(value)) {
96  params[skey] = PyFloat_AsDouble(value);
97  } else if (PyUnicode_Check(value)) {
98  params[skey] = getPyString(value);
99  } else if (PyList_Check(value)) { // assume everything is same value as first value
100  if (PyList_Size(value) > 0) {
101 
102  auto vec0{PyList_GetItem(value, 0)};
103 
104  if (PyLong_Check(vec0)) {
105  std::vector<int> vals;
106 
107  for (auto j{0}; j < PyList_Size(value); j++)
108  vals.push_back(PyLong_AsLong(PyList_GetItem(value, j)));
109 
110  params[skey] = vals;
111 
112  } else if (PyFloat_Check(vec0)) {
113  std::vector<double> vals;
114 
115  for (auto j{0}; j < PyList_Size(value); j++)
116  vals.push_back(PyFloat_AsDouble(PyList_GetItem(value, j)));
117 
118  params[skey] = vals;
119 
120  } else if (PyUnicode_Check(vec0)) {
121  std::vector<std::string> vals;
122  for (Py_ssize_t j = 0; j < PyList_Size(value); j++){
123  PyObject* elem = PyList_GetItem(value , j );
124  vals.push_back(getPyString(elem));
125  }
126 
127  params[skey] = vals;
128 
129  } else {
130 
131  // RECURSION zoinks!
132  // If the objects stored in the list doesn't
133  // satisfy any of the above conditions, just
134  // create a vector of parameters objects
135  std::vector< Parameters > vals;
136  for (auto j{0}; j < PyList_Size(value); ++j) {
137 
138  auto elem{PyList_GetItem( value, j )};
139 
140  vals.emplace_back();
141  vals.back().setParameters( getMembers(elem) );
142 
143  }
144  params[skey] = vals;
145 
146  } //type of object in python list
147  } //python list has non-zero size
148  } else {
149  //object got here, so we assume
150  //it is a higher level object
151  //(same logic as last option for a list)
152 
153  // RECURSION zoinks!
154  Parameters val;
155  val.setParameters( getMembers(value) );
156 
157  params[skey] = val;
158 
159  } //python object type
160  } //loop through python dictionary
161 
162  return std::move(params);
163  }

165  ConfigurePython::ConfigurePython(const std::string& pythonScript, char* args[], int nargs) {
166 
167  //assumes that nargs >= 0
168  // this is true always because we error out if no python script has been found
169 
170  std::string cmd = pythonScript;
171  if (pythonScript.rfind("/") != std::string::npos) {
172  cmd = pythonScript.substr(pythonScript.rfind("/") + 1);
173  }
174  cmd = cmd.substr(0, cmd.find(".py"));
175 
176  //python needs the argument list as if you are on the command line
177  // targs = [ script , arg0 , arg1 , ... ] ==> len(targs) = nargs+1
178  //PySys_SetArgvEx uses wchar_t instead of char in python3
179  wchar_t** targs = new wchar_t*[nargs+1];
180  targs[0] = Py_DecodeLocale(pythonScript.c_str(), NULL);
181  for (int i = 0; i < nargs; i++)
182  targs[i+1] = Py_DecodeLocale(args[i], NULL);
183 
184  //name our program after the script that is being run
185  Py_SetProgramName(targs[0]);
186 
187  //start up python interpreter
188  Py_Initialize();
189 
190  //The third argument to PySys_SetArgvEx tells python to import
191  //the args and add the directory of the first argument to
192  //the PYTHONPATH
193  //This way, the command to import the module just needs to be
194  //the name of the python script
195  PySys_SetArgvEx(nargs+1, targs, 1);
196 
197  //the following line is what actually runs the script
198  PyObject* script = PyImport_ImportModule(cmd.c_str());
199 
200  // script has been run so we can
201  // free up arguments to python script
202  for(int i = 0; i < nargs+1; i++)
203  PyMem_RawFree(targs[i]);
204  delete [] targs;
205 
206  if (script == 0) {
207  PyErr_Print();
208  EXCEPTION_RAISE("ConfigureError", "Problem loading python script");
209  }
210 
211  PyObject* pCMod = PyObject_GetAttrString(script, "ldmxcfg");
212  Py_DECREF(script); //don't need the script anymore
213  if (pCMod == 0) {
214  PyErr_Print();
215  EXCEPTION_RAISE("ConfigureError", "Problem loading python script");
216  }
217 
218  PyObject* pProcessClass = PyObject_GetAttrString(pCMod, "Process");
219  Py_DECREF(pCMod); //don't need the config module anymore
220  if (pProcessClass == 0) {
221  PyErr_Print();
222  EXCEPTION_RAISE("ConfigureError",
223  "Process object not defined. This object is required to run ldmx-app."
224  );
225  }
226 
227  PyObject* pProcess = PyObject_GetAttrString(pProcessClass, "lastProcess");
228  Py_DECREF(pProcessClass); //don't need the Process class anymore
229  if (pProcess == 0) {
230  //wasn't able to get lastProcess class member
231  PyErr_Print();
232  EXCEPTION_RAISE("ConfigureError",
233  "Process object not defined. This object is required to run."
234  );
235  } else if ( pProcess == Py_None ) {
236  //lastProcess was left undefined
237  EXCEPTION_RAISE("ConfigureError",
238  "Process object not defined. This object is required to run."
239  );
240  }
241 
242  //okay, now we have fully imported the script and gotten the handle
243  //to the last Process object defined in the script.
244  //We can now look at pProcess and get all of our parameters out of it.
245 
246  configuration_.setParameters(getMembers(pProcess));
247 
248  //all done with python nonsense
249  //delete one parent python object
250  //MEMORY still not sure if this is enough, but not super worried about it
251  // because this only happens once per run
252  Py_DECREF(pProcess);
253  //close up python interpreter
254  if ( Py_FinalizeEx() < 0 ) {
255  PyErr_Print();
256  EXCEPTION_RAISE(
257  "PyError",
258  "I wasn't able to close up the python interpreter!"
259  );
260  }
261  }
