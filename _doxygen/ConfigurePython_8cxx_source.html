76  static std::map< std::string, std::any > getMembers(PyObject* object) {
77 
78  PyObject* dictionary{PyObject_GetAttrString(object, "__dict__")};
79 
80  if ( dictionary == 0 ) {
81  EXCEPTION_RAISE(
82  "ObjFail",
83  "Python Object does not have __dict__ member"
84  );
85  }
86 
87  PyObject *key(0), *value(0);
88  Py_ssize_t pos = 0;
89 
90  std::map < std::string, std::any > params;
91 
92  while (PyDict_Next(dictionary, &pos, &key, &value)) {
93 
94  std::string skey{getPyString(key)};
95 
96  if (PyLong_Check(value)) {
97  if (PyBool_Check(value)) {
98  params[skey] = bool(PyLong_AsLong(value));
99  } else {
100  params[skey] = int(PyLong_AsLong(value));
101  }
102  } else if (PyFloat_Check(value)) {
103  params[skey] = PyFloat_AsDouble(value);
104  } else if (PyUnicode_Check(value)) {
105  params[skey] = getPyString(value);
106  } else if (PyList_Check(value)) { // assume everything is same value as first value
107  if (PyList_Size(value) > 0) {
108 
109  auto vec0{PyList_GetItem(value, 0)};
110 
111  if (PyLong_Check(vec0)) {
112  std::vector<int> vals;
113 
114  for (auto j{0}; j < PyList_Size(value); j++)
115  vals.push_back(PyLong_AsLong(PyList_GetItem(value, j)));
116 
117  params[skey] = vals;
118 
119  } else if (PyFloat_Check(vec0)) {
120  std::vector<double> vals;
121 
122  for (auto j{0}; j < PyList_Size(value); j++)
123  vals.push_back(PyFloat_AsDouble(PyList_GetItem(value, j)));
124 
125  params[skey] = vals;
126 
127  } else if (PyUnicode_Check(vec0)) {
128  std::vector<std::string> vals;
129  for (Py_ssize_t j = 0; j < PyList_Size(value); j++){
130  PyObject* elem = PyList_GetItem(value , j );
131  vals.push_back(getPyString(elem));
132  }
133 
134  params[skey] = vals;
135 
136  } else {
137 
138  // RECURSION zoinks!
139  // If the objects stored in the list doesn't
140  // satisfy any of the above conditions, just
141  // create a vector of parameters objects
142  std::vector< Parameters > vals;
143  for (auto j{0}; j < PyList_Size(value); ++j) {
144 
145  auto elem{PyList_GetItem( value, j )};
146 
147  vals.emplace_back();
148  vals.back().setParameters( getMembers(elem) );
149 
150  }
151  params[skey] = vals;
152 
153  } //type of object in python list
154  } //python list has non-zero size
155  } else {
156  //object got here, so we assume
157  //it is a higher level object
158  //(same logic as last option for a list)
159 
160  // RECURSION zoinks!
161  Parameters val;
162  val.setParameters( getMembers(value) );
163 
164  params[skey] = val;
165 
166  } //python object type
167  } //loop through python dictionary
168 
169  return std::move(params);
170  }

172  ConfigurePython::ConfigurePython(const std::string& pythonScript, char* args[], int nargs) {
173 
174  //assumes that nargs >= 0
175  // this is true always because we error out if no python script has been found
176 
177  std::string cmd = pythonScript;
178  if (pythonScript.rfind("/") != std::string::npos) {
179  cmd = pythonScript.substr(pythonScript.rfind("/") + 1);
180  }
181  cmd = cmd.substr(0, cmd.find(".py"));
182 
183  //python needs the argument list as if you are on the command line
184  // targs = [ script , arg0 , arg1 , ... ] ==> len(targs) = nargs+1
185  //PySys_SetArgvEx uses wchar_t instead of char in python3
186  wchar_t** targs = new wchar_t*[nargs+1];
187  targs[0] = Py_DecodeLocale(pythonScript.c_str(), NULL);
188  for (int i = 0; i < nargs; i++)
189  targs[i+1] = Py_DecodeLocale(args[i], NULL);
190 
191  //name our program after the script that is being run
192  Py_SetProgramName(targs[0]);
193 
194  //start up python interpreter
195  Py_Initialize();
196 
197  //The third argument to PySys_SetArgvEx tells python to import
198  //the args and add the directory of the first argument to
199  //the PYTHONPATH
200  //This way, the command to import the module just needs to be
201  //the name of the python script
202  PySys_SetArgvEx(nargs+1, targs, 1);
203 
204  //the following line is what actually runs the script
205  PyObject* script = PyImport_ImportModule(cmd.c_str());
206 
207  // script has been run so we can
208  // free up arguments to python script
209  for(int i = 0; i < nargs+1; i++)
210  PyMem_RawFree(targs[i]);
211  delete [] targs;
212 
213  if (script == 0) {
214  PyErr_Print();
215  EXCEPTION_RAISE("ConfigureError", "Problem loading python script");
216  }
217 
218  PyObject* pCMod = PyObject_GetAttrString(script, "ldmxcfg");
219  Py_DECREF(script); //don't need the script anymore
220  if (pCMod == 0) {
221  PyErr_Print();
222  EXCEPTION_RAISE("ConfigureError", "Problem loading python script");
223  }
224 
225  PyObject* pProcessClass = PyObject_GetAttrString(pCMod, "Process");
226  Py_DECREF(pCMod); //don't need the config module anymore
227  if (pProcessClass == 0) {
228  PyErr_Print();
229  EXCEPTION_RAISE("ConfigureError",
230  "Process object not defined. This object is required to run ldmx-app."
231  );
232  }
233 
234  PyObject* pProcess = PyObject_GetAttrString(pProcessClass, "lastProcess");
235  Py_DECREF(pProcessClass); //don't need the Process class anymore
236  if (pProcess == 0) {
237  //wasn't able to get lastProcess class member
238  PyErr_Print();
239  EXCEPTION_RAISE("ConfigureError",
240  "Process object not defined. This object is required to run."
241  );
242  } else if ( pProcess == Py_None ) {
243  //lastProcess was left undefined
244  EXCEPTION_RAISE("ConfigureError",
245  "Process object not defined. This object is required to run."
246  );
247  }
248 
249  //okay, now we have fully imported the script and gotten the handle
250  //to the last Process object defined in the script.
251  //We can now look at pProcess and get all of our parameters out of it.
252 
253  configuration_.setParameters(getMembers(pProcess));
254 
255  //all done with python nonsense
256  //delete one parent python object
257  //MEMORY still not sure if this is enough, but not super worried about it
258  // because this only happens once per run
259  Py_DECREF(pProcess);
260  //close up python interpreter
261  if ( Py_FinalizeEx() < 0 ) {
262  PyErr_Print();
263  EXCEPTION_RAISE(
264  "PyError",
265  "I wasn't able to close up the python interpreter!"
266  );
267  }

332  }

334  if (!histograms.empty()) {
335  ep->getHistoDirectory();
336  ep->createHistograms( histograms );
337  }
338  ep->configure(proc);
339  process->addToSequence(ep);
340  }
341 
342  return process;
