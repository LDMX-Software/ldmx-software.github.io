52  void DarkBremFilter::stepping(const G4Step* step) {
53 
54  // Get the track associated with this step.
55  G4Track* track = step->GetTrack();
56 
57  // Only process the primary electron track
58  if (track->GetParentID() != 0) return;
59 
60  // get the PDGID of the track.
61  G4int pdgID = track->GetParticleDefinition()->GetPDGEncoding();
62 
63  // Make sure that the particle being processed is an electron.
64  if (pdgID != 11) return;
65 
66  // Get the volume the particle is in.
67  G4VPhysicalVolume* volume = track->GetVolume();
68  G4String volumeName = volume->GetName();
69 
70  // If the particle isn't in the given volume, don't continue with the processing.
71  if (not volumeName.contains(volumeName_.c_str())) return;
72 
73  // Get the particle type.
74  G4String particleName = track->GetParticleDefinition()->GetParticleName();
75 
76  if (step->GetPostStepPoint()->GetStepStatus() == fGeomBoundary) {
77  // primary electron is exiting the volume.
78 
79  // Get the particles daughters.
80  const G4TrackVector* secondaries = step->GetSecondary();
81 
82  if (secondaries->size() == 0) {
83  // If the particle didn't produce any secondaries, stop processing the event.
84  if ( verbosity_ > 1 ) {
85  std::cout << "[ DarkBremFilter ]: "
86  << "Primary did not produce secondaries --> Killing primary track!"
87  << std::endl;
88  }
89 
90  track->SetTrackStatus(fKillTrackAndSecondaries);
91  G4RunManager::GetRunManager()->AbortEvent();
92  return;
93  } else if (not hasAPrime(secondaries)) {
94  // If the particle din't produce an A Prime, stop processing the event
95  if ( verbosity_ > 1 ) {
96  std::cout << "[ DarkBremFilter ]: "
97  << "No dark brem in " << volumeName_ << " --> Aborting event."
98  << std::endl;
99  }
100 
101  track->SetTrackStatus(fKillTrackAndSecondaries);
102  G4RunManager::GetRunManager()->AbortEvent();
103  return;
104  }
105 
106  } else if (step->GetPostStepPoint()->GetKineticEnergy() == 0) {
107  //primary electron stopped inside of volume
108  const G4TrackVector* secondaries = step->GetSecondary();
109  if(not hasAPrime( secondaries )){
110  // If the particle din't produce an A Prime, stop processing the event
111  if ( verbosity_ > 1 ) {
112  std::cout << "[ DarkBremFilter ]: "
113  << "Electron never made it out of the " << volumeName_ << " --> Killing all tracks!"
114  << std::endl;
115  }
116 
117  track->SetTrackStatus(fKillTrackAndSecondaries);
118  G4RunManager::GetRunManager()->AbortEvent();
119  return;
120  }
121  } //check if particle is leaving volume or stopped within it
122 
123  }//stepping
