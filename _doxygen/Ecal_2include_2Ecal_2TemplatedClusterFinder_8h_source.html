16template <class WeightClass>
17
18class TemplatedClusterFinder {
19 public:
20 void add(const ldmx::EcalHit* eh, const ldmx::EcalHexReadout& hex) {
21 clusters_.push_back(WorkingCluster(eh, hex));
22 }
23
24 static bool compClusters(const WorkingCluster& a, const WorkingCluster& b) {
25 return a.centroid().E() > b.centroid().E();
26 }
27
28 void cluster(double seed_threshold, double cutoff) {
29 int ncluster = clusters_.size();
30 double minwgt = cutoff;
31
32 std::sort(clusters_.begin(), clusters_.end(), compClusters);
33 do {
34 bool any = false;
35 size_t mi(0), mj(0);
36
37 int nseeds = 0;
38
39 for (size_t i = 0; i < clusters_.size(); i++) {
40 if (clusters_[i].empty()) continue;
41
42 bool iseed = (clusters_[i].centroid().E() >= seed_threshold);
43 if (iseed) {
44 nseeds++;
45 } else {
46 // Since we sorted initially if we find a hit below seed threshold
47 // it is guaranteed that there will be no seeds after.
48 break;
49 }
50
51 for (size_t j = i + 1; j < clusters_.size(); j++) {
52 if (clusters_[j].empty() ||
53 (!iseed && clusters_[j].centroid().E() < seed_threshold))
54 continue;
55 double wgt = wgt_(clusters_[i], clusters_[j]);
56 if (!any || wgt < minwgt) {
57 any = true;
58 minwgt = wgt;
59 mi = i;
60 mj = j;
61 }
62 }
63 }
64
65 nseeds_ = nseeds;
66 transitionWeights_.insert(std::pair<int, double>(ncluster, minwgt));
67
68 if (any && minwgt < cutoff) {
69 // put the bigger one in mi
70 if (clusters_[mi].centroid().E() < clusters_[mj].centroid().E()) {
71 std::swap(mi, mj);
72 }
73 // now we have the smallest, merge
74 clusters_[mi].add(clusters_[mj]);
75 clusters_[mj].clear();
76 // decrement cluster count
77 ncluster--;
78 }
79
80 } while (minwgt < cutoff && ncluster > 1);
81 finalwgt_ = minwgt;
82 }
83
84 double getYMax() const { return finalwgt_; }
85
86 int getNSeeds() const { return nseeds_; }
87
88 std::map<int, double> getWeights() const { return transitionWeights_; }
89
90 std::vector<WorkingCluster> getClusters() const { return clusters_; }
91
92 private:
93 WeightClass wgt_;
94 double finalwgt_;
95 int nseeds_;
96 std::map<int, double> transitionWeights_;
97 std::vector<WorkingCluster> clusters_;
98};
99} // namespace ecal
