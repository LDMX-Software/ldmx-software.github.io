13void XsecBiasingOperator::StartRun() {
14 if (this->getParticleToBias().compare("gamma") == 0) {
15 processManager_ = G4Gamma::GammaDefinition()->GetProcessManager();
16 } else if (this->getParticleToBias().compare("e-") == 0) {
17 processManager_ = G4Electron::ElectronDefinition()->GetProcessManager();
18 } else if (this->getParticleToBias().compare("neutron") == 0) {
19 processManager_ = G4Neutron::NeutronDefinition()->GetProcessManager();
20 } else if (this->getParticleToBias().compare("kaon0L") == 0) {
21 processManager_ = G4KaonZeroLong::KaonZeroLongDefinition()->GetProcessManager();
22 } else {
23 EXCEPTION_RAISE("BiasSetup", "Invalid particle type '" +
24 this->getParticleToBias() + "'.");
25 }
26
27 std::cout << "[ XsecBiasingOperator ]: Biasing particles of type "
28 << this->getParticleToBias() << std::endl;
29
30 if (processIsBiased(this->getProcessToBias())) {
31 xsecOperation_ =
32 new G4BOptnChangeCrossSection("changeXsec-" + this->getProcessToBias());
33 } else {
34 EXCEPTION_RAISE("BiasSetup",
35 this->getProcessToBias() +
36 " is not found in list of biased processes!");
37 }
38}

40bool XsecBiasingOperator::processIsBiased(std::string process) {
41 // Loop over all processes and check if the given process is being
42 // biased.
43 const G4BiasingProcessSharedData* sharedData =
44 G4BiasingProcessInterface::GetSharedData(processManager_);
45 if (sharedData) {
46 for (size_t iprocess = 0;
47 iprocess < (sharedData->GetPhysicsBiasingProcessInterfaces()).size();
48 ++iprocess) {
49 const G4BiasingProcessInterface* wrapperProcess =
50 (sharedData->GetPhysicsBiasingProcessInterfaces())[iprocess];
51
52 if (wrapperProcess->GetWrappedProcess()->GetProcessName().compareTo(
53 process) == 0) {
54 return true;
55 }
56 }
57 }
58 return false;
59}
