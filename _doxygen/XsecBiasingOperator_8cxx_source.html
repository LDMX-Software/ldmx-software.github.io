37 bool XsecBiasingOperator::processIsBiased(std::string process) {
38  // Loop over all processes and check if the given process is being
39  // biased.
40  const G4BiasingProcessSharedData* sharedData =
41  G4BiasingProcessInterface::GetSharedData(processManager_);
42  if (sharedData) {
43  for (size_t iprocess = 0;
44  iprocess < (sharedData->GetPhysicsBiasingProcessInterfaces()).size();
45  ++iprocess) {
46  const G4BiasingProcessInterface* wrapperProcess =
47  (sharedData->GetPhysicsBiasingProcessInterfaces())[iprocess];
48 
49  if (wrapperProcess->GetWrappedProcess()->GetProcessName().compareTo(
50  process) == 0) {
51  return true;
52  }
53  }
54  }
55  return false;
56 }
