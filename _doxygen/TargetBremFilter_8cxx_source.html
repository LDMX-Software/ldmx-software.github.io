49  void TargetBremFilter::stepping(const G4Step* step) {
50 
51  // Get the track associated with this step.
52  auto track{step->GetTrack()};
53 
54  // Only process the primary electron track
55  if (track->GetParentID() != 0) return;
56 
57  // Get the PDG ID of the track and make sure it's an electron. If
58  // another particle type is found, thrown an exception.
59  if (auto pdgID{track->GetParticleDefinition()->GetPDGEncoding()}; pdgID != 11) return;
60 
61  // Get the region the particle is currently in. Continue processing
62  // the particle only if it's in the target region.
63  if (auto region{track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
64  region.compareTo("target") != 0) return;
65 
66 
67  // Check if the electron will be exiting the target
68  if (auto volume{track->GetNextVolume()->GetName()}; volume.compareTo("recoil_PV") == 0) {
69 
70  // If the recoil electron
71  if (track->GetMomentum().mag() >= recoilMaxPThreshold_) {
72 
73  track->SetTrackStatus(fKillTrackAndSecondaries);
74  G4RunManager::GetRunManager()->AbortEvent();
75  return;
76 
77  }
78 
79  // Get the electron secondries
80  bool hasBremCandidate = false;
81  if (auto secondaries = step->GetSecondary(); secondaries->size() == 0) {
82 
83  track->SetTrackStatus(fKillTrackAndSecondaries);
84  G4RunManager::GetRunManager()->AbortEvent();
85  return;
86  } else {
87 
88  for (auto& secondary_track : *secondaries) {
89  G4String processName = secondary_track->GetCreatorProcess()->GetProcessName();
90 
91  if (processName.compareTo("eBrem") == 0
92  && secondary_track->GetKineticEnergy() > bremEnergyThreshold_) {
93 
94 
95  if (secondary_track->GetUserInformation() == nullptr) {
96  secondary_track->SetUserInformation(new UserTrackInformation());
97  }
98  auto trackInfo{static_cast< UserTrackInformation* >(secondary_track->GetUserInformation())};
99  trackInfo->tagBremCandidate();
100  trackInfo->setVertexVolume(secondary_track->GetVolume()->GetName());
101 
102  auto event{G4EventManager::GetEventManager()};
103  if (event->GetUserInformation() == nullptr) {
104  event->SetUserInformation(new UserEventInformation());
105  }
106  static_cast< UserEventInformation* >(event->GetUserInformation())->incBremCandidateCount();
107 
108  hasBremCandidate = true;
109  }
110  }
111  }
112 
113  if (!hasBremCandidate) {
114 
115  track->SetTrackStatus(fKillTrackAndSecondaries);
116  G4RunManager::GetRunManager()->AbortEvent();
117  return;
118  }
119 
120  // Check if the recoil electron should be killed. If not, postpone
121  // its processing until the brem gamma has been processed.
122  if (killRecoil_) track->SetTrackStatus(fStopAndKill);
123  else track->SetTrackStatus(fSuspend);
124 
125  } else if (step->GetPostStepPoint()->GetKineticEnergy() == 0) {
126 
127  track->SetTrackStatus(fKillTrackAndSecondaries);
128  G4RunManager::GetRunManager()->AbortEvent();
129  return;
130  }
131  }
