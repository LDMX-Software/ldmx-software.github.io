20void TruthSeedProcessor::configure(framework::config::Parameters &parameters) {
21 out_trk_coll_name_ =
22 parameters.getParameter<std::string>("trk_coll_name", "RecoilTruthSeeds");
23 pdg_ids_ = parameters.getParameter<std::vector<int> >("pdg_ids", {11});

26 sim_hits_ = parameters.getParameter<std::string>("sim_hits", "RecoilSimHits");
27 n_min_hits_ = parameters.getParameter<int>("n_min_hits", 7);
28 z_min_ = parameters.getParameter<double>("z_min", -9999); // mm
29 track_id_ = parameters.getParameter<int>("track_id", -9999);
30 pz_cut_ = parameters.getParameter<double>("pz_cut", -9999); // MeV
31 p_cut_ = parameters.getParameter<double>("p_cut", 0.);
32 p_cutMax_ = parameters.getParameter<double>("p_cutMax", 100000.); // MeV
33 //k0_sel_ = parameters.getParameter<bool>("k0_sel", false);
34 p_cutEcal_ = parameters.getParameter<double>("p_cut_ecal", -1.); // MeV
35}

37// Look at the scoring planes trackID
38// -> The trackID is the unique identification of the particles in the
39// generated event
40// -> 1 is the primary electron since it's the first one generated
41// -> For brehmsstrahlung / Energy Loss due to Ionization, the outgoing
42// electron trackID is the same of the incoming
43// --- > The recoil electron will still trackID == 1
44// --- > Get the TargetScoringPlaneHits. Filter on z > 4.5mm, Get the
45// TrackID==1 and get the (x,y,z,px,py,pz).
46
47// -> For eN interactions G4 is not able to follow the particle through the
48// interaction. New particles will be created
49// --- > Look through the recoil sim hits, get the track ID from those hits and
50// *then* find the TargetScoringPlaneHits with that trackID to do truth
51// matching.
52// --- > There will be other trackIDs in the hits. Pick the ones that are
53// electrons, and pick the highest energy ones.
54
55// One thing that I can check is the endpoint of the trackID==1 particle to
56// understand what kind of interaction the particle went through.
57
58// Truth based initial track parameters
59
60// In the case of Recoil tracking take only TrackID==1 and the generation
61// surface is the obtained from the TargetScoringPlane

63void TruthSeedProcessor::produce(framework::Event &event) {
64 nevents_++;
65
66 auto start = std::chrono::high_resolution_clock::now();
67
68 // Tagger truth seeds
69 std::vector<ldmx::Track> tagger_truth_seeds_{};
70
71 // Retrieve the scoring plane hits
72 const std::vector<ldmx::SimTrackerHit> scoring_hits =
73 event.getCollection<ldmx::SimTrackerHit>(scoring_hits_);
74
75 // Retrieve the scoring plane hits at the ECAL
76 const std::vector<ldmx::SimTrackerHit> scoring_hits_ecal =
77 event.getCollection<ldmx::SimTrackerHit>("EcalScoringPlaneHits");
78
79 // Retrieve the particle map to get the tagger seeds
80 auto particleMap{event.getMap<int, ldmx::SimParticle>("SimParticles")};
81 ldmx::SimParticle gen_e = particleMap[1];
82 Acts::Vector3 gen_e_pos{gen_e.getVertex().data()};
83 Acts::Vector3 gen_e_mom{gen_e.getMomentum().data()};
84 Acts::ActsScalar gen_e_time = 0.;
85
86 gen_e_pos = tracking::sim::utils::Ldmx2Acts(gen_e_pos);
87 gen_e_mom = tracking::sim::utils::Ldmx2Acts(gen_e_mom);
88
89 // Grabbing only the electrons for the tagger truth seeds
90 Acts::ActsScalar q = -1 * Acts::UnitConstants::e;
91 Acts::FreeVector tagger_free_params =
92 tracking::sim::utils::toFreeParameters(gen_e_pos, gen_e_mom, q);
93
94 std::shared_ptr<const Acts::PerigeeSurface> tagger_gen_surface =
95 Acts::Surface::makeShared<Acts::PerigeeSurface>(
96 Acts::Vector3(tagger_free_params[Acts::eFreePos0],
97 tagger_free_params[Acts::eFreePos1],
98 tagger_free_params[Acts::eFreePos2]));
99
100 // Transform the free parameters to the bound parameters
101 auto tagger_bound_params = Acts::detail::transformFreeToBoundParameters(
102 tagger_free_params, *tagger_gen_surface, gctx_)
103 .value();
104
105 // Check the parameters at the perigee
106 std::shared_ptr<const Acts::PerigeeSurface> tagger_per_surface =
107 Acts::Surface::makeShared<Acts::PerigeeSurface>(
108 Acts::Vector3(-700, 0., 0.));
109
110 auto tagger_bound_params_perigee =
111 Acts::detail::transformFreeToBoundParameters(tagger_free_params,
112 *tagger_per_surface, gctx_)
113 .value();
114
115 ldmx_log(debug) << "(TRUTH) Tagger bound params to the -700,0.,0. surface"<<std::endl
116 << tagger_bound_params_perigee;
117
118
119 // Try something reasonable
120 Acts::BoundVector stddev;
121 stddev[Acts::eBoundLoc0] = 500 * Acts::UnitConstants::um;
122 stddev[Acts::eBoundLoc1] = 1 * Acts::UnitConstants::mm;
123 stddev[Acts::eBoundTime] = 1000 * Acts::UnitConstants::ns;
124 stddev[Acts::eBoundPhi] = 2 * Acts::UnitConstants::degree;
125 stddev[Acts::eBoundTheta] = 2 * Acts::UnitConstants::degree;
126
127 double p_tagger =
128 sqrt(gen_e_mom(0) * gen_e_mom(0) + gen_e_mom(1) * gen_e_mom(1) +
129 gen_e_mom(2) * gen_e_mom(2)) *
130 Acts::UnitConstants::MeV;
131 // 50% of uncertainty on momentum from seed fit // Passing 2 GeV, Expected:
132 // 500 MeV for 4 GeV electrons
133 double sigma_p_tagger = 0.5 * p_tagger * Acts::UnitConstants::GeV;
134 double sigma_qop_tagger = (1. / p_tagger) * (1. / p_tagger) * sigma_p_tagger;
135 stddev[Acts::eBoundQOverP] = sigma_qop_tagger;
136
137 Acts::BoundSymMatrix tagger_bound_cov =
138 stddev.cwiseProduct(stddev).asDiagonal();
139
140 ldmx::Track tagger_seedTrack = ldmx::Track();
141 tagger_seedTrack.setPerigeeLocation(tagger_free_params[Acts::eFreePos0],
142 tagger_free_params[Acts::eFreePos1],
143 tagger_free_params[Acts::eFreePos2]);
144 tagger_seedTrack.setChi2(0.);
145 tagger_seedTrack.setNhits(0.);
146 tagger_seedTrack.setNdf(0);
147 tagger_seedTrack.setNsharedHits(0.);
148
149 std::vector<double> tagger_ldmx_cov;
150 tracking::sim::utils::flatCov(tagger_bound_cov, tagger_ldmx_cov);
151 tagger_seedTrack.setPerigeeParameters(
152 tracking::sim::utils::convertActsToLdmxPars(tagger_bound_params));
153 tagger_seedTrack.setPerigeeCov(tagger_ldmx_cov);
154
155 tagger_truth_seeds_.push_back(tagger_seedTrack);
156
157 event.add("TaggerTruthSeeds", tagger_truth_seeds_);
158
159 if (scoring_hits.size() < 1) return;
160
161 // Retrieve the sim hits in the tracker of interest
162 const std::vector<ldmx::SimTrackerHit> sim_hits =
163 event.getCollection<ldmx::SimTrackerHit>(sim_hits_);
164
165 // TODO:: change to indices instead objects
166 std::vector<ldmx::SimTrackerHit> selected_sp_hits;
167
168 for (auto &t_sp_hit : scoring_hits) {
169 // Clean some of the hits we don't want
170 if (t_sp_hit.getPosition()[2] < z_min_) continue;
171
172 // Check if the track_id was requested
173 if (track_id_ > 0 && t_sp_hit.getTrackID() != track_id_) continue;
174
175 // Check if we are requesting particular particles
176 if (std::find(pdg_ids_.begin(), pdg_ids_.end(), t_sp_hit.getPdgID()) ==
177 pdg_ids_.end())
178 continue;
179
180 Acts::Vector3 t_sp_p{t_sp_hit.getMomentum()[0], t_sp_hit.getMomentum()[1],
181 t_sp_hit.getMomentum()[2]};
182
183 // p cut
184 if (p_cut_ >= 0. && t_sp_p.norm() < p_cut_) continue;
185
186 // p cut Max
187 if (p_cut_ < 100000. && t_sp_p.norm() > p_cutMax_) continue;
188
189 // pz cut
190 if (pz_cut_ > -9999 && t_sp_p(2) < pz_cut_) continue;
191
192 // Check the ecal scoring plane
193 bool passEcalScoringPlane = true;
194
195 if (p_cutEcal_ > 0) { // only check if we care about it.
196
197 for (auto &e_sp_hit : scoring_hits_ecal) {
198 if (e_sp_hit.getTrackID() == t_sp_hit.getTrackID() &&
199 e_sp_hit.getPdgID() == t_sp_hit.getPdgID()) {
200 Acts::Vector3 e_sp_p{e_sp_hit.getMomentum()[0],
201 e_sp_hit.getMomentum()[1],
202 e_sp_hit.getMomentum()[2]};
203
204 if (e_sp_p.norm() < p_cutEcal_) passEcalScoringPlane = false;
205
206 // Skip the rest of the scoring plane hits since we already found the
207 // track we care about
208 break;
209
210 } // check that the hit belongs to the inital particle from the target
211 // scoring plane hit
212 } // loop on Ecal scoring plane hits
213 } // pcutEcal
214
215 if (!passEcalScoringPlane) continue;
216
217 // add the point
218 selected_sp_hits.push_back(t_sp_hit);
219 }
220
221 ldmx_log(debug) << "Selected scoring hits::" << selected_sp_hits.size();
222
223
224 std::vector<ldmx::Track> truth_seeds_{};
225
226 for (auto &sel_hit : selected_sp_hits) {
227 int seed_particle_hits = 0;
228
229 // Let's check how many hits this particle leaves in the recoil
230 for (auto &sim_hit : sim_hits)
231 if (sim_hit.getTrackID() == sel_hit.getTrackID()) seed_particle_hits++;
232
233 if (seed_particle_hits < n_min_hits_) continue;
234
235 Acts::Vector3 gen_pos{sel_hit.getPosition()[0], sel_hit.getPosition()[1],
236 sel_hit.getPosition()[2]};
237 Acts::Vector3 gen_mom{sel_hit.getMomentum()[0], sel_hit.getMomentum()[1],
238 sel_hit.getMomentum()[2]};
239
240 // Rotate into the ACTS Frame
241 gen_pos = tracking::sim::utils::Ldmx2Acts(gen_pos);
242 gen_mom = tracking::sim::utils::Ldmx2Acts(gen_mom);
243
244 // Check the pdg id
245 Acts::ActsScalar q = -1 * Acts::UnitConstants::e;
246
247 // Electrons or muons
248 if (abs(sel_hit.getPdgID()) == 11 || abs(sel_hit.getPdgID() == 13)) {
249 q = sel_hit.getPdgID() > 0 ? -1 * Acts::UnitConstants::e
250 : Acts::UnitConstants::e;
251 }
252
253 // Whatever else: pi, K ...
254 else {
255 q = sel_hit.getPdgID() > 0 ? Acts::UnitConstants::e
256 : -1 * Acts::UnitConstants::e;
257 }
258
259 ldmx_log(debug) << "Preparing seed from hit:" << std::endl
260 << sel_hit.getPdgID() << " " << sel_hit.getPosition()[0] << ","
261 << sel_hit.getPosition()[1] << "," << sel_hit.getPosition()[2]
262 << std::endl
263 << q << " " << sel_hit.getMomentum()[0] << ","
264 << sel_hit.getMomentum()[1] << "," << sel_hit.getMomentum()[2];
265
266 Acts::FreeVector free_params =
267 tracking::sim::utils::toFreeParameters(gen_pos, gen_mom, q);
268 std::shared_ptr<const Acts::PerigeeSurface> gen_surface =
269 Acts::Surface::makeShared<Acts::PerigeeSurface>(Acts::Vector3(
270 free_params[Acts::eFreePos0], free_params[Acts::eFreePos1],
271 free_params[Acts::eFreePos2]));
272 auto bound_params = Acts::detail::transformFreeToBoundParameters(
273 free_params, *gen_surface, gctx_)
274 .value();
275
276 // Try something reasonable
277 Acts::BoundVector recoil_stddev;
278 recoil_stddev[Acts::eBoundLoc0] = 500 * Acts::UnitConstants::um;
279 recoil_stddev[Acts::eBoundLoc1] = 1 * Acts::UnitConstants::mm;
280 recoil_stddev[Acts::eBoundTime] = 1000 * Acts::UnitConstants::ns;
281 recoil_stddev[Acts::eBoundPhi] = 2 * Acts::UnitConstants::degree;
282 recoil_stddev[Acts::eBoundTheta] = 5 * Acts::UnitConstants::degree;
283
284 double p = sqrt(gen_mom(0) * gen_mom(0) + gen_mom(1) * gen_mom(1) +
285 gen_mom(2) * gen_mom(2)) *
286 Acts::UnitConstants::MeV;
287
288 // 50% of uncertainty on momentum from seed fit // Passing 2 GeV, Expected:
289 // 500 MeV for 4 GeV electrons
290 double sigma_p = 0.5 * p * Acts::UnitConstants::GeV;
291 double sigma_qop = (1. / p) * (1. / p) * sigma_p;
292
293 recoil_stddev[Acts::eBoundQOverP] = sigma_qop;
294 Acts::BoundSymMatrix bound_cov =
295 recoil_stddev.cwiseProduct(recoil_stddev).asDiagonal();
296
297 // Form the seed track for the event bus
298
299 ldmx::Track seedTrack = ldmx::Track();
300 seedTrack.setPerigeeLocation(free_params[Acts::eFreePos0],
301 free_params[Acts::eFreePos1],
302 free_params[Acts::eFreePos2]);
303 seedTrack.setChi2(0.);
304 seedTrack.setNhits(seed_particle_hits);
305 seedTrack.setNdf(0);
306 seedTrack.setNsharedHits(0.);
307
308 std::vector<double> ldmx_cov;
309 tracking::sim::utils::flatCov(bound_cov, ldmx_cov);
310 seedTrack.setPerigeeParameters(
311 tracking::sim::utils::convertActsToLdmxPars(bound_params));
312 seedTrack.setPerigeeCov(ldmx_cov);
313
314 truth_seeds_.push_back(seedTrack);
315
316 } // selected hits
317
318 // Do the k0 selection.
319 // std::vector<ldmx::Track> k0_decay_pions;
320
321 // if (k0_sel_) {
322 // if (truth_seeds_.size() == 2 && truth_seeds_.at(0).charge() *
323 // truth_seeds_.at(1).charge() < 0) {
324 // TLorentzVector p1, p2;
325 // p1.SetXYZM(truth_seeds_.at(0).momentum()(0),
326 // truth_seeds_.at(0).momentum()(1),
327 // truth_seeds_.at(0).momentum()(2),
328 // pion_mass);
329 //
330 // p2.SetXYZM(truth_seeds_.at(1).momentum()(0),
331 // truth_seeds_.at(1).momentum()(1),
332 // truth_seeds_.at(1).momentum()(2),
333 // pion_mass);
334 // }
335 // }// k0 selection
336
337 event.add(out_trk_coll_name_, truth_seeds_);
338
339 auto end = std::chrono::high_resolution_clock::now();
340 // long long microseconds =
341 // std::chrono::duration_cast<std::chrono::microseconds>(end-start).count();
342 auto diff = end - start;
343 processing_time_ += std::chrono::duration<double, std::milli>(diff).count();
344}
