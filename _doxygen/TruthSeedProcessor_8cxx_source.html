51void TruthSeedProcessor::createTruthTrack(const std::vector<double> &pos_vec,
52 const std::vector<double> &p_vec,
53 int charge,
54 ldmx::TruthTrack& trk) {
55 // Get the position and momentum of the particle at the point of creation.
56 // This only works for the incident electron when creating a tagger tracker
57 // seed. For the recoil tracker, the scoring plane position will need to
58 // be used. For other particles created in the target or tracker planes,
59 // this version of the method can also be used.
60 // These are just Eigen vectors defined as
61 // Eigen::Matrix<double, kSize, 1>;
62 Acts::Vector3 pos{pos_vec.data()};
63 Acts::Vector3 mom{p_vec.data()};
64 double time{0.};
65
66 // Rotate the position and momentum into the ACTS frame.
67 pos = tracking::sim::utils::Ldmx2Acts(pos);
68 mom = tracking::sim::utils::Ldmx2Acts(mom);
69
70 // Get the charge of the particle.
71 // TODO: Add function that uses the PDG ID to calculate this.
72 double q{charge * Acts::UnitConstants::e};
73
74 // Transform the position, momentum and charge to free parameters.
75 auto free_params{tracking::sim::utils::toFreeParameters(pos, mom, q)};
76
77 // Create a line surface at the perigee. The perigee position is extracted
78 // from a particle's vertex or the particle's position at a specific
79 // scoring plane.
80 auto gen_surface{Acts::Surface::makeShared<Acts::PerigeeSurface>(
81 Acts::Vector3(free_params[Acts::eFreePos0], free_params[Acts::eFreePos1],
82 free_params[Acts::eFreePos2]))};
83
84 // Transform the parameters to local positions on the perigee surface.
85 auto bound_params{Acts::detail::transformFreeToBoundParameters(
86 free_params, *gen_surface, gctx_)
87 .value()};
88
89 // Create the seed track object.
90 trk.setPerigeeLocation(free_params[Acts::eFreePos0],
91 free_params[Acts::eFreePos1],
92 free_params[Acts::eFreePos2]);
93
94 trk.setPerigeeParameters(
95 tracking::sim::utils::convertActsToLdmxPars(bound_params));
96
97
98 trk.setPosition(pos(0),pos(1),pos(2));
99 trk.setMomentum(mom(0),mom(1),mom(2));
100
101}

103bool TruthSeedProcessor::scoringPlaneHitFilter(
104 const ldmx::SimTrackerHit &hit,
105 const std::vector<ldmx::SimTrackerHit> &ecal_sp_hits) {
106 // Clean some of the hits we don't want
107 if (hit.getPosition()[2] < z_min_) return false;
108
109 // Check if the track_id was requested
110 if (track_id_ > 0 && hit.getTrackID() != track_id_) return false;
111
112 // Check if we are requesting particular particles
113 if (std::find(pdg_ids_.begin(), pdg_ids_.end(), hit.getPdgID()) ==
114 pdg_ids_.end())
115 return false;
116
117 Acts::Vector3 p_vec{hit.getMomentum()[0], hit.getMomentum()[1],
118 hit.getMomentum()[2]};
119
120 // p cut
121 if (p_cut_ >= 0. && p_vec.norm() < p_cut_) return false;
122
123 // p cut Max
124 if (p_cut_ < 100000. && p_vec.norm() > p_cut_max_) return false;
125
126 // pz cut
127 if (pz_cut_ > -9999 && p_vec(2) < pz_cut_) return false;
128
129 // Check the ecal scoring plane
130 bool pass_ecal_scoring_plane = true;
131
132 if (p_cut_ecal_ > 0) { // only check if we care about it.
133
134 for (auto &e_sp_hit : ecal_sp_hits) {
135 if (e_sp_hit.getTrackID() == hit.getTrackID() &&
136 e_sp_hit.getPdgID() == hit.getPdgID()) {
137 Acts::Vector3 e_sp_p{e_sp_hit.getMomentum()[0],
138 e_sp_hit.getMomentum()[1],
139 e_sp_hit.getMomentum()[2]};
140
141 if (e_sp_p.norm() < p_cut_ecal_) pass_ecal_scoring_plane = false;
142
143 // Skip the rest of the scoring plane hits since we already found the
144 // track we care about
145 break;
146
147 } // check that the hit belongs to the inital particle from the target
148 // scoring plane hit
149 } // loop on Ecal scoring plane hits
150 } // pcutEcal
151
152 if (!pass_ecal_scoring_plane) return false;
153
154 return true;
155}

157void TruthSeedProcessor::produce(framework::Event &event) {
158
159 //Retrieve the particleMap
160 auto particleMap{event.getMap<int, ldmx::SimParticle>("SimParticles")};
161
162 //Retrieve the target scoring hits
163 // Information is extracted using the
164 // scoring plane hit left by the particle at the target.
165
166 const std::vector<ldmx::SimTrackerHit> scoring_hits{
167 event.getCollection<ldmx::SimTrackerHit>(scoring_hits_coll_name_)};
168
169 // Retrieve the scoring plane hits at the ECAL
170 const std::vector<ldmx::SimTrackerHit> scoring_hits_ecal{
171 event.getCollection<ldmx::SimTrackerHit>("EcalScoringPlaneHits")};
172
173 // Retrieve the sim hits in the tracker of interest
174 const std::vector<ldmx::SimTrackerHit> sim_hits =
175 event.getCollection<ldmx::SimTrackerHit>(recoil_sim_hits_coll_name_);
176
177
178
179 // Create a mapping from the selected scoring plane hit objects to the number
180 // of hits they associated particle creates in the tracker.
181 // In this way I can keep track of the amount of hits left by each trackID
182
183 std::unordered_map<int, int> hit_count_map;
184 for (auto &sim_hit : sim_hits) {
185 if (!hit_count_map.count(sim_hit.getTrackID()))
186 hit_count_map[sim_hit.getTrackID()] = 1;
187
188 hit_count_map[sim_hit.getTrackID()]++;
189 }
190
191 // to keep track of how many sim particles leave hits on the s-plane
192 std::vector<int> recoil_sh_idxs;
193 std::unordered_map<int,std::vector<int> > recoil_sh_count_map;
194
195 int idx_taggerhit = -1; //We are interested only in the trackID==1 electron
196
197
198 //Target scoring hits for Tagger will have Z<0, Recoil scoring hits will have Z>0
199 for (unsigned int i_sh=0; i_sh < scoring_hits.size(); i_sh++) {
200
201 const ldmx::SimTrackerHit& hit = scoring_hits.at(i_sh);
202 Acts::Vector3 p_vec{hit.getMomentum()[0],hit.getMomentum()[1],hit.getMomentum()[2]};
203
204 double tagger_p_max = 0.;
205
206 //Check if it is a tagger track going fwd that passes basic cuts
207 if (hit.getPosition()[2] < 0.) {
208 //Tagger selection cuts
209 //Negative scoring plane hit, with momentum > p_cut
210 if (p_vec(2) < 0. ||
211 p_vec.norm() < p_cut_ ||
212 hit.getPdgID() != 11 ||
213 hit.getTrackID() != 1)
214 continue;
215
216 if (p_vec.norm() > tagger_p_max) {
217 idx_taggerhit = i_sh;
218 tagger_p_max = p_vec.norm();
219 }
220 }//Tagger loop
221
222 //Check the recoil hits
223 else {
224
225 //Recoil selection cuts
226 //Positive scoring plane hit, forward direction with momentum > p_cut
227 if (p_vec(2) < 0. ||
228 p_vec.norm() < p_cut_ )
229 continue;
230
231 //Check that the hit was left by a charged particle
232 if (abs(particleMap[hit.getTrackID()].getCharge()) < 1e-8)
233 continue;
234
235 recoil_sh_count_map[hit.getTrackID()].push_back(i_sh);
236
237
238 }//Recoil
239 }// loop on Target scoring plane hits
240
241
242 for (std::pair<int,std::vector<int>> element : recoil_sh_count_map) {
243
244 std::sort(element.second.begin(),element.second.end(),
245 [&](const int idx1, int idx2) -> bool {
246
247 const ldmx::SimTrackerHit& hit1 = scoring_hits.at(idx1);
248 const ldmx::SimTrackerHit& hit2 = scoring_hits.at(idx2);
249
250 Acts::Vector3 phit1{hit1.getMomentum()[0],hit1.getMomentum()[1],hit1.getMomentum()[2]};
251 Acts::Vector3 phit2{hit2.getMomentum()[0],hit2.getMomentum()[1],hit2.getMomentum()[2]};
252
253 return phit1.norm() > phit2.norm();
254 }
255 );
256 }
257
258 std::vector<ldmx::TruthTrack> tagger_truth_tracks;
259 std::vector<ldmx::TruthTrack> recoil_truth_tracks;
260 //Define the perigee_surface at 0.0.0
261 auto perigee_surface{Acts::Surface::makeShared<Acts::PerigeeSurface>(Acts::Vector3(0.,0.,0.))};
262 //Define the target_surface
263 auto target_surface = tracking::sim::utils::unboundSurface(0.);
264
265 if (idx_taggerhit != -1) {
266
267 const ldmx::SimTrackerHit& hit = scoring_hits.at(idx_taggerhit);
268 const ldmx::SimParticle& phit = particleMap[hit.getTrackID()];
269
270 if (hit_count_map[hit.getTrackID()] > n_min_hits_) {
271
272 ldmx::TruthTrack truth_tagger_track;
273 createTruthTrack(phit, hit, truth_tagger_track);
274 //get track state at the generation point
275 //ldmx::TruthTrack::TrackState ts_beamOrigin(phit,"beam_origin");
276 //propagate track to target
277 //trackExtrapolator(truth_tagger_track, perigee_surface);
278
279 //truth_tagger_track.addTrackState(ts_beamOrigin);
280 tagger_truth_tracks.push_back(truth_tagger_track);
281 }
282 }
283
284
285 for (std::pair<int,std::vector<int>> element : recoil_sh_count_map) {
286
287 //Only take the first entry of the vector: it should be the scoring plane hit with the highest momentum.
288 const ldmx::SimTrackerHit& hit = scoring_hits.at(element.second.at(0));
289 const ldmx::SimParticle& phit = particleMap[hit.getTrackID()];
290
291 //Findable particle selection
292 if (hit_count_map[hit.getTrackID()] < n_min_hits_)
293 continue;
294
295 ldmx::TruthTrack truth_recoil_track;
296 createTruthTrack(phit,hit,truth_recoil_track);
297 recoil_truth_tracks.push_back(truth_recoil_track);
298 }
299
300 /*
301 std::cout<<"TAGGER TRUTH TRACKS"<<std::endl;
302 for (auto tt : tagger_truth_tracks) {
303 std::cout<<tt<<std::endl;
304 }
305 std::cout<<"#########"<<std::endl;
306
307 std::cout<<"RECOIL TRUTH TRACKS"<<std::endl;
308 for (auto tt : recoil_truth_tracks) {
309 std::cout<<tt<<std::endl;
310 }
311 std::cout<<"#########"<<std::endl;
312 */
313
314 event.add("TaggerTruthTracks",tagger_truth_tracks);
315 event.add("RecoilTruthTracks",recoil_truth_tracks);
316
317}
