27TEST_CASE("BinaryIO", "[Packing][functionality]") {
28 SECTION("Unformatted Data") {
29 std::string test_file{"test.raw"};
30 // test_vec and test_wide have the same data content
31 // but will be stored differently depending on endian-ness of system
32 std::vector<uint16_t> test_vec = { 0xFFFF, 0x00FF, 0xFF00, 0xF0F0 };
33 uint64_t test_wide = 0xFFFF00FFFF00F0F0;
34
35 SECTION("Write") {
36 packing::utility::Writer w(test_file);
37
38 CHECK(w << test_vec);
39 CHECK(w << test_wide);
40 }
41
42 SECTION("Read") {
43 packing::utility::Reader r(test_file);
44
45 std::vector<uint16_t> read_vec;
46 uint64_t read_wide, dummy;
47 CHECK(r.read(read_vec, 4));
48 CHECK(r >> read_wide);
49 CHECK_FALSE(r >> dummy);
50 CHECK(r.eof());
51 CHECK(read_vec == test_vec);
52 CHECK(read_wide == test_wide);
53 }
54 }

132
133 std::vector<uint32_t> data = {

136 0xCCCCCCCC,
137 0xDDDDDDDD,
