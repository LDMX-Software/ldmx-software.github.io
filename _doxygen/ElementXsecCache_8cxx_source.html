5G4double ElementXsecCache::get(G4double energy, G4double A, G4double Z) {
6 key_t key = computeKey(energy, A, Z);
7 if (the_cache_.find(key) == the_cache_.end()) {
8 if (model_.get() == nullptr) {
9 throw std::runtime_error(
10 "ElementXsecCache not given a model to calculate cross "
11 "sections with.");
12 }
13 the_cache_[key] = model_->ComputeCrossSectionPerAtom(energy, A, Z);
14 }
15 return the_cache_.at(key);
16}

18void ElementXsecCache::stream(std::ostream& o) const {
19 o << "A [au],Z [protons],Energy [MeV],Xsec [pb]\n"
20 << std::setprecision(std::numeric_limits<double>::digits10 +
21 1); // maximum precision
22 for (auto const& cache_entry : the_cache_) {
23 const key_t& key = cache_entry.first;
24 const double& xsec = cache_entry.second;
25 key_t E = key % MAX_E;
26 key_t A = ((key - E) / MAX_E) % MAX_A;
27 key_t Z = ((key - E) / MAX_E - A) / MAX_A;
28 o << A << "," << Z << "," << E << "," << xsec / CLHEP::picobarn << "\n";
29 }
30 o << std::endl;
31}
