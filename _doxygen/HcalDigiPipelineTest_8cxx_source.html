74class isCloseEnough : public Catch::MatcherBase<double> {
75 private:
77 double truth_;

80 const double max_absolute_diff_;

84
85 public:
91 isCloseEnough(double const &truth, double const &abs_diff,
92 double const &rel_diff)
93 : truth_{truth},
94 max_absolute_diff_{abs_diff},
95 max_relative_diff_{rel_diff} {}
96
104 bool match(const double &daq) const override {
105 return (daq == Approx(truth_).epsilon(max_relative_diff_) or
106 daq == Approx(truth_).margin(max_absolute_diff_));
107 }
108
112 virtual std::string describe() const override {
113 std::ostringstream ss;
114 ss << "is within an absolute difference of " << max_absolute_diff_
115 << " OR a relative difference of " << max_relative_diff_ << " with "
116 << truth_;
117 return ss.str();
118 }

129class HcalFakeSimHits : public framework::Producer {
133 // Based on the current gain settings for the ADC readout mode
134 // we will reach saturation ~ 20 MeV ~ 290 PEs
135 const double maxEnergy_ = 200 * PE_ENERGY; // ~ 13 MeV
136
141 const double minEnergy_ = 4 * PE_ENERGY;
147 const double energyStep_ = (maxEnergy_ - minEnergy_) / NUM_TEST_SIM_HITS;
148
150 double currEnergy_ = minEnergy_;
151
152 public:
153 HcalFakeSimHits(const std::string &name, framework::Process &p)
154 : framework::Producer(name, p) {}
155 ~HcalFakeSimHits() {}
156
157 void beforeNewRun(ldmx::RunHeader &header) {
158 header.setDetectorName("ldmx-det-v12");
159 }
160
161 void produce(framework::Event &event) final override {
162 // put in a single sim hit
163 std::vector<ldmx::SimCalorimeterHit> pretendSimHits(1);
164
165 // We hard-code the position of one hit: back hcal, layer 1, strip 31
166 // This real simHit position is obtained by looking at calorimeter
167 // SimHits of a 4 GeV muon shoot through the beamline
168 ldmx::HcalID id(0, 1, 31);
169 pretendSimHits[0].setPosition(-6.70265, 3.70265, 879); // mm
170 pretendSimHits[0].setID(id.raw());
171 pretendSimHits[0].addContrib(
172 -1, // incidentID
173 -1, // trackID
174 0, // pdg ID
175 currEnergy_, // edep
176 2.96628 // time - 299mm is about 1ns from target and in middle of HCal
177 );
178
179 // needs to be correct collection name
180 // REQUIRE_NOTHROW(event.add("HcalSimHits", pretendSimHits));
181 REQUIRE_NOTHROW(event.add("HcalFakeSimHits", pretendSimHits));
182 currEnergy_ += energyStep_;
183
184 return;
185 }
186}; // HcalFakeSimHits
