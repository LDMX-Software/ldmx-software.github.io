59void APrimePhysics::ConstructProcess() {
60 // add process to electron if we are enabled
61 if (enable_) {
62 auto model{
63 parameters_.getParameter<framework::config::Parameters>("model")};
64 auto model_name{model.getParameter<std::string>("name")};
65 if (model_name == "vertex_library" or model_name == "g4db") {
66 static const std::map<std::string, g4db::G4DarkBreMModel::ScalingMethod>
67 method_lut = {
68 {"forward_only",
69 g4db::G4DarkBreMModel::ScalingMethod::ForwardOnly},
70 {"cm_scaling", g4db::G4DarkBreMModel::ScalingMethod::CMScaling},
71 {"undefined", g4db::G4DarkBreMModel::ScalingMethod::Undefined}};
72 auto scaling_method_it{
73 method_lut.find(model.getParameter<std::string>("method"))};
74 if (scaling_method_it == method_lut.end()) {
75 EXCEPTION_RAISE(
76 "BadConf",
77 "Unrecognized scaling method '" +
78 model.getParameter<std::string>("method") +
79 "',"
80 " options are 'forward_only', 'cm_scaling', or 'undefined'.");
81 }
82 // Note: The process variable isn't used here, but creating the
83 // G4DarkBremsstahlung object has side-effects
84 process_ = std::make_unique<G4DarkBremsstrahlung>(
85 std::make_shared<g4db::G4DarkBreMModel>(
86 model.getParameter<std::string>("library_path"),
87 false /* dark brem off muons instead of electrons - we
88 always DB off electrons here */
89 ,
90 model.getParameter<double>("threshold"),
91 model.getParameter<double>("epsilon"), scaling_method_it->second),
92 parameters_.getParameter<bool>("only_one_per_event"),
93 1., /* global bias - should use bias operator instead */
94 parameters_.getParameter<bool>("cache_xsec"));
95 process_->RegisterStorageMechanism(store_element_z);
96 } else {
97 EXCEPTION_RAISE("BadConf",
98 "Unrecognized model name '" + model_name + "'.");
99 }
100 G4cout << "[ APrimePhysics ] : Initialization of dark brem complete"
101 << G4endl;
102 }
103}
