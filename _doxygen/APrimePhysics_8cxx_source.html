39 void APrimePhysics::ConstructProcess() {
40  // add process to electron if LHE file has been provided
41  if (enable_) {
42  /*
43  * In G4 speak, a "discrete" process is one that only happens at the end of
44  * steps. we want the DB to be discrete because it is not a "slow braking"
45  * like ionization, the electron suddenly has the interaction and loses a
46  * lot of its energy.
47  *
48  * The first argument to this function is the process we are adding.
49  * The process manager handles cleaning up the processes,
50  * so we just give it a new pointer.
51  * The second argument is the "ordering" index.
52  * This index determines when the process is called w.r.t. the other
53  * processes that could be called at the end of the step. Not providing the
54  * second argument means that the ordering index is given a default value of
55  * 1000 which seems to be safely above all the internal/default processes.
56  */
57  G4Electron::ElectronDefinition()->GetProcessManager()->AddDiscreteProcess(
58  new G4eDarkBremsstrahlung(parameters_));
59  }
60 }
