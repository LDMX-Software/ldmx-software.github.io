42 auto model{parameters_.getParameter<framework::config::Parameters>("model")};

44 if (model_name == "vertex_library" or model_name == "g4db") {
45 static const std::map<std::string,g4db::G4DarkBreMModel::ScalingMethod> method_lut = {
46 { "forward_only", g4db::G4DarkBreMModel::ScalingMethod::ForwardOnly },
47 { "cm_scaling" , g4db::G4DarkBreMModel::ScalingMethod::CMScaling },
48 { "undefined" , g4db::G4DarkBreMModel::ScalingMethod::Undefined }
49 };
50 auto scaling_method_it{method_lut.find(model.getParameter<std::string>("method"))};
51 if (scaling_method_it == method_lut.end()) {
52 EXCEPTION_RAISE("BadConf","Unrecognized scaling method '"+
53 model.getParameter<std::string>("method") + "',"
54 " options are 'forward_only', 'cm_scaling', or 'undefined'.");
55 }
56 auto proc = new G4DarkBremsstrahlung(
57 std::make_shared<g4db::G4DarkBreMModel>(
58 model.getParameter<std::string>("library_path"),
59 false /* dark brem off muons instead of electrons - we always DB off electrons here */,
60 model.getParameter<double>("threshold"),
61 model.getParameter<double>("epsilon"),
62 scaling_method_it->second),
63 parameters_.getParameter<bool>("only_one_per_event"),
64 1., /* global bias - should use bias operator instead */
65 parameters_.getParameter<bool>("cache_xsec"));
66 } else {
67 EXCEPTION_RAISE("BadConf",
68 "Unrecognized model name '"+model_name+"'.");
69 }
70 G4cout << "[ APrimePhysics ] : Initialization of dark brem complete" << G4endl;
71 }
