31 std::vector<ProductTag> Event::searchProducts(
32  const std::string& namematch, const std::string& passmatch,
33  const std::string& typematch) const {
34  std::vector<ProductTag> retval;
35 
36  regex_t reg_name, reg_pass, reg_type;
37 
38  if (!regcomp(&reg_name, (namematch.empty() ? (".*") : (namematch.c_str())),
39  REG_EXTENDED | REG_ICASE | REG_NOSUB)) {
40  if (!regcomp(&reg_pass, (passmatch.empty() ? (".*") : (passmatch.c_str())),
41  REG_EXTENDED | REG_ICASE | REG_NOSUB)) {
42  if (!regcomp(&reg_type,
43  (typematch.empty() ? (".*") : (typematch.c_str())),
44  REG_EXTENDED | REG_ICASE | REG_NOSUB)) {
45  // all passed expressions are valid regular expressions
46  const std::vector<ProductTag>& products = getProducts();
47  for (std::vector<ProductTag>::const_iterator i = products.begin();
48  i != products.end(); i++) {
49  if (!regexec(&reg_name, i->name().c_str(), 0, 0, 0) &&
50  !regexec(&reg_pass, i->passname().c_str(), 0, 0, 0) &&
51  !regexec(&reg_type, i->type().c_str(), 0, 0, 0))
52  retval.push_back(*i);
53  }
54 
55  regfree(&reg_type);
56  } else {
57  EXCEPTION_RAISE("InvalidRegex",
58  "The passed type regex '" + typematch +
59  "' is not a valid regular expression");
60  }
61  regfree(&reg_pass);
62  } else {
63  EXCEPTION_RAISE("InvalidRegex",
64  "The passed name regex '" + namematch +
65  "' is not a valid regular expression");
66  }
67  regfree(&reg_name);
68  } else {
69  EXCEPTION_RAISE("RegexErr", "The passed passname regex '" + passmatch +
70  "' is not a valid regular expression.");
71  }
72 
73  return retval;
74 }

84 void Event::setInputTree(TTree* tree) {
85  inputTree_ = tree;
86  entries_ = inputTree_->GetEntriesFast();
87 
88  // in some cases, setInputTree is called more than once,
89  // so reset branch listing before starting
90  products_.clear();
91  knownLookups_.clear(); //reset caching of empty pass requests
92  bus_.everybodyOff();
93 
94  // put in EventHeader (only one without pass name)
95  products_.emplace_back(ldmx::EventHeader::BRANCH, "", "ldmx::EventHeader");
96 
97  // find the names of all the existing branches
98  TObjArray* branches = inputTree_->GetListOfBranches();
99  for (int i = 0; i < branches->GetEntriesFast(); i++) {
100  std::string brname = branches->At(i)->GetName();
101  if (brname != ldmx::EventHeader::BRANCH) {
102  size_t j = brname.find("_");
103  auto br = dynamic_cast<TBranchElement*>(branches->At(i));
104  // can't determine type if branch isn't
105  // the higher-level TBranchElement type
106  // Only occurs if the type on the bus is one of:
107  // bool, short, int, long, float, double (BSILFD)
108  products_.emplace_back(
109  brname.substr(0, j), // collection name is before '_'
110  brname.substr(j + 1), // pass name is after
111  br ? br->GetClassName() : "BSILFD");
112  }
113  }
114 }
