31std::vector<ProductTag> Event::searchProducts(
32 const std::string& namematch, const std::string& passmatch,
33 const std::string& typematch) const {
34 std::vector<ProductTag> retval;
35
36 regex_t reg_name, reg_pass, reg_type;
37
38 if (!regcomp(&reg_name, (namematch.empty() ? (".*") : (namematch.c_str())),
39 REG_EXTENDED | REG_ICASE | REG_NOSUB)) {
40 if (!regcomp(&reg_pass, (passmatch.empty() ? (".*") : (passmatch.c_str())),
41 REG_EXTENDED | REG_ICASE | REG_NOSUB)) {
42 if (!regcomp(&reg_type,
43 (typematch.empty() ? (".*") : (typematch.c_str())),
44 REG_EXTENDED | REG_ICASE | REG_NOSUB)) {
45 // all passed expressions are valid regular expressions
46 const std::vector<ProductTag>& products = getProducts();
47 for (std::vector<ProductTag>::const_iterator i = products.begin();
48 i != products.end(); i++) {
49 if (!regexec(&reg_name, i->name().c_str(), 0, 0, 0) &&
50 !regexec(&reg_pass, i->passname().c_str(), 0, 0, 0) &&
51 !regexec(&reg_type, i->type().c_str(), 0, 0, 0))
52 retval.push_back(*i);
53 }
54
55 regfree(&reg_type);
56 } else {
57 EXCEPTION_RAISE("InvalidRegex",
58 "The passed type regex '" + typematch +
59 "' is not a valid regular expression");
60 }
61 regfree(&reg_pass);
62 } else {
63 EXCEPTION_RAISE("InvalidRegex",
64 "The passed name regex '" + namematch +
65 "' is not a valid regular expression");
66 }
67 regfree(&reg_name);
68 } else {
69 EXCEPTION_RAISE("RegexErr", "The passed passname regex '" + passmatch +
70 "' is not a valid regular expression.");
71 }
72
73 return retval;
74}

84void Event::setInputTree(TTree* tree) {
85 inputTree_ = tree;
86
87 // in some cases, setInputTree is called more than once,
88 // so reset branch listing before starting
89 products_.clear();
90 knownLookups_.clear(); //reset caching of empty pass requests
91 bus_.everybodyOff();
92
93 // put in EventHeader (only one without pass name)
94 products_.emplace_back(ldmx::EventHeader::BRANCH, "", "ldmx::EventHeader");
95
96 // find the names of all the existing branches
97 TObjArray* branches = inputTree_->GetListOfBranches();
98 for (int i = 0; i < branches->GetEntriesFast(); i++) {
99 std::string brname = branches->At(i)->GetName();
100 if (brname != ldmx::EventHeader::BRANCH) {
101 size_t j = brname.find("_");
102 auto br = dynamic_cast<TBranchElement*>(branches->At(i));
103 // can't determine type if branch isn't
104 // the higher-level TBranchElement type
105 // Only occurs if the type on the bus is one of:
106 // bool, short, int, long, float, double (BSILFD)
107 products_.emplace_back(
108 brname.substr(0, j), // collection name is before '_'
109 brname.substr(j + 1), // pass name is after
110 br ? br->GetClassName() : "BSILFD");
111 }
112 }
113}
