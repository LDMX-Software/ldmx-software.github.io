5void HcalInefficiencyAnalyzer::analyze(const framework::Event &event) {
6 const auto hcalSimHits = event.getCollection<ldmx::SimCalorimeterHit>(
7 hcalSimHitsCollection_, hcalSimHitsPassName_);
8 const auto hcalRecHits = event.getCollection<ldmx::HcalHit>(
9 hcalRecHitsCollection_, hcalRecHitsPassName_);
10
11 const int failedVeto{999};
12 // Check veto for each section, combined side hcal veto
13 std::vector<int> firstLayersHit{failedVeto, failedVeto, failedVeto,
14 failedVeto, failedVeto};
15
16 const std::vector<std::string> sectionNames{"back", "top", "bottom", "right",
17 "left"};
18 for (const auto hit : hcalRecHits) {
19 const ldmx::HcalID id{hit.getID()};
20 const auto section{id.section()};
21 const auto z{hit.getZPos()};
22 const auto layer{id.layer()};
23 if (hitPassesVeto(hit, section)) {
24 if (layer < firstLayersHit[section]) {
25 firstLayersHit[section] = layer;
26 }
27 }
28 }
29
30 bool vetoedByBack{firstLayersHit[ldmx::HcalID::HcalSection::BACK] !=
31 failedVeto};
32 bool vetoedByTop{firstLayersHit[ldmx::HcalID::HcalSection::TOP]};
33 bool vetoedByBottom{firstLayersHit[ldmx::HcalID::HcalSection::BOTTOM]};
34 bool vetoedByRight{firstLayersHit[ldmx::HcalID::HcalSection::RIGHT]};
35 bool vetoedByLeft{firstLayersHit[ldmx::HcalID::HcalSection::LEFT]};
36 bool vetoedBySide{vetoedByTop || vetoedByBottom || vetoedByRight ||
37 vetoedByLeft};
38
39 for (int section{0}; section < firstLayersHit.size(); ++section) {
40 const auto layer{firstLayersHit[section]};
41 const auto sectionName{sectionNames[section]};
42 if (layer != failedVeto) {
43 histograms_.fill("inefficiency_" + sectionName, layer);
44 histograms_.fill("efficiency", section);
45 }
46 }
47 if (vetoedByBack || vetoedBySide) {
48 histograms_.fill("efficiency", vetoCategories::any);
49 if (vetoedByBack && vetoedBySide) {
50 histograms_.fill("efficiency", vetoCategories::both);
51 } else if (vetoedByBack && !vetoedBySide) {
52 histograms_.fill("efficiency", vetoCategories::back_only);
53 } else if (vetoedBySide && !vetoedByBack) {
54 histograms_.fill("efficiency", vetoCategories::side_only);
55 }
56 } else {
57 histograms_.fill("efficiency", vetoCategories::neither);
58 }
59}
