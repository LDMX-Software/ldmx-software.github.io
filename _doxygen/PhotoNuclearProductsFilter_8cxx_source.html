30void PhotoNuclearProductsFilter::stepping(const G4Step* step) {
31 // Get the track associated with this step.
32 auto track{step->GetTrack()};
33
34 // Get the track info and check if this track has been tagged as the
35 // photon that underwent a photo-nuclear reaction. Only those tracks
36 // tagged as PN photos will be processed. The track is currently only
37 // tagged by the UserAction ECalProcessFilter which needs to be run
38 // before this UserAction.
39 auto trackInfo{simcore::UserTrackInformation::get(track)};
40 if ((trackInfo != nullptr) && !trackInfo->isPNGamma()) return;
41
42 // Get the PN photon daughters.
43 auto secondaries{step->GetSecondary()};
44
45 // Loop through all of the secondaries and check for the product of
46 // interest. This is done by getting the PDG ID of a daughter and
47 // checking that it's in the vector of PDG IDs passed to this
48 // UserAction.
49 bool productFound{false};
50 for (const auto& secondary : *secondaries) {
51 // Get the PDG ID of the track
52 auto pdgID{std::abs(secondary->GetParticleDefinition()->GetPDGEncoding())};
53
54 // Check if the PDG ID is in the list of products of interest
55 if (std::find(productsPdgID_.begin(), productsPdgID_.end(), pdgID) !=
56 productsPdgID_.end()) {
57 productFound = true;
58 break;
59 }
60 }
61
62 // If the product of interest was not found, kill the track and abort
63 // the event.
64 if (!productFound) {
65 track->SetTrackStatus(fKillTrackAndSecondaries);
66 G4RunManager::GetRunManager()->AbortEvent();
67 return;
68 }
69
70 // Once the PN gamma has been procesed, untag it so its not reprocessed
71 // again.
72 trackInfo->tagPNGamma(false);
73}
