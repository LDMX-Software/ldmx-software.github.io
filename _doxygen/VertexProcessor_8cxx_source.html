18void VertexProcessor::onProcessStart() {
19 gctx_ = Acts::GeometryContext();
20 bctx_ = Acts::MagneticFieldContext();

22 h_m_ = new TH1F("m", "m", 100, 0., 1.);
23 h_m_truthFilter_ = new TH1F("m_filter", "m", 100, 0., 1.);
24 h_m_truth_ = new TH1F("m_truth", "m_truth", 100, 0., 1.);

26 auto localToGlobalBin_xyz = [](std::array<size_t, 3> bins,
27 std::array<size_t, 3> sizes) {
28 return (bins[0] * (sizes[1] * sizes[2]) + bins[1] * sizes[2] +
29 bins[2]); // xyz - field space
30 // return (bins[1] * (sizes[2] * sizes[0]) + bins[2] * sizes[0] + bins[0]);
31 // //zxy
32 };
33
34 InterpolatedMagneticField3 map = makeMagneticFieldMapXyzFromText(
35 std::move(localToGlobalBin_xyz), field_map_,
36 1. * Acts::UnitConstants::mm, // default scale for axes length
37 1000. * Acts::UnitConstants::T, // The map is in kT, so scale it to T
38 false, // not symmetrical
39 true // rotate the axes to tracking frame
40 );
41
42 sp_interpolated_bField_ =
43 std::make_shared<InterpolatedMagneticField3>(std::move(map));
44 ;

59void VertexProcessor::produce(framework::Event &event) {
60 // TODO:: Move this to an external file
61 // And move all this to a single time per processor not for each event!!
62
63 nevents_++;
64 auto start = std::chrono::high_resolution_clock::now();
65 auto &&stepper = Acts::EigenStepper<>{sp_interpolated_bField_};
66
67 // Set up propagator with void navigator
68 propagator_ = std::make_shared<VoidPropagator>(stepper);
69
70 // Track linearizer in the proximity of the vertex location
71 using Linearizer = Acts::HelicalTrackLinearizer<VoidPropagator>;
72 Linearizer::Config linearizerConfig(sp_interpolated_bField_, propagator_);
73 Linearizer linearizer(linearizerConfig);
74
75 // Set up Billoir Vertex Fitter
76 using VertexFitter =
77 Acts::FullBilloirVertexFitter<Acts::BoundTrackParameters, Linearizer>;
78
79 VertexFitter::Config vertexFitterCfg;
80
81 VertexFitter billoirFitter(vertexFitterCfg);
82
83 VertexFitter::State state(sp_interpolated_bField_->makeCache(bctx_));
84
85 // Unconstrained fit
86 // See
87 // https://github.com/acts-project/acts/blob/main/Tests/UnitTests/Core/Vertexing/FullBilloirVertexFitterTests.cpp#L149
88 // For constraint implementation
89
90 Acts::VertexingOptions<Acts::BoundTrackParameters> vfOptions(gctx_, bctx_);
91
92 // Retrieve the track collection
93 const std::vector<ldmx::Track> tracks =
94 event.getCollection<ldmx::Track>(trk_coll_name_);
95
96 // Retrieve the truth seeds
97 const std::vector<ldmx::Track> seeds =
98 event.getCollection<ldmx::Track>("RecoilTruthSeeds");
99
100 if (tracks.size() < 1) return;
101
102 // Transform the EDM ldmx::tracks to the format needed by ACTS
103 std::vector<Acts::BoundTrackParameters> billoir_tracks;
104
105 // TODO:: The perigee surface should be common between all tracks.
106 // So should only be created once in principle.
107
108 std::shared_ptr<Acts::PerigeeSurface> perigeeSurface =
109 Acts::Surface::makeShared<Acts::PerigeeSurface>(Acts::Vector3(
110 tracks.front().getPerigeeX(), tracks.front().getPerigeeY(),
111 tracks.front().getPerigeeZ()));
112
113 for (unsigned int iTrack = 0; iTrack < tracks.size(); iTrack++) {
114 Acts::BoundVector paramVec;
115 paramVec << tracks.at(iTrack).getD0(), tracks.at(iTrack).getZ0(),
116 tracks.at(iTrack).getPhi(), tracks.at(iTrack).getTheta(),
117 tracks.at(iTrack).getQoP(), tracks.at(iTrack).getT();
118
119 Acts::BoundSymMatrix covMat =
120 tracking::sim::utils::unpackCov(tracks.at(iTrack).getPerigeeCov());
121
122 billoir_tracks.push_back(Acts::BoundTrackParameters(
123 perigeeSurface, paramVec, std::move(covMat)));
124 }
125
126 // Select exactly 2 tracks
127 if (billoir_tracks.size() != 2) {
128 return;
129 }
130
131 if (billoir_tracks.at(0).charge() * billoir_tracks.at(1).charge() > 0) return;
132
133 // Pion mass hypothesis
134 double pion_mass = 139.570 * Acts::UnitConstants::MeV;
135
136 TLorentzVector p1, p2;
137 p1.SetXYZM(billoir_tracks.at(0).momentum()(0),
138 billoir_tracks.at(0).momentum()(1),
139 billoir_tracks.at(0).momentum()(2), pion_mass);
140
141 p2.SetXYZM(billoir_tracks.at(1).momentum()(0),
142 billoir_tracks.at(1).momentum()(1),
143 billoir_tracks.at(1).momentum()(2), pion_mass);
144
145 std::vector<TLorentzVector> pion_seeds;
146
147 if (seeds.size() == 2) {
148 for (int iSeed = 0; iSeed < seeds.size(); iSeed++) {
149 std::shared_ptr<Acts::PerigeeSurface> perigeeSurface =
150 Acts::Surface::makeShared<Acts::PerigeeSurface>(Acts::Vector3(
151 seeds.at(iSeed).getPerigeeX(), seeds.at(iSeed).getPerigeeY(),
152 seeds.at(iSeed).getPerigeeZ()));
153
154 Acts::BoundVector paramVec;
155 paramVec << seeds.at(iSeed).getD0(), seeds.at(iSeed).getZ0(),
156 seeds.at(iSeed).getPhi(), seeds.at(iSeed).getTheta(),
157 seeds.at(iSeed).getQoP(), seeds.at(iSeed).getT();
158
159 Acts::BoundSymMatrix covMat =
160 tracking::sim::utils::unpackCov(seeds.at(iSeed).getPerigeeCov());
161
162 auto boundSeedParams = Acts::BoundTrackParameters(
163 perigeeSurface, paramVec, std::move(covMat));
164
165 TLorentzVector pion4v;
166 pion4v.SetXYZM(boundSeedParams.momentum()(0),
167 boundSeedParams.momentum()(1),
168 boundSeedParams.momentum()(2), pion_mass);
169
170 pion_seeds.push_back(pion4v);
171 } // loops on seeds
172
173 h_m_truth_->Fill((pion_seeds.at(0) + pion_seeds.at(1)).M());
174 }
175
176 if ((pion_seeds.size() == 2) &&
177 (pion_seeds.at(0) + pion_seeds.at(1)).M() > 0.490 &&
178 (pion_seeds.at(0) + pion_seeds.at(1)).M() < 0.510) {
179 // Check if the tracks have opposite charge
180 h_m_truthFilter_->Fill((p1 + p2).M());
181 }
182
183 h_m_->Fill((p1 + p2).M());
184
185 auto end = std::chrono::high_resolution_clock::now();
186 // long long microseconds =
187 // std::chrono::duration_cast<std::chrono::microseconds>(end-start).count();
188 auto diff = end - start;
189 processing_time_ += std::chrono::duration<double, std::milli>(diff).count();
190}
