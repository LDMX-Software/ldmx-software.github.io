8void TrigScintTrackProducer::configure(framework::config::Parameters &ps) {
9 maxDelta_ = ps.getParameter<double>(
10 "delta_max"); // max distance to consider adding in a cluster to track
11 seeding_collection_ = ps.getParameter<std::string>(
12 "seeding_collection"); // probably tagger pad, "TriggerPadTagClusters"
13 input_collections_ = ps.getParameter<std::vector<std::string>>(
14 "further_input_collections"); // {"TriggerPadUpClusters" ,
15 // "TriggerPadDnClusters" }
16 output_collection_ = ps.getParameter<std::string>("output_collection");
17 passName_ = ps.getParameter<std::string>("input_pass_name");
18 verbose_ = ps.getParameter<int>("verbosity");
19 vertBarStartIdx_ = ps.getParameter<int>("vertical_bar_start_index");
20 nBarsY_ = ps.getParameter<int>("number_horizontal_bars");
21 barWidth_y_ = ps.getParameter<double>("horizontal_bar_width");
22 barGap_y_ = ps.getParameter<double>("horizontal_bar_gap");
23 nBarsX_ = ps.getParameter<int>("number_vertical_bars");
24 barWidth_x_ = ps.getParameter<double>("vertical_bar_width");
25 barGap_x_ = ps.getParameter<double>("vertical_bar_gap");
26 skipLast_ = ps.getParameter<bool>("allow_skip_last_collection");
27
28
29 // TO DO: allow any number of input collections
30
31 if (verbose_) {
32 ldmx_log(info) << "In TrigScintTrackProducer: configure done!" << std::endl;
33 ldmx_log(info) << "Got parameters: \nSeeding: " << seeding_collection_
34 << "\nTolerance: " << maxDelta_
35 << "\nInput: " << input_collections_.at(0) << " and "
36 << input_collections_.at(1)
37 << "\nInput pass name: " << passName_
38 << "\nAllow tracks with no hit in last collection: " << skipLast_
39 << "\nVertical bar start index: " << vertBarStartIdx_
40 << "\nNumber of horizontal bars: " << nBarsY_
41 << "\nHorizontal bar width: " << barWidth_y_
42 << "\nHorizontal bar gap: " << barGap_y_
43 << "\nNumber of vertical bars: " << nBarsX_
44 << "\nVertical bar width: " << barWidth_x_
45 << "\nVertical bar gap: " << barGap_x_
46 << "\nOutput: " << output_collection_
47 << "\nVerbosity: " << verbose_;
48 }
49
50
51 yConvFactor_ = (barWidth_y_+barGap_y_)/2.; //each bar only goes half this distance up (overlap/zig-zag)
52 yStart_ = -(nBarsY_*(barWidth_y_+barGap_y_) - barGap_y_)/2.; //half height of pad
53 xConvFactor_ = barWidth_x_+barGap_x_; //each bar goes entire distance sideways (no overlap)
54 xStart_ =-(nBarsX_*(barWidth_x_+barGap_x_) - barGap_x_)/2.; //half width of pad
55
56 return;

467ldmx::TrigScintTrack TrigScintTrackProducer::makeTrack(
468 std::vector<ldmx::TrigScintCluster> clusters) {
469 // for now let's keep a straight, unweighted centroid
470 // consider the possibility that at least one cluster has a centroid
471 // identically == 0. then we need to shift them by 1 if we want to do energy
472 // weighted track centroid later. but no need now
473 ldmx::TrigScintTrack tr;
474 float centroid = 0;
475 float centroidX = 0;
476 float centroidY = 0;
477 float beamEfrac = 0;
478 float pe = 0;
479 for (uint i = 0; i < clusters.size(); i++) {
480 centroid += (clusters.at(i)).getCentroid();
481 centroidX += (clusters.at(i)).getCentroidX();
482 centroidY += (clusters.at(i)).getCentroidY();
483 tr.addConstituent(clusters.at(i));
484 beamEfrac += (clusters.at(i)).getBeamEfrac();
485 pe += (clusters.at(i)).getPE();
486 }
487 centroid /= clusters.size();
488 centroidX /= clusters.size();
489 if (centroid >= vertBarStartIdx_) {
490 if (verbose_) {
491 ldmx_log(debug) << " -- In makeTrack made vertical bar track with centroid "
492 << centroid << " and y flag sum " << centroidY ;
493 //try commenting this to check if that helps with an out-of-bounds problem
494 // << " from clusters with y centroids";
495 //for (uint i = 0; i < clusters.size(); i++)
496 // ldmx_log(debug) << "\tpad " << i << ": centroidY "
497 // << (clusters.at(i)).getCentroidY();
498 }
499 //then the sum of centroid y is 0, 2, 4 or 6
500 // we have 4 divisions, so, the center of it should be divNb/8
501 // (or rather, that's where channel nBars/8 begins)
502 // and then a factor 2 for the zig-zag pattern
503 centroidY = (centroidY+1)*2*nBarsY_/8.;
504 // TODO: here we could instead just use quadrant indices 0-3 by dividing by 2
505 // but that would mean that in the raw, x and y track centroidY would mean different things
506 if (verbose_)
507 ldmx_log(debug) << " -- new centroidY = " << centroidY ;
508 }
509 else
510 centroidY /= clusters.size();
511
512 beamEfrac /= clusters.size();
513 pe /= clusters.size();
514
515 float residual = 0;
516 for (uint i = 0; i < clusters.size(); i++)
517 residual += ((clusters.at(i)).getCentroid() - centroid) *
518 ((clusters.at(i)).getCentroid() - centroid);
519 residual = sqrt((float)(residual / clusters.size()));
520
521 tr.setCentroid(centroid);
522 tr.setCentroidX(centroidX);
523 tr.setCentroidY(centroidY);
524 tr.setResidual(residual);
525 tr.setBeamEfrac(beamEfrac);
526 tr.setPE(pe);
527
528 if (verbose_) {
529 ldmx_log(debug) << " -- In makeTrack made track with centroid "
530 << centroid << " and residual " << residual
531 << " and pe " << pe
532 << " from clusters with centroids";
533 for (uint i = 0; i < clusters.size(); i++)
534 ldmx_log(debug) << "\tpad " << i << ": centroid "
535 << (clusters.at(i)).getCentroid();
536 }
537
538 return tr;
539}

544 void TrigScintTrackProducer::matchXYTracks(
545 std::vector<ldmx::TrigScintTrack> &tracks) {
546
547 //map quadrant nb to track (can be multiple per quadrant)
548 std::multimap <int, int > yIdxQuadMap; //key = quad, val = track index in collection
549 std::multimap <int, int > xIdxQuadMap;
550
551 std::multimap <int, ldmx::TrigScintTrack > yQuadMap;
552 std::multimap <int, ldmx::TrigScintTrack > xQuadMap;
553 //map track in quadrant back to index in entire track collection
554 //used for updating collection track variables
555 std::map <ldmx::TrigScintTrack, int > yTrackMap;
556 std::map <ldmx::TrigScintTrack, int > xTrackMap;
557
558 uint trkIdx = -1;
559 for ( auto trk : tracks ) {
560 trkIdx++;
561 //1. get the y bar tracks with centroidX = -1
562 if ( trk.getCentroidX() == -1) {
563 if (verbose_)
564 ldmx_log(debug) << " -- In matchXYTracks found y track at " << trk.getCentroidY() << "; mapping to quad " << (int)trk.getCentroidY()/8 << " with trk index " << trkIdx ;
565 //2. order them... or map them to quadrants. note that there are 2 layers so 2*nBarsY_/4 channels per quadrant
566 yQuadMap.insert(std::make_pair( (int)(trk.getCentroidY()/8), trk));
567 yTrackMap[ trk ] = trkIdx;
568 yIdxQuadMap.insert(std::make_pair( (int)(trk.getCentroidY()/8), trkIdx));
569
570 }
571 else { //3. get the remaining tracks (from vertical bars) and map them (back) to (middle of) quadrants
572 xQuadMap.insert(std::make_pair( (int)(trk.getCentroidY()/8), trk));
573 xTrackMap[ trk ] = trkIdx;
574 xIdxQuadMap.insert(std::make_pair( (int)(trk.getCentroidY()/8), trkIdx));
575 if (verbose_)
576 ldmx_log(debug) << " -- In matchXYTracks found x track at (x,y) = (" << trk.getCentroidX() << ", " << trk.getCentroidY() << "); mapping to quad " << (int)trk.getCentroidY()/8 << " with trk index " << trkIdx ;
577 }
578 }
579
580
581 // 4a1) here use the geometry? if we can assume perfect alignment we can take width and nBars and take nBars/2 as origin
582 // --- now do the matching ---
583
584 //if there is no useful matching to be done: these are the pad width wide numbers
585 float x0 = 0;
586 float sx0 = fabs(xStart_); // this should be half the pad... could also set it to full beam spot width
587 float sy0 = fabs(yStart_)/4.; // yStart_ is half the pad, so this should be half a quadrant
588
589 //assume at least one y track. will have to figure out if there is ever a reason to use an isolated x track in its place.
590 for ( auto yitr = yQuadMap.begin(); yitr != yQuadMap.end(); ++yitr) {
591 int nYinQuad = yQuadMap.count( (*yitr).first );
592 int nXinQuad = xQuadMap.count( (*yitr).first );
593 float y, sy, x, x1, x2, sx1, sx2, y1, y2, sy1, sy2;
594 //quad midpoint:
595 float y0 = (*yitr).first*8 + sy0;
596 float sx = 1./2*xConvFactor_; //rely on x precision being one single bar width; always used unless x is undeterminable
597
598 //check all x first
599 //do the easiest first:
600 if (nXinQuad == 0 ) { // then there's no hope of setting a better x here
601 // just use the beam spot width... and center of pad
602 x = x0;
603 sx = sx0;
604 if (verbose_)
605 ldmx_log(debug) << "\t\t\t no x info in quad " << (*yitr).first << "; will set x to middle of pad, pad half-width as precision: set (x, sx)=(" << x << ", " << sx << ")";
606 } //0 x tracks in quadrant
607 else if (nXinQuad == 1 ) { //slightly harder: 1 x track -- might be easy if it's just one y track; if several, need to think about overlaps. but in overlap case, just revert to setting x0 and sx0, when we know
608 auto xitr = xQuadMap.find( (*yitr).first );
609 x = ((*xitr).second).getCentroidX() * xConvFactor_ + xStart_;
610
611 if (verbose_)
612 ldmx_log(debug) << "\t\t\t 1 x in quad " << (*yitr).first << ", getting (x, sx)=(" << x << ", " << sx << ")";
613 }// 1 x track in quadrant
614 else if (nXinQuad == 2) { // finally if we have two tracks, get x1 and x2 and decide later how to use them
615 // don't think we want to experiment with discerning three overlapping tracks, so not >= 2
616 // continue; //debugging: skip for now -- didn't help
617 auto xitr1 = xQuadMap.lower_bound( (*yitr).first );
618 auto xitr2 = xQuadMap.upper_bound( (*yitr).first );
619 xitr2--; //upper_bound points to next element
620
621 if (xitr1 != xitr2) {// should be true already but...
622 x1 = ((*xitr1).second).getCentroidX() * xConvFactor_ + xStart_;
623 x2 = ((*xitr2).second).getCentroidX() * xConvFactor_ + xStart_;
624 sx1 = xConvFactor_/2.; //1 bar width
625 sx2 = sx1;
626 x = (x1+x2)/2.;
627 sx = fabs(x1-x2)/2*xConvFactor_; //rely on x precision being one single pad width
628 if (verbose_)
629 ldmx_log(debug) << "\t\t -- 2 x in quad: setting y track x coordinate to midpoint";
630 }
631 }// if 2 x tracks in quad
632
633
634 //ok! over y:
635 // can skip 0 y case by construction
636 if (nYinQuad == 1 ){ // we can already now tell what the y coordinate and its precision is
637 y = ((*yitr).second).getCentroidY() * yConvFactor_ + yStart_;
638 sy = ((*yitr).second).getResidual() * yConvFactor_;
639 if (sy == 0 )
640 sy = 1./2*yConvFactor_; //if all clusters lined up, assign precision of 1 bar width
641
642 if (nXinQuad <= 1 ) {
643 //4. every quadrant which just has one of each --> done ;
644 // b) set the sx, sy of the x track now, using the residuals from the other
645 // b1) special case: no x tracks; then x, sx have been set above
646 auto xidx = xIdxQuadMap.find( (*yitr).first );
647 auto yidx = yIdxQuadMap.find( (*yitr).first );
648 tracks.at( (*xidx).second).setPosition(x,y);
649 tracks.at( (*xidx).second).setSigmaXY(sx, sy);
650 tracks.at( (*yidx).second).setPosition(x,y);
651 tracks.at( (*yidx).second).setSigmaXY(sx, sy);
652 if (verbose_)
653 ldmx_log(debug) << "\t\t\t in quad " << (*yitr).first << ", set (x, y) = (" << x << ", " << y << ") and (sx, sy) = "<< sx << ", " << sy << ")";
654 continue;
655 }
656 }// 1 y, 0 or 1 x track in quadrant
657
658 if (verbose_)
659 ldmx_log(debug) << "\t\t in quad " << (*yitr).first << ", not single x,y tracks: " << nXinQuad << " of x and "<< nYinQuad << " of y";
660
661 if (nYinQuad == 2) {// let's start here and see if we can do >= 2 later
662 //here one could do sth to avoid checking the other y track again in the outermost loop over y
663 auto yitr1 = yQuadMap.lower_bound( (*yitr).first );
664 auto yitr2 = yQuadMap.upper_bound( (*yitr).first );
665 yitr2--; //back up once
666 y1 = ((*yitr1).second).getCentroidY() * yConvFactor_ + yStart_;
667 y2 = ((*yitr2).second).getCentroidY() * yConvFactor_ + yStart_;
668 sy1 = ((*yitr1).second).getResidual()*yConvFactor_;
669 sy2 = ((*yitr2).second).getResidual()*yConvFactor_;
670 y = (y1+y2)/2.;
671 sy = fabs(y1-y2)/2*yConvFactor_;
672 if (verbose_)
673 ldmx_log(debug) << "\t\t -- 2 y in quad: setting x track y coordinate to midpoint";
674 } //2y in quad
675
676 if (nYinQuad == 1 && nXinQuad == 2) {// don't think we want to experiment with discerning three overlapping tracks, so not >= 2
677
678 //first: set the y track coordinates to x = the mid of x tracks, y = y of y track
679 auto yidx = yIdxQuadMap.find( (*yitr).first );
680 tracks.at( (*yidx).second).setPosition(x,y);
681 tracks.at( (*yidx).second).setSigmaXY(sx, sy);
682
683 int minOverlapPE_ = 250;
684 if (((*yitr).second).getPE() < minOverlapPE_ ) {
685 // can't tell, really, that either of these belong to the y track. so.
686 // let them keep their own x coordinate but set y to quadrant midpoint, with uncertainty +/- half quadrant width (1/8 of pad height)
687 y = y0;
688 sy = sy0;
689 if (verbose_)
690 ldmx_log(debug) << "\t\t -- Can't tell which x track should be matched to single y track. Setting both x track coordinates to y quadrant value:" ;
691 }// if can't assume overlap
692 else
693 if (verbose_)
694 ldmx_log(debug) << "\t\t -- Found large PE count (" << ((*yitr).second).getPE() << " > " << minOverlapPE_
695 << "), suggesting overlap! Setting both x track coordinates to y track value:" ;
696
697 // consider making two x tracks out if this one, and, anyway have to set their average as the y track x cocordinate
698 //EXPERIMENTAL : apply only to x tracks, which can be disregarded for electron counting
699 if (verbose_) ldmx_log(debug) << "\t\t -- (x1, x2, y) = ("
700 << x1 << ", " << x2 << ", " << y << ") and (sx1, sx2, sy) = "
701 << sx1 << ", " << sx2 << ", "<< sy << ")";
702
703 //now set x track coordinates according to overlap check result
704 auto xidx1 = xIdxQuadMap.lower_bound( (*yitr).first );
705 auto xidx2 = xIdxQuadMap.upper_bound( (*yitr).first );
706 xidx2--; //upper_bound points to (last+1) element
707 tracks.at( (*xidx1).second).setPosition(x1,y);
708 tracks.at( (*xidx1).second).setSigmaXY(sx1, sy);
709 tracks.at( (*xidx2).second).setPosition(x2,y);
710 tracks.at( (*xidx2).second).setSigmaXY(sx2, sy);
711
712 }// 1 y, 2 x tracks in the quadrant
713 else if (nYinQuad == 2 && nXinQuad == 1) {
714 // 5b) if there are more y than x: could be an overlap
715
716 //first: set the x track coordinates to x = x of x track, y = the mid of y tracks
717 auto xidx = xIdxQuadMap.find( (*yitr).first );
718 tracks.at( (*xidx).second).setPosition(x,y);
719 tracks.at( (*xidx).second).setSigmaXY(sx, sy);
720
721 auto xitr = xQuadMap.lower_bound( (*yitr).first );
722 int minOverlapPE_ = 300;
723 if (((*xitr).second).getPE() < minOverlapPE_ ) {
724 if (verbose_)
725 ldmx_log(debug) << "\t\t just 1 x track with not-unusual PE in the quad -- can't match; setting mid-point values for x ";
726 x = x0;
727 sx = sx0 ;
728 }//if can't assume overlap
729 else {
730 // consider making two x tracks out if this one, and, anyway have to set their average as the y track x cocordinate
731 //EXPERIMENTAL : apply only to x tracks, which can be disregarded for electron counting
732 if (verbose_)
733 ldmx_log(debug) << "\t\t -- Found large PE count (" << ((*xitr).second).getPE() << " > " << minOverlapPE_
734 << ") in x track, suggesting overlap! Setting both y track coordinates to x track value:" ;
735 }// if can assume overlap
736 if (verbose_)
737 ldmx_log(debug) << "\t\t -- (x, y1, y2) = ("
738 << x << ", " << y1 << ", " << y2 << ") and (sx, sy1, sy2) = "
739 << sx << ", " << sy1 << ", "<< sy2 << ")";
740
741 auto yidx1 = yIdxQuadMap.lower_bound( (*yitr).first );
742 auto yidx2 = yIdxQuadMap.upper_bound( (*yitr).first );
743 yidx2--; //upper_bound points to next element
744 tracks.at( (*yidx1).second).setPosition(x,y1);
745 tracks.at( (*yidx1).second).setSigmaXY(sx, sy1);
746 tracks.at( (*yidx2).second).setPosition(x,y2);
747 tracks.at( (*yidx2).second).setSigmaXY(sx, sy2);
748
749 }// 2 y and 1 x track in quad
750 else if ( nYinQuad == 2 && nXinQuad == 2 ) {
751 //MIDPONTS ALL OVER!
752 auto xidx1 = xIdxQuadMap.lower_bound( (*yitr).first );
753 auto xidx2 = xIdxQuadMap.upper_bound( (*yitr).first );
754 xidx2--;
755 auto yidx1 = yIdxQuadMap.lower_bound( (*yitr).first );
756 auto yidx2 = yIdxQuadMap.upper_bound( (*yitr).first );
757 yidx2--;
758
759 if ( yIdxQuadMap.find( (*yitr).first )== yIdxQuadMap.end() )
760 ldmx_log(error) << "The two y tracks in the same quadrant at " << (*yitr).first << " appear to not be found in the y track map! investigate. Note that yidx1.first = " << (*yidx1).first << " and yidx2.first = " << (*yidx2).first ;
761 else {
762 tracks.at( (*xidx1).second).setPosition(x1,y);
763 tracks.at( (*xidx1).second).setSigmaXY(sx1, sy);
764 tracks.at( (*xidx2).second).setPosition(x2,y);
765 tracks.at( (*xidx2).second).setSigmaXY(sx2, sy);
766
767 tracks.at( (*yidx1).second).setPosition(x,y1);
768 tracks.at( (*yidx1).second).setSigmaXY(sx, sy1);
769 tracks.at( (*yidx2).second).setPosition(x,y2);
770 tracks.at( (*yidx2).second).setSigmaXY(sx, sy2);
771
772 if (verbose_)
773 ldmx_log(debug) << "\t\t -- in a 2 x 2 situaiton; midpoint y: " << y << " for both x tracks, midpoint x: " << x << " for both y tracks";
774 }
775 } // if 2 y, 2 x tracks
776
777 if ( nXinQuad > 2) {
778 if (verbose_)
779 ldmx_log(debug) << "\t\t -*-*-*- more than 2 x tracks in the same quad -- nothing done about the x,y coordinates in this situation -- implement if needed!!" ;
780 }
781 if ( nYinQuad > 2) {
782 if (verbose_)
783 ldmx_log(debug) << "\t\t -*-*-*- more than 2 y tracks in the same quad -- nothing done about the x,y coordinates in this situation -- implement if needed!!" ;
784 }
785
786 }// over y tracks
787
788 yQuadMap.clear();
789 xQuadMap.clear();
790
791
