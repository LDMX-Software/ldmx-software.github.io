321 ldmx::TrigScintTrack TrigScintTrackProducer::makeTrack(
322  std::vector<ldmx::TrigScintCluster> clusters) {
323  // for now let's keep a straight, unweighted centroid
324  // consider the possibility that at least one cluster has a centroid
325  // identically == 0. then we need to shift them by 1 if we want to do energy
326  // weighted track centroid later. but no need now
327  ldmx::TrigScintTrack tr;
328  float centroid = 0;
329  float beamEfrac = 0;
330  for (uint i = 0; i < clusters.size(); i++) {
331  centroid += (clusters.at(i)).getCentroid();
332  tr.addConstituent(clusters.at(i));
333  beamEfrac += (clusters.at(i)).getBeamEfrac();
334  }
335  centroid /= clusters.size();
336  beamEfrac /= clusters.size();
337 
338  float residual = 0;
339  for (uint i = 0; i < clusters.size(); i++)
340  residual += ((clusters.at(i)).getCentroid() - centroid) *
341  ((clusters.at(i)).getCentroid() - centroid);
342  residual = sqrt((float)(residual / clusters.size()));
343 
344  tr.setCentroid(centroid);
345  tr.setResidual(residual);
346  tr.setBeamEfrac(beamEfrac);
347 
348  if (verbose_) {
349  ldmx_log(debug) << " -- In makeTrack made track with centroid "
350  << centroid << " and residual " << residual
351  << " from clusters with centroids";
352  for (uint i = 0; i < clusters.size(); i++)
353  ldmx_log(debug) << "\tpad " << i << ": centroid "
354  << (clusters.at(i)).getCentroid();
355  ldmx_log(debug);
356  }
357 
358  return tr;
359 }
