29void TrigScintRecHitProducer::produce(framework::Event &event) {
30 // initialize QIE object for linearizing ADCs
31 SimQIE qie;
32
33 // Ensure the sample of interest <4
34 /* // this assumes we are in well-behaved simulation land, not test beam wilderness
35 if(sample_of_interest_>3) {
36 ldmx_log(error)<<"sample_of_interest_ should be one of 0,1,2,3\n"
37 <<"Currently, sample_of_interest = "<<sample_of_interest_
38 <<"\n";
39 return;
40 }
41 */
42
43 // looper over sim hits and aggregate energy depositions
44 // for each detID
45 const auto digis{event.getCollection<trigscint::TrigScintQIEDigis>(
46 inputCollection_, inputPassName_)};
47
48 std::vector<ldmx::TrigScintHit> trigScintHits;
49 for (const auto &digi : digis) {
50 ldmx::TrigScintHit hit;
51 auto adc{digi.getADC()};
52 auto tdc{digi.getTDC()};
53
54 hit.setModuleID(0);
55 hit.setBarID(digi.getChanID());
56 hit.setBeamEfrac(-1.);
57
58 hit.setAmplitude(qie.ADC2Q(adc[sample_of_interest_]) +
59 qie.ADC2Q(adc[sample_of_interest_+1])); // femptocoulombs
60
61 if (tdc[sample_of_interest_] > 49)
62 hit.setTime(-999.);
63 else
64 hit.setTime(tdc[sample_of_interest_] * 0.5);
65
66 hit.setEnergy((qie.ADC2Q(adc[sample_of_interest_])+
67 qie.ADC2Q(adc[sample_of_interest_+1])-pedestal_)
68 *6250. / gain_ * mevPerMip_ / pePerMip_); // MeV
69 hit.setPE((qie.ADC2Q(adc[sample_of_interest_])+
70 qie.ADC2Q(adc[sample_of_interest_+1]) - pedestal_)
71 *6250. / gain_);
72 trigScintHits.push_back(hit);
73 }
74 // Create the container to hold the
75 // digitized trigger scintillator hits.
76
77 event.add(outputCollection_, trigScintHits);
78}
