32 
33  // Ensure the sample of interest <4

40 
41  // looper over sim hits and aggregate energy depositions

44  inputCollection_, inputPassName_)};
45 
46  std::vector<ldmx::TrigScintHit> trigScintHits;
47  for (const auto &digi : digis) {

50  auto tdc{digi.getTDC()};
51 
52  hit.setModuleID(0);

54  hit.setBeamEfrac(-1.);
55 
56  hit.setAmplitude(qie.ADC2Q(adc[sample_of_interest_]) +
57  qie.ADC2Q(adc[sample_of_interest_+1])); // femptocoulombs
58 
59  if (tdc[sample_of_interest_] > 49)
60  hit.setTime(-999.);
61  else
62  hit.setTime(tdc[sample_of_interest_] * 0.5);
63 
64  hit.setEnergy((qie.ADC2Q(adc[sample_of_interest_])+
65  qie.ADC2Q(adc[sample_of_interest_+1])-pedestal_)
66  *6250. / gain_ * mevPerMip_ / pePerMip_); // MeV
67  hit.setPE((qie.ADC2Q(adc[sample_of_interest_])+
68  qie.ADC2Q(adc[sample_of_interest_+1]) - pedestal_)
69  *6250. / gain_);

71  }
