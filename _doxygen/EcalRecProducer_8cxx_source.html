90  int numDigiHits = ecalDigis.getNumDigis();
91  //loop through digis
92  for ( unsigned int iDigi = 0; iDigi < numDigiHits; iDigi++ ) {
93 
94  //Right now, hard-coded to only use one sample in EcalDigiProducer
95  //TODO: expand to multiple samples per digi
96  EcalDigiSample sample = (ecalDigis.getDigi( iDigi )).at(0);
97  EcalID id(sample.rawID_);
98 
99  //ID to real space position
100  double x,y,z;
101  ecalHexReadout_->getCellAbsolutePosition( id , x , y , z );
102 
103  //get energy and time estimate from digi information
104  double siEnergy, hitTime;
105 
106  //TODO: Energy estimate from N samples can (and should be) refined
107  //TOA is the time of arrival with respect to the 25ns clock window
108  double timeRelClock25 = sample.toa_*(25./1024); //ns
109  hitTime = timeRelClock25;
110 
111  //ADC - voltage measurement at a specific time of the pulse
112  // Pulse Shape:
113  // p[0]/(1.0+exp(p[1](t-p[2]+p[3]-p[4])))/(1.0+exp(p[5]*(t-p[6]+p[3]-p[4])))
114  // p[0] = amplitude to be fit (TBD)
115  // p[1] = -0.345 shape parameter - rate of up slope
116  // p[2] = 70.6547 shape parameter - time of up slope relative to shape fit
117  // p[3] = 77.732 shape parameter - time of peak relative to shape fit
118  // p[4] = peak time to be fit (TBD)
119  // p[5] = 0.140068 shape parameter - rate of down slope
120  // p[6] = 87.7649 shape paramter - time of down slope relative to shape fit
121  //These measurements can be used to fit the pulse shape if TOT is not available
122 
123  //TOT - number of clock ticks that pulse was over threshold
124  // this is related to the amplitude of the pulse through some convoluted relation using the pulse shape
125  // the amplitude of the pulse is related to the energy deposited
126  // TODO actually have mutliple samples instead of having adc_t_ count number of samples over threshold
127  siEnergy = convertTOT( sample.adc_t_*1024 + sample.tot_ );
128  //printf( "%6d Clocks and %6d tot --> %6.2f MeV\n" , sample.adc_t_ , sample.tot_ , siEnergy );
129 
130  //incorporate layer weights
131  int layer = id.layer();
132  double recHitEnergy =
133  ( (siEnergy / MIP_SI_RESPONSE)*layerWeights_.at(layer)+siEnergy )*secondOrderEnergyCorrection_;
134 
135  //copy over information to rec hit structure in new collection
136  EcalHit recHit;
137  recHit.setID( id.raw() );
138  recHit.setXPos( x );
139  recHit.setYPos( y );
140  recHit.setZPos( z );
141  recHit.setAmplitude( siEnergy );
142  recHit.setEnergy( recHitEnergy );
143  recHit.setTime( hitTime );
144 
145  ecalRecHits.push_back( recHit );
146  }
