18void SingleSubsystemUnpacker::produce(framework::Event& event) {
19 if (!reader_ or reader_.eof()) abortEvent();
20
21 std::vector<uint8_t> buff;
22 if(!reader_.read(buff, num_bytes_per_event_)) {
23 EXCEPTION_RAISE("MalForm",
24 "Raw file provided was unable to read "+std::to_string(num_bytes_per_event_)
25 +" bytes in an event.");
26 }
27 event.add(output_name_, buff);
