60void TrigScintQIEDigiProducer::produce(framework::Event& event) {
61 // Need to handle seeding on the first event
62 if (random_.get() == nullptr) {
63 const auto& rseed = getCondition<framework::RandomNumberSeedService>(
64 framework::RandomNumberSeedService::CONDITIONS_OBJECT_NAME);
65 const auto& rseed2 = getCondition<framework::RandomNumberSeedService>(
66 framework::RandomNumberSeedService::CONDITIONS_OBJECT_NAME);
67
68 random_ = std::make_unique<TRandom3>(rseed.getSeed(outputCollection_));
69
70 // Initialize SimQIE instance with
71 // pedestal, electronic noise and the random seed
72 smq_ = new SimQIE(pedestal_, elec_noise_,
73 rseed2.getSeed(outputCollection_ + "SimQIE"));
74
75 smq_->setGain(sipm_gain_);
76 smq_->setFreq(s_freq_);
77 smq_->setNTimeSamples(maxts_);
78 smq_->setTDCThreshold(tdc_thr_);
79 }
80
81 // To simulate multiple pulses coming at different times, SiPMS
82 float TrueEdep[stripsPerArray_];
83 Expo* ex[stripsPerArray_] = {0};
84 for (int i = 0; i < stripsPerArray_; i++) {
85 // Set the pulse shape with fixed parameters given by config. file
86 ex[i] = new Expo(pulse_params_[0], pulse_params_[1]);
87 TrueEdep[i] = 0;
88 }
89
90 // loop over sim hits and aggregate energy depositions for each detID
91 const auto simHits{event.getCollection<ldmx::SimCalorimeterHit>(
92 inputCollection_, inputPassName_)};
93
94 for (const auto& simHit : simHits) {
95 ldmx::TrigScintID id(simHit.getID());
96
97 ldmx_log(info) << "Processing sim hit with bar ID: " << id.bar();
98
99 // Simulating the noise corresponding to uncertainity in
100 // detecting scintillating photons.
101 // Poissonian distribution with mean = mean PEs generated
102 double PulseAmp =
103 random_->Poisson(simHit.getEdep() / mevPerMip_ * pePerMip_);
104
105 // Adding a pulse for every sim hit recorded.
106 // time offset = global offset+simhit time
107 ex[id.bar()]->AddPulse(toff_overall_ + simHit.getTime(), PulseAmp);
108
109 // incrementing true energy deposited in appropriate bar.
110 TrueEdep[id.bar()] += simHit.getEdep();
111 }
112
113 // A container to hold the digitized trigger scintillator hits.
114 std::vector<trigscint::TrigScintQIEDigis> QDigis;
115
116 double TotalNoise = meanNoise_ * maxts_;
117
118 // time period[ns] = 1000/sampling freq.[MHz]
119 double SamplingTime = 1000 / s_freq_;
120
121 // Loop over all the bars available.
122 for (int bar_id = 0; bar_id < stripsPerArray_; bar_id++) {
123 // Dark current simulation
124 // e-hole pairs may be generated at random times in SiPM
125 // due to thermal fluctuations.
126 // Every e- thus generated, mimicks a Photo Electron.
127 // Hence we will creat 1PE pulses for each electron generated.
128 int n_noise_pulses = random_->Poisson(TotalNoise);
129 for (int i = 0; i < n_noise_pulses; i++) {
130 ex[bar_id]->AddPulse(random_->Uniform(0, maxts_ * SamplingTime), 1);
131 }
132
133 // Storing the "good" digis
134 if (smq_->PulseCut(ex[bar_id])) {
135 trigscint::TrigScintQIEDigis QIEInfo;
136
137 QIEInfo.setChanID(bar_id);
138 QIEInfo.setADC(smq_->Out_ADC(ex[bar_id]));
139 QIEInfo.setTDC(smq_->Out_TDC(ex[bar_id]));
140 QIEInfo.setCID(smq_->CapID(ex[bar_id]));
141
142 QDigis.push_back(QIEInfo);
143 }
144 }
145 event.add(outputCollection_, QDigis);
146}
