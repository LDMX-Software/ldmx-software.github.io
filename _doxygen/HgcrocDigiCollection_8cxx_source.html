7 HgcrocDigiCollection::Sample::Sample(bool tot_progress, bool tot_complete,
8 int firstMeas, int seconMeas, int toa, int version) {
9
10 version_ = version;
11 if (version_ == 2) {
12 // version 2 HGC ROC had a much simpler word structure
13 // 12-bit TOT | 10-bit TOA | 10-bit ADCt
14 // we assume that firstMeas is adct and secondMeas is tot
15 word_ =
16 (((seconMeas > 0xfff ? 0xfff : seconMeas) & 0xfff) << 20)
17 +(((firstMeas > TEN_BIT_MASK ? TEN_BIT_MASK : firstMeas) & TEN_BIT_MASK) << 10)
18 +((toa > TEN_BIT_MASK ? TEN_BIT_MASK : toa) & TEN_BIT_MASK);
19
20 } else {
21 if (not tot_progress and tot_complete) {
22 // the 12 bit internal tot measurement needs to
23 // be packed into a 10 bit int
24 if (seconMeas > 512)
25 seconMeas =
26 512 + seconMeas / 8; // lost some precision but can go higher
27 }
28
29 // check if over largest number possible ==> set to largest if over (don't
30 // want wrapping) and then do bit shifting nonsense to code the measurements
31 // into the 32-bit word set last measurement to TOA
32 word_ =
33 (tot_progress << FIRSTFLAG_POS) + (tot_complete << SECONFLAG_POS) +
34 (((firstMeas > TEN_BIT_MASK ? TEN_BIT_MASK : firstMeas) & TEN_BIT_MASK)
35 << FIRSTMEAS_POS) +
36 (((seconMeas > TEN_BIT_MASK ? TEN_BIT_MASK : seconMeas) & TEN_BIT_MASK)
37 << SECONMEAS_POS) +
38 (((toa > TEN_BIT_MASK ? TEN_BIT_MASK : toa) & TEN_BIT_MASK));
39 }
40 }

65 void HgcrocDigiCollection::addDigi(
66 unsigned int id, const std::vector<HgcrocDigiCollection::Sample> &digi) {
67 if (digi.size() != this->getNumSamplesPerDigi()) {
68 std::cerr << "[ WARN ] [ HgcrocDigiCollection ] Input list of samples "
69 "has size '"
70 << digi.size()
71 << "' that does not match the number of samples per digi '"
72 << this->getNumSamplesPerDigi() << "'!." << std::endl;
73 return;
74 }
75
76 channelIDs_.push_back(id);
77 for (auto const &s : digi) samples_.push_back(s.raw());
78
79 return;
80 }
