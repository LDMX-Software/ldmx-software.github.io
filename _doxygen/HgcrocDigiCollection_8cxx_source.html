8  HgcrocDigiCollection::Sample::Sample(bool tot_progress, bool tot_complete, int firstMeas, int seconMeas, int toa) {
9 
10  if ( not tot_progress and tot_complete ) {
11  //the 12 bit internal tot measurement needs to
12  // be packed into a 10 bit int
13  if ( seconMeas > 512 )
14  seconMeas = 512 + seconMeas/8; //lost some precision but can go higher
15  }
16 
17  //check if over largest number possible ==> set to largest if over (don't want wrapping)
18  //and then do bit shifting nonsense to code the measurements into the 32-bit word
19  //set last measurement to TOA
20  word_ = (tot_progress << FIRSTFLAG_POS)
21  + (tot_complete << SECONFLAG_POS)
22  + ( (( firstMeas > TEN_BIT_MASK ? TEN_BIT_MASK : firstMeas) & TEN_BIT_MASK) << FIRSTMEAS_POS )
23  + ( (( seconMeas > TEN_BIT_MASK ? TEN_BIT_MASK : seconMeas) & TEN_BIT_MASK) << SECONMEAS_POS )
24  + ( (( toa > TEN_BIT_MASK ? TEN_BIT_MASK : toa ) & TEN_BIT_MASK) );
25 
26  }
