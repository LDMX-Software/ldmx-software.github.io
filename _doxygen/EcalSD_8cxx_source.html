26G4bool EcalSD::ProcessHits(G4Step* aStep, G4TouchableHistory*) {
27 static const int layer_depth = 2; // index depends on GDML implementation
28 const auto& geometry = getCondition<ldmx::EcalGeometry>(
29 ldmx::EcalGeometry::CONDITIONS_OBJECT_NAME);
30
31 // Get the edep from the step.
32 G4double edep = aStep->GetTotalEnergyDeposit();
33
34 // Skip steps with no energy dep which come from non-Geantino particles.
35 if (edep == 0.0 and not isGeantino(aStep)) {
36 if (verboseLevel > 2) {
37 G4cout << "CalorimeterSD skipping step with zero edep." << G4endl
38 << G4endl;
39 }
40 return false;
41 }
42
43 // Compute the hit position
44 G4StepPoint* prePoint = aStep->GetPreStepPoint();
45 G4StepPoint* postPoint = aStep->GetPostStepPoint();
46 G4ThreeVector position =
47 0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
48
49 // Create the ID for the hit.
50 int cpynum = aStep->GetPreStepPoint()
51 ->GetTouchableHandle()
52 ->GetHistory()
53 ->GetVolume(layer_depth)
54 ->GetCopyNo();
55 int layerNumber;
56 layerNumber = cpynum / 7;
57 int module_position = cpynum % 7;
69 // fastest, but need to trust module number between GDML and EcalGeometry
70 // match
71 ldmx::EcalID id =
72 geometry.getID(position[0], position[1], layerNumber, module_position);
73
74 // medium, only need to trust z-layer positions in GDML and EcalGeometry match
75 // helpful for debugging any issues where transverse position is not
76 // matching between the GDML and EcalGeometry
77 // ldmx::EcalID id = geometry.getID(position[0], position[1], layerNumber);
78
79 // slowest, completely rely on EcalGeometry
80 // this is helpful for validating the EcalGeometry implementation and
81 // configuration since this will be called with any hit position that
82 // is inside of the configured SD volumes from Geant4's point of view
83 // ldmx::EcalID id = geometry.getID(position[0], position[1], position[2]);
84
85 if (hits_.find(id) == hits_.end()) {
86 // hit in empty cell
87 auto& hit = hits_[id];
88 hit.setID(id.raw());
98 auto [x, y, z] = geometry.getPosition(id);
99 hit.setPosition(x, y, z);
100 }
101
102 auto& hit = hits_[id];
103
104 // hit variables
105 auto track = aStep->GetTrack();
106 auto time = track->GetGlobalTime();
107 auto track_id = track->GetTrackID();
108 auto pdg = track->GetParticleDefinition()->GetPDGEncoding();
109
110 if (enableHitContribs_) {
111 int contrib_i = hit.findContribIndex(track_id, pdg);
112 if (compressHitContribs_ and contrib_i != -1) {
113 hit.updateContrib(contrib_i, edep, time);
114 } else {
115 hit.addContrib(getTrackMap().findIncident(track_id), track_id, pdg, edep,
116 time);
117 }
118 } else {
119 // no hit contribs and hit already exists
120 hit.setEdep(hit.getEdep() + edep);
121 if (time < hit.getTime() or hit.getTime() == 0) {
122 hit.setTime(time);
123 }
124 }
125
126 return true;
127}

129void EcalSD::saveHits(framework::Event& event) {
130 // squash hits into list
131 std::vector<ldmx::SimCalorimeterHit> hits;
132 hits.reserve(hits_.size());
133 for (const auto& [id, hit] : hits_) hits.push_back(hit);
134 event.add(COLLECTION_NAME, hits);
135}
