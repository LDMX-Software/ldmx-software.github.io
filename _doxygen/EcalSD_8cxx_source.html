80  G4bool EcalSD::ProcessHits(G4Step* aStep, G4TouchableHistory*) {
81 
82  // Determine if current particle of this step is a Geantino.
83  G4ParticleDefinition* pdef = aStep->GetTrack()->GetDefinition();
84  bool isGeantino = false;
85  if (pdef == G4Geantino::Definition() || pdef == G4ChargedGeantino::Definition()) {
86  isGeantino = true;
87  }
88 
89  // Get the edep from the step.
90  G4double edep = aStep->GetTotalEnergyDeposit();
91 
92  // Skip steps with no energy dep which come from non-Geantino particles.
93  if (edep == 0.0 && !isGeantino) {
94  if (verboseLevel > 2) {
95  G4cout << "CalorimeterSD skipping step with zero edep." << G4endl << G4endl;
96  }
97  return false;
98  }
99 
100  // Create a new cal hit.
101  G4CalorimeterHit* hit = new G4CalorimeterHit();
102 
103  // Set the edep.
104  hit->setEdep(edep);
105 
106  // Compute the hit position using the utility function.
107  G4ThreeVector hitPosition = getHitPosition(aStep);
108  hit->setPosition(hitPosition.x(), hitPosition.y(), hitPosition.z());
109 
110  // Set the global time.
111  hit->setTime(aStep->GetTrack()->GetGlobalTime());
112 
113  // Create the ID for the hit.
114  int cpynum = aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()->GetVolume(layerDepth_)->GetCopyNo();
115  int layerNumber;
116  layerNumber = int(cpynum/7);
117  int module_position = cpynum%7;
118 
119  int cellModuleID = hitMap_->getCellModuleID(hitPosition[0], hitPosition[1]);
120  int cellID = (hitMap_->separateID(cellModuleID)).first;
121  detID_->setFieldValue(1, layerNumber);
122  detID_->setFieldValue(2, module_position);
123  detID_->setFieldValue(3, cellID);
124  hit->setID(detID_->pack());
125 
126  // Set the track ID on the hit.
127  hit->setTrackID(aStep->GetTrack()->GetTrackID());
128 
129  // Set the PDG code from the track.
130  hit->setPdgCode(aStep->GetTrack()->GetParticleDefinition()->GetPDGEncoding());
131 
132  if (this->verboseLevel > 2) {
133  G4cout << "Created new SimCalorimeterHit in detector " << this->GetName() << " with subdet ID " << subdet_ << " and layer " << layerNumber << " and cellid " << cellID << " module position " << module_position << " ...";
134  hit->Print();
135  G4cout << G4endl;
136  }
137 
138  // Insert the hit into the hits collection.
139  hitsCollection_->insert(hit);
140 
141  return true;
142  }

144  G4ThreeVector EcalSD::getHitPosition(G4Step* aStep) {
145 
149  G4StepPoint* prePoint = aStep->GetPreStepPoint();
150  G4StepPoint* postPoint = aStep->GetPostStepPoint();
151  G4ThreeVector position = 0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
152 
153  /*
154  * Get the volume position in global coordinates, which for the ECal is the center of
155  * the front face of the sensor.
156  */
157  G4ThreeVector volumePosition = aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()->GetTopTransform().Inverse().TransformPoint(G4ThreeVector());
158 
159  // Get the solid from this step.
160  G4VSolid* solid = prePoint->GetTouchableHandle()->GetVolume()->GetLogicalVolume()->GetSolid();
161  auto it = polyMap_.find(solid);
162  G4Polyhedron* poly;
163  if (it == polyMap_.end()) {
164  poly = solid->CreatePolyhedron();
165  polyMap_[solid] = poly;
166  } else {
167  poly = polyMap_[solid];
168  }
169 
173  G4Point3D iNodes[4];
174  G4int n;
175  poly->GetFacet(1, n, iNodes);
176  G4double zstart = iNodes[1][2];
177  G4double zend = iNodes[0][2];
178  G4double zmid = (zstart - zend) / 2;
179  position.setZ(volumePosition.z() + zmid);
180 
181  return position;
182  }
