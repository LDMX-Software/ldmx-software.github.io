24 G4bool EcalSD::ProcessHits(G4Step* aStep, G4TouchableHistory*) {
25  const ldmx::EcalHexReadout& hitMap =
26  conditionsIntf_.getCondition<ldmx::EcalHexReadout>(
27  ldmx::EcalHexReadout::CONDITIONS_OBJECT_NAME);
28 
29  // Determine if current particle of this step is a Geantino.
30  G4ParticleDefinition* pdef = aStep->GetTrack()->GetDefinition();
31  bool isGeantino = false;
32  if (pdef == G4Geantino::Definition() ||
33  pdef == G4ChargedGeantino::Definition()) {
34  isGeantino = true;
35  }
36 
37  // Get the edep from the step.
38  G4double edep = aStep->GetTotalEnergyDeposit();
39 
40  // Skip steps with no energy dep which come from non-Geantino particles.
41  if (edep == 0.0 && !isGeantino) {
42  if (verboseLevel > 2) {
43  G4cout << "CalorimeterSD skipping step with zero edep." << G4endl
44  << G4endl;
45  }
46  return false;
47  }
48 
49  // Create a new cal hit.
50  G4CalorimeterHit* hit = new G4CalorimeterHit();
51 
52  // Set the edep.
53  hit->setEdep(edep);
54 
55  // Compute the hit position using the utility function.
56  G4ThreeVector hitPosition = getHitPosition(aStep);
57  hit->setPosition(hitPosition.x(), hitPosition.y(), hitPosition.z());
58 
59  // Set the global time.
60  hit->setTime(aStep->GetTrack()->GetGlobalTime());
61 
62  // Create the ID for the hit.
63  int cpynum = aStep->GetPreStepPoint()
64  ->GetTouchableHandle()
65  ->GetHistory()
66  ->GetVolume(layerDepth_)
67  ->GetCopyNo();
68  int layerNumber;
69  layerNumber = int(cpynum / 7);
70  int module_position = cpynum % 7;
71 
72  ldmx::EcalID partialId =
73  hitMap.getCellModuleID(hitPosition[0], hitPosition[1]);
74  ldmx::EcalID id(layerNumber, module_position, partialId.cell());
75  hit->setID(id.raw());
76 
77  // Set the track ID on the hit.
78  hit->setTrackID(aStep->GetTrack()->GetTrackID());
79 
80  // Set the PDG code from the track.
81  hit->setPdgCode(aStep->GetTrack()->GetParticleDefinition()->GetPDGEncoding());
82 
83  if (this->verboseLevel > 2) {
84  G4cout << "Created new SimCalorimeterHit in detector " << this->GetName()
85  << " with subdet ID " << id << " ...";
86  hit->Print();
87  G4cout << G4endl;
88  }
89 
90  // Insert the hit into the hits collection.
91  hitsCollection_->insert(hit);
92 
93  return true;
94 }

96 G4ThreeVector EcalSD::getHitPosition(G4Step* aStep) {
100  G4StepPoint* prePoint = aStep->GetPreStepPoint();
101  G4StepPoint* postPoint = aStep->GetPostStepPoint();
102  G4ThreeVector position =
103  0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
104 
105  /*
106  * Get the volume position in global coordinates, which for the ECal is the
107  * center of the front face of the sensor.
108  */
109  G4ThreeVector volumePosition = aStep->GetPreStepPoint()
110  ->GetTouchableHandle()
111  ->GetHistory()
112  ->GetTopTransform()
113  .Inverse()
114  .TransformPoint(G4ThreeVector());
115 
116  // Get the solid from this step.
117  G4VSolid* solid = prePoint->GetTouchableHandle()
118  ->GetVolume()
119  ->GetLogicalVolume()
120  ->GetSolid();
121  auto it = polyMap_.find(solid);
122  G4Polyhedron* poly;
123  if (it == polyMap_.end()) {
124  poly = solid->CreatePolyhedron();
125  polyMap_[solid] = poly;
126  } else {
127  poly = polyMap_[solid];
128  }
129 
133  G4Point3D iNodes[4];
134  G4int n;
135  poly->GetFacet(1, n, iNodes);
136  G4double zstart = iNodes[1][2];
137  G4double zend = iNodes[0][2];
138  G4double zmid = (zstart - zend) / 2;
139  position.setZ(volumePosition.z() + zmid);
140 
141  return position;
142 }
