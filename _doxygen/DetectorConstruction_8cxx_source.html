128void DetectorConstruction::ConstructSDandField() {
129 // Biasing operators were created in RunManager::setupPhysics
130 // which is called before G4RunManager::Initialize
131 // which is where this method ends up being called.
132
133 auto bops{simcore::PluginFactory::getInstance().getBiasingOperators()};
134 for (simcore::XsecBiasingOperator* bop : bops) {
135 logical_volume_tests::Test includeVolumeTest{nullptr};
136 if (bop->getVolumeToBias().compare("ecal") == 0) {
137 includeVolumeTest = &logical_volume_tests::isInEcal;
138 } else if (bop->getVolumeToBias().compare("old_ecal") == 0) {
139 includeVolumeTest = &logical_volume_tests::isInEcalOld;
140 } else if (bop->getVolumeToBias().compare("target") == 0) {
141 includeVolumeTest = &logical_volume_tests::isInTargetOnly;
142 } else if (bop->getVolumeToBias().compare("target_region") == 0) {
143 includeVolumeTest = &logical_volume_tests::isInTargetRegion;
144 } else if (bop->getVolumeToBias().compare("hcal") == 0) {
145 includeVolumeTest = &logical_volume_tests::isInHcal;
146 } else {
147 std::cerr << "[ DetectorConstruction ] : "
148 << "WARN - Requested volume to bias '" << bop->getVolumeToBias()
149 << "' is not recognized. Will attach volumes based on if their"
150 << " name contains the volume to bias." << std::endl;
151 includeVolumeTest = &logical_volume_tests::nameContains;
152 }
153
154 for (G4LogicalVolume* volume : *G4LogicalVolumeStore::GetInstance()) {
155 auto volume_name = volume->GetName();
156 if (includeVolumeTest(volume, bop->getVolumeToBias())) {
157 bop->AttachTo(volume);
158 std::cout << "[ DetectorConstruction ]: "
159 << "Attaching biasing operator " << bop->GetName()
160 << " to volume " << volume->GetName() << std::endl;
161 } // BOP attached to target or ecal
162 } // loop over volumes
163 } // loop over biasing operators
164}
