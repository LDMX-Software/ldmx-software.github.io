129 for (auto& det : sens_dets) {
130 // create
131 auto sd = SensitiveDetector::Factory::get().make(
132 det.getParameter<std::string>("class_name"),
133 det.getParameter<std::string>("instance_name"),
134 conditions_interface_, det);
135 // attach to volumes
136 for (G4LogicalVolume* volume : *G4LogicalVolumeStore::GetInstance()) {
137 if (sd->isSensDet(volume)) {
138 std::cout << "[ DetectorConstruction ] : "
139 << "Attaching " << sd->GetName()

141 volume->SetSensitiveDetector(sd);
142 }
143 }

146 // Biasing operators were created in RunManager::setupPhysics
147 // which is called before G4RunManager::Initialize
148 // which is where this method ends up being called.
149 simcore::XsecBiasingOperator::Factory::get().apply([](auto bop) {
150 logical_volume_tests::Test includeVolumeTest{nullptr};
151 if (bop->getVolumeToBias().compare("ecal") == 0) {
152 includeVolumeTest = &logical_volume_tests::isInEcal;
153 } else if (bop->getVolumeToBias().compare("old_ecal") == 0) {
154 includeVolumeTest = &logical_volume_tests::isInEcalOld;
155 } else if (bop->getVolumeToBias().compare("target") == 0) {
156 includeVolumeTest = &logical_volume_tests::isInTargetOnly;
157 } else if (bop->getVolumeToBias().compare("target_region") == 0) {
158 includeVolumeTest = &logical_volume_tests::isInTargetRegion;
159 } else if (bop->getVolumeToBias().compare("hcal") == 0) {
160 includeVolumeTest = &logical_volume_tests::isInHcal;
161 } else {
162 std::cerr << "[ DetectorConstruction ] : "
163 << "WARN - Requested volume to bias '" << bop->getVolumeToBias()
164 << "' is not recognized. Will attach volumes based on if their"
165 << " name contains the volume to bias." << std::endl;
166 includeVolumeTest = &logical_volume_tests::nameContains;
167 }
168
169 for (G4LogicalVolume* volume : *G4LogicalVolumeStore::GetInstance()) {
170 auto volume_name = volume->GetName();
171 if (includeVolumeTest(volume, bop->getVolumeToBias())) {
172 bop->AttachTo(volume);
173 std::cout << "[ DetectorConstruction ]: "
174 << "Attaching biasing operator " << bop->GetName()
175 << " to volume " << volume->GetName() << std::endl;
176 } // BOP attached to target or ecal
177 } // loop over volumes
178 }); // loop over biasing operators
179}
