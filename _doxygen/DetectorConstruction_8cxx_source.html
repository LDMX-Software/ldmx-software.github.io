113 void DetectorConstruction::ConstructSDandField() {
114  // Biasing operators were created in RunManager::setupPhysics
115  // which is called before G4RunManager::Initialize
116  // which is where this method ends up being called.
117 
118  auto bops{simcore::PluginFactory::getInstance().getBiasingOperators()};
119  for (simcore::XsecBiasingOperator* bop : bops) {
120  logical_volume_tests::Test includeVolumeTest{nullptr};
121  if (bop->getVolumeToBias().compare("ecal") == 0) {
122  includeVolumeTest = &logical_volume_tests::isInEcal;
123  } else if (bop->getVolumeToBias().compare("old_ecal") == 0) {
124  includeVolumeTest = &logical_volume_tests::isInEcalOld;
125  } else if (bop->getVolumeToBias().compare("target") == 0) {
126  includeVolumeTest = &logical_volume_tests::isInTargetOnly;
127  } else if (bop->getVolumeToBias().compare("target_region") == 0) {
128  includeVolumeTest = &logical_volume_tests::isInTargetRegion;
129  } else {
130  std::cerr << "[ DetectorConstruction ] : "
131  << "WARN - Requested volume to bias '" << bop->getVolumeToBias()
132  << "' is not recognized. Will attach volumes based on if their"
133  << " name contains the volume to bias." << std::endl;
134  includeVolumeTest = &logical_volume_tests::nameContains;
135  }
136 
137  for (G4LogicalVolume* volume : *G4LogicalVolumeStore::GetInstance()) {
138  auto volume_name = volume->GetName();
139  if (includeVolumeTest(volume, bop->getVolumeToBias())) {
140  bop->AttachTo(volume);
141  std::cout << "[ DetectorConstruction ]: "
142  << "Attaching biasing operator " << bop->GetName()
143  << " to volume " << volume->GetName() << std::endl;
144  } // BOP attached to target or ecal
145  } // loop over volumes
146  } // loop over biasing operators
147 }
