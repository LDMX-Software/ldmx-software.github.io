45static double energy(const std::vector<double>& p, const double& m) {
46 return sqrt(p.at(0)*p.at(0)+ p.at(1)*p.at(1)+ p.at(2)*p.at(2)+ m*m);

62void NtuplizeDarkBremInteraction::analyze(const framework::Event& e) {
63 const auto& particle_map{e.getMap<int,ldmx::SimParticle>("SimParticles")};
64 const ldmx::SimParticle *recoil{nullptr}, *aprime{nullptr}, *beam{nullptr};
65 for (const auto& [track_id, particle] : particle_map) {
66 if (track_id == 1) beam = &particle;
67 if (particle.getProcessType() == ldmx::SimParticle::ProcessType::eDarkBrem) {
68 if (particle.getPdgID() == 62) {
69 if (aprime != nullptr) {
70 EXCEPTION_RAISE("BadEvent", "Found multiple A' in event.");
71 }
72 aprime = &particle;
73 } else {
74 recoil = &particle;
75 }
76 }
77 }
78
79 if (recoil == nullptr and aprime == nullptr) {
80 /* dark brem did not occur during the simulation
81 * IF PROPERLY CONFIGURED, this occurs because the simulation
82 * exhausted the maximum number of tries to get a dark brem
83 * to occur. We just leave early so that the entries in the
84 * ntuple are the unphysical numeric minimum.
85 *
86 * This can also happen during development, so I leave a debug
87 * printout here to be uncommented when developing the dark
88 * brem simulation.
89 std::cout << "Event " << e.getEventNumber()
90 << " did not have a dark brem occur within it." << std::endl;
91 */
92 return;
93 }
94
95 if (recoil == nullptr or aprime == nullptr or beam == nullptr) {
96 // we are going to end processing so let's take our time to
97 // construct a nice error message
98 std::stringstream err_msg;
99 err_msg
100 << "Unable to final all necessary particles for DarkBrem interaction."
101 << " Missing: [ "
102 << (recoil == nullptr ? "recoil " : "")
103 << (aprime == nullptr ? "aprime " : "")
104 << (beam == nullptr ? "beam " : "")
105 << "]" << std::endl;
106 EXCEPTION_RAISE("BadEvent", err_msg.str());
107 return;
108 }
109
110 const auto& recoil_p = recoil->getMomentum();
111 const auto& aprime_p = aprime->getMomentum();
112
113 std::vector<double> incident_p = recoil_p;
114 for (std::size_t i{0}; i < recoil_p.size(); ++i) incident_p[i] += aprime_p.at(i);
115
116 double incident_energy = energy(incident_p, recoil->getMass());
117 double recoil_energy = energy(recoil_p, recoil->getMass());
118 double visible_energy = (beam->getEnergy() - incident_energy) + recoil_energy;
119
120 ntuple_.setVar<double>("x", aprime->getVertex().at(0));
121 ntuple_.setVar<double>("y", aprime->getVertex().at(1));
122 ntuple_.setVar<double>("z", aprime->getVertex().at(2));
123 ntuple_.setVar<double>("weight", e.getEventWeight());
124 ntuple_.setVar<int>("incident_pdg", recoil->getPdgID());
125 ntuple_.setVar<int>("incident_genstatus", -1);
126 ntuple_.setVar<double>("incident_mass", recoil->getMass());
127 ntuple_.setVar<double>("incident_energy", incident_energy);
128 ntuple_.setVar<double>("incident_px", incident_p.at(0));
129 ntuple_.setVar<double>("incident_py", incident_p.at(1));
130 ntuple_.setVar<double>("incident_pz", incident_p.at(2));
131 ntuple_.setVar<int>("recoil_pdg", recoil->getPdgID());
132 ntuple_.setVar<int>("recoil_genstatus", recoil->getGenStatus());
133 ntuple_.setVar<double>("recoil_mass", recoil->getMass());
134 ntuple_.setVar<double>("recoil_energy", recoil_energy);
135 ntuple_.setVar<double>("recoil_px", recoil_p.at(0));
136 ntuple_.setVar<double>("recoil_py", recoil_p.at(1));
137 ntuple_.setVar<double>("recoil_pz", recoil_p.at(2));
138 ntuple_.setVar<int>("aprime_pdg", aprime->getPdgID());
139 ntuple_.setVar<int>("aprime_genstatus", aprime->getGenStatus());
140 ntuple_.setVar<double>("aprime_mass", aprime->getMass());
141 ntuple_.setVar<double>("aprime_energy", energy(aprime_p,aprime->getMass()));
142 ntuple_.setVar<double>("aprime_px", aprime_p.at(0));
143 ntuple_.setVar<double>("aprime_py", aprime_p.at(1));
144 ntuple_.setVar<double>("aprime_pz", aprime_p.at(2));
145 ntuple_.setVar<double>("beam_energy", beam->getEnergy());
146 ntuple_.setVar<double>("visible_energy", visible_energy);
147}
