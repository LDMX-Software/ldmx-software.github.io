57  template <typename T>
58  void addVar(const std::string& tname, const std::string& vname) {
59 
60  // Check if the variable exists in the map. If it does, throw
61  // an exception.
62  if (variables_.count(vname) != 0) {
63  EXCEPTION_RAISE(
64  "NtupleManager",
65  "A variable with name " + vname + " has already been defined.");
66  }
67 
68  // Check if a tree named 'tname' has already been created. If
69  // not, throw an exception.
70  if (trees_.count(tname) == 0)
71  EXCEPTION_RAISE(
72  "NtupleManager",
73  "A tree with name " + tname + " has already been created."
74  );
75 
76  // Initialize the variable to the minimum limit of that type.
77  variables_[vname] = T(std::numeric_limits<T>::min());
78 
79  // Get the type name
80  std::string typeName = typeid(std::get<T>(variables_[vname])).name();
81 
82  // The type name in C++ for basic types are lowercase:
83  // s, i, f, d, l
84  // while in ROOT it is uppercase:
85  // S, I, F, D, L
86  // so we need the call to toupper
87  for ( char& c : typeName ) c = toupper(c);
88 
89  // Add the variable to the ROOT tree
90  trees_[tname]->Branch(vname.c_str(), &std::get<T>(variables_[vname]),
91  (vname + "/" + typeName).c_str());
92  }
