55  template <typename T>
56  void addVar(const std::string& tname, const std::string& vname) {
57  // Check if the variable exists in the map. If it does, throw
58  // an exception.
59  if (variables_.count(vname) != 0) {
60  EXCEPTION_RAISE("NtupleManager", "A variable with name " + vname +
61  " has already been defined.");
62  }
63 
64  // Check if a tree named 'tname' has already been created. If
65  // not, throw an exception.
66  if (trees_.count(tname) == 0)
67  EXCEPTION_RAISE("NtupleManager", "A tree with name " + tname +
68  " has already been created.");
69 
70  // Initialize the variable to the minimum limit of that type.
71  variables_[vname] = T(std::numeric_limits<T>::min());
72 
73  // Get the type name
74  std::string typeName = typeid(std::get<T>(variables_[vname])).name();
75 
76  // The type name in C++ for basic types are lowercase:
77  // s, i, f, d, l
78  // while in ROOT it is uppercase:
79  // S, I, F, D, L
80  // so we need the call to toupper
81  for (char& c : typeName) c = toupper(c);
82 
83  // Add the variable to the ROOT tree
84  trees_[tname]->Branch(vname.c_str(), &std::get<T>(variables_[vname]),
85  (vname + "/" + typeName).c_str());
86  }
