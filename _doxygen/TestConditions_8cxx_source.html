26 void matchesAll(const conditions::DoubleTableCondition& a,
27  const conditions::DoubleTableCondition& b) {
28  matchesMeta(a, b);
29  for (unsigned int i = 0; i < a.getRowCount(); i++) {
30  std::pair<unsigned int, std::vector<double>> ar = a.getRow(i);
31  std::pair<unsigned int, std::vector<double>> br = b.getRow(i);
32  REQUIRE(ar.first == br.first);
33  for (unsigned int ic = 0; ic < a.getColumnCount(); ic++) {
34  // std::cout << i << ',' << ic << "=>" << ar.second[ic] << " " <<
35  // br.second[ic] << std::endl;
36  REQUIRE(fabs(ar.second[ic] - br.second[ic]) /
37  std::max(1e-5, (ar.second[ic] + br.second[ic]) / 2) <
38  1e-5); // limited precision of course
39  }
40  }
41 }
