46HcalDetectorMap::HcalDetectorMap(const std::string& connections_table, bool want_d2e)
47 : framework::ConditionsObject(CONDITIONS_OBJECT_NAME),
48 ldmx::ElectronicsMap<ldmx::HcalElectronicsID, ldmx::HcalDigiID>(want_d2e) {
49
50 this->clear();
51 conditions::StreamCSVLoader csv(connections_table);
52 while (csv.nextRow()) {
63 int polarfire = csv.getInteger("Polarfire");
64 int hgcroc = csv.getInteger("HGCROC");
65 int chan = csv.getInteger("Channel");
66
87 int link = 2*hgcroc; // link count starts from 0
88 if (chan >= 36) {
89 link += 1;
90 chan -= 36;
91 }
92
93 // one quadbar groups 4 strips and each quadbar is connected to 2 CMBs - one in each end
94 int end = csv.getInteger("CMB")%2;
95 if(end==0) end=1; // negative end (?)
96 else end=0; // positive end (?)
97
98 // The Bar field in the map represents a numbering *within a quadbar* not
99 // the strip number.
100 //
101 // This gives correct range for the strip numbers [0,7] and [0,11] but the
102 // actual order of the quadbars has to be checked
103
104 auto QuadBar {csv.getInteger("QuadBar")};
105 auto BarNumber {csv.getInteger("Bar")};
106 // Strip = (4 - Bar) + 4 * (QuadBar - 1)
107 // QuadBar starts at 1 so we need to subtract to get first value
108 // to be zero
109 auto strip { (BarNumber - 1) + 4 * (QuadBar - 1)}; //This is flipped from the original
110 if(csv.getInteger("Plane")%2 == 1) strip = (4 - BarNumber) + 4 * (QuadBar - 1); //Plane 2 is backward by mistake
111 ldmx::HcalDigiID detid(0 /*section - only one section during test beam*/,
112 csv.getInteger("Plane") /*layer*/,
113 strip /*strip*/,
114 end /*end*/);
115 ldmx::HcalElectronicsID eleid(
116 polarfire /*polarfire fpga*/,
117 link /*elink*/,
118 chan /*channel*/);
119
120 if (this->exists(eleid)) {
121 std::stringstream ss;
122 ss << "Two different mappings for electronics channel " << eleid;
123 EXCEPTION_RAISE("DuplicateMapping", ss.str());
124 }
125 this->addEntry(eleid, detid);
126 }
127}
