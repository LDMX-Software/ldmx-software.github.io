10 void QIEEncoder::configure(
11 framework::config::Parameters &ps) {
12 // Configure this instance of the encoder
13 outputCollection_ = ps.getParameter<std::string>("output_collection");
14 inputCollection_ = ps.getParameter<std::string>("input_collection");
15 inputPassName_ = ps.getParameter<std::string>("input_pass_name");
16 channelMapFileName_ = ps.getParameter<std::string>("channel_map_file");
17 nChannels_ = ps.getParameter<int>("number_channels");
18 verbose_ = ps.getParameter<bool>("verbose");
19
20 ldmx_log(debug) << "In configure, got parameters:" <<
21 "\noutput_collection = " << outputCollection_ <<
22 "\ninput_collection = " << inputCollection_ <<
23 "\ninput_pass_name = " << inputPassName_ <<
24 "\nchannel_map_file = " << channelMapFileName_ <<
25 "\nnumber_channels = " << nChannels_ <<
26 "\nverbose = " << verbose_ ;
27
28 //set up channel mapping
29 channelMapFile_.open(channelMapFileName_, std::ios::in);
30 if (! channelMapFile_.is_open() ){
31 EXCEPTION_RAISE("BadMapFile", "The channel mapping file cannot be opened."); // <-- appears this needs implementing first
32 ldmx_log(fatal) << "The channel mapping file cannot be opened.";
33 return;
34 }
35 int chID, elID;
36 while (! channelMapFile_.eof() ){
37 channelMapFile_ >> elID >> chID ;
38 // for test beam, we will only know the elecID from
39 // the position of the word in the stream.
40 // so these need to be strictly ordered in the map.
41 // barID can always be set, or looked up, as a property of the digi.
42
43 //make this based on channel ID. this is like looking up the position in a vector of a certain value. but it's fine
44 channelMap_.insert(std::pair<int,int>(chID, elID));
45 ldmx_log(debug) << "elID " << elID << " chID " << chID ;
46 }
47 if (elID != nChannels_ - 1 )
48 ldmx_log(fatal) << "The set number of channels " << nChannels_ << " seems not to match the number from the map (+1) :" << elID;
49 channelMapFile_.close();
50
51 return;
52 }

55 void QIEEncoder::produce(framework::Event &event) {
56 ldmx_log(debug)
57 << "QIEEncoder: produce() starts! Event number: "
58 << event.getEventHeader().getEventNumber();
59
60 std::vector <trigscint::QIEStream> qieOuts;
61 int nSamp = QIEStream::NUM_SAMPLES ;
62 std::vector < int > initVec(nSamp, 0);
63 ldmx_log(debug) << "num samples = " << nSamp;
64
65 //we're keeping a list ordered in elec ID since this is the order we'll use to write them to stream
66 for (int iQ = 0; iQ < nChannels_ ; iQ++) {
67 QIEStream qieOut;
68 qieOut.setADC(initVec);
69 qieOut.setTDC(initVec);
70 qieOut.setCID(initVec);
71 qieOut.setElectronicsID( iQ ); //assume id is index //channelMap_.at(iQ) );
72
73 qieOuts.push_back(qieOut);
74 }
75
76 ldmx_log(debug) << "Looking up input collection " << inputCollection_ << "_" << inputPassName_;
77 const auto digis{event.getCollection<trigscint::TrigScintQIEDigis>(
78 inputCollection_, inputPassName_)};
79 ldmx_log(debug) << "Got input collection" << inputCollection_ << "_" << inputPassName_;
80
81 bool isCIDunsync = false; //if there is a mismatch between CID reported by channels within the same time sample
82 bool isCIDskipped = false; //if there is a gap in the CID increment.
83 bool isCRC0malformed = false; //if there was an issue with CRC from fiber0
84 bool isCRC1malformed = false; //if there was an issue with CRC from fiber1
85
86 int firstCID=-1;
87 ldmx_log(debug) << "entering loop over digis " ;
88 for (auto &digi: digis) {
89 int bar = digi.getChanID();
90 auto itr = channelMap_.find( bar );
91 if ( itr == channelMap_.end() ) {// yikes! didn't find the bar in the map
92 ldmx_log(fatal) << "Couldn't find an entry for bar " << bar << "; check the (choice of) channel map!. Exiting event " << event.getEventHeader().getEventNumber();
93 return;
94 }
95 int idx = itr->second; //here we're just using the order. no actual elID is assumed.
96 qieOuts.at(idx).setChannelID( bar );
97 qieOuts.at(idx).setElectronicsID( idx );
98 // ldmx_log(debug) << "Channel "<< bar << " elec ID " << unsigned(qieOuts.at(idx).getElectronicsID()) ;
99 ldmx_log(debug) << "Channel "<< bar << " elec ID " << qieOuts.at(idx).getElectronicsID() ;
100 std::vector <int> LEtdcs; //make the LE (Leading Edge) truncation explicit
101 std::vector <uint8_t> cids;
102 for (int iS = 0; iS < nSamp; iS++) {
103 int tdc= digi.getTDC().at(iS);
104 int cid= digi.getCID().at(iS);
105 if ( cids.size()>0 && (cid%4)!=( (cids.back()+1)%4) ) {
106 //by construction shouldn't happen in simulation. still, explicitly
107 //checking here, considering any future changes to our CID simulation.
108 isCIDskipped = true;
109 }
110 if (verbose_) { //all this is only useful for debugging
111 std::vector <uint8_t> adcs;
112 int adc= digi.getADC().at(iS);
113 uint8_t mant = adc%64 ;
114 uint8_t exp = adc/64 ;
115 ldmx_log(debug) << "\tSample " << iS << std::left << std::setw(6) << " ADC " << adc << ",\texp " << unsigned(exp) << " mant " << unsigned(mant) <<
116 ",\tTDC = " << tdc << ", LE TDC = " << std::bitset<8>(tdc/16) << " and capID= " << cid ; // << std::endl;
117 adcs.push_back( 64*exp+mant);
118 ldmx_log(debug) << "Combined ADC: " << std::showbase << std::bitset<8>(adcs.back()) << " and original adc " << std::bitset<8>(adc) << std::dec ;
119 }//if verbose
120
121 tdc/=16; // do LE (leading edge) TDC
122 LEtdcs.push_back( tdc );
123 cids.push_back( (uint8_t)cid );
124 }// over samples
125 if (firstCID == -1)
126 firstCID=cids.back(); //just store the 5th one, doesn't matter, if all channels are aligned then cids should match at any given time sample
127 if (firstCID!=cids.back())
128 isCIDunsync=true; //any one channel not aligned is enough to set this bool
129
130 qieOuts.at(idx).setADC( digi.getADC() );
131 qieOuts.at(idx).setTDC( LEtdcs );
132 }//over digis
133 if (isCIDunsync)
134 ldmx_log(debug) << "Found unsynced CIDs!" ;
135 if (isCIDskipped)
136 ldmx_log(info) << "Found skipped CIDs!" ;
137
138
139 //data format:
140 // RM ID. we don't set it for testbeam. in LDMX let's say it's 0 for tag, 1 for up, 2 for dn -- later this can be read using moduleID...
141 // 16 bit trigger ID.
142 // 4 bits of flags, then 4 reserved 0 for now
143 // some 8-bit error word/checksum.
144 // all channel 8-bit ADCs.
145 // all channel TDCS.
146 // done.
147
148 uint16_t triggerID =event.getEventHeader().getEventNumber();
149 uint16_t header0;
150 uint8_t randomChecksum = 30; // really, this is just a random number for now. TODO> implement a checksum
151 uint8_t flags; //we use this to contain the reserved 0's for now
152 //put it all in, at the assigned position
153 flags |= (isCRC0malformed << QIEStream::CRC0_ERR_POS);
154 flags |= (isCRC1malformed << QIEStream::CRC1_ERR_POS);
155 flags |= (isCIDunsync<< QIEStream::CID_UNSYNC_POS);
156 flags |= (isCIDskipped<< QIEStream::CID_SKIP_POS);
157 ldmx_log(debug) << "FLAGS: " << std::bitset<8>(flags) ;
158
159 // std::vector <int> outWord;
160 std::vector <uint8_t> outWord;
161 std::vector <uint8_t> triggerIDwords;
162 //for (int iW = 0; iW < QIEStream::TRIGID_LEN_BYTES; iW++) { //if the LSB comes first in the 2-byte word (it shouldn't)
163 for (int iW = QIEStream::TRIGID_LEN_BYTES-1; iW >= 0; iW--) { //assume the whole 2B are written as a single 16 bits word
164 uint8_t tIDword = triggerID >> iW*8; //shift by a byte at a time
165 triggerIDwords.push_back( tIDword );
166 outWord.push_back( tIDword);
167 }
168 // outWord.push_back( triggerID);
169
170 outWord.push_back( flags );
171 outWord.push_back( randomChecksum);
172
173 if (verbose_) {
174 std::cout << "header word " ;
175 for (auto word: outWord)
176 std::cout << std::bitset<8>(word) << " " ;
177 std::cout << std::endl;
178 }
179
180 //now write this in sequence: ADC of all channels, then TDC; repeat for all samples
181 for (int iS = 0; iS < nSamp; iS++) {
182 for (int iQ = 0; iQ < nChannels_ ; iQ++) {
183 outWord.push_back( qieOuts.at(iQ).getADC().at(iS) ) ;
184 }//over channels : ADC
185 for (int iQ = 0; iQ < nChannels_ ; iQ++) {
186 outWord.push_back( qieOuts.at(iQ).getTDC().at(iS) ) ;
187 }//over channels: TDC
188 }//over time samples
189
190
191 //in verbose mode, print this all to screen
192 if (verbose_) {
193 std::cout << "total word " ;
194 int widx = 0;
195 int iWstart = std::max( std::max(QIEStream::ERROR_POS, QIEStream::CHECKSUM_POS),
196 QIEStream::TRIGID_POS+(QIEStream::TRIGID_LEN_BYTES)) +1; //probably overkill :D should be 4
197 for (auto word: outWord) {
198 if ((widx - iWstart)%nChannels_ == 0 ) {
199 int sample = (widx - iWstart)/nChannels_ ;
200 if ( sample%2 == 0 )
201 std::cout << "\n sample " << sample/2 << " | ";
202 else
203 std::cout << "\n TDC: " ;
204 }
205 std::cout << (unsigned)word << " " ;
206 //std::cout << std::bitset<8>(word) << " " ; //for binary output format
207 widx++;
208 }
209 std::cout << std::endl;
210 }//if verbose
211
212 event.add(outputCollection_, outWord);
213
214 }
