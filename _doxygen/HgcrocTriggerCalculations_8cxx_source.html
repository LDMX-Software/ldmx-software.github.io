8HgcrocTriggerConditions::HgcrocTriggerConditions(
9 const conditions::IntegerTableCondition &ict, bool validate)
10 : ict_{ict} {
11 if (!validate) return;
12 if (ict_.getColumnCount() < 5) {
13 EXCEPTION_RAISE(
14 "ConditionsException",
15 "Inconsistent condition for HgcrocTriggerConditions :" + ict.getName());
16 }
17 std::vector<std::string> expected_colnames;
18 expected_colnames.push_back("ADC_PEDESTAL");
19 expected_colnames.push_back("ADC_THRESHOLD");
20 expected_colnames.push_back("TOT_PEDESTAL");
21 expected_colnames.push_back("TOT_THRESHOLD");
22 expected_colnames.push_back("TOT_GAIN");
23
24 for (size_t i = 0; i < 5; i++) {
25 if (ict_.getColumnNames()[i] != expected_colnames[i]) {
26 EXCEPTION_RAISE("ConditionsException",
27 "Expected column '" + expected_colnames[i] + "', got '" +
28 ict_.getColumnNames()[i] + "'");
29 }
30 }
31}

33unsigned int HgcrocTriggerCalculations::singleChannelCharge(
34 int adc, int tot, int adc_ped, int adc_thresh, int tot_ped, int tot_thresh,
35 int tot_gain) {
36 unsigned int charge_adc = 0;
37
38 if (adc > (adc_ped + adc_thresh))
39 charge_adc = adc - adc_ped; // otherwise zero
40
41 unsigned int eff_tot{0};
42 if (tot > tot_thresh && tot > tot_ped)
43 eff_tot = tot - tot_ped;
44 else
45 eff_tot = tot_thresh - tot_ped;
46
47 unsigned int charge_tot{0};
48 charge_tot = eff_tot * tot_gain;
49
50 unsigned int charge_final = (tot != 0) ? (charge_tot) : (charge_adc);
51
52 return charge_final;
53}

59void HgcrocTriggerCalculations::addDigi(unsigned int id, unsigned int tid,
60 int adc, int tot) {
61 unsigned int charge = singleChannelCharge(
62 adc, tot, conditions_.adcPedestal(id), conditions_.adcThreshold(id),
63 conditions_.totPedestal(id), conditions_.totThreshold(id),
64 conditions_.totGain(id));
65 if (charge > 0) {
66 std::map<unsigned int, unsigned int>::iterator ic = linearCharge_.find(tid);
67 if (ic == linearCharge_.end())
68 linearCharge_[tid] = charge;
69 else
70 ic->second += charge;
71 }
72}

74void HgcrocTriggerCalculations::compressDigis(int cells_per_trig) {
75 int shift;
76 if (cells_per_trig == 4)
77 shift = 1;
78 else if (cells_per_trig == 9)
79 shift = 3;
80 else {
81 EXCEPTION_RAISE("HgcrocTriggerException",
82 "Invalid number of precision cells per trigger cell: " +
83 std::to_string(cells_per_trig));
84 }
85
86 for (auto ilinear : linearCharge_) {
87 unsigned int lcharge = ilinear.second;
88 lcharge = lcharge >> shift;
89 uint8_t ccharge = ldmx::HgcrocTrigDigi::linear2Compressed(lcharge);
90 compressedCharge_[ilinear.first] = ccharge;
91 }
92}
