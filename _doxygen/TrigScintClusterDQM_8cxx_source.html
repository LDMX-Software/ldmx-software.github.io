44void TrigScintClusterDQM::analyze(const framework::Event &event) {
45 // Get the collection of TrigScintCluster digitized clusters if the exists
46 const std::vector<ldmx::TrigScintCluster> TrigScintClusters =
47 event.getCollection<ldmx::TrigScintCluster>(clusterCollectionName_,
48 passName_);
49
50 double totalPE{0};
51 // Loop through all TrigScint clusters in the event
52 for (const ldmx::TrigScintCluster &cluster : TrigScintClusters) {
53 histograms_.fill("pe", cluster.getPE());
54 histograms_.fill("energy", cluster.getEnergy());
55 histograms_.fill("cluster_time", cluster.getTime());
56 histograms_.fill("centroid", cluster.getCentroid());
57 histograms_.fill("n_hits", cluster.getNHits());
58 histograms_.fill("seed", cluster.getSeed());
59 histograms_.fill("beamEfrac", cluster.getBeamEfrac());
60
61 histograms_.fill("x", cluster.getCentroidX());
62 histograms_.fill("y", cluster.getCentroidY());
63 histograms_.fill("z", cluster.getCentroidZ());
64
65 totalPE += cluster.getPE();
66 }
67
68 histograms_.fill("n_clusters", TrigScintClusters.size());
69 histograms_.fill("total_pe", totalPE);
70}
