70void lhe(boost::iostreams::filtering_istream& reader, int aprime_lhe_id, std::map<double, std::vector<OutgoingKinematics>>& lib) {
71 std::string line;
72 while (std::getline(reader,line)) {
73 std::istringstream iss(line);
74 int ptype, state;
75 double skip, px, py, pz, E, M;
76 if (iss >> ptype >> state >> skip >> skip >> skip >> skip
77 >> px >> py >> pz >> E >> M) {
78 if ((ptype == 11 or ptype == 13) && (state == -1)) {
79 double incident_energy = E;
80 double e_px, e_py, e_pz, a_px, a_py, a_pz, e_E, a_E, e_M, a_M;
81 for (int i = 0; i < 2; i++) {
82 std::getline(reader,line);
83 }
84 std::istringstream jss(line);
85 jss >> ptype >> state >> skip >> skip >> skip >> skip
86 >> e_px >> e_py >> e_pz >> e_E >> e_M;
87 if ((ptype == 11 or ptype == 13) && (state == 1)) { // Find a final state lepton
88 for (int i = 0; i < 2; i++) {
89 std::getline(reader,line);
90 }
91 std::istringstream kss(line);
92 kss >> ptype >> state >> skip >> skip >> skip >> skip
93 >> a_px >> a_py >> a_pz >> a_E >> a_M;
94 if (ptype == aprime_lhe_id and state == 1) {
95 OutgoingKinematics evnt;
96 double cmpx = a_px + e_px;
97 double cmpy = a_py + e_py;
98 double cmpz = a_pz + e_pz;
99 double cmE = a_E + e_E;
100 evnt.lepton = CLHEP::HepLorentzVector(e_px, e_py, e_pz, e_E);
101 evnt.centerMomentum = CLHEP::HepLorentzVector(cmpx, cmpy, cmpz, cmE);
102 evnt.E = incident_energy;
103 lib[incident_energy].push_back(evnt);
104 } // get a prime kinematics
105 } // check for final state
106 } // check for particle type and state
107 } // able to get momentum/energy numbers
108 } // while getting lines
109}

136void csv(boost::iostreams::filtering_istream& reader, std::map<double, std::vector<OutgoingKinematics>>& lib) {
137 std::string line;
138 // skip the header line
139 if (not std::getline(reader, line)) {
140 throw std::runtime_error("Empty CSV file.");
141 }
142 // read in all non-empty lines
143 while (std::getline(reader, line) and not line.empty()) {
144 std::istringstream lss{line};
145 std::vector<double> vals;
146 std::string cell;
147 while (std::getline(lss,cell,',')) {
148 vals.push_back(std::stod(cell));
149 }
150 if (not lss and cell.empty()) vals.push_back(-9999);
151 if (vals.size() != 9) {
152 throw std::runtime_error("Malformed row in CSV file: not exactly 9 columns");
153 }
154 OutgoingKinematics ok;
155 ok.E = vals[0];
156 ok.lepton = CLHEP::HepLorentzVector(vals[2], vals[3], vals[4], vals[1]);
157 ok.centerMomentum = CLHEP::HepLorentzVector(vals[6], vals[7], vals[8], vals[5]);
158 lib[ok.E].push_back(ok);
159 }
160}
