80void lhe(boost::iostreams::filtering_istream& reader, int aprime_lhe_id, std::map<int, std::map<double, std::vector<OutgoingKinematics>>>& lib) {
81 std::string line;
82 int target_Z{-1};
83 while (std::getline(reader,line)) {
84 std::istringstream iss(line);
85 if (line.find("Znuc") != std::string::npos) {
86 int param_id;
87 double Znuc_param;
88 if (iss >> param_id >> Znuc_param) {
89 target_Z = int(Znuc_param);
90 } else {
91 throw std::runtime_error("Unable to deduce target Z from line '"+line+"'.");
92 }
93 }
94 int ptype, state;
95 double skip, px, py, pz, E, M;
96 if (iss >> ptype >> state >> skip >> skip >> skip >> skip
97 >> px >> py >> pz >> E >> M) {
98 if ((ptype == 11 or ptype == 13) && (state == -1)) {
99 double incident_energy = E;
100 double e_px, e_py, e_pz, a_px, a_py, a_pz, e_E, a_E, e_M, a_M;
101 for (int i = 0; i < 2; i++) {
102 std::getline(reader,line);
103 }
104 std::istringstream jss(line);
105 jss >> ptype >> state >> skip >> skip >> skip >> skip
106 >> e_px >> e_py >> e_pz >> e_E >> e_M;
107 if ((ptype == 11 or ptype == 13) && (state == 1)) { // Find a final state lepton
108 for (int i = 0; i < 2; i++) {
109 std::getline(reader,line);
110 }
111 std::istringstream kss(line);
112 kss >> ptype >> state >> skip >> skip >> skip >> skip
113 >> a_px >> a_py >> a_pz >> a_E >> a_M;
114 if (ptype == aprime_lhe_id and state == 1) {
115 OutgoingKinematics evnt;
116 double cmpx = a_px + e_px;
117 double cmpy = a_py + e_py;
118 double cmpz = a_pz + e_pz;
119 double cmE = a_E + e_E;
120 evnt.lepton = CLHEP::HepLorentzVector(e_px, e_py, e_pz, e_E);
121 evnt.centerMomentum = CLHEP::HepLorentzVector(cmpx, cmpy, cmpz, cmE);
122 evnt.E = incident_energy;
123 if (target_Z < 0) {
124 throw std::runtime_error("Did not deduce target Z before starting to deduce events.");
125 }
126 lib[target_Z][incident_energy].push_back(evnt);
127 } // get a prime kinematics
128 } // check for final state
129 } // check for particle type and state
130 } // able to get momentum/energy numbers
131 } // while getting lines
132}

160void csv(boost::iostreams::filtering_istream& reader, std::map<int, std::map<double, std::vector<OutgoingKinematics>>>& lib) {
161 std::string line;
162 // skip the header line
163 if (not std::getline(reader, line)) {
164 throw std::runtime_error("Empty CSV file.");
165 }
166 // read in all non-empty lines
167 while (std::getline(reader, line) and not line.empty()) {
168 std::istringstream lss{line};
169 std::vector<double> vals;
170 std::string cell;
171 while (std::getline(lss,cell,',')) {
172 vals.push_back(std::stod(cell));
173 }
174 if (not lss and cell.empty()) vals.push_back(-9999);
175 if (vals.size() != 10) {
176 throw std::runtime_error("Malformed row in CSV file: not exactly 9 columns");
177 }
178 OutgoingKinematics ok;
179 int target_Z = vals[0]; // implicit drop of any decimal points
180 ok.E = vals[1];
181 ok.lepton = CLHEP::HepLorentzVector(vals[3], vals[4], vals[5], vals[2]);
182 ok.centerMomentum = CLHEP::HepLorentzVector(vals[7], vals[8], vals[9], vals[6]);
183 lib[target_Z][ok.E].push_back(ok);
184 }
185}

189void parseLibrary(const std::string& path, int aprime_lhe_id, std::map<int, std::map<double, std::vector<OutgoingKinematics>>>& lib) {
190 if (hasEnding(path, ".csv") or hasEnding(path, ".csv.gz") or hasEnding(path, ".lhe") or hasEnding(path, ".lhe.gz")) {
207 boost::iostreams::filtering_istream reader;
208 if (hasEnding(path, ".gz")) reader.push(boost::iostreams::gzip_decompressor());
209 reader.push(boost::iostreams::file_source(path));
210 if (hasEnding(path, ".csv") or hasEnding(path, ".csv.gz")) parse::csv(reader, lib);

212 } else {
219 DIR *dir; // handle to opened directory

221 if ((dir = opendir(path.c_str())) != NULL) {
222 // directory can be opened

224 std::string fp = path + '/' + std::string(ent->d_name);
225 if (hasEnding(fp,".lhe") or hasEnding(fp, ".lhe.gz")
226 or hasEnding(fp,".csv") or hasEnding(fp, ".csv.gz")) {
232 parseLibrary(fp, aprime_lhe_id, lib);
233 }
234 }
235 closedir(dir);
236 } else {
241 throw std::runtime_error("Unable to open '"+path+"' as a directory.");
242 }
