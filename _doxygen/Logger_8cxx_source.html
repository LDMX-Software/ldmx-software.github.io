29void open(const level termLevel, const level fileLevel,
30 const std::string &fileName) {
31 // some helpful types
32 typedef sinks::text_ostream_backend ourSinkBack_t;
33 typedef sinks::synchronous_sink<ourSinkBack_t> ourSinkFront_t;
34
35 // allow our logs to access common attributes, the ones availabe are
36 // "LineID" : counter increments for each record being made (terminal or
37 // file) "TimeStamp" : time the log message was created "ProcessID" : machine
38 // ID for the process that is running "ThreadID" : machine ID for the thread
39 // the message is in
40 log::add_common_attributes();
41
42 // get the core logging service
43 boost::shared_ptr<log::core> core = log::core::get();
44
45 // file sink is optional
46 // don't even make it if no fileName is provided
47 if (not fileName.empty()) {
48 boost::shared_ptr<ourSinkBack_t> fileBack =
49 boost::make_shared<ourSinkBack_t>();
50 fileBack->add_stream(boost::make_shared<std::ofstream>(fileName));
51
52 boost::shared_ptr<ourSinkFront_t> fileSink =
53 boost::make_shared<ourSinkFront_t>(fileBack);
54
55 // this is where the logging level is set
56 fileSink->set_filter(log::expressions::attr<level>("Severity") >=
57 fileLevel);
58
59 // TODO change format to something helpful
60 // Currently:
61 // [ Channel ] int severity : message
62 fileSink->set_formatter([](const log::record_view &view,
63 log::formatting_ostream &os) {
64 os
65 // <<
66 // log::extract<boost::date_time::int_adapter>(
67 // "TimeStamp" , view )
68 << " [ " << log::extract<std::string>("Channel", view) << " ] "
69 << /*humanReadableLevel.at*/ (log::extract<level>("Severity", view))
70 << " : " << view[log::expressions::smessage];
71 });
72
73 core->add_sink(fileSink);
74 } // file set to pass something
75
76 // terminal sink is always created
77 boost::shared_ptr<ourSinkBack_t> termBack =
78 boost::make_shared<ourSinkBack_t>();
79 termBack->add_stream(boost::shared_ptr<std::ostream>(
80 &std::cout, // point this stream to std::cout
81 boost::null_deleter() // don't let boost delete std::cout
82 ));
83 // flushes message to screen **after each message**
84 termBack->auto_flush(true);
85
86 boost::shared_ptr<ourSinkFront_t> termSink =
87 boost::make_shared<ourSinkFront_t>(termBack);
88
89 // translate integer level to enum
90 termSink->set_filter(log::expressions::attr<level>("Severity") >= termLevel);
91
92 // TODO change format to something helpful
93 // Currently:
94 // [ Channel ](int severity) : message
95 termSink->set_formatter([](const log::record_view &view,
96 log::formatting_ostream &os) {
97 os
98 // <<
99 // log::extract<boost::date_time::int_adapter>(
100 // "TimeStamp" , view )
101 << " [ " << log::extract<std::string>("Channel", view) << " ] "
102 << /*humanReadableLevel.at*/ (log::extract<level>("Severity", view))
103 << " : " << view[log::expressions::smessage];
104 });
105
106 core->add_sink(termSink);
107
108 return;
109
110} // open
