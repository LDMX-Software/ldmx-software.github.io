95  void EcalDigiProducer::produce(Event& event) {
96 
97  //get simulated ecal hits from Geant4
98  // the class EcalHitIO in the SimApplication module handles the translation from G4CalorimeterHits to SimCalorimeterHits
99  // this class ensures that only one SimCalorimeterHit is generated per cell, but
100  // multiple "contributions" are still handled within SimCalorimeterHit
101  auto ecalSimHits{event.getCollection<SimCalorimeterHit>(EventConstants::ECAL_SIM_HITS)};
102 
103  //Empty collection to be filled
104  EcalDigiCollection ecalDigis;
105  ecalDigis.setNumSamplesPerDigi( nADCs_ );
106  ecalDigis.setSampleOfInterestIndex( iSOI_ );
107 
108  std::set<int> simHitIDs;
109  for (auto const& simHit : ecalSimHits ) {
110 
111  std::vector<double> energyDepositions, simulatedTimes;
112  for ( int iContrib = 0; iContrib < simHit.getNumberOfContribs(); iContrib++ ) {
113  energyDepositions.push_back( simHit.getContrib( iContrib ).edep );
114  simulatedTimes.push_back( simHit.getContrib( iContrib ).time );
115  }
116 
117  int hitID = simHit.getID();
118  simHitIDs.insert( hitID );
119 
120  std::vector<EcalDigiSample> digiToAdd;
121  if ( constructDigis( energyDepositions , simulatedTimes , digiToAdd ) ) {
122  for ( auto& sample : digiToAdd ) sample.rawID_ = hitID;
123  ecalDigis.addDigi( digiToAdd );
124  }
125  }
126 
127  //put noise into some empty channels
128  int numEmptyChannels = TOTAL_NUM_CHANNELS - ecalDigis.getNumDigis();
129  EcalDetectorID detID;
130  auto noiseHitAmplitudes{noiseGenerator_->generateNoiseHits(numEmptyChannels)};
131  for ( double noiseHit : noiseHitAmplitudes ) {
132 
133  //generate detector ID for noise hit
134  //making sure that it is in an empty channel
135  int noiseID;
136  do {
137  int layerID = noiseInjector_->Integer(NUM_ECAL_LAYERS);
138  int moduleID= noiseInjector_->Integer(NUM_HEX_MODULES_PER_LAYER);
139  int cellID = noiseInjector_->Integer(CELLS_PER_HEX_MODULE);
140  detID.setFieldValue( 1 , layerID );
141  detID.setFieldValue( 2 , moduleID );
142  detID.setFieldValue( 3 , cellID );
143  noiseID = detID.pack();
144  } while ( simHitIDs.find( noiseID ) != simHitIDs.end() );
145 
146  //get a time for this noise hit
147  double hitTime = noiseInjector_->Uniform( EcalDigiProducer::CLOCK_CYCLE );
148 
149  std::vector<EcalDigiSample> digiToAdd;
150  std::vector<double> noiseEnergies( 1 , noiseHit ), noiseTimes( 1 , hitTime );
151  if ( constructDigis( noiseEnergies , noiseTimes , digiToAdd ) ) {
152  for ( auto& sample : digiToAdd ) sample.rawID_ = noiseID;
153  ecalDigis.addDigi( digiToAdd );
154  }
155 
156  }
157 
158  event.add("EcalDigis", ecalDigis );
159 
160  return;
161  } //produce

163  bool EcalDigiProducer::constructDigis(
164  const std::vector<double> &energies,
165  const std::vector<double> &times,
166  std::vector<EcalDigiSample> &digiToAdd) {
167 
168  digiToAdd.clear(); //make sure it is clean
169  digiToAdd.resize( nADCs_ ); //fill with required number of samples (default constructed)
170 
171  //First we emulate the ROC response by constructing
172  // a pulse from the timing/energy info and then measuring
173  // it at 25ns increments
174  //total energy and average tiem of contribs inside timing window
175  double energyInWindow = 0.0;
176  double timeInWindow = 0.0;
177  for ( int iContrib = 0; iContrib < energies.size(); iContrib++ ) {
178 
179  if ( times.at(iContrib) < 0 or times.at(iContrib) > EcalDigiProducer::CLOCK_CYCLE*nADCs_ ) {
180  //invalid contribution - outside time range or time is unset
181  continue;
182  }
183 
184  energyInWindow += energies.at(iContrib);
185  timeInWindow += energies.at(iContrib) * times.at(iContrib);
186  }
187  if ( energyInWindow > 0. ) timeInWindow /= energyInWindow; //energy weighted average
188 
189  //put noise onto pulse parameters
190  //TODO this is putting noise ontop of noise, is this a problem? (I don't think so, but maybe)
191  energyInWindow += noiseInjector_->Gaus( 0.0 , noiseRMS_/gain_ );
192  //this 100. was chosen arbitrarily
193  //TODO choose the width of the timing jitter more realistically --> python parameter?
194  timeInWindow += noiseInjector_->Gaus( 0.0 , EcalDigiProducer::CLOCK_CYCLE / 100. );
195 
196  //set time in the window to zero if noise pushed it below zero
197  //TODO better (more physical) method for handling this case?
198  if ( timeInWindow < 0. ) timeInWindow = 0.;
199 
200  //setup up pulse by changing the amplitude and timing parameters
201  pulseFunc_.SetParameter( 0 , gain_*energyInWindow ); //set amplitude to gain * energy
202  pulseFunc_.SetParameter( 4 , timeInWindow ); //set time of peak to simulated hit time
203 
204  //skip the hit if the peak of the pulse after noise is less than readoutThreshold_
205  //TODO do ADC counts in this case instead of skipping it
206  if ( pulseFunc_.Eval( timeInWindow ) < readoutThreshold_ ) return false;
207 
208  //measure time of arrival (TOA) and time under threshold (TUT) from pulse
209  // TOA: earliest possible measure for crossing threshold line
210  // TUT: latest possible measure for crossing threshold line
211 
212  double toa(0.);
213  // check if first half is just always above readout
214  if ( pulseFunc_.Eval( -nADCs_*EcalDigiProducer::CLOCK_CYCLE ) > readoutThreshold_ )
215  toa = -nADCs_*EcalDigiProducer::CLOCK_CYCLE; //always above --> toa is beginning of readout interval
216  else
217  toa = pulseFunc_.GetX(readoutThreshold_, -nADCs_*EcalDigiProducer::CLOCK_CYCLE, timeInWindow);
218 
219  double tut(0.);
220  // check if second half is just always above readout
221  if ( pulseFunc_.Eval( nADCs_*EcalDigiProducer::CLOCK_CYCLE ) > readoutThreshold_ )
222  tut = nADCs_*EcalDigiProducer::CLOCK_CYCLE; //always above --> tut is end of readout interval
223  else
224  tut = pulseFunc_.GetX(readoutThreshold_, timeInWindow, nADCs_*EcalDigiProducer::CLOCK_CYCLE);
225 
226  double tot = tut - toa;
227 
228  /*
229  std::cout << std::setw(6)
230  << energyInWindow << " MeV at " << timeInWindow << " ns --> "
231  << tot << " TOT " << toa << " TOA " << tut << " TUT" << std::endl;
232  */
233 
234  //conversion from ns to clock counts (converting to int implicitly)
235  int totalClockCounts = tot*(1024/EcalDigiProducer::CLOCK_CYCLE);
236 
237  //TODO: should expand digi response to multiple samples instead of just the SOI
238  //TODO currently using adc_t_ to count number of samples that clock is over threshold
239  // this is NOT how it is actually done on the chip
240  digiToAdd[iSOI_].adc_t_ = totalClockCounts / 1024; //Place Holder - not actually how response works
241  digiToAdd[iSOI_].adc_tm1_ = -1; //NOT IMPLEMENTED
242  digiToAdd[iSOI_].tot_ = totalClockCounts % 1024; //clock counts since last trigger clock (25ns clock)
243  digiToAdd[iSOI_].toa_ = toa * (1024/EcalDigiProducer::CLOCK_CYCLE); //conversion from ns to clock counts
244 
245  /*
246  std::cout << std::setw(6) << totalClockCounts
247  << " TOT --> " << digiToAdd[iSOI_].adc_t_ << " Clocks and "
248  << digiToAdd[iSOI_].tot_ << " tot" << std::endl;
249  */
250 
251  if ( makeConfigHists_ ) {
252  tot_SimE_->Fill( totalClockCounts , energyInWindow );
253  }
254 
255  return true;
256  }
