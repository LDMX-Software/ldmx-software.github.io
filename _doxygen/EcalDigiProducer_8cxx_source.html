26  void EcalDigiProducer::configure(Parameters& ps) {

28  //settings of readout chip

30  auto hgcrocParams = ps.getParameter<Parameters>("hgcroc");
31  hgcroc_ = std::make_unique<HgcrocEmulator>(hgcrocParams);
32  gain_ = hgcrocParams.getParameter<double>("gain");
33  pedestal_ = hgcrocParams.getParameter<double>("pedestal");

37 
38  // physical constants
39  // used to calculate unit conversions
40  MeV_ = ps.getParameter<double>("MeV");
41 
42  //Time -> clock counts conversion
43  // time [ns] * ( 2^10 / max time in ns ) = clock counts
44  ns_ = 1024./clockCycle_;
45 
46  // Configure generator that will produce noise hits in empty channels
47  readoutThreshold_ = hgcrocParams.getParameter<double>("readoutThreshold");
48  noiseGenerator_->setNoise(hgcrocParams.getParameter<double>("noiseRMS")); //rms noise in mV
49  noiseGenerator_->setPedestal(gain_*pedestal_); //mean noise amplitude (if using Gaussian Model for the noise) in mV
50  noiseGenerator_->setNoiseThreshold(readoutThreshold_); //threshold for readout in mV
51  }

63  }

81  //get simulated ecal hits from Geant4
82  // the class EcalHitIO in the SimApplication module handles the translation from G4CalorimeterHits to SimCalorimeterHits
83  // this class ensures that only one SimCalorimeterHit is generated per cell, but

89  */

92

95  /* debug printout

97  */

100  }

104

107  //transfer samples into the digi collection

111  }
112  }

115  * Noise Simulation on Empty Channels
116  *****************************************************************************************/
117  if ( noise_ ) {
118  //std::cout << "Noise Hits" << std::endl;
119  //put noise into some empty channels
120 
121  // geometry constants
122  // These are used in the noise generation so that we can randomly
123  // distribute the noise uniformly throughout the ECal channels.
124  const auto& hexGeom = getCondition<EcalHexReadout>(EcalHexReadout::CONDITIONS_OBJECT_NAME);
125  int nEcalLayers = hexGeom.getNumLayers();
126  int nModulesPerLayer = hexGeom.getNumModulesPerLayer();
127  int nCellsPerModule = hexGeom.getNumCellsPerModule();
128  int numEmptyChannels = nEcalLayers*nModulesPerLayer*nCellsPerModule - ecalDigis.getNumDigis();
129  //noise generator gives us a list of noise amplitudes [mV] that randomly populate the empty
130  //channels and are above the readout threshold
131  auto noiseHitAmplitudes{noiseGenerator_->generateNoiseHits(numEmptyChannels)};
132  std::vector<double> voltages( 1 , 0.), times( 1 , 0. );
133  for ( double noiseHit : noiseHitAmplitudes ) {
134 
135  //generate detector ID for noise hit
136  //making sure that it is in an empty channel
137  unsigned int noiseID;
138  do {
139  int layerID = noiseInjector_->Integer(nEcalLayers);
140  int moduleID = noiseInjector_->Integer(nModulesPerLayer);
141  int cellID = noiseInjector_->Integer(nCellsPerModule);
142  auto detID=EcalID(layerID, moduleID, cellID);
143  noiseID = detID.raw();
144  } while ( filledDetIDs.find( noiseID ) != filledDetIDs.end() );
145  filledDetIDs.insert( noiseID );
146  //std::cout << noiseID << " -> " << noiseHit + readoutThreshold_ - gain_*pedestal_ << std::endl;
147 
148  //get a time for this noise hit
149  times[0] = noiseInjector_->Uniform( clockCycle_ );
150 
151  //noise generator gives the amplitude above the readout threshold
152  // we need to convert it to the amplitdue above the pedestal
153  voltages[0] = noiseHit + readoutThreshold_ - gain_*pedestal_;
154 
155  std::vector<HgcrocDigiCollection::Sample> digiToAdd;
156  if ( hgcroc_->digitize( noiseID , voltages , times , digiToAdd ) ) {
157  ecalDigis.addDigi( noiseID , digiToAdd );
158  }
159  } //loop over noise amplitudes
160  } //if we should do the noise
161 
162  event.add("EcalDigis", ecalDigis );
163 
164  return;
165  } //produce
