55 void Simulator::configure(Parameters& parameters) {
56 
57  // parameters used to configure the simulation
58  parameters_ = parameters;
59 
60  // Set the verbosity level. The default level is 0.
61  verbosity_ = parameters_.getParameter< int >("verbosity");
62 
63  // If the verbosity level is set to 0,
64  // If the verbosity level is > 1, log everything to a file. Otherwise,
65  // dump the output. If a prefix has been specified, append it ot the
66  // log message.
67  auto loggingPrefix = parameters_.getParameter< std::string >("logging_prefix");
68  if ( verbosity_ == 0 ) sessionHandle_ = std::make_unique<BatchSession>();
69  else if ( verbosity_ > 1 ) {
70 
71  if ( loggingPrefix.empty() ) sessionHandle_ = std::make_unique<LoggedSession>();
72  else sessionHandle_ = std::make_unique<LoggedSession>( loggingPrefix + "_G4cout.log" , loggingPrefix + "_G4cerr.log" );
73 
74  }
75  if (sessionHandle_ != nullptr) uiManager_->SetCoutDestination( sessionHandle_.get() );
76 
77  // Instantiate the run manager.
78  runManager_ = std::make_unique<RunManager>(parameters, conditionsIntf_);
79 
80  // Instantiate the GDML parser and corresponding messenger owned and
81  // managed by DetectorConstruction
82  G4GDMLParser *parser = new G4GDMLParser;
83 
84  // Instantiate the class so cascade parameters can be set.
85  G4CascadeParameters::Instance();
86 
87  // Set the DetectorConstruction instance used to build the detector
88  // from the GDML description.
89  runManager_->SetUserInitialization( new DetectorConstruction( parser , parameters, conditionsIntf_ ) );
90 
91  // Parse the detector geometry and validate if specified.
92  auto detectorPath{parameters_.getParameter< std::string >("detector")};
93  auto validateGeometry{parameters_.getParameter< bool >("validate_detector")};
94  if ( verbosity_ > 0 ) {
95  std::cout << "[ Simulator ] : Reading in geometry from '" << detectorPath << "'... " << std::flush;
96  }
97  G4GeometryManager::GetInstance()->OpenGeometry();
98  parser->Read( detectorPath, validateGeometry );
99  runManager_->DefineWorldVolume( parser->GetWorldVolume() );
100 
101  auto preInitCommands = parameters_.getParameter< std::vector< std::string > >("preInitCommands" ,{} );
102  for ( const std::string& cmd : preInitCommands ) {
103  if ( allowed(cmd) ) {
104  int g4Ret = uiManager_->ApplyCommand( cmd );
105  if ( g4Ret > 0 ) {
106  EXCEPTION_RAISE(
107  "PreInitCmd",
108  "Pre Initialization command '" + cmd + "' returned a failue status from Geant4: " + std::to_string(g4Ret)
109  );
110  }
111  } else {
112  EXCEPTION_RAISE(
113  "PreInitCmd",
114  "Pre Initialization command '" + cmd + "' is not allowed because another part of Simulator handles it."

237 void Simulator::produce(ldmx::Event& event) {
238 
239  // Pass the current LDMX event object to the persistency manager. This
240  // is needed by the persistency manager to fill the current event.
241  persistencyManager_->setCurrentEvent( &event );
242 
243  // Generate and process a Geant4 event.
244  numEventsBegan_++;
245  runManager_->ProcessOneEvent( event.getEventHeader().getEventNumber() );
246 
247  // If a Geant4 event has been aborted, skip the rest of the processing
248  // sequence. This will immediately force the simulation to move on to
249  // the next event.
250  if ( runManager_->GetCurrentEvent()->IsAborted() ) {
251  runManager_->TerminateOneEvent(); //clean up event objects
252  this->abortEvent(); //get out of processors loop
253  }
254 
255  if ( this->getLogFrequency() > 0 and event.getEventHeader().getEventNumber() % this->getLogFrequency() == 0 ) {
256  //print according to log frequency and verbosity
257  if ( verbosity_ > 1 ) std::cout << "[ Simulator ] : Printing event contents:" << std::endl;
258  event.Print( verbosity_ );
259  }
260 
261  // Terminate the event. This checks if an event is to be stored or
262  // stacked for later.
263  numEventsCompleted_++;
264  runManager_->TerminateOneEvent();
265 
266  return;
267 }

305 void Simulator::onFileClose(EventFile&) {
306 
307  // End the current run and print out some basic statistics if verbose
308  // level > 0.
309  runManager_->TerminateEventLoop();
310 
311  // Pass the **real** number of events to the persistency manager
312  persistencyManager_->setNumEvents( numEventsBegan_ , numEventsCompleted_ );
313 
314  // Persist any remaining events, call the end of run action and
315  // terminate the Geant4 kernel.
316  runManager_->RunTermination();
317 
318  // Cleanup persistency manager
319  // Geant4 expects us to handle the persistency manager
320  // In order to avoid segfaulting nonsense, I delete it here
321  // so that it is deleted before the EventFile it references
322  // is deleted
323  persistencyManager_.reset( nullptr );
324 }

326 void Simulator::onProcessEnd() {
327 
328  std::cout << "[ Simulator ] : "
329  << "Started " << numEventsBegan_ << " events to produce "
330  << numEventsCompleted_ << " events."
331  << std::endl;
332 
333  // Delete Run Manager
334  // From Geant4 Basic Example B01:
335  // Job termination
336  // Free the store: user actions, physics list and detector descriptions are
337  // owned and deleted by the run manager, so they should not be deleted
338  // in the main() program
339  // This needs to happen here because otherwise, Geant4 objects are deleted twice:
340  // 1. When the histogram file is closed (all ROOT objects created during processing are put there because ROOT)
341  // 2. When Simulator is deleted because runManager_ is a unique_ptr
342  runManager_.reset( nullptr );
343 
344  // Delete the G4UIsession
345  // I don't think this needs to happen here, but since we are cleaning up loose ends...
346  sessionHandle_.reset( nullptr );
347 }

360 void Simulator::setSeeds(std::vector< int > seeds) {
361 
362  // If no seeds have been specified then return immediately.
363  if (seeds.empty()) return;
364 
365  // If seeds are specified, make sure that the container has at least
366  // two seeds. If not, throw an exception.
367  if (seeds.size() == 1) {
368  EXCEPTION_RAISE("ConfigurationException", "At least two seeds need to be specified.");
369  }
370 
371  // Create the array of seeds and pass them to G4Random. Currently,
372  // only 100 seeds can be specified at a time. If less than 100
373  // seeds are specified, the remaining slots are set to 0.
374  std::vector< long > seedVec(100, 0);
375  for (std::size_t index{0}; index < seeds.size(); ++index) seedVec[index] = static_cast<long>(seeds[index]);
376 
377  // Pass the array of seeds to the random engine.
378  G4Random::setTheSeeds(&seedVec[0]);
379 
380 }
