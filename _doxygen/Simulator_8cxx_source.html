53  void Simulator::configure(Parameters& parameters) {
54 
55  // parameters used to configure the simulation
56  parameters_ = parameters;
57 
58  // Set the run number. If not specified, the run number is set to 0.
59  int runNumber = parameters_.getParameter< int >( "runNumber" );
60  process_.setRunNumber( runNumber );
61 
62  // Set the verbosity level. The default level is 0.
63  verbosity_ = parameters_.getParameter< int >("verbosity");
64 
65  // If the verbosity level is set to 0,
66  // If the verbosity level is > 1, log everything to a file. Otherwise,
67  // dump the output. If a prefix has been specified, append it ot the
68  // log message.
69  auto loggingPrefix = parameters_.getParameter< std::string >("logging_prefix");
70  if ( verbosity_ == 0 ) sessionHandle_ = std::make_unique<BatchSession>();
71  else if ( verbosity_ > 1 ) {
72 
73  if ( loggingPrefix.empty() ) sessionHandle_ = std::make_unique<LoggedSession>();
74  else sessionHandle_ = std::make_unique<LoggedSession>( loggingPrefix + "_G4cout.log" , loggingPrefix + "_G4cerr.log" );
75 
76  }
77  if (sessionHandle_ != nullptr) uiManager_->SetCoutDestination( sessionHandle_.get() );
78 
79  // Instantiate the run manager.
80  runManager_ = std::make_unique<RunManager>(parameters);
81 
82  // Instantiate the GDML parser and corresponding messenger owned and
83  // managed by DetectorConstruction
84  G4GDMLParser *parser = new G4GDMLParser;
85 
86  // Instantiate the class so cascade parameters can be set.
87  G4CascadeParameters::Instance();
88 
89  // Set the DetectorConstruction instance used to build the detector
90  // from the GDML description.
91  runManager_->SetUserInitialization( new DetectorConstruction( parser , parameters ) );
92 
93  // Store the random numbers used to generate an event.
94  runManager_->SetRandomNumberStore( true );
95 
96  // Parse the detector geometry and validate if specified.
97  auto detectorPath{parameters_.getParameter< std::string >("detector")};
98  auto validateGeometry{parameters_.getParameter< bool >("validate_detector")};
99  if ( verbosity_ > 0 ) {
100  std::cout << "[ Simulator ] : Reading in geometry from '" << detectorPath << "'... " << std::flush;
101  }
102  G4GeometryManager::GetInstance()->OpenGeometry();
103  parser->Read( detectorPath, validateGeometry );
104  runManager_->DefineWorldVolume( parser->GetWorldVolume() );
105 
106  auto preInitCommands = parameters_.getParameter< std::vector< std::string > >("preInitCommands" );
107  for ( const std::string& cmd : preInitCommands ) {
108  if ( allowed(cmd) ) {
109  int g4Ret = uiManager_->ApplyCommand( cmd );
110  if ( g4Ret > 0 ) {
111  EXCEPTION_RAISE(
112  "PreInitCmd",
113  "Pre Initialization command '" + cmd + "' returned a failue status from Geant4: " + std::to_string(g4Ret)
114  );
115  }
116  } else {
117  EXCEPTION_RAISE(
118  "PreInitCmd",
119  "Pre Initialization command '" + cmd + "' is not allowed because another part of Simulator handles it."
120  );
121  }
122  }

131  void Simulator::produce(ldmx::Event& event) {
132 
133  // Pass the current LDMX event object to the persistency manager. This
134  // is needed by the persistency manager to fill the current event.
135  persistencyManager_->setCurrentEvent( &event );
136 
137  // Generate and process a Geant4 event.
138  runManager_->ProcessOneEvent( event.getEventHeader().getEventNumber() );
139 
140  // If a Geant4 event has been aborted, skip the rest of the processing
141  // sequence. This will immediately force the simulation to move on to
142  // the next event.
143  if ( runManager_->GetCurrentEvent()->IsAborted() ) { this->abortEvent(); }
144 
145  if ( process_.getLogFrequency() > 0 and event.getEventHeader().getEventNumber() % process_.getLogFrequency() == 0 ) {
146  //print according to log frequency and verbosity
147  if ( verbosity_ > 1 ) std::cout << "[ Simulator ] : Printing event contents:" << std::endl;
148  event.Print( verbosity_ );
149  }
150 
151  // Terminate the event. This checks if an event is to be stored or
152  // stacked for later.
153  runManager_->TerminateOneEvent();
154 
155  return;
156  }

158  void Simulator::onProcessStart() {
159 
160  //initialize run
161  runManager_->Initialize();
162 
163  auto randomSeeds = parameters_.getParameter<std::vector<int>>("randomSeeds");
164  if ( randomSeeds.size() > 1 ) {
165  //Geant4 allows for random seeds from 2 to 100
166  std::string cmd( "/random/setSeeds " );
167  for ( const int &seed : randomSeeds ) {
168  cmd += std::to_string(seed) + " ";
169  }
170  uiManager_->ApplyCommand( cmd );
171  }
172 
173  // Get the extra simulation configuring commands
174  auto postInitCommands = parameters_.getParameter< std::vector< std::string > >("postInitCommands");
175  for ( const std::string& cmd : postInitCommands ) {
176  if ( allowed(cmd) ) {
177  int g4Ret = uiManager_->ApplyCommand( cmd );
178  if ( g4Ret > 0 ) {
179  EXCEPTION_RAISE(
180  "PostInitCmd",
181  "Post Initialization command '" + cmd + "' returned a failue status from Geant4: " + std::to_string(g4Ret)
182  );
183  }
184  } else {
185  EXCEPTION_RAISE(
186  "PostInitCmd",
187  "Post Initialization command '" + cmd + "' is not allowed because another part of Simulator handles it."
188  );
189  }
190  }
191 
192  // Instantiate the scoring worlds including any parallel worlds.
193  runManager_->ConstructScoringWorlds();
194 
195  // Initialize the current run
196  runManager_->RunInitialization();
197 
198  // Initialize the event processing
199  runManager_->InitializeEventLoop( 1 );
200 
201  return;
202  }

215  // Geant4 expects us to handle the persistency manager

219  persistencyManager_.reset( nullptr );

222  void Simulator::onProcessEnd() {
223 
224  // Delete Run Manager
225  // From Geant4 Basic Example B01:
226  // Job termination
227  // Free the store: user actions, physics list and detector descriptions are
228  // owned and deleted by the run manager, so they should not be deleted
229  // in the main() program
230  // This needs to happen here because otherwise, Geant4 objects are deleted twice:
231  // 1. When the histogram file is closed (all ROOT objects created during processing are put there because ROOT)
232  // 2. When Simulator is deleted because runManager_ is a unique_ptr
233  runManager_.reset( nullptr );
234 
235  // Delete the G4UIsession
236  // I don't think this needs to happen here, but since we are cleaning up loose ends...
237  sessionHandle_.reset( nullptr );
238  }
