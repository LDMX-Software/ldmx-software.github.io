52void Simulator::configure(framework::config::Parameters& parameters) {
53 // parameters used to configure the simulation
54 parameters_ = parameters;
55
56 // Set the verbosity level. The default level is 0.
57 verbosity_ = parameters_.getParameter<int>("verbosity");
58
59 // If the verbosity level is set to 0,
60 // If the verbosity level is > 1, log everything to a file. Otherwise,
61 // dump the output. If a prefix has been specified, append it ot the
62 // log message.
63 auto loggingPrefix = parameters_.getParameter<std::string>("logging_prefix");
64 if (verbosity_ == 0)
65 sessionHandle_ = std::make_unique<BatchSession>();
66 else if (verbosity_ > 1) {
67 if (loggingPrefix.empty())
68 sessionHandle_ = std::make_unique<LoggedSession>();
69 else
70 sessionHandle_ = std::make_unique<LoggedSession>(
71 loggingPrefix + "_G4cout.log", loggingPrefix + "_G4cerr.log");
72 }
73 if (sessionHandle_ != nullptr)
74 uiManager_->SetCoutDestination(sessionHandle_.get());
75
76 // Instantiate the run manager.
77 runManager_ = std::make_unique<RunManager>(parameters, conditionsIntf_);
78
79 // Instantiate the GDML parser
80 auto parser{simcore::geo::ParserFactory::getInstance().createParser(
81 "gdml", parameters, conditionsIntf_)};
82
83 // Instantiate the class so cascade parameters can be set.
84 G4CascadeParameters::Instance();
85
86 // Set the DetectorConstruction instance used to build the detector
87 // from the GDML description.
88 runManager_->SetUserInitialization(
89 new DetectorConstruction(parser, parameters, conditionsIntf_));
90
91 G4GeometryManager::GetInstance()->OpenGeometry();
92 parser->read();
93 runManager_->DefineWorldVolume(parser->GetWorldVolume());
94
95 auto preInitCommands =
96 parameters_.getParameter<std::vector<std::string>>("preInitCommands", {});
97 for (const std::string& cmd : preInitCommands) {
98 if (allowed(cmd)) {
99 int g4Ret = uiManager_->ApplyCommand(cmd);
100 if (g4Ret > 0) {
101 EXCEPTION_RAISE("PreInitCmd",
102 "Pre Initialization command '" + cmd +
103 "' returned a failue status from Geant4: " +
104 std::to_string(g4Ret));
105 }
106 } else {
107 EXCEPTION_RAISE(
108 "PreInitCmd",
109 "Pre Initialization command '" + cmd +
110 "' is not allowed because another part of Simulator handles it.");
111 }
112 }
113}

123void Simulator::beforeNewRun(ldmx::RunHeader& header) {
124 // Get the detector header from the user detector construction
125 DetectorConstruction* detector =
126 static_cast<RunManager*>(RunManager::GetRunManager())
127 ->getDetectorConstruction();
128
129 if (!detector)
130 EXCEPTION_RAISE("SimSetup", "Detector not constructed before run start.");
131
132 header.setDetectorName(detector->getDetectorName());
133 header.setDescription(parameters_.getParameter<std::string>("description"));
134
135 header.setIntParameter("Save ECal Hit Contribs",
136 parameters_.getParameter<bool>("enableHitContribs"));
137 header.setIntParameter("Compress ECal Hit Contribs",
138 parameters_.getParameter<bool>("compressHitContribs"));
139 header.setIntParameter(
140 "Included Scoring Planes",
141 !parameters_.getParameter<std::string>("scoringPlanes").empty());
142 header.setIntParameter(
143 "Use Random Seed from Event Header",
144 parameters_.getParameter<bool>("rootPrimaryGenUseSeed"));
145
146 // lambda function for dumping 3-vectors into the run header
147 auto threeVectorDump = [&header](const std::string& name,
148 const std::vector<double>& vec) {
149 header.setFloatParameter(name + " X", vec.at(0));
150 header.setFloatParameter(name + " Y", vec.at(1));
151 header.setFloatParameter(name + " Z", vec.at(2));
152 };
153
154 auto beamSpotSmear{
155 parameters_.getParameter<std::vector<double>>("beamSpotSmear", {})};
156 if (!beamSpotSmear.empty())
157 threeVectorDump("Smear Beam Spot [mm]", beamSpotSmear);
158
159 // lambda function for dumping vectors of strings to the run header
160 auto stringVectorDump = [&header](const std::string& name,
161 const std::vector<std::string>& vec) {
162 int index = 0;
163 for (auto const& val : vec) {
164 header.setStringParameter(name + " " + std::to_string(++index), val);
165 }
166 };
167
168 stringVectorDump("Pre Init Command",
169 parameters_.getParameter<std::vector<std::string>>(
170 "preInitCommands", {}));
171 stringVectorDump("Post Init Command",
172 parameters_.getParameter<std::vector<std::string>>(
173 "postInitCommands", {}));
174
175 auto bops{PluginFactory::getInstance().getBiasingOperators()};
176 for (const XsecBiasingOperator* bop : bops) {
177 bop->RecordConfig(header);
178 }
179
180 auto dark_brem{
181 parameters_.getParameter<framework::config::Parameters>("dark_brem")};
182 if (dark_brem.getParameter<bool>("enable")) {
183 // the dark brem process is enabled, find it and then record its
184 // configuration
185 G4ProcessVector* electron_processes =
186 G4Electron::Electron()->GetProcessManager()->GetProcessList();
187 int n_electron_processes = electron_processes->size();
188 for (int i_process = 0; i_process < n_electron_processes; i_process++) {
189 G4VProcess* process = (*electron_processes)[i_process];
190 if (process->GetProcessName().contains(
191 darkbrem::G4eDarkBremsstrahlung::PROCESS_NAME)) {
192 // reset process to wrapped process if it is biased
193 if (dynamic_cast<G4BiasingProcessInterface*>(process))
194 process = dynamic_cast<G4BiasingProcessInterface*>(process)
195 ->GetWrappedProcess();
196 // record the process configuration to the run header
197 dynamic_cast<darkbrem::G4eDarkBremsstrahlung*>(process)->RecordConfig(
198 header);
199 break;
200 } // this process is the dark brem process
201 } // loop through electron processes
202 } // dark brem has been enabled
203
204 auto generators{
205 parameters_.getParameter<std::vector<framework::config::Parameters>>(
206 "generators")};
207 int counter = 0;
208 for (auto const& gen : generators) {
209 std::string genID = "Gen " + std::to_string(++counter);
210 auto className{gen.getParameter<std::string>("class_name")};
211 header.setStringParameter(genID + " Class", className);
212
213 if (className.find("simcore::ParticleGun") != std::string::npos) {
214 header.setFloatParameter(genID + " Time [ns]",
215 gen.getParameter<double>("time"));
216 header.setFloatParameter(genID + " Energy [GeV]",
217 gen.getParameter<double>("energy"));
218 header.setStringParameter(genID + " Particle",
219 gen.getParameter<std::string>("particle"));
220 threeVectorDump(genID + " Position [mm]",
221 gen.getParameter<std::vector<double>>("position"));
222 threeVectorDump(genID + " Direction",
223 gen.getParameter<std::vector<double>>("direction"));
224 } else if (className.find("simcore::MultiParticleGunPrimaryGenerator") !=
225 std::string::npos) {
226 header.setIntParameter(genID + " Poisson Enabled",
227 gen.getParameter<bool>("enablePoisson"));
228 header.setIntParameter(genID + " N Particles",
229 gen.getParameter<int>("nParticles"));
230 header.setIntParameter(genID + " PDG ID", gen.getParameter<int>("pdgID"));
231 threeVectorDump(genID + " Vertex [mm]",
232 gen.getParameter<std::vector<double>>("vertex"));
233 threeVectorDump(genID + " Momentum [MeV]",
234 gen.getParameter<std::vector<double>>("momentum"));
235 } else if (className.find("simcore::LHEPrimaryGenerator") !=
236 std::string::npos) {
237 header.setStringParameter(genID + " LHE File",
238 gen.getParameter<std::string>("filePath"));
239 } else if (className.find("simcore::RootCompleteReSim") !=
240 std::string::npos) {
241 header.setStringParameter(genID + " ROOT File",
242 gen.getParameter<std::string>("filePath"));
243 } else if (className.find("simcore::RootSimFromEcalSP") !=
244 std::string::npos) {
245 header.setStringParameter(genID + " ROOT File",
246 gen.getParameter<std::string>("filePath"));
247 header.setFloatParameter(genID + " Time Cutoff [ns]",
248 gen.getParameter<double>("time_cutoff"));
249 } else if (className.find("simcore::GeneralParticleSource") !=
250 std::string::npos) {
251 stringVectorDump(
252 genID + " Init Cmd",
253 gen.getParameter<std::vector<std::string>>("initCommands"));
254 } else {
255 ldmx_log(warn) << "Unrecognized primary generator '" << className << "'. "
256 << "Will not be saving details to RunHeader.";
257 }
258 }
259
260 // Set a string parameter with the Geant4 SHA-1.
261 if (G4RunManagerKernel::GetRunManagerKernel()) {
262 G4String g4Version{
263 G4RunManagerKernel::GetRunManagerKernel()->GetVersionString()};
264 header.setStringParameter("Geant4 revision", g4Version);
265 } else {
266 ldmx_log(warn) << "Unable to access G4 RunManager Kernel. Will not store "
267 "G4 Version string.";
268 }
269
270 header.setStringParameter("ldmx-sw revision", GIT_SHA1);
271}

283void Simulator::produce(framework::Event& event) {
284 // Pass the current LDMX event object to the persistency manager. This
285 // is needed by the persistency manager to fill the current event.
286 persistencyManager_->setCurrentEvent(&event);
287
288 // Generate and process a Geant4 event.
289 numEventsBegan_++;
290 runManager_->ProcessOneEvent(event.getEventHeader().getEventNumber());
291
292 // If a Geant4 event has been aborted, skip the rest of the processing
293 // sequence. This will immediately force the simulation to move on to
294 // the next event.
295 if (runManager_->GetCurrentEvent()->IsAborted()) {
296 runManager_->TerminateOneEvent(); // clean up event objects
297 this->abortEvent(); // get out of processors loop
298 }
299
300 if (this->getLogFrequency() > 0 and
301 event.getEventHeader().getEventNumber() % this->getLogFrequency() == 0) {
302 // print according to log frequency and verbosity
303 if (verbosity_ > 1) {
304 std::cout << "[ Simulator ] : Printing event contents:" << std::endl;
305 event.Print();
306 }
307 }
308
309 // Terminate the event. This checks if an event is to be stored or
310 // stacked for later.
311 numEventsCompleted_++;
312 runManager_->TerminateOneEvent();
313
314 return;
315}

317void Simulator::onProcessStart() {
318 // initialize run
319 runManager_->Initialize();
320
321 // Get the extra simulation configuring commands
322 auto postInitCommands = parameters_.getParameter<std::vector<std::string>>(
323 "postInitCommands", {});
324 for (const std::string& cmd : postInitCommands) {
325 if (allowed(cmd)) {
326 int g4Ret = uiManager_->ApplyCommand(cmd);
327 if (g4Ret > 0) {
328 EXCEPTION_RAISE("PostInitCmd",
329 "Post Initialization command '" + cmd +
330 "' returned a failue status from Geant4: " +
331 std::to_string(g4Ret));
332 }
333 } else {
334 EXCEPTION_RAISE(
335 "PostInitCmd",
336 "Post Initialization command '" + cmd +
337 "' is not allowed because another part of Simulator handles it.");
338 }
339 }
340
341 // Instantiate the scoring worlds including any parallel worlds.
342 runManager_->ConstructScoringWorlds();
343
344 // Initialize the current run
345 runManager_->RunInitialization();
346
347 // Initialize the event processing
348 runManager_->InitializeEventLoop(1);
349
350 return;
351}

353void Simulator::onFileClose(framework::EventFile&) {
354 // End the current run and print out some basic statistics if verbose
355 // level > 0.
356 runManager_->TerminateEventLoop();
357
358 // Pass the **real** number of events to the persistency manager
359 persistencyManager_->setNumEvents(numEventsBegan_, numEventsCompleted_);
360
361 // Persist any remaining events, call the end of run action and
362 // terminate the Geant4 kernel.
363 runManager_->RunTermination();
364
365 // Cleanup persistency manager
366 // Geant4 expects us to handle the persistency manager
367 // In order to avoid segfaulting nonsense, I delete it here
368 // so that it is deleted before the EventFile it references
369 // is deleted
370 persistencyManager_.reset(nullptr);
371}

373void Simulator::onProcessEnd() {
374 std::cout << "[ Simulator ] : "
375 << "Started " << numEventsBegan_ << " events to produce "
376 << numEventsCompleted_ << " events." << std::endl;
377
378 // Delete Run Manager
379 // From Geant4 Basic Example B01:
380 // Job termination
381 // Free the store: user actions, physics list and detector descriptions
382 // are owned and deleted by the run manager, so they should not be
383 // deleted in the main() program
384 // This needs to happen here because otherwise, Geant4 objects are deleted
385 // twice:
386 // 1. When the histogram file is closed (all ROOT objects created during
387 // processing are put there because ROOT)
388 // 2. When Simulator is deleted because runManager_ is a unique_ptr
389 runManager_.reset(nullptr);
390
391 // Delete the G4UIsession
392 // I don't think this needs to happen here, but since we are cleaning up loose
393 // ends...
394 sessionHandle_.reset(nullptr);
395}

397bool Simulator::allowed(const std::string& command) const {
398 for (const std::string& invalidSubstring : invalidCommands_) {
399 if (command.find(invalidSubstring) != std::string::npos) {
400 // found invalid substring in this command ==> NOT ALLOWED
401 return false;
402 }
403 }
404 // checked all invalid commands ==> ALLOWED
405 return true;
406}

408void Simulator::setSeeds(std::vector<int> seeds) {
409 // If no seeds have been specified then return immediately.
410 if (seeds.empty()) return;
411
412 // If seeds are specified, make sure that the container has at least
413 // two seeds. If not, throw an exception.
414 if (seeds.size() == 1) {
415 EXCEPTION_RAISE("ConfigurationException",
416 "At least two seeds need to be specified.");
417 }
418
419 // Create the array of seeds and pass them to G4Random. Currently,
420 // only 100 seeds can be specified at a time. If less than 100
421 // seeds are specified, the remaining slots are set to 0.
422 std::vector<long> seedVec(100, 0);
423 for (std::size_t index{0}; index < seeds.size(); ++index)
424 seedVec[index] = static_cast<long>(seeds[index]);
425
426 // Pass the array of seeds to the random engine.
427 G4Random::setTheSeeds(&seedVec[0]);
428}
