266  T getImpl(const std::string &collectionName,
267  const std::string &passName) const {
268  // get branch name
269  std::string branchName;
270  if (collectionName == ldmx::EventHeader::BRANCH)
271  branchName = collectionName;
272  else
273  branchName = makeBranchName(collectionName, passName);
274 
275  // if no passName, then find branchName by looking over known branches
276  if (passName.empty() && collectionName != ldmx::EventHeader::BRANCH) {
277  auto itKL = knownLookups_.find(collectionName);
278  if (itKL != knownLookups_.end())
279  branchName = itKL->second;
280  else {
281  // this collecitonName hasn't been found before
282  std::vector<std::vector<std::string>::const_iterator> matches;
283  branchName = collectionName + "_";
284  for (auto itBN = branchNames_.begin(); itBN != branchNames_.end();
285  itBN++) {
286  if (!itBN->compare(0, branchName.size(), branchName))
287  matches.push_back(itBN);
288  }
289  if (matches.empty()) {
290  // no matches found
291  EXCEPTION_RAISE("ProductNotFound",
292  "No product found for name '" + collectionName + "'");
293  } else if (matches.size() > 1) {
294  // more than one branch found
295  std::string names;
296  for (auto strs : matches) {
297  if (!names.empty()) names += ", ";
298  names += *strs;
299  }
300  EXCEPTION_RAISE("ProductAmbiguous",
301  "Multiple products found for name '" +
302  collectionName +
303  "' without specified pass name (" + names + ")");
304  } else {
305  // exactly one branch found
306  branchName = *matches.front();
307  knownLookups_[collectionName] = branchName;
308  }
309  }
310  }
311 
312  // get iterators to branch and collection
313  auto itBranch = branches_.find(branchName);
314  auto itPassenger = passengers_.find(branchName);
315 
316  if (itPassenger != passengers_.end()) {
317  if (itBranch != branches_.end()) {
318  // passenger and branch found
319  itBranch->second->GetEntry(ientry_);
320  // reading branches from input tree need to be manually updated
321  passengers_[branchName] = *((T *)(itBranch->second->GetObject()));
322  }
323  return std::get<T>(itPassenger->second);
324  } else if (inputTree_ == 0) {
325  // not found in loaded branches and there is no inputTree,
326  // so no hope of finding an unloaded object
327  EXCEPTION_RAISE("ProductNotFound", "No product found for name '" +
328  collectionName + "' and pass '" +
329  passName_ + "'");
330  }
331 
332  // find the active branch and update if necessary
333  if (itBranch != branches_.end()) {
334  // update buffers if needed
335  if (itBranch->second->GetReadEntry() != ientry_) {
336  itBranch->second->GetEntry(ientry_, 1);
337  }
338 
339  // check the objects map
340  if (itPassenger != passengers_.end())
341  return std::get<T>(itPassenger->second);
342 
343  // this case is hard (impossible?) to achieve
344  EXCEPTION_RAISE(
345  "ProductNotFound",
346  "A branch mis-match occurred. I'm not sure how I got here!");
347  } else {
348  // ok, maybe we've not loaded this yet, look for a branch
349  TBranchElement *branch = dynamic_cast<TBranchElement *>(
350  inputTree_->GetBranch(branchName.c_str()));
351  if (branch == 0) {
352  // inputTree doesn't have that branch
353  EXCEPTION_RAISE("ProductNotFound", "No product found for name '" +
354  collectionName + "' and pass '" +
355  passName_ + "'");
356  }
357  // ooh, new branch!
358  // load in the current entry
359  branch->SetStatus(1); // overrides any 'ignore' rules
360  branch->GetEntry((ientry_ < 0) ? (0) : (ientry_));
361 
362  // insert into maps of loaded branches and passengers
363  passengers_[branchName] = *((
364  T *)(branch->GetObject())); // this will fail if wrong type is passed
365  branches_[branchName] = branch;

367  return std::get<T>(passengers_.at(branchName));
