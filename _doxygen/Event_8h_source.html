248  const T& getObject(const std::string &collectionName,
249  const std::string &passName = "") const {
250  // get branch name
251  std::string branchName;
252  if (collectionName == ldmx::EventHeader::BRANCH) {
253  branchName = collectionName;
254  } else if (passName.empty()) {
255  //if no passName, then find branchName by looking over known products
256  if (knownLookups_.find(collectionName)==knownLookups_.end()) {
257  //this collectionName hasn't been found before
258  // wrap collection name in regex specifying that
259  // this collection name is the whole name and not a partial name
260  std::string coll_regex{"^"+collectionName+"$"};
261  auto matches = searchProducts(coll_regex,"","");
262  if (matches.empty()) {
263  // no matches found
264  EXCEPTION_RAISE("ProductNotFound",
265  "No product found for name '" + collectionName + "'");
266  } else if (matches.size() > 1) {
267  // more than one branch found
268  std::stringstream names;
269  for (auto strs : matches) { names << "\n" << strs; }
270  EXCEPTION_RAISE("ProductAmbiguous",
271  "Multiple products found for name '" +
272  collectionName +
273  "' without specified pass name :" + names.str() );
274  } else {
275  // exactly one branch found -> cache for later
276  knownLookups_[collectionName] =
277  makeBranchName(collectionName, matches.at(0).passname());
278  } //different options for number of possible branch matches
279  } //collection not in known lookups
280  branchName = knownLookups_.at(collectionName);
281  } else {
282  branchName = makeBranchName(collectionName, passName);
283  }
284 
285  // now we have determined the unique branch name to look for
286  // so we can start looking on the bus and the input tree
287  // (if it exists) for it
288  bool already_on_board{bus_.isOnBoard(branchName)};
289  if (not already_on_board and inputTree_) {
290  // branch is not on the bus but there is an input tree
291  // -> let's look for a new branch to load
292 
293  // default construct a new passenger
294  bus_.board<T>(branchName);
295 
296  // attempt to attach the new passenger to the input tree
297  TBranch* branch = bus_.attach(inputTree_,branchName,false);
298  if (branch == 0) {
299  //inputTree doesn't have that branch
300  EXCEPTION_RAISE(
301  "ProductNotFound",
302  "No product found for branch '"
303  + branchName
304  + "' on input tree."
305  );
306  }
307  // ooh, new branch!
308  branch->SetStatus(1); //overrides any 'ignore' rules
309  // load in the current entry
310  // this is necessary because getObject is called _after_ nextEvent
311  // (I think)
312  branch->GetEntry((ientry_<0)?(0):(ientry_));
313  } else if (not already_on_board) {
314  // not found in loaded branches and there is no inputTree,
315  // so no hope of finding an unloaded object
316  EXCEPTION_RAISE("ProductNotFound", "No product found for name '" +
317  collectionName + "' and pass '" +
318  passName_ + "'");
319  }
