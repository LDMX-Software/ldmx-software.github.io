144 template <typename T>
145 void add(const std::string &collectionName, T &obj) {
146 if (collectionName.find('_') != std::string::npos) {
147 EXCEPTION_RAISE("IllegalName",
148 "The product name '" + collectionName +
149 "' is illegal as it contains an underscore.");
150 }
151
152 // determine the branch name
153 std::string branchName;
154 if (collectionName == ldmx::EventHeader::BRANCH)
155 branchName = collectionName;
156 else
157 branchName = makeBranchName(collectionName);
158
159 if (branchesFilled_.find(branchName) != branchesFilled_.end()) {
160 EXCEPTION_RAISE("ProductExists",
161 "A product named '" + collectionName +
162 "' already exists in the event (has been loaded by a "
163 "previous producer in this process).");
164 }
165 branchesFilled_.insert(branchName);
166 // MEMORY add is leaking memory when given a vector (possible upon
167 // destruction of Event?) MEMORY add is 'conditional jump or move depends on
168 // uninitialised values' for all types of objects
169 // TTree::BranchImpRef or TTree::BronchExec
170 if (not bus_.isOnBoard(branchName)) {
171 // create a new branch for this collection
172
173 // have type T board bus under name 'branchName'
174 bus_.board<T>(branchName);
175
176 // type name (want to use branch element if possible)
177 std::string tname = typeid(obj).name();
178
179 if (outputTree_ and not shouldDrop(branchName)) {
180 // we are writing this branch to an output file, so let's
181 // attach this passenger to the output tree
182 TBranch *outBranch = bus_.attach(outputTree_, branchName, true);
183 // get type name from branch if possible,
184 // otherwise use compiler level type name (above)
185 std::string class_name{outBranch->GetClassName()};
186 if (not class_name.empty()) tname = class_name;
187 } // output tree exists or not
188
189 // check for cache entry to remove
190 auto it_known{knownLookups_.find(collectionName)};
191 if (it_known != knownLookups_.end())
192 knownLookups_.erase(it_known);
193
194 // add us to list of products
195 products_.emplace_back(collectionName, passName_, tname);
196 }
197
198 // copy input contents into bus passenger
199 try {
200 bus_.update(branchName, obj);
201 } catch(const std::bad_cast&) {
202 EXCEPTION_RAISE("TypeMismatch",
203 "Attempting to add an object whose type '" +
204 std::string(typeid(obj).name()) +
205 "' doesn't match the type stored in the collection.");
206 }
207
208 return;
209 }

249 template <typename T>
250 const T& getObject(const std::string &collectionName,
251 const std::string &passName = "") const {
252 // get branch name
253 std::string branchName;
254 if (collectionName == ldmx::EventHeader::BRANCH) {
255 branchName = collectionName;
256 } else if (passName.empty()) {
257 //if no passName, then find branchName by looking over known products
258 if (knownLookups_.find(collectionName)==knownLookups_.end()) {
259 //this collectionName hasn't been found before
260 // wrap collection name in regex specifying that
261 // this collection name is the whole name and not a partial name
262 std::string coll_regex{"^"+collectionName+"$"};
263 auto matches = searchProducts(coll_regex,"","");
264 if (matches.empty()) {
265 // no matches found
266 EXCEPTION_RAISE("ProductNotFound",
267 "No product found for name '" + collectionName + "'");
268 } else if (matches.size() > 1) {
269 // more than one branch found
270 std::stringstream names;
271 for (auto strs : matches) { names << "\n" << strs; }
272 EXCEPTION_RAISE("ProductAmbiguous",
273 "Multiple products found for name '" +
274 collectionName +
275 "' without specified pass name :" + names.str() );
276 } else {
277 // exactly one branch found -> cache for later
278 knownLookups_[collectionName] =
279 makeBranchName(collectionName, matches.at(0).passname());
280 } //different options for number of possible branch matches
281 } //collection not in known lookups
282 branchName = knownLookups_.at(collectionName);
283 } else {
284 branchName = makeBranchName(collectionName, passName);
285 }
286
287 // now we have determined the unique branch name to look for
288 // so we can start looking on the bus and the input tree
289 // (if it exists) for it
290 bool already_on_board{bus_.isOnBoard(branchName)};
291 if (not already_on_board and inputTree_) {
292 // branch is not on the bus but there is an input tree
293 // -> let's look for a new branch to load
294
295 // default construct a new passenger
296 bus_.board<T>(branchName);
297
298 // attempt to attach the new passenger to the input tree
299 TBranch* branch = bus_.attach(inputTree_,branchName,false);
300 if (branch == 0) {
301 //inputTree doesn't have that branch
302 EXCEPTION_RAISE(
303 "ProductNotFound",
304 "No product found for branch '"
305 + branchName
306 + "' on input tree."
307 );
308 }
309 // ooh, new branch!
310 branch->SetStatus(1); //overrides any 'ignore' rules
321 long long int ientry{inputTree_->GetReadEntry()};
322 if (ientry<0) {
323 //reached getObject without initializing inputTree's read entry
324 EXCEPTION_RAISE("InTreeInit",
325 "The input tree was un-initialized with read entry "
326 + std::to_string(ientry) + " when attempting to get '"
327 + branchName + "'.");
328 }
329 branch->GetEntry(ientry);
330 } else if (not already_on_board) {
331 // not found in loaded branches and there is no inputTree,
332 // so no hope of finding an unloaded object
333 EXCEPTION_RAISE("ProductNotFound", "No product found for name '" +
334 collectionName + "' and pass '" +
335 passName_ + "'");
336 }
337
338 // we've made sure the passenger is on the bus
339 // and the branch we are reading from (if we are reading)
340 // has been updated
341 // let's return the object that the passenger is carrying
342 try {
343 const T& obj = bus_.get<T>(branchName);
344 return obj;
345 } catch(const std::bad_cast&) {
346 EXCEPTION_RAISE(
347 "BadType",
348 "Trying to get product from '"
349 + branchName
350 + "' but asking for wrong type."
351 );
352 }
353 } // getObject
