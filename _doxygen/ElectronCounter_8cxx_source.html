34 void ElectronCounter::produce(framework::Event &event) {
35  int nElectrons = -1;
36 
37  if (useSimElectronCount_) {
38  if (nElectronsSim_ < 0) {
39  ldmx_log(fatal)
40  << "Can't use unset number of simulated electrons as electron count! "
41  "Set with 'simulated_electron_number' ";
42  return;
43  }
44  // then we just set it equal to simulated number and we're done
45  nElectrons = nElectronsSim_;
46  }
47  // Check if the collection of trig scint tracks exist. If not,
48  // don't bother processing the event.
49  else {
50  if (!event.exists(inputColl_, inputPassName_)) {
51  ldmx_log(fatal) << "Attemping to use non-existing input collection "
52  << inputColl_ << "_" << inputPassName_
53  << " to count electrons! Exiting.";
54  return;
55  }
56  // TODO, if cluster counting is needed: have two functions, one with tracks,
57  // one with clusters, and just call one or the other.
58 
59  // Get the collection of TS tracks
60  const std::vector<ldmx::TrigScintTrack> tracks =
61  event.getCollection<ldmx::TrigScintTrack>(inputColl_, inputPassName_);
62 
63  nElectrons = tracks.size();
64  ldmx_log(debug) << "Found " << tracks.size()
65  << " electrons (tracks) using input collection "
66  << inputColl_ << "_" << inputPassName_;
67  }
68  // add number of electrons to event header. allow for it to be unset (-1)
69  event.getEventHeader().setIntParameter("nElectrons", nElectrons);
70 }
