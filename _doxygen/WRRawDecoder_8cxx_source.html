78void WRRawDecoder::configure(framework::config::Parameters& ps) {

86void WRRawDecoder::onProcessStart() {
87 if (ntuplize_) {
88 getHistoDirectory();
89 tree_ = new TTree("wrraw","Flattened and decoded raw WR data");
90 tree_->Branch("run",&p.runNumber);
91 tree_->Branch("counter",&p.WRCounter);
92 tree_->Branch("channel",&p.channel);
93 tree_->Branch("seq_id",&p.seq_id);
94 tree_->Branch("sec",&p.sec);
95 tree_->Branch("coarse",&p.coarse);
96 tree_->Branch("frac",&p.frac);
97 }

100void WRRawDecoder::produce(framework::Event& event) {
101 // only add and fill when file able to readout packet
102 if (file_reader_ >> p) {
103 p.add(event,output_name_);
104 tree_->Fill();
105#ifdef DEBUG
106 std::cout << p << std::endl;
107#endif
108 }
109#ifdef DEBUG
110 else {
111 std::cout << "no more events" << std::endl;

113#endif
