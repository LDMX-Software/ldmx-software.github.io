378// Seed finder from Robert's in HPS
379// https://github.com/JeffersonLab/hps-java/blob/47712878302eb0c0374d077a208a6f8f0e2c3dc6/tracking/src/main/java/org/hps/recon/tracking/kalman/SeedTrack.java

382// yOrigin is the location along the beam about which we fit the seed helix
383// perigee_location is where the track parameters will be extracted
384
385ldmx::Track SeedFinderProcessor::SeedTracker(
386 const std::vector<ldmx::Measurement>& vmeas, double xOrigin,
387 const Acts::Vector3& perigee_location) {
388 // Fit a straight line in the non-bending plane and a parabola in the bending
389 // plane
390 // TODO:: use the actual errors from the measurement.
391
392 const int N = vmeas.size();
393 double uError = 0.06;
394 double vError = 40. / sqrt(12);
395
396 // Each measurement is treated as a 3D point, where the v direction is in the
397 // center of the strip with sigma equal to the length of the strip / sqrt(12).
398 // In this way it's easier to incorporate the tagger track extrapolation to
399 // the fit
400
401 Acts::ActsMatrix<5, 5> A = Acts::ActsMatrix<5, 5>::Zero();
402 Acts::ActsVector<5> Y = Acts::ActsVector<5>::Zero();
403
404 for (auto meas : vmeas) {
405 double xmeas = meas.getGlobalPosition()[0] - xOrigin;
406
407 // Get the surface
408 const Acts::Surface* hit_surface = ldmx_tg->getSurface(meas.getLayer());
409
410 // Get the global to local transformation
411 auto rot = hit_surface->transform(gctx_).rotation();
412 auto tr = hit_surface->transform(gctx_).translation();
413
414 auto rotl2g = rot.transpose();
415
416 // Only for saving purposes
417 Acts::Vector2 loc{meas.getLocalPosition()[0], 0.};
418
419 xhit_.push_back(xmeas);
420 yhit_.push_back(meas.getGlobalPosition()[1]);
421 zhit_.push_back(meas.getGlobalPosition()[2]);
422
423 Acts::ActsMatrix<2, 5> A_i;
424
425 A_i(0, 0) = rotl2g(0, 1);
426 A_i(0, 1) = rotl2g(0, 1) * xmeas;
427 A_i(0, 2) = rotl2g(0, 1) * xmeas * xmeas;
428 A_i(0, 3) = rotl2g(0, 2);
429 A_i(0, 4) = rotl2g(0, 2) * xmeas;
430
431 A_i(1, 0) = rotl2g(1, 1);
432 A_i(1, 1) = rotl2g(1, 1) * xmeas;
433 A_i(1, 2) = rotl2g(1, 1) * xmeas * xmeas;
434 A_i(1, 3) = rotl2g(1, 2);
435 A_i(1, 4) = rotl2g(1, 2) * xmeas;
436
437 // Fill the yprime vector
438 Acts::Vector2 offset = (rot.transpose() * tr).topRows<2>();
439 Acts::Vector2 xoffset = {rotl2g(0, 0) * xmeas, rotl2g(1, 0) * xmeas};
440
441 loc(0) = meas.getLocalPosition()[0];
442 loc(1) = 0.;
443
444 Acts::ActsMatrix<2, 2> W_i =
445 Acts::ActsMatrix<2, 2>::Zero(); // weight matrix
446
447 W_i(0, 0) = 1. / (uError * uError);
448 W_i(1, 1) = 1. / (vError * vError);
449
450 Acts::Vector2 Yprime_i = loc + offset - xoffset;
451
452 Acts::Vector2 yi = W_i * Yprime_i;
453 Y += (A_i.transpose()) * W_i * Yprime_i;
454
455 Acts::ActsMatrix<2, 5> WA_i = (W_i * A_i);
456 A += A_i.transpose() * WA_i;
457 }
458
459 Acts::ActsVector<5> B;
460 B = A.inverse() * Y;
461
462 b0_.push_back(B(0));
463 b1_.push_back(B(1));
464 b2_.push_back(B(2));
465 b3_.push_back(B(3));
466 b4_.push_back(B(4));
467
468 outputTree_->Fill();
469
470 Acts::ActsVector<5> hlx = Acts::ActsVector<5>::Zero();
471 Acts::ActsVector<3> ref{0., 0., 0.};
472 LineParabolaToHelix(B, hlx, ref);
473
474 double relativePerigeeX = perigee_location(0) - xOrigin;
475
476 std::shared_ptr<const Acts::PerigeeSurface> seed_perigee =
477 Acts::Surface::makeShared<Acts::PerigeeSurface>(Acts::Vector3(
478 relativePerigeeX, perigee_location(1), perigee_location(2)));
479
480 // in mm
481 Acts::Vector3 seed_pos{relativePerigeeX,
482 B(0) + B(1) * relativePerigeeX +
483 B(2) * relativePerigeeX * relativePerigeeX,
484 B(3) + B(4) * relativePerigeeX};
485 Acts::Vector3 dir{1, B(1) + 2 * B(2) * relativePerigeeX, B(4)};
486 dir /= dir.norm();
487
488 // Momentum at xmeas
489 double p =
490 0.3 * 1.5 * (1. / (2. * abs(B(2)))) * 0.001; // R in meters, p in GeV
491 // std::cout<<"Momentum "<< p*dir << std::endl;
492
493 // Convert it to MeV since that's what TrackUtils assumes
494 Acts::Vector3 seed_mom = p * dir / Acts::UnitConstants::MeV;
495 Acts::ActsScalar q =
496 B(2) < 0 ? -1 * Acts::UnitConstants::e : +1 * Acts::UnitConstants::e;
497 Acts::FreeVector seed_free =
498 tracking::sim::utils::toFreeParameters(seed_pos, seed_mom, q);
499
500 auto bound_params = Acts::detail::transformFreeToBoundParameters(
501 seed_free, *seed_perigee, gctx_)
502 .value();
503
504 ldmx_log(debug)
505 << "bound parameters at perigee location" << std::endl
506 << bound_params;
507
508
509 Acts::BoundVector stddev;
510 double sigma_p = 0.75 * p * Acts::UnitConstants::GeV;
511 stddev[Acts::eBoundLoc0] = 2 * Acts::UnitConstants::mm;
512 stddev[Acts::eBoundLoc1] = 5 * Acts::UnitConstants::mm;
513 stddev[Acts::eBoundTime] = 1000 * Acts::UnitConstants::ns;
514 stddev[Acts::eBoundPhi] = 5 * Acts::UnitConstants::degree;
515 stddev[Acts::eBoundTheta] = 5 * Acts::UnitConstants::degree;
516 stddev[Acts::eBoundQOverP] = (1. / p) * (1. / p) * sigma_p;
517
518 Acts::BoundSymMatrix bound_cov = stddev.cwiseProduct(stddev).asDiagonal();
519
520 ldmx::Track trk = ldmx::Track();
521 trk.setPerigeeLocation(perigee_location(0), perigee_location(1),
522 perigee_location(2));
523 trk.setChi2(0.);
524 trk.setNhits(5);
525 trk.setNdf(0);
526 trk.setNsharedHits(0);
527 std::vector<double> v_seed_params(
528 (bound_params).data(),
529 bound_params.data() + bound_params.rows() * bound_params.cols());
530 std::vector<double> v_seed_cov;
531 tracking::sim::utils::flatCov(bound_cov, v_seed_cov);
532 trk.setPerigeeParameters(v_seed_params);
533 trk.setPerigeeCov(v_seed_cov);
534
535 Acts::BoundTrackParameters seedParameters(seed_perigee,
536 std::move(bound_params), bound_cov);
537
538 return trk;
539}

541// To augment with the covariance matrix.
542// The following formulas only works if the center if x0 where the
543// linear+parabola function is evaluated is 0. It also assumes the axes
544// orientation according to the ACTS needs. phi0 is the angle of the tangent of
545// the track at the perigee. Positive counterclockwise. Theta is positive from
546// the z axis to the bending plane. (pi/2 for tracks along the x axis)
547
548void SeedFinderProcessor::LineParabolaToHelix(
549 const Acts::ActsVector<5> parameters, Acts::ActsVector<5>& helix_parameters,
550 Acts::Vector3 ref) {
551 double R = 0.5 / abs(parameters(2));
552 double xc = R * parameters(1);
553 double p2 = 0.5 * parameters(1) * parameters(1);
554 double factor = parameters(2) < 0 ? -1 : 1;
555 double yc =
556 parameters(0) +
557 factor *
558 (R * (1 - p2)); //+ or minus solution. I chose beta0 - R ( ... ),
559 // because the curvature is negative for electrons.
560 // The sign need to be chosen by the sign of B(2)
561 double theta0 = atan2(yc, xc);
562 double k = parameters(2) < 0 ? (-1. / R) : 1. / R;
563 double d0 = R - xc / cos(theta0);
564 double phi0 = theta0 + 1.57079632679;
565 double tanL = parameters(4) * cos(theta0);
566 double theta = 1.57079632679 - atan(tanL);
567 double z0 = parameters(3) + (d0 * tanL * tan(theta0));
568 double qOp = factor / (0.3 * 1.5 * R * 0.001);
569
570 // std::cout<<d0<<" "<<z0<<" "<<phi0<<" "<<theta<<" "<<qOp<<std::endl;
571}
