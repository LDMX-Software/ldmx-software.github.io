152ldmx::Track SeedFinderProcessor::SeedTracker(
153 const std::vector<ldmx::Measurement>& vmeas, double xOrigin,
154 const Acts::Vector3& perigee_location) {
155 // Fit a straight line in the non-bending plane and a parabola in the bending
156 // plane
157 // TODO:: use the actual errors from the measurement.
158
159 const int N = vmeas.size();
160 double uError = 0.006;
161 double vError = 40. / sqrt(12);
162
163 // Each measurement is treated as a 3D point, where the v direction is in the
164 // center of the strip with sigma equal to the length of the strip / sqrt(12).
165 // In this way it's easier to incorporate the tagger track extrapolation to
166 // the fit
167
168 Acts::ActsMatrix<5, 5> A = Acts::ActsMatrix<5, 5>::Zero();
169 Acts::ActsVector<5> Y = Acts::ActsVector<5>::Zero();
170
171 for (auto meas : vmeas) {
172 double xmeas = meas.getGlobalPosition()[0] - xOrigin;
173
174 // Get the surface
175 const Acts::Surface* hit_surface = ldmx_tg->getSurface(meas.getLayerID());
176
177 // Get the global to local transformation
178 auto rot = hit_surface->transform(gctx_).rotation();
179 auto tr = hit_surface->transform(gctx_).translation();
180
181 auto rotl2g = rot.transpose();
182
183 // Only for saving purposes
184 Acts::Vector2 loc{meas.getLocalPosition()[0], 0.};
185
186 xhit_.push_back(xmeas);
187 yhit_.push_back(meas.getGlobalPosition()[1]);
188 zhit_.push_back(meas.getGlobalPosition()[2]);
189
190 Acts::ActsMatrix<2, 5> A_i;
191
192 A_i(0, 0) = rotl2g(0, 1);
193 A_i(0, 1) = rotl2g(0, 1) * xmeas;
194 A_i(0, 2) = rotl2g(0, 1) * xmeas * xmeas;
195 A_i(0, 3) = rotl2g(0, 2);
196 A_i(0, 4) = rotl2g(0, 2) * xmeas;
197
198 A_i(1, 0) = rotl2g(1, 1);
199 A_i(1, 1) = rotl2g(1, 1) * xmeas;
200 A_i(1, 2) = rotl2g(1, 1) * xmeas * xmeas;
201 A_i(1, 3) = rotl2g(1, 2);
202 A_i(1, 4) = rotl2g(1, 2) * xmeas;
203
204 // Fill the yprime vector
205 Acts::Vector2 offset = (rot.transpose() * tr).topRows<2>();
206 Acts::Vector2 xoffset = {rotl2g(0, 0) * xmeas, rotl2g(1, 0) * xmeas};
207
208 loc(0) = meas.getLocalPosition()[0];
209 loc(1) = 0.;
210
211 Acts::ActsMatrix<2, 2> W_i =
212 Acts::ActsMatrix<2, 2>::Zero(); // weight matrix
213
214 W_i(0, 0) = 1. / (uError * uError);
215 W_i(1, 1) = 1. / (vError * vError);
216
217 Acts::Vector2 Yprime_i = loc + offset - xoffset;
218
219 Acts::Vector2 yi = W_i * Yprime_i;
220 Y += (A_i.transpose()) * W_i * Yprime_i;
221
222 Acts::ActsMatrix<2, 5> WA_i = (W_i * A_i);
223 A += A_i.transpose() * WA_i;
224 }
225
226 Acts::ActsVector<5> B;
227 B = A.inverse() * Y;
228
229 b0_.push_back(B(0));
230 b1_.push_back(B(1));
231 b2_.push_back(B(2));
232 b3_.push_back(B(3));
233 b4_.push_back(B(4));
234
235 Acts::ActsVector<5> hlx = Acts::ActsVector<5>::Zero();
236 Acts::ActsVector<3> ref{0., 0., 0.};
237 LineParabolaToHelix(B, hlx, ref);
238
239 double relativePerigeeX = perigee_location(0) - xOrigin;
240
241 std::shared_ptr<const Acts::PerigeeSurface> seed_perigee =
242 Acts::Surface::makeShared<Acts::PerigeeSurface>(Acts::Vector3(
243 relativePerigeeX, perigee_location(1), perigee_location(2)));
244
245 // in mm
246 Acts::Vector3 seed_pos{relativePerigeeX,
247 B(0) + B(1) * relativePerigeeX +
248 B(2) * relativePerigeeX * relativePerigeeX,
249 B(3) + B(4) * relativePerigeeX};
250 Acts::Vector3 dir{1, B(1) + 2 * B(2) * relativePerigeeX, B(4)};
251 dir /= dir.norm();
252
253 // Momentum at xmeas
254 double p =
255 0.3 * bfield_ * (1. / (2. * abs(B(2)))) * 0.001; // R in meters, p in GeV
256 // std::cout<<"Momentum "<< p*dir << std::endl;
257
258 // Convert it to MeV since that's what TrackUtils assumes
259 Acts::Vector3 seed_mom = p * dir / Acts::UnitConstants::MeV;
260 Acts::ActsScalar q =
261 B(2) < 0 ? -1 * Acts::UnitConstants::e : +1 * Acts::UnitConstants::e;
262
263
264 // Linear intersection with the perigee line. TODO:: Use propagator instead
265 // Project the position on the surface.
266 // This is mainly necessary for the perigee surface, where
267 // the mean might not fulfill the perigee condition.
268
269 auto intersection =
270 (*seed_perigee).intersect(gctx_, seed_pos, dir, false);
271
272 Acts::FreeVector seed_free =
273 tracking::sim::utils::toFreeParameters(intersection.intersection.position, seed_mom, q);
274
275 auto bound_params = Acts::detail::transformFreeToBoundParameters(
276 seed_free, *seed_perigee, gctx_)
277 .value();
278
279 ldmx_log(debug) << "bound parameters at perigee location" << std::endl
280 << bound_params;
281
282 Acts::BoundVector stddev;
283 double sigma_p = 0.75 * p * Acts::UnitConstants::GeV;
284 stddev[Acts::eBoundLoc0] = 2 * Acts::UnitConstants::mm;
285 stddev[Acts::eBoundLoc1] = 5 * Acts::UnitConstants::mm;
286 stddev[Acts::eBoundTime] = 1000 * Acts::UnitConstants::ns;
287 stddev[Acts::eBoundPhi] = 5 * Acts::UnitConstants::degree;
288 stddev[Acts::eBoundTheta] = 5 * Acts::UnitConstants::degree;
289 stddev[Acts::eBoundQOverP] = (1. / p) * (1. / p) * sigma_p;
290
291 Acts::BoundSymMatrix bound_cov = stddev.cwiseProduct(stddev).asDiagonal();
292
293 ldmx::Track trk = ldmx::Track();
294 trk.setPerigeeLocation(perigee_location(0), perigee_location(1),
295 perigee_location(2));
296 trk.setChi2(0.);
297 trk.setNhits(5);
298 trk.setNdf(0);
299 trk.setNsharedHits(0);
300 std::vector<double> v_seed_params(
301 (bound_params).data(),
302 bound_params.data() + bound_params.rows() * bound_params.cols());
303 std::vector<double> v_seed_cov;
304 tracking::sim::utils::flatCov(bound_cov, v_seed_cov);
305 trk.setPerigeeParameters(v_seed_params);
306 trk.setPerigeeCov(v_seed_cov);
307
308 Acts::BoundTrackParameters seedParameters(seed_perigee,
309 std::move(bound_params), bound_cov);
310
311 return trk;
312}

314// To augment with the covariance matrix.
315// The following formulas only works if the center if x0 where the
316// linear+parabola function is evaluated is 0. It also assumes the axes
317// orientation according to the ACTS needs. phi0 is the angle of the tangent of
318// the track at the perigee. Positive counterclockwise. Theta is positive from
319// the z axis to the bending plane. (pi/2 for tracks along the x axis)
320
321void SeedFinderProcessor::LineParabolaToHelix(
322 const Acts::ActsVector<5> parameters, Acts::ActsVector<5>& helix_parameters,
323 Acts::Vector3 ref) {
324 double R = 0.5 / abs(parameters(2));
325 double xc = R * parameters(1);
326 double p2 = 0.5 * parameters(1) * parameters(1);
327 double factor = parameters(2) < 0 ? -1 : 1;
328 double yc =
329 parameters(0) +
330 factor *
331 (R * (1 - p2)); //+ or minus solution. I chose beta0 - R ( ... ),
332 // because the curvature is negative for electrons.
333 // The sign need to be chosen by the sign of B(2)
334 double theta0 = atan2(yc, xc);
335 double k = parameters(2) < 0 ? (-1. / R) : 1. / R;
336 double d0 = R - xc / cos(theta0);
337 double phi0 = theta0 + 1.57079632679;
338 double tanL = parameters(4) * cos(theta0);
339 double theta = 1.57079632679 - atan(tanL);
340 double z0 = parameters(3) + (d0 * tanL * tan(theta0));
341 double qOp = factor / (0.3 * bfield_ * R * 0.001);
342
343 // std::cout<<d0<<" "<<z0<<" "<<phi0<<" "<<theta<<" "<<qOp<<std::endl;
344}

407void SeedFinderProcessor::FindSeedsFromMap(std::vector<ldmx::Track>& seeds) {
408 std::vector<ldmx::Measurement> meas_for_seeds;
409 meas_for_seeds.reserve(5);
410
411 std::map<int, std::vector<const ldmx::Measurement*>>::iterator groups_iter =
412 groups_map.begin();
413 std::vector<const ldmx::Measurement*>::iterator meas_iter;
414
415 // Vector of iterators
416
417 constexpr size_t K = 5;
418 std::vector<std::vector<const ldmx::Measurement*>::iterator> it;
419 it.reserve(K);
420
421 unsigned int ikey = 0;
422 for (auto& key : groups_map) {
423 it[ikey] = key.second.begin();
424 ikey++;
425 }
426
427 // K vectors in an array v[0],v[1].... v[K-1]
428
429 while (it[0] != groups_iter->second.end()) {
430 // process the pointed-to elements
431
432 /*
433 for (int j=0; j<K; j++) {
434 const ldmx::Measurement* meas = (*(it[j]));
435 std::cout<<meas->getGlobalPosition()[0]<<","
436 <<meas->getGlobalPosition()[1]<<","
437 <<meas->getGlobalPosition()[2]<<","<<std::endl;
438 }
439 */
440
441 std::vector<ldmx::Measurement> meas_for_seeds;
442 meas_for_seeds.reserve(5);
443
444 ldmx_log(debug)<<" Grouping ";
445
446 for (int j = 0; j < K; j++) {
447 const ldmx::Measurement* meas = (*(it[j]));
448 meas_for_seeds.push_back(*meas);
449 }
450
451 std::sort(meas_for_seeds.begin(), meas_for_seeds.end(),
452 [](const ldmx::Measurement& m1, const ldmx::Measurement& m2) {
453 return m1.getGlobalPosition()[0] < m2.getGlobalPosition()[0];
454 });
455
456 if (meas_for_seeds.size() < 5) {
457 nmissing_++;
458 return;
459 }
460
461 ldmx_log(debug)<<"seedTrack";
462
463 ldmx::Track seedTrack =
464 SeedTracker(meas_for_seeds, meas_for_seeds.at(2).getGlobalPosition()[0],
465 Acts::Vector3(perigee_location_[0], perigee_location_[1],
466 perigee_location_[2]));
467
468 bool fail = false;
469 // Remove failed fits
470
471 if (1. / abs(seedTrack.getQoP()) < pmin_) {
472 nfailpmin_++;
473 fail = true;
474 } else if (1. / abs(seedTrack.getQoP()) > pmax_) {
475 nfailpmax_++;
476 fail = true;
477 } else if (abs(seedTrack.getZ0()) > z0max_) {
478 nfailz0max_++;
479 fail = true;
480 } else if (seedTrack.getD0() < d0min_) {
481 nfaild0min_++;
482 fail = true;
483 } else if (seedTrack.getD0() > d0max_) {
484 nfaild0max_++;
485 fail = true;
486 }
487
488 if (!fail) {
489
490 if (truthMatchingTool_->configured()) {
491 auto truthInfo = truthMatchingTool_->TruthMatch(meas_for_seeds);
492 seedTrack.setTrackID(truthInfo.trackID);
493 seedTrack.setPdgID(truthInfo.pdgID);
494 seedTrack.setTruthProb(truthInfo.truthProb);
495 }
496
497 seeds.push_back(seedTrack);
498 }
499
500 else {
501 b0_.pop_back();
502 b1_.pop_back();
503 b2_.pop_back();
504 b3_.pop_back();
505 b4_.pop_back();
506 }
507
508 // Go to next combination
509 ldmx_log(debug)<<"Go to the next combination";
510
511 ++it[K - 1];
512 for (int i = K - 1;
513 (i > 0) && (it[i] == (std::next(groups_iter, i))->second.end()); --i) {
514 it[i] = std::next(groups_iter, i)->second.begin();
515 ++it[i - 1];
516 }
517 }
518} // find seeds
