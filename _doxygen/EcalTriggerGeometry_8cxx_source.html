17EcalTriggerGeometry::EcalTriggerGeometry(int symmetry,
18 const ldmx::EcalGeometry* ecalGeom)
19 : ConditionsObject(CONDITIONS_OBJECT_NAME),
20 symmetry_{symmetry},
21 ecalGeometry_{ecalGeom} {
22 if ((symmetry_ & MODULES_MASK) == INPLANE_IDENTICAL) {
23 // first set is the same regardless of alignment...
24 int tcell = 0;
26 for (int v = 1; v <= 10; v += 3) {
27 for (int u = 1; u <= 10; u += 3) {
28 ldmx::EcalTriggerID tid(0, 0, tcell);
29 std::vector<ldmx::EcalID> pids;
30 for (int du = -1; du <= 1; du++) {
31 for (int dv = -1; dv <= 1; dv++) {
32 ldmx::EcalID pid(0, 0, u + du, v + dv);
33 precision2trigger_[pid] = tid;
34 pids.push_back(pid);
35 }
36 }
37 trigger2precision_[tid] = pids;
38 tcell++;
39 }
40 }
42 for (int v = 13; v <= 22; v += 3) {
43 for (int u = v - 10; u <= v; u += 3) {
44 ldmx::EcalTriggerID tid(0, 0, tcell);
45 std::vector<ldmx::EcalID> pids;
46 for (int dv = -1; dv <= 1; dv++) {
47 for (int du = -1; du <= 1; du++) {
48 ldmx::EcalID pid(0, 0, u + du + dv,
49 v + dv); // changes directions here
50 precision2trigger_[pid] = tid;
51 pids.push_back(pid);
52 }
53 }
54 trigger2precision_[tid] = pids;
55 tcell++;
56 }
57 }
58 // right side
59 for (int v = 2; v <= 22; v += 3) {
60 int irow = (v - 2) / 3;
61 for (int icol = 0; icol <= std::min(irow, 3); icol++) {
62 if (irow - icol >= 4) continue;
63 ldmx::EcalTriggerID tid(0, 0, tcell);
64 std::vector<ldmx::EcalID> pids;
65 int u = 13 + 3 * icol;
66 for (int dv = -1; dv <= 1; dv++) {
67 for (int du = -1; du <= 1; du++) {
68 ldmx::EcalID pid(0, 0, u + du, v + du + dv);
69 precision2trigger_[pid] = tid;
70 pids.push_back(pid);
71 }
72 }
73 trigger2precision_[tid] = pids;
74 tcell++;
75 }
76 }
77
78 } else {
79 // raise an exception...
80 }
81}

83std::vector<ldmx::EcalID> EcalTriggerGeometry::contentsOfTriggerCell(
84 ldmx::EcalTriggerID triggerCell) const {
85 ldmx::EcalTriggerID effId;
86 std::vector<ldmx::EcalID> retval;
87 if ((symmetry_ & MODULES_MASK) == INPLANE_IDENTICAL) {
88 effId = ldmx::EcalTriggerID(0, 0, triggerCell.triggercell());
89 }
90 auto ptr = trigger2precision_.find(effId);
91 if (ptr != trigger2precision_.end()) {
92 for (auto idz : ptr->second) {
93 retval.push_back(
94 ldmx::EcalID(triggerCell.layer(), triggerCell.module(), idz.cell()));
95 }
96 }
97 return retval;
98}

100ldmx::EcalID EcalTriggerGeometry::centerInTriggerCell(
101 ldmx::EcalTriggerID triggerCell) const {
102 ldmx::EcalTriggerID effId;
103 if ((symmetry_ & MODULES_MASK) == INPLANE_IDENTICAL) {
104 effId = ldmx::EcalTriggerID(0, 0, triggerCell.triggercell());
105 }
106 auto ptr = trigger2precision_.find(effId);
107 if (ptr == trigger2precision_.end()) {
108 std::stringstream ss;
109 ss << "Unable to find trigger cell " << triggerCell;
110 EXCEPTION_RAISE("EcalGeometryException", ss.str());
111 }
112
113 return ldmx::EcalID(triggerCell.layer(), triggerCell.module(),
114 ptr->second[4].cell());
115}

117ldmx::EcalTriggerID EcalTriggerGeometry::belongsTo(
118 ldmx::EcalID precisionCell) const {
119 ldmx::EcalID effId;
120 if ((symmetry_ & MODULES_MASK) == INPLANE_IDENTICAL) {
121 effId = ldmx::EcalID(0, 0, precisionCell.cell());
122 }
123 auto ptr = precision2trigger_.find(effId);
124 if (ptr == precision2trigger_.end()) {
125 return ldmx::EcalTriggerID(0, 0, 0); // not ideal
126 } else {
127 return ldmx::EcalTriggerID(precisionCell.layer(), precisionCell.module(),
128 ptr->second.triggercell());
129 }
130}
