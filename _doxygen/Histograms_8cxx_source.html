73void HistogramHelper::create(const std::string& name, const std::string& xLabel,
74 const std::vector<double>& bins) {
75 std::string fullName = name_ + "_" + name;
76
77 // copy bin edges into a C98 form acceptable by ROOT
78 int nbins = bins.size() - 1;
79 double* binEdges = new double[bins.size()];
80 for (unsigned int iBin = 0; iBin < bins.size(); iBin++)
81 binEdges[iBin] = bins.at(iBin);
82
83 auto hist = new TH1F(fullName.c_str(), fullName.c_str(), nbins, binEdges);
84
85 delete[] binEdges; // cleanup
86
87 // Set the title
88 hist->SetTitle("");
89
90 // Set the x-axis label
91 hist->GetXaxis()->SetTitle(xLabel.c_str());
92 hist->GetXaxis()->CenterTitle();
93
94 // Insert it into the pool of histograms for later use
95 HistogramPool::getInstance().insert(fullName, hist);
