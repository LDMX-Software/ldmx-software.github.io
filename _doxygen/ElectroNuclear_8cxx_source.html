30 << std::endl;*/
31
32 std::string currentProcess =
33 callingProcess->GetWrappedProcess()->GetProcessName();
34 if (currentProcess.compare(this->getProcessToBias()) == 0) {
35 G4double interactionLength =
36 callingProcess->GetWrappedProcess()->GetCurrentInteractionLength();
37 /*std::cout << "[ ElectroNuclearXsecBiasingOperator ]: "
38 << "EN Interaction length: "
39 << interactionLength << std::endl;*/
40
41 double enXsecUnbiased = 1. / interactionLength;
42 /*std::cout << "[ ElectroNuclearXsecBiasingOperator ]: Unbiased EN xsec: "
43 << enXsecUnbiased << std::endl;*/
44
45 double enXsecBiased = enXsecUnbiased * factor_;
46 /*std::cout << "[ ElectroNuclearXsecBiasingOperator ]: Biased EN xsec: "
47 << enXsecBiased << std::endl;*/
48
49 return BiasedXsec(enXsecBiased);
50 } else
51 return 0;
