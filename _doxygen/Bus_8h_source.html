510 TBranch* attachBasic(TTree* tree, const std::string& branch_name,
511 bool can_create) {
512 TBranch* branch = tree->GetBranch(branch_name.c_str());
513 if (branch) {
514 // branch already exists
515 // set the object the branch should read/write from/to
516 branch->SetAddress(baggage_);
517 } else if (can_create) {
518 static const std::map<std::string, std::string> cpp_to_root_type_name =
519 {{"b", "O"}, {"s", "S"}, {"i", "I"},
520 {"l", "L"}, {"f", "F"}, {"d", "D"}};
521 // branch doesnt exist and we are allowed to make a new one
522 std::string cpp_type = typeid(*baggage_).name();
523 branch = tree->Branch(
524 branch_name.c_str(), baggage_,
525 (branch_name + "/" + cpp_to_root_type_name.at(cpp_type)).c_str());
526 }
527 return branch;
528 }
