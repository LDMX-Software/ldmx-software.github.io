9EventPacket::EventPacket(uint32_t id, const std::map<uint16_t,std::vector<uint32_t>>& unwrapped_subsys_data)
10 : id_{id}, event_length_in_words_{0} {
11 crc_ok_ = true;
12
13 utility::CRC crc;
14
15 for (auto const&[subsys_id, subsys_data] : unwrapped_subsys_data) {
16 subsys_data_.emplace_back(id_,subsys_id,subsys_data);
17 auto& subsys = subsys_data_.back();
18 crc << subsys;
19 event_length_in_words_ += subsys.header().size()+subsys.data().size()+subsys.tail().size();
20 }
21
22 crc_ = crc.get();
23}

25std::vector<uint32_t> EventPacket::header() const {
26 uint32_t word = ((subsys_data_.size() & utility::mask<16>) << 16)
27 + ((event_length_in_words_ & utility::mask<15>) << 1)
28 + crc_ok_;
29 return { id_ , word };
30}

32std::vector<uint32_t> EventPacket::tail() const {

36utility::Reader& EventPacket::read(utility::Reader& r) {
37 r >> id_;
38
39 uint32_t word;
40 r >> word;
41
42 uint16_t num_subsys = (word >> 16) & utility::mask<16>;
43 event_length_in_words_ = (word >> 1) & utility::mask<15>;
44 crc_ok_ = word & utility::mask<1>;
45
46 r.read(subsys_data_, num_subsys);
47
48 r >> crc_;
49}
