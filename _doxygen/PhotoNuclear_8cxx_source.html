30G4VBiasingOperation* PhotoNuclear::ProposeOccurenceBiasingOperation(
31 const G4Track* track, const G4BiasingProcessInterface* callingProcess) {
32 /*std::cout << "[ PhotoNuclearXsecBiasingOperator ]: "
33 << "Kinetic energy: " << track->GetKineticEnergy()
34 << " MeV" << std::endl;*/
35
36 // if we want to only bias children of primary, leave if this track is NOT a
37 // child of the primary
38 if (only_children_of_primary_ and track->GetParentID() != 1) return 0;
39
40 // is this track too low energy to be biased?
41 if (track->GetKineticEnergy() < threshold_) return 0;
42
43 /*std::cout << "[ PhotoNuclearXsecBiasingOperator ]: "
44 << "Calling process: "
45 << callingProcess->GetWrappedProcess()->GetProcessName()
46 << std::endl;*/
47
48 std::string currentProcess =
49 callingProcess->GetWrappedProcess()->GetProcessName();
50 if (currentProcess.compare(this->getProcessToBias()) == 0) {
51 G4double interactionLength =
52 callingProcess->GetWrappedProcess()->GetCurrentInteractionLength();
53 /*std::cout << "[ PhotoNuclearXsecBiasingOperator ]: "
54 << "PN Interaction length: "
55 << interactionLength << std::endl;*/
56
57 pnXsecUnbiased_ = 1. / interactionLength;
58 /*std::cout << "[ PhotoNuclearXsecBiasingOperator ]: Unbiased PN xsec: "
59 << pnXsecUnbiased_ << std::endl;*/
60
61 pnXsecBiased_ = pnXsecUnbiased_ * factor_;
62 /*std::cout << "[ PhotoNuclearXsecBiasingOperator ]: Biased PN xsec: "
63 << pnXsecBiased_ << std::endl;*/
64
65 return BiasedXsec(pnXsecBiased_);
66
67 } else if ((currentProcess.compare(CONVERSION_PROCESS) == 0) and
68 down_bias_conv_) {
69 G4double interactionLength =
70 callingProcess->GetWrappedProcess()->GetCurrentInteractionLength();
71 /*std::cout << "[ PhotoNuclearXsecBiasingOperator ]: "
72 << "EM Interaction length: "
73 << interactionLength << std::endl;*/
74
75 double emXsecUnbiased = 1. / interactionLength;
76 /*std::cout << "[ PhotoNuclearXsecBiasingOperator ]: Unbiased EM xsec: "
77 << emXsecUnbiased << std::endl;*/
78
79 double emXsecBiased = std::max(
80 emXsecUnbiased + pnXsecUnbiased_ - pnXsecBiased_, pnXsecUnbiased_);
81 if (emXsecBiased == pnXsecUnbiased_) {
82 G4cout << "[ PhotoNuclearXsecBiasingOperator ]: [ WARNING ]: "
83 << "Biasing factor is too large." << std::endl;
84 }
85 /*std::cout << "[ PhotoNuclearXsecBiasingOperator ]: Biased EM xsec: "
86 << emXsecBiased << std::endl;*/
87
88 emXsecOperation->SetBiasedCrossSection(emXsecBiased);
89 emXsecOperation->Sample();
90
91 return emXsecOperation;
92
93 } else
94 return 0;
95}
