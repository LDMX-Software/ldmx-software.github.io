64  void EcalHexReadout::buildCellMap() {
69  TH2Poly gridMap;
70 
71  // make hexagonal grid [boundary is rectangle] larger than the module
72  double gridMinX = 0., gridMinY = 0.; //start at the origin
73  int numXCells=0, numYCells=0;
74 
75  //first x-cell is only a half
76  gridMinX -= cellr_;
77  numXCells++;
78  while ( gridMinX > -1*moduleR_ ) {
79  gridMinX -= 2*cellr_; //decrement x by cell center-to-flat diameter
80  numXCells++;
81  }
82  while ( gridMinY > -1*moduler_ ) {
83  //decrement y by cell center-to-corner radius
84  // alternate between a full corner-to-corner diameter
85  // and a side of a cell (center-to-corner radius)
86  if (numYCells%2==0) gridMinY -= 1*cellR_;
87  else gridMinY -= 2*cellR_;
88  numYCells++;
89  }
90  //only counted one half of the cells
91  numXCells *= 2;
92  numYCells *= 2;
93 
94  gridMap.Honeycomb(gridMinX , gridMinY, cellR_, numXCells , numYCells);
95 
96  if(verbose_>0){
97  std::cout << std::endl;
98  std::cout << std::setprecision(2)
99  << "[buildCellMap] cell rmin: " << cellr_ << " cell rmax: " << cellR_
100  << " (gridMinX,gridMinY) = (" << gridMinX << "," << gridMinY << ")"
101  << " (numXCells,numYCells) = (" << numXCells << "," << numYCells << ")" << std::endl;
102  }
103 
104  // copy cells lying within module boundaries to a module grid
105  TListIter next(gridMap.GetBins()); // a TH2Poly is a TList of TH2PolyBin
106  TH2PolyBin *polyBin = 0;
107  TGraph * poly = 0; // a polygon returned by TH2Poly is a TGraph
108  int ecalMapID = 0; // ecalMap cell IDs go from 0 to N-1, not equal to original grid cell ID.
109  while( (polyBin = (TH2PolyBin*)next()) ){
110 
111  //these bins are coming from the honeycomb
112  // grid so we assume that they are regular
113  // hexagons i.e. has 6 vertices
114  poly = (TGraph*)polyBin->GetPolygon();
115 
116  // decide whether to copy polygon to new map.
117  // use all vertices in case of cut-off edge polygons.
118  int numVerticesInside = 0;
119  double vertex_x[6], vertex_y[6]; //vertices of the cell
120  bool isinside[6]; //which vertices are inside
121  if(verbose_>2) std::cout << "[buildCellMap] Cell vertices" << std::endl;
122  for(unsigned i = 0 ; i < 6; i++){
123  poly->GetPoint(i,vertex_x[i],vertex_y[i]);
124  if (verbose_>2) {
125  std::cout << " vtx # " << i << std::endl;
126  std::cout << " vtx x,y " << vertex_x[i] << " " << vertex_y[i] << std::endl;
127  }
128  isinside[i] = isInside(vertex_x[i]/moduleR_, vertex_y[i]/moduleR_);
129  if (isinside[i]) numVerticesInside++;
130  }
131 
132  if(numVerticesInside > 1){
133  // Include this cell if more than one of its vertices is inside the module hexagon
134  double actual_x[8], actual_y[8];
135  int num_vertices{0};
136  if ( numVerticesInside < 6 ) {
137  // This cell is stradling the edge of the module
138  // and is NOT cleanly cut by module edge
139 
140  if(verbose_>1) {
141  std::cout << "[buildCellMap] Polygon " << ecalMapID
142  << " has vertices poking out of module hexagon." << std::endl;
143  }
144 
145  // loop through vertices
146  for ( int i = 0; i < 6; i++ ) {
147  int up = i==5 ? 0 : i+1;
148  int dn = i==0 ? 5 : i-1;
149  if ( isinside[i] and (not isinside[up] or not isinside[dn]) ) {
150  //this vertex is inside the module hexagon and is adjacent to a vertex outside
151  // ==> project this vertex onto the nearest edge of the module hexagon
152 
153  // determine which side of hexagon we should project onto
154  double edge_origin_x, edge_origin_y;
155  double edge_dest_x, edge_dest_y;
156  if ( vertex_x[i] < -moduleR_/2. ) {
157  //sloped edge on negative-x side
158  edge_origin_x = -1.*moduleR_;
159  edge_origin_y = 0.;
160  edge_dest_x = -0.5*moduleR_;
161  edge_dest_y = moduler_;
162  } else if ( vertex_x[i] > moduleR_/2. ) {
163  //sloped edge on positive-x side
164  edge_origin_x = 0.5*moduleR_;
165  edge_origin_y = moduler_;
166  edge_dest_x = moduleR_;
167  edge_dest_y = 0.;
168  } else {
169  //flat edge at top
170  edge_origin_x = 0.5*moduleR_;
171  edge_origin_y = moduler_;
172  edge_dest_x = -0.5*moduleR_;
173  edge_dest_y = moduler_;
174  }
175 
176  // flip to bottom half if below x-axis
177  if ( vertex_y[i] < 0 ) {
178  edge_dest_y *= -1;
179  edge_origin_y *= -1;
180  }
181 
182  // get edge slope vector
183  double edge_slope_x = edge_dest_x - edge_origin_x;
184  double edge_slope_y = edge_dest_y - edge_origin_y;
185 
186  if(verbose_>2) {
187  std::cout << "Vertex " << i
188  << " is inside and adjacent to a vertex outside the module." << std::endl;
189  std::cout << "Working on edge with slope (" << edge_slope_x << "," << edge_slope_y << ")"
190  << " and origin (" << edge_origin_x << "," << edge_origin_y << ")" << std::endl;
191  }
192 
193  //project vertices adjacent to the vertex outside the module onto the module edge
194  double projection_factor =
195  ( (vertex_x[i]-edge_origin_x)*edge_slope_x
196  + (vertex_y[i]-edge_origin_y)*edge_slope_y )
197  / (edge_slope_x*edge_slope_x + edge_slope_y*edge_slope_y);
198 
199  double proj_x = edge_origin_x + projection_factor*edge_slope_x;
200  double proj_y = edge_origin_y + projection_factor*edge_slope_y;
201 
202  if ( not isinside[up] ) {
203  //the next point is outside
204  actual_x[num_vertices] = vertex_x[i];
205  actual_y[num_vertices] = vertex_y[i];
206  actual_x[num_vertices+1] = proj_x;
207  actual_y[num_vertices+1] = proj_y;
208  } else {
209  //the previous point was outside
210  actual_x[num_vertices] = proj_x;
211  actual_y[num_vertices] = proj_y;
212  actual_x[num_vertices+1] = vertex_x[i];
213  actual_y[num_vertices+1] = vertex_y[i];
214  }
215  num_vertices += 2;
216 
217  if(verbose_>2) {
218  std::cout << "New Vertex " << i
219  << " : (" << vertex_x[i] << "," << vertex_y[i] << ")" << std::endl;
220  }
221  } else {
222  actual_x[num_vertices] = vertex_x[i];
223  actual_y[num_vertices] = vertex_y[i];
224  num_vertices++;
225  } //should we project or not
226  } //loop through vertices
227  } else {
228  //all 6 inside, just copy the vertices over
229  num_vertices = 6;
230  for ( int i = 0; i < 6; i++ ) {
231  actual_x[i] = vertex_x[i];
232  actual_y[i] = vertex_y[i];
233  }
234  } //if numVerticesInside is less than 5
235 
236  //ecalMap_ needs to have its own copy of the polygon TGraph
237  // otherwise, we get a seg fault when EcalHexReadout is destructed
238  // because the polygon that was copied over from gridMap is deleted at the end of this function
239  ecalMap_.AddBin( num_vertices , actual_x , actual_y );
240 
241  double x = (polyBin->GetXMax() + polyBin->GetXMin()) / 2.;
242  double y = (polyBin->GetYMax() + polyBin->GetYMin()) / 2.;
243  if(verbose_>1) {
244  std::cout << "[buildCellMap] Copying poly with ID " << polyBin->GetBinNumber()
245  << " and (x,y) (" << std::setprecision(2) << x << "," << y << ")" << std::endl;
246  }
247  //save cell location as center of ENTIRE hexagon
248  cellPositionMap_[ecalMapID] = std::pair<double,double>(x,y);
249  ecalMapID++; //incrememnt cell ID
250  } // if num vertices inside is > 1
251  } //loop over larger grid spanning module hexagon
252 
253  if(verbose_>0) std::cout << std::endl;
254  return;
255  }

325  double EcalHexReadout::distanceToEdge(EcalID cellModuleID) const {
326  // https://math.stackexchange.com/questions/1210572/find-the-distance-to-the-edge-of-a-hexagon
327  int cellID = cellModuleID.cell();
328  std::pair<double,double> cellLocation = getCellCenterRelative(cellID);
329  double x = fabs(cellLocation.first); // bring to first quadrant
330  double y = fabs(cellLocation.second);
331  double r = sqrt(x*x+y*y);
332  double theta = (r > 1E-3) ? fabs(std::atan(y/x)) : 0;
333  if(x < moduleR_/2.) return (moduler_ - y); // closest line is straight vertical to top edge
334  double dist = sqrt(3.)*moduleR_/(std::sin(theta) + sqrt(3.)*std::cos(theta));
335  return dist;
336  }

338  bool EcalHexReadout::isInside(double normX, double normY) const {
339  if(verbose_>2) std::cout << TString::Format("[isInside] Checking if normXY=(%.2f,%.2f) is inside.",normX,normY) << std::endl;
340  normX = fabs(normX), normY = fabs(normY);
341  double xvec = -1, yvec = -1./sqrt(3);
342  double xref = 0.5, yref = sqrt(3)/2.;
343  if( (normX > 1.) || (normY > yref) ){
344  if(verbose_>2) std::cout << "[isInside] they are outside quadrant." << std::endl;
345  return false;
346  }
347  double dotProd = (xvec*(normX-xref) + yvec*(normY-yref));
348  if(verbose_>2) std::cout << TString::Format("[isInside] they are inside quadrant. Dot product (>0 is inside): %.2f ", dotProd) << std::endl;
349  return (dotProd > 0.);
350  }
