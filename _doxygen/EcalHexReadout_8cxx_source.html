13  EcalHexReadout::EcalHexReadout(double moduleMinR, double gap, unsigned nCellsWide,
14  const std::vector<double> &layerZPositions, double ecalFrontZ)
15  : layerZPositions_(layerZPositions), ecalFrontZ_(ecalFrontZ) {
16 
17  // ORIENTATION ASSUMPTIONS:
18  // modules are oriented flat side down. cells are oriented corner side down.
19  // module corners are one cell -- the 'single mousebite' cell layout.
20  // cells along center horizontal line form a neat row and share vertical edges.
21  // SOME GEOMETRY:
22  // hexagons have two radii:
23  // r (half of flat-to-flat width) and R (half of corner-to-corner width).
24  // r = (sqrt(3)/2)R and s = R, where s is the length of an edge.
25  // for seven ecal modules oriented flat-side-down, maximum x and y extents are:
26  // deltaY = 6r' + 2g = 3sqrt(3)R' + 2g
27  // deltaX = 4R' + s' + 2g/cos(30 deg) = 5R' + 4g/sqrt(3)
28  // where g is uniform gap width between modules, and primed variables correspond to modules.
29  // THIS GRID:
30  // column-to-column distance in a grid such as ours is 2r = sqrt(3)R.
31  // row-to-row distance is 1.5R (easy to observe that twice that distance = 3R)
32  // a cell will be centered at (0,0).
33 
34  assert(nCellsWide % 2 == 1); // calculations rely on centering a cell at (0,0)
35 
36  // calculate module radius reduction due to module edges cutting off outer cell corners.
37  // r = cell center-to-flat (i.e. minimum) radius
38  // R = cell center-to-corner (i.e. maximum) radius
39  // s = R = cell edge length
40  // nCellsWide = count of cells along horizontal center row
41  // moduleMaxR = module center-to-corner radius
42  // moduleMinR = module center-to-flat radius
43  // lengthWide = total length of CELLS along horizontal center row
44  // x = distance from module's corner to outside flat edge of cell it resides in
45  // THEN
46  // nCellsWide*(2*r) = lengthWide = 2*moduleMaxR + 2*x
47  // AND tan(30 deg) = x/(s/2), or x = s/(2*sqrt(3)),
48  // SO nCellsWide*r = moduleMaxR + R/(2*sqrt(3)) = moduleMaxR + r/3
49  // SO r = moduleMaxR/(nCellsWide - 1/3) = (1/2)*(2*moduleMaxR/(nCellsWide - 1/3))
50  // AND lengthWide = nCellsWide*2*r = nCellsWide*(2*moduleMaxR/(nCellsWide - 1/3))
51  //
52  // Test: moduleMaxR = 1., nCellsWide = 1
53  // r = 1/(1-1/3) = 1.5, lengthWide = 3
54  // Checked with a ruler - correct.
55 
56  gap_ = gap;
57  moduleR_ = moduleMinR*(2/sqrt(3));
58  moduler_ = moduleMinR;
59  nCellsWide_ = nCellsWide;
60  cellr_ = moduleR_/(nCellsWide - 1./3.);
61  cellR_ = (2./sqrt(3.))*cellr_;
62  lengthWide_ = nCellsWide*2.*cellr_;
63  if(verbose_>0){
64  std::cout << std::endl << "[EcalHexReadout] Verbosity set in header to " << verbose_ << std::endl;
65  std::cout << TString::Format("Building module map with gap %.2f, lengthWide %.2f, nCellsWide %d ",gap_,lengthWide_,nCellsWide_) << std::endl;
66  std::cout << TString::Format(" min/max radii of cell %.2f %.2f and module %.2f %.2f",cellr_,cellR_,moduler_,moduleR_) << std::endl;
67  }
68 
69  buildModuleMap();
70  buildCellMap();
71  buildCellModuleMap();
72  buildNeighborMaps();
73  if(verbose_>0){ std::cout << std::endl; }
74  }

78  // module IDs are 0 for ecal center, 1 at 12 o'clock, and clockwise till 6 at 11 o'clock.
79  double C_PI = 3.14159265358979323846; // or TMath::Pi(), #define, atan(), ...
80  modulePositionMap_[0] = std::pair<double,double>(0.,0.);
81  for(unsigned id = 1 ; id < 7 ; id++){
82  double x = (2.*moduler_+gap_)*sin( (id-1)*(C_PI/3.) );
83  double y = (2.*moduler_+gap_)*cos( (id-1)*(C_PI/3.) );
84  modulePositionMap_[id] = std::pair<double,double>(x,y);
85  if(verbose_>2) std::cout << TString::Format(" id %d is at (%.2f, %.2f)",id,x,y) << std::endl;

118  // make hexagonal grid [boundary is rectangle] larger than the module

120  if( (nCellsWide_-1) % 4 == 0) gridCellsWide += 2; // parity case
121  columnDistance_ = 2*cellr_;

123  double gridWidth = (gridCellsWide)*columnDistance_;

125  gridMap.Honeycomb( -gridWidth/2, -gridHeight/2, cellR_, gridCellsWide, gridCellsWide);
126 
127  if(verbose_>0){
128  std::cout << std::endl;

132  gridCellsWide, gridWidth, gridHeight) << std::endl;
133  }

135  // copy cells lying within module boundaries to a module grid
136  std::vector<int> cellIdCopied(gridMap.GetNumberOfBins());

138  TH2PolyBin *polyBin = 0;
139  TGraph * poly = 0; // a polygon returned by TH2Poly is a TGraph

169  ecalMap_.AddBin( poly->GetN() , poly->GetX() , poly->GetY() );
170  cellPositionMap_[ecalMapID] = std::pair<double,double>(x,y);

233  double EcalHexReadout::distanceToEdge(int cellModuleID) const {
234  // https://math.stackexchange.com/questions/1210572/find-the-distance-to-the-edge-of-a-hexagon
235  int cellID = separateID(cellModuleID).first;
236  XYCoords cellLocation = getCellCenterRelative(cellID);
237  double x = fabs(cellLocation.first); // bring to first quadrant
238  double y = fabs(cellLocation.second);
239  double r = sqrt(x*x+y*y);
240  double theta = (r > 1E-3) ? fabs(std::atan(y/x)) : 0;
241  if(x < moduleR_/2.) return (moduler_ - y); // closest line is straight vertical to top edge
242  double dist = sqrt(3.)*moduleR_/(std::sin(theta) + sqrt(3.)*std::cos(theta));
243  return dist;
244  }

246  bool EcalHexReadout::isInside(double normX, double normY) const {
247  if(verbose_>2) std::cout << TString::Format("[isInside] Checking if normXY=(%.2f,%.2f) is inside.",normX,normY) << std::endl;
248  normX = fabs(normX), normY = fabs(normY);
249  double xvec = -1, yvec = -1./sqrt(3);
250  double xref = 0.5, yref = sqrt(3)/2.;
251  if( (normX > 1.) || (normY > yref) ){
252  if(verbose_>2) std::cout << "[isInside] they are outside quadrant." << std::endl;
253  return false;
254  }
255  double dotProd = (xvec*(normX-xref) + yvec*(normY-yref));
256  if(verbose_>2) std::cout << TString::Format("[isInside] they are inside quadrant. Dot product (>0 is inside): %.2f ", dotProd) << std::endl;
257  return (dotProd > 0.);
258  }
