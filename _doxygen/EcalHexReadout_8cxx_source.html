47 void EcalHexReadout::buildModuleMap() {
48  if (verbose_ > 0) {
49  std::cout
50  << std::endl
51  << "[buildModuleMap] Building module position map for module min r of "
52  << moduler_ << " and gap of " << gap_ << std::endl;
53  }
54 
55  // module IDs are 0 for ecal center, 1 at 12 o'clock, and clockwise till 6 at
56  // 11 o'clock.
57  double C_PI = 3.14159265358979323846; // or TMath::Pi(), #define, atan(), ...
58  modulePositionMap_[0] = std::pair<double, double>(0., 0.);
59  for (unsigned id = 1; id < 7; id++) {
60  double x = (2. * moduler_ + gap_) * sin((id - 1) * (C_PI / 3.));
61  double y = (2. * moduler_ + gap_) * cos((id - 1) * (C_PI / 3.));
62  modulePositionMap_[id] = std::pair<double, double>(x, y);
63  if (verbose_ > 2)
64  std::cout << TString::Format(" id %d is at (%.2f, %.2f)", id, x, y)
65  << std::endl;
66  }
67  if (verbose_ > 0) std::cout << std::endl;
68 }

70 void EcalHexReadout::buildCellMap() {
75  TH2Poly gridMap;
76 
77  // make hexagonal grid [boundary is rectangle] larger than the module
78  double gridMinX = 0., gridMinY = 0.; // start at the origin
79  int numXCells = 0, numYCells = 0;
80 
81  // first x-cell is only a half
82  gridMinX -= cellr_;
83  numXCells++;
84  while (gridMinX > -1 * moduleR_) {
85  gridMinX -= 2 * cellr_; // decrement x by cell center-to-flat diameter
86  numXCells++;
87  }
88  while (gridMinY > -1 * moduler_) {
89  // decrement y by cell center-to-corner radius
90  // alternate between a full corner-to-corner diameter
91  // and a side of a cell (center-to-corner radius)
92  if (numYCells % 2 == 0)
93  gridMinY -= 1 * cellR_;
94  else
95  gridMinY -= 2 * cellR_;
96  numYCells++;
97  }
98  // only counted one half of the cells
99  numXCells *= 2;
100  numYCells *= 2;
101 
102  gridMap.Honeycomb(gridMinX, gridMinY, cellR_, numXCells, numYCells);
103 
104  if (verbose_ > 0) {
105  std::cout << std::endl;
106  std::cout << std::setprecision(2) << "[buildCellMap] cell rmin: " << cellr_
107  << " cell rmax: " << cellR_ << " (gridMinX,gridMinY) = ("
108  << gridMinX << "," << gridMinY << ")"
109  << " (numXCells,numYCells) = (" << numXCells << "," << numYCells
110  << ")" << std::endl;
111  }
112 
113  // copy cells lying within module boundaries to a module grid
114  TListIter next(gridMap.GetBins()); // a TH2Poly is a TList of TH2PolyBin
115  TH2PolyBin* polyBin = 0;
116  TGraph* poly = 0; // a polygon returned by TH2Poly is a TGraph
117  int ecalMapID = 0; // ecalMap cell IDs go from 0 to N-1, not equal to
118  // original grid cell ID.
119  while ((polyBin = (TH2PolyBin*)next())) {
120  // these bins are coming from the honeycomb
121  // grid so we assume that they are regular
122  // hexagons i.e. has 6 vertices
123  poly = (TGraph*)polyBin->GetPolygon();
124 
125  // decide whether to copy polygon to new map.
126  // use all vertices in case of cut-off edge polygons.
127  int numVerticesInside = 0;
128  double vertex_x[6], vertex_y[6]; // vertices of the cell
129  bool isinside[6]; // which vertices are inside
130  if (verbose_ > 2) std::cout << "[buildCellMap] Cell vertices" << std::endl;
131  for (unsigned i = 0; i < 6; i++) {
132  poly->GetPoint(i, vertex_x[i], vertex_y[i]);
133  if (verbose_ > 2) {
134  std::cout << " vtx # " << i << std::endl;
135  std::cout << " vtx x,y " << vertex_x[i] << " " << vertex_y[i]
136  << std::endl;
137  }
138  isinside[i] = isInside(vertex_x[i] / moduleR_, vertex_y[i] / moduleR_);
139  if (isinside[i]) numVerticesInside++;
140  }
141 
142  if (numVerticesInside > 1) {
143  // Include this cell if more than one of its vertices is inside the module
144  // hexagon
145  double actual_x[8], actual_y[8];
146  int num_vertices{0};
147  if (numVerticesInside < 6) {
148  // This cell is stradling the edge of the module
149  // and is NOT cleanly cut by module edge
150 
151  if (verbose_ > 1) {
152  std::cout << "[buildCellMap] Polygon " << ecalMapID
153  << " has vertices poking out of module hexagon."
154  << std::endl;
155  }
156 
157  // loop through vertices
158  for (int i = 0; i < 6; i++) {
159  int up = i == 5 ? 0 : i + 1;
160  int dn = i == 0 ? 5 : i - 1;
161  if (isinside[i] and (not isinside[up] or not isinside[dn])) {
162  // this vertex is inside the module hexagon and is adjacent to a
163  // vertex outside
164  // ==> project this vertex onto the nearest edge of the module
165  // hexagon
166 
167  // determine which side of hexagon we should project onto
168  double edge_origin_x, edge_origin_y;
169  double edge_dest_x, edge_dest_y;
170  if (vertex_x[i] < -moduleR_ / 2.) {
171  // sloped edge on negative-x side
172  edge_origin_x = -1. * moduleR_;
173  edge_origin_y = 0.;
174  edge_dest_x = -0.5 * moduleR_;
175  edge_dest_y = moduler_;
176  } else if (vertex_x[i] > moduleR_ / 2.) {
177  // sloped edge on positive-x side
178  edge_origin_x = 0.5 * moduleR_;
179  edge_origin_y = moduler_;
180  edge_dest_x = moduleR_;
181  edge_dest_y = 0.;
182  } else {
183  // flat edge at top
184  edge_origin_x = 0.5 * moduleR_;
185  edge_origin_y = moduler_;
186  edge_dest_x = -0.5 * moduleR_;
187  edge_dest_y = moduler_;
188  }
189 
190  // flip to bottom half if below x-axis
191  if (vertex_y[i] < 0) {
192  edge_dest_y *= -1;
193  edge_origin_y *= -1;
194  }
195 
196  // get edge slope vector
197  double edge_slope_x = edge_dest_x - edge_origin_x;
198  double edge_slope_y = edge_dest_y - edge_origin_y;
199 
200  if (verbose_ > 2) {
201  std::cout
202  << "Vertex " << i
203  << " is inside and adjacent to a vertex outside the module."
204  << std::endl;
205  std::cout << "Working on edge with slope (" << edge_slope_x << ","
206  << edge_slope_y << ")"
207  << " and origin (" << edge_origin_x << ","
208  << edge_origin_y << ")" << std::endl;
209  }
210 
211  // project vertices adjacent to the vertex outside the module onto
212  // the module edge
213  double projection_factor =
214  ((vertex_x[i] - edge_origin_x) * edge_slope_x +
215  (vertex_y[i] - edge_origin_y) * edge_slope_y) /
216  (edge_slope_x * edge_slope_x + edge_slope_y * edge_slope_y);
217 
218  double proj_x = edge_origin_x + projection_factor * edge_slope_x;
219  double proj_y = edge_origin_y + projection_factor * edge_slope_y;
220 
221  if (not isinside[up]) {
222  // the next point is outside
223  actual_x[num_vertices] = vertex_x[i];
224  actual_y[num_vertices] = vertex_y[i];
225  actual_x[num_vertices + 1] = proj_x;
226  actual_y[num_vertices + 1] = proj_y;
227  } else {
228  // the previous point was outside
229  actual_x[num_vertices] = proj_x;
230  actual_y[num_vertices] = proj_y;
231  actual_x[num_vertices + 1] = vertex_x[i];
232  actual_y[num_vertices + 1] = vertex_y[i];
233  }
234  num_vertices += 2;
235 
236  if (verbose_ > 2) {
237  std::cout << "New Vertex " << i << " : (" << vertex_x[i] << ","
238  << vertex_y[i] << ")" << std::endl;
239  }
240  } else {
241  actual_x[num_vertices] = vertex_x[i];
242  actual_y[num_vertices] = vertex_y[i];
243  num_vertices++;
244  } // should we project or not
245  } // loop through vertices
246  } else {
247  // all 6 inside, just copy the vertices over
248  num_vertices = 6;
249  for (int i = 0; i < 6; i++) {
250  actual_x[i] = vertex_x[i];
251  actual_y[i] = vertex_y[i];
252  }
253  } // if numVerticesInside is less than 5
254 
255  // ecalMap_ needs to have its own copy of the polygon TGraph
256  // otherwise, we get a seg fault when EcalHexReadout is destructed
257  // because the polygon that was copied over from gridMap is deleted at
258  // the end of this function
259  ecalMap_.AddBin(num_vertices, actual_x, actual_y);
260 
261  double x = (polyBin->GetXMax() + polyBin->GetXMin()) / 2.;
262  double y = (polyBin->GetYMax() + polyBin->GetYMin()) / 2.;
263  if (verbose_ > 1) {
264  std::cout << "[buildCellMap] Copying poly with ID "
265  << polyBin->GetBinNumber() << " and (x,y) ("
266  << std::setprecision(2) << x << "," << y << ")" << std::endl;
267  }
268  // save cell location as center of ENTIRE hexagon
269  cellPositionMap_[ecalMapID] = std::pair<double, double>(x, y);
270  ecalMapID++; // incrememnt cell ID
271  } // if num vertices inside is > 1
272  } // loop over larger grid spanning module hexagon
273 
274  if (verbose_ > 0) std::cout << std::endl;
275  return;
276 }

302 void EcalHexReadout::buildNeighborMaps() {
316  NNMap_.clear();
317  NNNMap_.clear();
318  for (auto const& centerChannel : cellModulePositionMap_) {
319  EcalID centerID = centerChannel.first;
320  double centerX = centerChannel.second.first;
321  double centerY = centerChannel.second.second;
322  for (auto const& probeChannel : cellModulePositionMap_) {
323  EcalID probeID = probeChannel.first;
324  double probeX = probeChannel.second.first;
325  double probeY = probeChannel.second.second;
326  double dist = sqrt((probeX - centerX) * (probeX - centerX) +
327  (probeY - centerY) * (probeY - centerY));
328  if (dist > 1 * cellr_ && dist <= 3. * cellr_) {
329  NNMap_[centerID].push_back(probeID);
330  } else if (dist > 3. * cellr_ && dist <= 4.5 * cellr_) {
331  NNNMap_[centerID].push_back(probeID);
332  }
333  }
334  if (verbose_ > 1)
335  std::cout << TString::Format("Found %d NN and %d NNN for cellModuleID ",
336  int(NNMap_[centerID].size()),
337  int(NNNMap_[centerID].size()))
338  << centerID
339  << TString::Format(" with x,y (%.2f,%.2f)", centerX, centerY)
340  << std::endl;
341  }
342  if (verbose_ > 2) {
343  double specialX =
344  0.5 * moduleR_ - 0.5 * cellr_; // center of cell which is upper-right
345  // corner of center module
346  double specialY = moduler_ - 0.5 * cellR_;

357  }
358  for (auto centerNNN : NNNMap_.at(specialCellModuleID)) {
359  std::cout << " NNN " << centerNNN

361  getCellCenterAbsolute(centerNNN).first,
362  getCellCenterAbsolute(centerNNN).second)
363  << std::endl;
364  }
365  std::cout << TString::Format(

376 double EcalHexReadout::distanceToEdge(EcalID cellModuleID) const {
377  // https://math.stackexchange.com/questions/1210572/find-the-distance-to-the-edge-of-a-hexagon
378  int cellID = cellModuleID.cell();
379  std::pair<double, double> cellLocation = getCellCenterRelative(cellID);
380  double x = fabs(cellLocation.first); // bring to first quadrant
381  double y = fabs(cellLocation.second);
382  double r = sqrt(x * x + y * y);
383  double theta = (r > 1E-3) ? fabs(std::atan(y / x)) : 0;
384  if (x < moduleR_ / 2.)
385  return (moduler_ - y); // closest line is straight vertical to top edge
386  double dist =
387  sqrt(3.) * moduleR_ / (std::sin(theta) + sqrt(3.) * std::cos(theta));
388  return dist;
389 }

391 bool EcalHexReadout::isInside(double normX, double normY) const {
392  if (verbose_ > 2)
393  std::cout << TString::Format(
394  "[isInside] Checking if normXY=(%.2f,%.2f) is inside.",
395  normX, normY)
396  << std::endl;
397  normX = fabs(normX), normY = fabs(normY);
398  double xvec = -1, yvec = -1. / sqrt(3);
399  double xref = 0.5, yref = sqrt(3) / 2.;
400  if ((normX > 1.) || (normY > yref)) {
401  if (verbose_ > 2)
402  std::cout << "[isInside] they are outside quadrant." << std::endl;
403  return false;
404  }
405  double dotProd = (xvec * (normX - xref) + yvec * (normY - yref));
406  if (verbose_ > 2)
407  std::cout << TString::Format(
408  "[isInside] they are inside quadrant. Dot product (>0 is "
409  "inside): %.2f ",
410  dotProd)
411  << std::endl;
412  return (dotProd > 0.);
413 }
