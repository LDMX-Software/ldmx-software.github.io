15LHEEvent* LHEReader::readNextEvent() {
16 std::string line;
17 bool foundEventElement = false;
18 while (getline(ifs_, line)) {
19 if (line == "<event>") {
20 foundEventElement = true;
21 break;
22 }
23 }
24
25 if (!foundEventElement) {
26 std::cerr << "WARNING: No next <event> element was found by the LHE reader."
27 << std::endl;
28 return NULL;
29 }
30
31 getline(ifs_, line);
32
33 LHEEvent* nextEvent = new LHEEvent(line);
34
35 while (getline(ifs_, line)) {
36 if (line == "</event>") {
37 break;
38 }
39
40 if (line.find("#") == std::string::npos) { // not a comment line
41 LHEParticle* particle = new LHEParticle(line);
42 nextEvent->addParticle(particle);
43 } else {
44 if (line.find("#vertex") != std::string::npos) {
45 nextEvent->setVertex(line);
46 }
47 }
48 }
49
50 const std::vector<LHEParticle*>& particles = nextEvent->getParticles();
51 int particleIndex = 0;
52 for (std::vector<LHEParticle*>::const_iterator it = particles.begin();
53 it != particles.end(); it++) {
54 LHEParticle* particle = (*it);
55 if (particle->getMOTHUP(0) != 0) {
56 int mother1 = particle->getMOTHUP(0);
57 int mother2 = particle->getMOTHUP(1);
58 if (mother1 > 0) {
59 particle->setMother(0, particles[mother1 - 1]);
60 }
61 if (mother2 > 0) {
62 particle->setMother(1, particles[mother2 - 1]);
63 }
64 }
65 ++particleIndex;
66 }
67
68 return nextEvent;
69}
