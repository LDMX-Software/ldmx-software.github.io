71void NtuplizeHgcrocDigiCollection::analyze(const framework::Event& event) {
72 // get the reconstruction parameters
73 auto pedestal_table{getCondition<conditions::IntegerTableCondition>(pedestal_table_)};
74
75 ldmxsw_event_ = event.getEventNumber();
76 if (already_aligned_) {
77 aligned_ = event.getObject<bool>(input_name_+"Aligned", input_pass_);
78 } else {
79 aligned_ = false;
80 version_ = event.getObject<int>(input_name_+"Version", input_pass_);
81 pf_event_ = event.getObject<int>(input_name_+"Number", input_pass_);
82 pf_ticks_ = event.getObject<int>(input_name_+"Ticks", input_pass_);
83 pf_spill_ = event.getObject<int>(input_name_+"Spill", input_pass_);
84 }
85
86 auto const& digis{
87 event.getObject<ldmx::HgcrocDigiCollection>(input_name_, input_pass_)};
88 for (std::size_t i_digi{0}; i_digi < digis.size(); i_digi++) {
89 auto d{digis.getDigi(i_digi)};
90 raw_id_ = static_cast<int>(d.id());
91 if (using_eid_) {
92 ldmx::HcalElectronicsID eid(d.id());
93 fpga_ = eid.fiber();
94 link_ = eid.elink();
95 channel_ = eid.channel();
96 index_ = eid.index();
97 } else {
98 ldmx::HcalDigiID detid(d.id());
99 section_ = detid.section();
100 layer_ = detid.layer();
101 strip_ = detid.strip();
102 end_ = detid.end();
103 }
104
105 for (i_sample_ = 0; i_sample_ < digis.getNumSamplesPerDigi(); i_sample_++) {
106 tot_prog_ = d.at(i_sample_).isTOTinProgress();
107 tot_comp_ = d.at(i_sample_).isTOTComplete();
108 tot_ = d.at(i_sample_).tot();
109 toa_ = d.at(i_sample_).toa();
110 int adc_t = d.at(i_sample_).adc_t();
111 raw_adc_ = adc_t;
112 adc_ = adc_t - pedestal_table.get(d.id(), 0);
113 flat_tree_->Fill();
114 }
115 }
116}
