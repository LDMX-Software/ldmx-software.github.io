310  BoundingBox DetectorGeometry::getBoundingBox( const std::vector<HcalHit> &hitVec ) const {
311 
312  std::vector<double> pointSum ( 3 , 0.0 ); //sums of weighted coordinates
313  std::vector<double> weightSum( 3 , 0.0 ); //sums of weights for each coordinate
314 
315  //calculate real space point for each hit
316  for ( const HcalHit &hit : hitVec ) {
317 
318  BoundingBox box = getBoundingBox( hit );
319 
320  //Add weighted values to sums
321  double weight;
322  for ( unsigned int iC = 0; iC < 3; iC++ ) {
323 
324  double cer = abs(box[iC].second - box[iC].first)/2.0;
325 
326  weight = 1.0 / ( cer*cer );
327  weightSum[ iC ] += weight;
328  pointSum[ iC ] += weight*( ( box[iC].second + box[iC].first )/2.0 );
329  }
330  } //go through hitVec
331 
332  //Construct final BoundingBox
333  BoundingBox hbox;
334  for ( int iC = 0; iC < 3; iC++ ) {
335  double c = pointSum[ iC ] / weightSum[ iC ];
336  double ec = 1.0 / sqrt( weightSum[ iC ] );
337  hbox.emplace_back( c - ec , c + ec );
338  }
339 
340  return hbox;
341  }
