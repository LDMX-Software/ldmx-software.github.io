249BoundingBox DetectorGeometry::getBoundingBox(const ldmx::HcalHit &hit) const {
250 // pairs that will go into BoundingBox
251 std::pair<double, double> X(0, 0), Y(0, 0), Z(0, 0);
252
253 ldmx::HcalID id(hit.getID());
254 ldmx::HcalID::HcalSection section = (ldmx::HcalID::HcalSection)id.section();
255 int layer = id.layer();
256 int strip = id.strip();
257
258 // calculate center of layer,strip with respect to detector section
259 double layercenter =
260 layer * hcalLayerThickness_.at(section) + 0.5 * hcalThicknessScint_;
261 double stripcenter = (strip + 0.5) * hcalWidthScint_;
262
263 // calculate error in layer,strip position
264 double elayer = 0.5 * hcalThicknessScint_;
265 double estrip = 0.5 * hcalWidthScint_;
266
267 double x, y, z;
268 if (section == ldmx::HcalID::HcalSection::BACK) {
269 z = hcalZeroLayer_.at(section) + layercenter;
270 Z.first = z - elayer;
271 Z.second = z + elayer;
272
273 // only horizontal layers implemented currently
274 if (false) { //( (layer ^ hcalParityVertical_) & 1) == 0 ) { //checks for
275 // same parity
276 // Vertical Layers
277
278 x = -hcalZeroStrip_.at(section) + stripcenter;
279 X.first = x - estrip;
280 X.second = x + estrip;
281
282 y = hit.getYPos();
283 Y.first = y - hcalUncertaintyTimingPos_;
284 Y.second = y + hcalUncertaintyTimingPos_;
285
286 } else {
287 // Horizontal Layers
288
289 x = hit.getXPos();
290 X.first = x - hcalUncertaintyTimingPos_;
291 X.second = x + hcalUncertaintyTimingPos_;
292
293 y = -1 * hcalZeroStrip_.at(section) + stripcenter;
294 Y.first = y - estrip;
295 Y.second = y + estrip;
296
297 } // calculate depending on layer
298
299 } else {
300 z = hcalZeroStrip_.at(section) + stripcenter;
301 Z.first = z - estrip;
302 Z.second = z + estrip;
303
304 if (section == ldmx::HcalID::HcalSection::TOP or
305 section == ldmx::HcalID::HcalSection::BOTTOM) {
306 x = hit.getXPos();
307 X.first = x - hcalUncertaintyTimingPos_;
308 X.second = x + hcalUncertaintyTimingPos_;
309
310 y = hcalZeroLayer_.at(section) + layercenter;
311 if (section == ldmx::HcalID::HcalSection::BOTTOM) {
312 y *= -1;
313 }
314
315 Y.first = y - elayer;
316 Y.second = y + elayer;
317
318 } else if (section == ldmx::HcalID::HcalSection::LEFT or
319 section == ldmx::HcalID::HcalSection::RIGHT) {
320 y = hit.getYPos();
321 Y.first = y - hcalUncertaintyTimingPos_;
322 Y.second = y + hcalUncertaintyTimingPos_;
323
324 x = hcalZeroLayer_.at(section) + layercenter;
325 if (section == ldmx::HcalID::HcalSection::RIGHT) {
326 x *= -1;
327 }
328
329 X.first = x - elayer;
330 X.second = x + elayer;
331
332 } else {
333 std::cerr
334 << "[ DetectorGeometry::getBoundingBox ] : Unknown Hcal Section!"
335 << std::endl;
336 std::cerr << " Returning a valid BoundingBox but with values that are "
337 "all zero."
338 << std::endl;
339 } // side hcal
340
341 } // calculate depending on section
342
343 BoundingBox hbox;
344 hbox.push_back(X);
345 hbox.push_back(Y);
346 hbox.push_back(Z);
347 return hbox;
348}

350BoundingBox DetectorGeometry::getBoundingBox(
351 const std::vector<ldmx::HcalHit> &hitVec) const {
352 std::vector<double> pointSum(3, 0.0); // sums of weighted coordinates
353 std::vector<double> weightSum(3, 0.0); // sums of weights for each coordinate
354
355 // calculate real space point for each hit
356 for (const ldmx::HcalHit &hit : hitVec) {
357 BoundingBox box = getBoundingBox(hit);
358
359 // Add weighted values to sums
360 double weight;
361 for (unsigned int iC = 0; iC < 3; iC++) {
362 double cer = abs(box[iC].second - box[iC].first) / 2.0;
363
364 weight = 1.0 / (cer * cer);
365 weightSum[iC] += weight;
366 pointSum[iC] += weight * ((box[iC].second + box[iC].first) / 2.0);
367 }
368 } // go through hitVec
369
370 // Construct final BoundingBox
371 BoundingBox hbox;
372 for (int iC = 0; iC < 3; iC++) {
373 double c = pointSum[iC] / weightSum[iC];
374 double ec = 1.0 / sqrt(weightSum[iC]);
375 hbox.emplace_back(c - ec, c + ec);
376 }
377
378 return hbox;
379}

381BoundingBox DetectorGeometry::getBoundingBox(
382 ldmx::HcalID::HcalSection section) const {
383 std::pair<double, double> X(0, 0), Y(0, 0), Z(0, 0);
384
385 double total_strip_width = hcalNStrips_.at(section) * hcalWidthScint_;
386 double total_thickness =
387 hcalNLayers_.at(section) * hcalLayerThickness_.at(section);
388 if (section == ldmx::HcalID::HcalSection::BACK) {
389 X.first = -hcalZeroStrip_.at(ldmx::HcalID::HcalSection::BACK);
390 X.second = X.first + total_strip_width;
391
392 Y.first = -hcalLengthScint_.at(ldmx::HcalID::HcalSection::BACK) / 2.0;
393 Y.second = hcalLengthScint_.at(ldmx::HcalID::HcalSection::BACK) / 2.0;
394
395 Z.first = hcalZeroLayer_.at(ldmx::HcalID::HcalSection::BACK);
396 Z.second = Z.first + total_thickness;
397
398 } else {
399 Z.first = hcalZeroStrip_.at(section);
400 Z.second = Z.first + total_strip_width;
401
402 if (section == ldmx::HcalID::HcalSection::LEFT) {
403 X.first = hcalZeroLayer_.at(ldmx::HcalID::HcalSection::LEFT);
404 X.second = X.first + total_thickness;
405
406 Y.second = hcalZeroLayer_.at(ldmx::HcalID::HcalSection::TOP);
407 Y.first = Y.second - hcalLengthScint_.at(ldmx::HcalID::HcalSection::LEFT);
408
409 } else if (section == ldmx::HcalID::HcalSection::RIGHT) {
410 X.second = -hcalZeroLayer_.at(ldmx::HcalID::HcalSection::RIGHT);
411 X.first = X.second - total_thickness;
412
413 Y.first = -hcalZeroLayer_.at(ldmx::HcalID::HcalSection::BOTTOM);
414 Y.second = Y.first + hcalLengthScint_.at(ldmx::HcalID::HcalSection::RIGHT);
415
416 } else if (section == ldmx::HcalID::HcalSection::TOP) {
417 Y.first = hcalZeroLayer_.at(ldmx::HcalID::HcalSection::TOP);
418 Y.second = Y.first + total_thickness;
419
420 X.first = -hcalZeroLayer_.at(ldmx::HcalID::HcalSection::RIGHT);
421 X.second = X.first + hcalLengthScint_.at(ldmx::HcalID::HcalSection::TOP);
422
423 } else if (section == ldmx::HcalID::HcalSection::BOTTOM) {
424 Y.second = -hcalZeroLayer_.at(ldmx::HcalID::HcalSection::BOTTOM);
425 Y.first = Y.second - total_thickness;
426
427 X.second = hcalZeroLayer_.at(ldmx::HcalID::HcalSection::LEFT);
428 X.first = X.second - hcalLengthScint_.at(ldmx::HcalID::HcalSection::BOTTOM);
429
430 } else {
431 std::cerr << "[ Warning ] : Unrecognized ldmx::HcalID::HcalSection in "
432 "DetectorGeometry::getBoundingBox."
433 << std::endl;
434 std::cerr << " Will return an incorrect geometry description!"
435 << std::endl;
436 }
437 }
438
439 BoundingBox boundingbox;
440 boundingbox.push_back(X);
441 boundingbox.push_back(Y);
442 boundingbox.push_back(Z);
443
444 return boundingbox;
445}

518BoundingBox DetectorGeometry::getBoundingBox(
519 const ldmx::SimTrackerHit &recoilHit) const {
520 int layerID = recoilHit.getLayerID();
521 int moduleID = recoilHit.getModuleID();
522 int combined = layerID * 10 + moduleID;
523
524 BoundingBox bbox;
525 if (recoilModulePos_.find(combined) == recoilModulePos_.end()) {
526 std::cerr
527 << "[ Warning ] : DetectorGeometry::getBoundingBox : Input layerID ("
528 << layerID << ") and input moduleID (" << moduleID
529 << ") are not included in the geometry!" << std::endl;
530 return bbox;
531 }
532
533 std::vector<float> hitPos = recoilHit.getPosition();
534
535 double xWidth = 1.0;
536 double yWidth = recoilStereoStripLength_;
537 if (layerID > 8) {
538 yWidth = recoilMonoStripLength_;
539 }
540
541 // we have to un-rotate the x-position of the hit, so we can rotate it later
542 // with the drawer
543 double rotAngle = this->getRotAngle(layerID, moduleID);
544 double xPos = hitPos.at(0) * cos(-rotAngle) - hitPos.at(1) * sin(-rotAngle);
545
546 bbox.emplace_back(xPos - xWidth / 2., xPos + xWidth / 2.);
547 bbox.emplace_back(recoilModulePos_.at(combined).at(1) - yWidth / 2.,
548 recoilModulePos_.at(combined).at(1) + yWidth / 2.);
549 bbox.emplace_back(
550 recoilModulePos_.at(combined).at(2) - recoilSensorThickness_ / 2.,
551 recoilModulePos_.at(combined).at(2) + recoilSensorThickness_ / 2.);
552
