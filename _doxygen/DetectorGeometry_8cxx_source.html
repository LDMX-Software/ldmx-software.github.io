316  BoundingBox DetectorGeometry::getBoundingBox( const std::vector<HcalHit> &hitVec ) const {
317 
318  std::vector<double> pointSum ( 3 , 0.0 ); //sums of weighted coordinates
319  std::vector<double> weightSum( 3 , 0.0 ); //sums of weights for each coordinate
320 
321  //calculate real space point for each hit
322  for ( const HcalHit &hit : hitVec ) {
323 
324  BoundingBox box = getBoundingBox( hit );
325 
326  //Add weighted values to sums
327  double weight;
328  for ( unsigned int iC = 0; iC < 3; iC++ ) {
329 
330  double cer = abs(box[iC].second - box[iC].first)/2.0;
331 
332  weight = 1.0 / ( cer*cer );
333  weightSum[ iC ] += weight;
334  pointSum[ iC ] += weight*( ( box[iC].second + box[iC].first )/2.0 );
335  }
336  } //go through hitVec
337 
338  //Construct final BoundingBox
339  BoundingBox hbox;
340  for ( int iC = 0; iC < 3; iC++ ) {
341  double c = pointSum[ iC ] / weightSum[ iC ];
342  double ec = 1.0 / sqrt( weightSum[ iC ] );
343  hbox.emplace_back( c - ec , c + ec );
344  }
345 
346  return hbox;
347  }
