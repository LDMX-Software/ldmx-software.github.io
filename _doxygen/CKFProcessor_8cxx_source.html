52 // Setup a constant magnetic field
53 const auto constBField = std::make_shared<Acts::ConstantBField>(b_field);

55 // TODO:: Move this to an external file
56 auto localToGlobalBin_xyz = [](std::array<size_t, 3> bins,
57 std::array<size_t, 3> sizes) {
58 return (bins[0] * (sizes[1] * sizes[2]) + bins[1] * sizes[2] +
59 bins[2]); // xyz - field space
60 // return (bins[1] * (sizes[2] * sizes[0]) + bins[2] * sizes[0] + bins[0]);
61 // //zxy
62 };
63
64 // Setup a interpolated bfield map
65 const auto map = std::make_shared<InterpolatedMagneticField3>(
66 makeMagneticFieldMapXyzFromText(
67 std::move(localToGlobalBin_xyz), field_map_,
68 1. * Acts::UnitConstants::mm, // default scale for axes length
69 1000. * Acts::UnitConstants::T, // The map is in kT, so scale it to T
70 false, // not symmetrical
71 true // rotate the axes to tracking frame
72 ));
73
74 // Setup the steppers
75 const auto stepper = Acts::EigenStepper<>{map};
76 const auto const_stepper = Acts::EigenStepper<>{constBField};
77 const auto multi_stepper = Acts::MultiEigenStepperLoop{map};
78
79 // Setup the navigator
80 Acts::Navigator::Config navCfg{tGeometry};
81 navCfg.resolveMaterial = true;
82 navCfg.resolvePassive = true;
83 navCfg.resolveSensitive = true;
84 navCfg.boundaryCheckLayerResolving = false;
85 const Acts::Navigator navigator(navCfg);
86
87 // Setup the propagators
88 propagator_ = const_b_field_
89 ? std::make_unique<CkfPropagator>(const_stepper, navigator)
90 : std::make_unique<CkfPropagator>(stepper, navigator);
91 //auto gsf_propagator = GsfPropagator(multi_stepper, navigator);
92
93 // Setup the fitters // you can add a second argument with a unique pointer to the logger you want
94 //ckf_ = std::make_unique<std::decay_t<decltype(*ckf_)>>(*propagator_,Acts::getDefaultLogger("CKF", Acts::Logging::VERBOSE));
95 ckf_ = std::make_unique<std::decay_t<decltype(*ckf_)>>(*propagator_);
96 kf_ = std::make_unique<std::decay_t<decltype(*kf_)>>(*propagator_);
97
98 //gsf_ = std::make_unique<std::decay_t<decltype(*gsf_)>>(
99 // std::move(gsf_propagator));
100
101 // Setup the propagator steps writer
102 tracking::sim::PropagatorStepWriter::Config cfg;
103 cfg.filePath = steps_outfile_path_;

105 writer_ = std::make_unique<tracking::sim::PropagatorStepWriter>(cfg);
