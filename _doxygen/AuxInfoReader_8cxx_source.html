142  void AuxInfoReader::assignAuxInfoToVolumes() {
143  const G4LogicalVolumeStore* lvs = G4LogicalVolumeStore::GetInstance();
144  std::vector<G4LogicalVolume*>::const_iterator lvciter;
145  for (lvciter = lvs->begin(); lvciter != lvs->end(); lvciter++) {
146  G4GDMLAuxListType auxInfo = parser_->GetVolumeAuxiliaryInformation(*lvciter);
147  if (auxInfo.size() > 0) {
148 
149  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux = auxInfo.begin(); iaux != auxInfo.end(); iaux++) {
150 
151  G4String auxType = iaux->type;
152  G4String auxVal = iaux->value;
153  G4String auxUnit = iaux->unit;
154 
155  G4LogicalVolume* lv = (*lvciter);
156 
157  if (auxType == "SensDet") {
158  G4String sdName = auxVal;
159  G4VSensitiveDetector* sd = G4SDManager::GetSDMpointer()->FindSensitiveDetector(sdName);
160  if (sd != NULL) {
161  lv->SetSensitiveDetector(sd);
162  //G4cout << "Assigned SD " << sd->GetName() << " to " << lv->GetName() << G4endl;
163  } else {
164  EXCEPTION_RAISE( "MissingInfo" , "Unknown SensDet in volume's auxiliary info: " + std::string(sdName.data()) );
165  }
166  } else if (auxType == "MagneticField") {
167  G4String magFieldName = auxVal;
168  G4MagneticField* magField = MagneticFieldStore::getInstance()->getMagneticField(magFieldName);
169  if (magField != NULL) {
170  G4FieldManager* mgr = new G4FieldManager(magField);
171  lv->SetFieldManager(mgr, true /* FIXME: hard-coded to force field manager to daughters */);
172  //G4cout << "Assigned magnetic field " << magFieldName << " to volume " << lv->GetName() << G4endl;
173  } else {
174  EXCEPTION_RAISE( "MissingInfo" , "Unknown MagneticField ref in volume's auxiliary info: " + std::string(magFieldName.data()) );
175  }
176  } else if (auxType == "Region") {
177  G4String regionName = auxVal;
178  G4Region* region = G4RegionStore::GetInstance()->GetRegion(regionName);
179  if (region != NULL) {
180  region->AddRootLogicalVolume(lv);
181  //G4cout << "Added volume " << lv->GetName() << " to region " << regionName << G4endl;
182  } else {
183  EXCEPTION_RAISE( "MissingInfo" , "Reference region '" + std::string(regionName.data()) + "' was not found!" );
184  }
185  } else if (auxType == "VisAttributes") {
186  G4String visName = auxVal;
187  G4VisAttributes* visAttributes = VisAttributesStore::getInstance()->getVisAttributes(visName);
188  if (visAttributes != NULL) {
189  lv->SetVisAttributes(visAttributes);
190  //G4cout << "Assigned VisAttributes " << visName << " to volume " << lv->GetName() << G4endl;
191  } else {
192  EXCEPTION_RAISE( "MissingInfo" , "Referenced VisAttributes '" + std::string(visName.data()) + "' was not found!" );
193  }
194  }

200  void AuxInfoReader::createMagneticField(G4String magFieldName, const G4GDMLAuxListType* auxInfoList) {
201 
202  // Find type of the mag field.
203  G4String magFieldType("");
204  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux = auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
205 
206  G4String auxType = iaux->type;
207  G4String auxVal = iaux->value;
208 
209  if (auxType == "MagneticFieldType") {
210  magFieldType = auxVal;
211  break;
212  }
213  }
214 
215  if (magFieldType == "") {
216  EXCEPTION_RAISE( "MissingInfo" , "Missing MagFieldType for magnetic field definition." );
217  }
218 
219  G4MagneticField* magField = NULL;
220 
221  // Create a uniform mag field using the built-in Geant4 type.
222  if (magFieldType == "G4UniformMagField") {
223  string::size_type sz;
224  double bx, by, bz;
225  bx = by = bz = 0.;
226  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux = auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
227 
228  G4String auxType = iaux->type;
229  G4String auxVal = iaux->value;
230  G4String auxUnit = iaux->unit;
231 
232  G4String expr = auxVal + "*" + auxUnit;
233  if (auxType == "bx") {
234  bx = eval_->Evaluate(expr);
235  } else if (auxType == "by") {
236  by = eval_->Evaluate(expr);
237  } else if (auxType == "bz") {
238  bz = eval_->Evaluate(expr);
239  }
240  }
241  G4ThreeVector fieldComponents(bx, by, bz);
242  magField = new G4UniformMagField(fieldComponents);
243 
244  //G4cout << "Created G4UniformMagField " << magFieldName << " with field components " << fieldComponents << G4endl << G4endl;
245 
246  // Create a global 3D field map by reading from a data file.
247  } else if (magFieldType == "MagneticFieldMap3D") {
248 
249  string fileName;
250  double offsetX, offsetY, offsetZ;
251 
252  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux = auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
253 
254  G4String auxType = iaux->type;
255  G4String auxVal = iaux->value;
256  G4String auxUnit = iaux->unit;
257 
258  G4String expr = auxVal + "*" + auxUnit;
259 
260  if (auxType == "File") {
261  fileName = auxVal;
262  } else if (auxType == "OffsetX") {
263  offsetX = eval_->Evaluate(expr);
264  } else if (auxType == "OffsetY") {
265  offsetY = eval_->Evaluate(expr);
266  } else if (auxType == "OffsetZ") {
267  offsetZ = eval_->Evaluate(expr);
268  }
269  }
270 
271  if (fileName.size() == 0) {
272  EXCEPTION_RAISE( "MissingInfo" , "File info with field data was not provided." );
273  }
274 
275  // Create new 3D field map.
276  magField = new MagneticFieldMap3D(fileName.c_str(), offsetX, offsetY, offsetZ);
277 
278  // Assign field map as global field.
279  G4FieldManager* fieldMgr = G4TransportationManager::GetTransportationManager()->GetFieldManager();
280  if (fieldMgr->GetDetectorField() != nullptr) {
281  EXCEPTION_RAISE( "MisAssign" , "Global mag field was already assigned." );
282  }
283  fieldMgr->SetDetectorField(magField);
284  fieldMgr->CreateChordFinder(magField);
285 
286  } else {
287  EXCEPTION_RAISE( "UnknownType" , "Unknown MagFieldType '" + std::string(magFieldType.data()) + "' in auxiliary info." );
288  }
289 
290  MagneticFieldStore::getInstance()->addMagneticField(magFieldName, magField);
291  }
