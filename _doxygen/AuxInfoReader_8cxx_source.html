47void AuxInfoReader::readGlobalAuxInfo() {
48 const G4GDMLAuxListType* auxInfoList = parser_->GetAuxList();
49 for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
50 auxInfoList->begin();
51 iaux != auxInfoList->end(); iaux++) {
52 G4String auxType = iaux->type;
53 G4String auxVal = iaux->value;
54 G4String auxUnit = iaux->unit;
55
56 if (auxType == "SensDet") {
57 createSensitiveDetector(auxVal, iaux->auxList);
58 } else if (auxType == "MagneticField") {
59 createMagneticField(auxVal, iaux->auxList);
60 } else if (auxType == "Region") {
61 createRegion(auxVal, iaux->auxList);
62 } else if (auxType == "VisAttributes") {
63 createVisAttributes(auxVal, iaux->auxList);
64 } else if (auxType == "DetectorVersion") {
65 createDetectorHeader(auxVal, iaux->auxList);
66 }
67 }
68 return;
69}

71void AuxInfoReader::createSensitiveDetector(
72 G4String theSensDetName, const G4GDMLAuxListType* auxInfoList) {
73 // G4cout << "Creating SensitiveDetector " << theSensDetName << G4endl;
74
75 G4String sdType("");
76 G4String hcName("");
77 G4String idName("");
78 int subdetID = -1;
79 int layerDepth = -1;
80 int verbose = 0;
81 for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
82 auxInfoList->begin();
83 iaux != auxInfoList->end(); iaux++) {
84 G4String auxType = iaux->type;
85 G4String auxVal = iaux->value;
86 G4String auxUnit = iaux->unit;
87
88 // G4cout << "auxType: " << auxType << ", auxVal: " << auxVal << ", auxUnit:
89 // " << auxUnit << G4endl;
90
91 if (auxType == "SensDetType") {
92 sdType = auxVal;
93 } else if (auxType == "HitsCollection") {
94 hcName = auxVal;
95 } else if (auxType == "Verbose") {
96 verbose = atoi(auxVal.c_str());
97 } else if (auxType == "SubdetID") {
98 subdetID = atoi(auxVal.c_str());
99 } else if (auxType == "DetectorID") {
100 idName = auxVal;
101 } else if (auxType == "LayerDepth") {
102 layerDepth = atoi(auxVal.c_str());
103 }
104 }
105
106 if (sdType == "") {
107 EXCEPTION_RAISE("MissingInfo", "The SensDet is missing the SensDetType.");
108 }
109
110 if (hcName == "") {
111 EXCEPTION_RAISE("MissingInfo",
112 "The SensDet is missing the HitsCollection.");
113 }
114
115 if (subdetID <= 0) {
116 EXCEPTION_RAISE("BadID", "The SubdetID '" + std::to_string(subdetID) +
117 "' is missing or invalid.");
118 }
119
120 /*
121 * Build the Sensitive Detector, and re-assign the detID if applicable
122 */
123 G4VSensitiveDetector* sd = 0;
124
125 if (sdType == "TrackerSD") {
126 sd = new TrackerSD(theSensDetName, hcName, subdetID);
127 } else if (sdType == "EcalSD") {
128 sd = new EcalSD(theSensDetName, hcName, subdetID, conditionsIntf_);
129 } else if (sdType == "HcalSD") {
130 sd = new HcalSD(theSensDetName, hcName, subdetID);
131 } else if (sdType == "ScoringPlaneSD") {
132 sd = new ScoringPlaneSD(theSensDetName, hcName, subdetID);
133 } else if (sdType == "TrigScintSD") {
134 sd = new TrigScintSD(theSensDetName, hcName, subdetID);
135 } else {
136 EXCEPTION_RAISE("DetType", "Unknown SensitiveDetector type: " + sdType);
137 }
138
139 /*
140 * Fix layer depth if the Sensitive Detector is not the Tracker
141 */
142 if (sdType != "TrackerSD" && layerDepth != -1) {
143 ((CalorimeterSD*)sd)->setLayerDepth(layerDepth);
144 }
145 sd->SetVerboseLevel(verbose);
146
147 // G4cout << "Created " << sdType << " " << theSensDetName << " with hits
148 // collection " << hcName << " and verbose level " << verbose << G4endl <<
149 // G4endl;
150}

152void AuxInfoReader::assignAuxInfoToVolumes() {
153 const G4LogicalVolumeStore* lvs = G4LogicalVolumeStore::GetInstance();
154 std::vector<G4LogicalVolume*>::const_iterator lvciter;
155 for (lvciter = lvs->begin(); lvciter != lvs->end(); lvciter++) {
156 G4GDMLAuxListType auxInfo =
157 parser_->GetVolumeAuxiliaryInformation(*lvciter);
158 if (auxInfo.size() > 0) {
159 for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
160 auxInfo.begin();
161 iaux != auxInfo.end(); iaux++) {
162 G4String auxType = iaux->type;
163 G4String auxVal = iaux->value;
164 G4String auxUnit = iaux->unit;
165
166 G4LogicalVolume* lv = (*lvciter);
167
168 if (auxType == "SensDet") {
169 G4String sdName = auxVal;
170 G4VSensitiveDetector* sd =
171 G4SDManager::GetSDMpointer()->FindSensitiveDetector(sdName);
172 if (sd != NULL) {
173 lv->SetSensitiveDetector(sd);
174 // G4cout << "Assigned SD " << sd->GetName() << " to " <<
175 // lv->GetName() << G4endl;
176 } else {
177 EXCEPTION_RAISE("MissingInfo",
178 "Unknown SensDet in volume's auxiliary info: " +
179 std::string(sdName.data()));
180 }
181 } else if (auxType == "MagneticField") {
182 G4String magFieldName = auxVal;
183 G4MagneticField* magField =
184 MagneticFieldStore::getInstance()->getMagneticField(magFieldName);
185 if (magField != NULL) {
186 G4FieldManager* mgr = new G4FieldManager(magField);
187 lv->SetFieldManager(mgr, true /* FIXME: hard-coded to force field manager to daughters */);
188 // G4cout << "Assigned magnetic field " << magFieldName << " to
189 // volume " << lv->GetName() << G4endl;
190 } else {
191 EXCEPTION_RAISE(
192 "MissingInfo",
193 "Unknown MagneticField ref in volume's auxiliary info: " +
194 std::string(magFieldName.data()));
195 }
196 } else if (auxType == "Region") {
197 G4String regionName = auxVal;
198 G4Region* region =
199 G4RegionStore::GetInstance()->GetRegion(regionName);
200 if (region != NULL) {
201 region->AddRootLogicalVolume(lv);
202 // G4cout << "Added volume " << lv->GetName() << " to region " <<
203 // regionName << G4endl;
204 } else {
205 EXCEPTION_RAISE("MissingInfo", "Reference region '" +
206 std::string(regionName.data()) +
207 "' was not found!");
208 }
209 } else if (auxType == "VisAttributes") {
210 G4String visName = auxVal;
211 G4VisAttributes* visAttributes =
212 VisAttributesStore::getInstance()->getVisAttributes(visName);
213 if (visAttributes != NULL) {
214 lv->SetVisAttributes(visAttributes);
215 // G4cout << "Assigned VisAttributes " << visName << " to volume "
216 // << lv->GetName() << G4endl;
217 } else {
218 EXCEPTION_RAISE("MissingInfo", "Referenced VisAttributes '" +
219 std::string(visName.data()) +
220 "' was not found!");
221 }
222 }
223 }
224 }
225 }

228void AuxInfoReader::createMagneticField(G4String magFieldName,
229 const G4GDMLAuxListType* auxInfoList) {
230 // Find type of the mag field.
231 G4String magFieldType("");
232 for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
233 auxInfoList->begin();
234 iaux != auxInfoList->end(); iaux++) {
235 G4String auxType = iaux->type;
236 G4String auxVal = iaux->value;
237
238 if (auxType == "MagneticFieldType") {
239 magFieldType = auxVal;
240 break;
241 }
242 }
243
244 if (magFieldType == "") {
245 EXCEPTION_RAISE("MissingInfo",
246 "Missing MagFieldType for magnetic field definition.");
247 }
248
249 G4MagneticField* magField = NULL;
250
251 // Create a uniform mag field using the built-in Geant4 type.
252 if (magFieldType == "G4UniformMagField") {
253 std::string::size_type sz;
254 double bx, by, bz;
255 bx = by = bz = 0.;
256 for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
257 auxInfoList->begin();
258 iaux != auxInfoList->end(); iaux++) {
259 G4String auxType = iaux->type;
260 G4String auxVal = iaux->value;
261 G4String auxUnit = iaux->unit;
262
263 G4String expr = auxVal + "*" + auxUnit;
264 if (auxType == "bx") {
265 bx = eval_->Evaluate(expr);
266 } else if (auxType == "by") {
267 by = eval_->Evaluate(expr);
268 } else if (auxType == "bz") {
269 bz = eval_->Evaluate(expr);
270 }
271 }
272 G4ThreeVector fieldComponents(bx, by, bz);
273 magField = new G4UniformMagField(fieldComponents);
274
275 // G4cout << "Created G4UniformMagField " << magFieldName << " with field
276 // components " << fieldComponents << G4endl << G4endl;
277
278 // Create a global 3D field map by reading from a data file.
279 } else if (magFieldType == "MagneticFieldMap3D") {
280 std::string fileName;
281 double offsetX, offsetY, offsetZ;
282
283 for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
284 auxInfoList->begin();
285 iaux != auxInfoList->end(); iaux++) {
286 G4String auxType = iaux->type;
287 G4String auxVal = iaux->value;
288 G4String auxUnit = iaux->unit;
289
290 G4String expr = auxVal + "*" + auxUnit;
291
292 if (auxType == "File") {
293 fileName = auxVal;
294 } else if (auxType == "OffsetX") {
295 offsetX = eval_->Evaluate(expr);
296 } else if (auxType == "OffsetY") {
297 offsetY = eval_->Evaluate(expr);
298 } else if (auxType == "OffsetZ") {
299 offsetZ = eval_->Evaluate(expr);
300 }
301 }
302
303 if (fileName.size() == 0) {
304 EXCEPTION_RAISE("MissingInfo",
305 "File info with field data was not provided.");
306 }
307
308 // Create new 3D field map.
309 magField =
310 new MagneticFieldMap3D(fileName.c_str(), offsetX, offsetY, offsetZ);
311
312 // Assign field map as global field.
313 G4FieldManager* fieldMgr =
314 G4TransportationManager::GetTransportationManager()->GetFieldManager();
315 if (fieldMgr->GetDetectorField() != nullptr) {
316 EXCEPTION_RAISE("MisAssign", "Global mag field was already assigned.");
317 }
318 fieldMgr->SetDetectorField(magField);
319 fieldMgr->CreateChordFinder(magField);
320
321 } else {
322 EXCEPTION_RAISE("UnknownType", "Unknown MagFieldType '" +
323 std::string(magFieldType.data()) +
324 "' in auxiliary info.");
325 }
326
327 MagneticFieldStore::getInstance()->addMagneticField(magFieldName, magField);
328}

358void AuxInfoReader::createVisAttributes(G4String name,
359 const G4GDMLAuxListType* auxInfoList) {
360 G4double rgba[4] = {1., 1., 1., 1.};
361 G4bool visible = true;
362 G4bool dauInvisible = false;
363 G4bool forceWireframe = false;
364 G4bool forceSolid = false;
365 G4double lineWidth = 1.0;
366 G4VisAttributes::LineStyle lineStyle = G4VisAttributes::unbroken;
367
368 for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
369 auxInfoList->begin();
370 iaux != auxInfoList->end(); iaux++) {
371 G4String auxType = iaux->type;
372 G4String auxVal = iaux->value;
373 G4String auxUnit = iaux->unit;
374
375 if (auxType == "R") {
376 rgba[0] = atof(auxVal.c_str());
377 } else if (auxType == "G") {
378 rgba[1] = atof(auxVal.c_str());
379 } else if (auxType == "B") {
380 rgba[2] = atof(auxVal.c_str());
381 } else if (auxType == "A") {
382 rgba[3] = atof(auxVal.c_str());
383 } else if (auxType == "Style") {
384 if (auxVal == "wireframe") {
385 forceWireframe = true;
386 } else if (auxVal == "solid") {
387 forceSolid = true;
388 }
389 } else if (auxType == "DaughtersInvisible") {
390 if (auxVal == "true") {
391 dauInvisible = true;
392 } else if (auxVal == "false") {
393 dauInvisible = false;
394 }
395 } else if (auxType == "Visible") {
396 if (auxVal == "true") {
397 visible = true;
398 } else if (auxVal == "false") {
399 visible = false;
400 }
401 } else if (auxType == "LineStyle") {
402 if (auxVal == "unbroken") {
403 lineStyle = G4VisAttributes::unbroken;
404 } else if (auxVal == "dashed") {
405 lineStyle = G4VisAttributes::dashed;
406 } else if (auxVal == "dotted") {
407 lineStyle = G4VisAttributes::dotted;
408 }
409 } else if (auxType == "LineWidth") {
410 lineWidth = atof(auxVal.c_str());
411 }
412 }
413
414 G4VisAttributes* visAttributes = new G4VisAttributes();
415 visAttributes->SetColor(rgba[0], rgba[1], rgba[2], rgba[3]);
416 visAttributes->SetVisibility(visible);
417 visAttributes->SetDaughtersInvisible(dauInvisible);
418 visAttributes->SetForceWireframe(forceWireframe);
419 visAttributes->SetForceSolid(forceSolid);
420 visAttributes->SetLineWidth(lineWidth);
421 visAttributes->SetLineStyle(lineStyle);
422 VisAttributesStore::getInstance()->addVisAttributes(name, visAttributes);
423
424 // G4cout << "Created VisAttributes " << name << G4endl << (*visAttributes) <<
425 // G4endl << G4endl;
426}
