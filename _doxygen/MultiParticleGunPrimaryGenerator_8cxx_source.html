36void MultiParticleGunPrimaryGenerator::GeneratePrimaryVertex(G4Event* anEvent) {
37 int cur_mpg_pdgid = mpgPdgID_;
38 G4ThreeVector cur_mpg_vertex = mpgVertex_;
39 G4ThreeVector cur_mpg_momentum = mpgMomentum_;
40
41 // current number of vertices in the event!
42 int curNVertices = anEvent->GetNumberOfPrimaryVertex();
43
44 double nInteractionsInput = mpgNParticles_;
45 int nInteractions = nInteractionsInput;
46 if (mpgEnablePoisson_) {
47 nInteractions = 0;
48 while (nInteractions == 0) { // keep generating a random poisson until > 0,
49 // no point in generator 0 vertices...
50 nInteractions = random_->Poisson(nInteractionsInput);
51 }
52 }
53
54 // make a for loop
55 for (int i = 0; i < (nInteractions - curNVertices); ++i) {
56 G4PrimaryVertex* curvertex =
57 new G4PrimaryVertex(cur_mpg_vertex, 0.); // second input is t0
58 // curvertex->SetPosition(0. * mm,0. * mm,-10. * mm);
59 curvertex->SetWeight(1.);
60
61 G4PrimaryParticle* primary =
62 new G4PrimaryParticle(cur_mpg_pdgid, cur_mpg_momentum.x(),
63 cur_mpg_momentum.y(), cur_mpg_momentum.z());
64
65 UserPrimaryParticleInformation* primaryInfo =
66 new UserPrimaryParticleInformation();
67 primaryInfo->setHepEvtStatus(1.);
68 primary->SetUserInformation(primaryInfo);
69
70 curvertex->SetPrimary(primary);
71 anEvent->AddPrimaryVertex(curvertex);
72 }
73}
