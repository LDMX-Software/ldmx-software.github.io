109  void PhotoNuclearDQM::analyze(const Event& event) {
110 
111  // Get the particle map from the event. If the particle map is empty,
112  // don't process the event.
113  auto particleMap{event.getMap<int,SimParticle>("SimParticles")};
114  if (particleMap.size() == 0) return;
115 
116  // Get the recoil electron
117  auto [ trackID, recoil ] = Analysis::getRecoil(particleMap);
118 
119  histograms_.fill("recoil_vertex_x",recoil->getVertex()[0]);
120  histograms_.fill("recoil_vertex_y",recoil->getVertex()[1]);
121  histograms_.fill("recoil_vertex_z",recoil->getVertex()[2]);
122  histograms_.fill("recoil_vertex_x:recoil_vertex_y",
123  recoil->getVertex()[0],
124  recoil->getVertex()[1]);
125 
126  // Use the recoil electron to retrieve the gamma that underwent a
127  // photo-nuclear reaction.
128  auto pnGamma{Analysis::getPNGamma(particleMap, recoil, 2500.)};
129  if (pnGamma == nullptr) {
130  std::cout << "[ PhotoNuclearDQM ]: PN Daughter is lost, skipping." << std::endl;
131  return;
132  }
133 
134  histograms_.fill("pn_particle_mult",pnGamma->getDaughters().size());
135  histograms_.fill("pn_gamma_energy",pnGamma->getEnergy());
136  histograms_.fill("pn_gamma_int_z",pnGamma->getEndPoint()[2]);
137  histograms_.fill("pn_gamma_vertex_x",pnGamma->getVertex()[0]);
138  histograms_.fill("pn_gamma_vertex_y",pnGamma->getVertex()[1]);
139  histograms_.fill("pn_gamma_vertex_z",pnGamma->getVertex()[2]);
140 
141  double lke{-1}, lt{-1};
142  double lpke{-1}, lpt{-1};
143  double lnke{-1}, lnt{-1};
144  double lpike{-1}, lpit{-1};
145 
146  std::vector<const SimParticle*> pnDaughters;
147 
148  // Loop through all of the PN daughters and extract kinematic
149  // information.
150  for (const auto& daughterTrackID : pnGamma->getDaughters() ) {
151 
152  //skip daughters that weren't saved
153  if ( particleMap.count( daughterTrackID ) == 0 ) continue;
154 
155  auto daughter{&(particleMap.at(daughterTrackID))};
156 
157  // Get the PDG ID
158  auto pdgID{daughter->getPdgID()};
159 
160  // Ignore photons and nuclei
161  if (pdgID == 22 || pdgID > 10000) continue;
162 
163  // Calculate the kinetic energy
164  double ke{daughter->getEnergy() - daughter->getMass()};
165 
166  std::vector<double> vec{daughter->getMomentum()};
167  TVector3 pvec(vec[0], vec[1], vec[2]);
168 
169  // Calculate the polar angle
170  auto theta{pvec.Theta()*(180/3.14159)};
171 
172  if (lke < ke) { lke = ke; lt = theta; }
173 
174  if ((pdgID == 2112) && (lnke < ke)) { lnke = ke; lnt = theta; }
175 
176  if ((pdgID == 2212) && (lpke < ke)) { lpke = ke; lpt = theta; }
177 
178  if ( ( (abs(pdgID) == 211) || (pdgID == 111) ) && (lpike < ke) ) {
179  lpike = ke;
180  lpit = theta;
181  }
182 
183  pnDaughters.push_back(daughter);
184  }
185 
186  histograms_.fill("hardest_ke",lke);
187  histograms_.fill("hardest_theta",lt);
188  histograms_.fill("h_ke_h_theta",lke, lt);
189  histograms_.fill("hardest_p_ke",lpke);
190  histograms_.fill("hardest_p_theta",lpt);
191  histograms_.fill("hardest_n_ke",lnke);
192  histograms_.fill("hardest_n_theta",lnt);
193  histograms_.fill("hardest_pi_ke",lpike);
194  histograms_.fill("hardest_pi_theta",lpit);
195 
196  // Classify the event
197  auto eventType{classifyEvent(pnDaughters, 200)};
198  auto eventType500MeV{classifyEvent(pnDaughters, 500)};
199  auto eventType2000MeV{classifyEvent(pnDaughters, 2000)};
200 
201  auto eventTypeComp{classifyCompactEvent(pnGamma, pnDaughters, 200)};
202  auto eventTypeComp500MeV{classifyCompactEvent(pnGamma, pnDaughters, 200)};
203  auto eventTypeComp2000MeV{classifyCompactEvent(pnGamma, pnDaughters, 200)};
204 
205  histograms_.fill("event_type",eventType);
206  histograms_.fill("event_type_500mev",eventType500MeV);
207  histograms_.fill("event_type_2000mev",eventType2000MeV);
208 
209  histograms_.fill("event_type_compact",eventTypeComp);
210  histograms_.fill("event_type_compact_500mev",eventTypeComp500MeV);
211  histograms_.fill("event_type_compact_2000mev",eventTypeComp2000MeV);
212 
213  double slke{-9999};
214  double nEnergy{-9999}, energyDiff{-9999}, energyFrac{-9999};
215 
216  if (eventType == 1 || eventType == 17 || eventType == 16 || eventType == 18 || eventType == 2) {
217 
218  std::sort (pnDaughters.begin(), pnDaughters.end(), [] (const auto& lhs, const auto& rhs)
219  {
220  double lhs_ke = lhs->getEnergy() - lhs->getMass();
221  double rhs_ke = rhs->getEnergy() - rhs->getMass();
222  return lhs_ke > rhs_ke;
223  });
224 
225 
226  nEnergy = pnDaughters[0]->getEnergy() - pnDaughters[0]->getMass();
227  slke = -9999;
228  if (pnDaughters.size() > 1) slke = pnDaughters[1]->getEnergy() - pnDaughters[1]->getMass();
229  energyDiff = pnGamma->getEnergy() - nEnergy;
230  energyFrac = nEnergy/pnGamma->getEnergy();
231 
232  if (eventType == 1) {
233  histograms_.fill("1n_ke:2nd_h_ke",nEnergy, slke);
234  histograms_.fill("1n_neutron_energy",nEnergy);
235  histograms_.fill("1n_energy_diff",energyDiff);
236  histograms_.fill("1n_energy_frac",energyFrac);
237  } else if (eventType == 2) {
238  histograms_.fill("2n_n2_energy",slke);
239  auto energyFrac2n = (nEnergy + slke)/pnGamma->getEnergy();
240  histograms_.fill("2n_energy_frac",energyFrac2n);
241  histograms_.fill("2n_energy_other",pnGamma->getEnergy() - energyFrac2n);
242 
243  } else if (eventType == 17) {
244  histograms_.fill("1kp_ke:2nd_h_ke",nEnergy, slke);
245  histograms_.fill("1kp_energy",nEnergy);
246  histograms_.fill("1kp_energy_diff",energyDiff);
247  histograms_.fill("1kp_energy_frac",energyFrac);
248  } else if (eventType == 16 || eventType == 18) {
249  histograms_.fill("1k0_ke:2nd_h_ke",nEnergy, slke);
250  histograms_.fill("1k0_energy",nEnergy);
251  histograms_.fill("1k0_energy_diff",energyDiff);
252  histograms_.fill("1k0_energy_frac",energyFrac);
253  }
254 
255  auto nPdgID{abs(pnDaughters[0]->getPdgID())};
256  auto nEventType{-10};
257  if (nPdgID == 2112) nEventType = 1;
258  else if (nPdgID == 2212) nEventType = 2;
259  else if (nPdgID == 211) nEventType = 3;
260  else if (nPdgID == 111) nEventType = 4;
261 
262  histograms_.fill("1n_event_type",nEventType);
263 
264  }

267  int PhotoNuclearDQM::classifyEvent(const std::vector< const SimParticle*> daughters, double threshold) {
268 
269  short n{0}, p{0}, pi{0}, pi0{0}, exotic{0}, k0l{0}, kp{0}, k0s{0},
270  lambda{0};
271 
272  // Loop through all of the PN daughters and extract kinematic
273  // information.
274  for (const auto& daughter : daughters ) {
275 
276  // Calculate the kinetic energy
277  auto ke{daughter->getEnergy() - daughter->getMass()};
278 
279  // If the kinetic energy is below threshold, continue
280  if (ke <= threshold) continue;
281 
282  // Get the PDG ID
283  auto pdgID{abs(daughter->getPdgID())};
284 
285  if (pdgID == 2112) n++;
286  else if (pdgID == 2212) p++;
287  else if (pdgID == 211) pi++;
288  else if (pdgID == 111) pi0++;
289  else if (pdgID == 130) k0l++;
290  else if (pdgID == 321) kp++;
291  else if (pdgID == 310) k0s++;
292  else exotic++;
293  }
294 
295  int kaons = k0l + kp + k0s;
296  int nucleons = n + p;
297  int pions = pi + pi0;
298  int count = nucleons + pions + exotic + kaons ;
299  int count_a = p + pions + exotic + kaons ;
300  int count_b = pions + exotic+ kaons;
301  int count_c = nucleons + pi0 + exotic+ kaons;
302  int count_d = n + pi0 + exotic+ k0l + kp + k0s;
303  int count_e = p + pi0 + exotic+ k0l + kp + k0s;
304  int count_f = n + p + pi + exotic+ k0l + kp + k0s;
305  int count_g = n + pi + pi0 + exotic+ k0l + kp + k0s;
306  int count_h = n + p + pi + pi0 + k0l + kp + k0s;
307  int count_i = p + pi + exotic+ k0l + kp + k0s;
308  int count_j = n + pi + exotic+ k0l + kp + k0s;
309  int count_k = nucleons + pions + exotic + kp + k0s;
310  int count_l = nucleons + pions + exotic + k0l + k0s;
311  int count_m = nucleons + pions + exotic + kp + k0l;
312  int count_n = pi0 + exotic + kaons;
313  int count_o = pi + exotic + kaons;
314  int count_p = exotic + kaons;
315 
316  if (count == 0) return 0; // Nothing hard
317 
318  if (n == 1) {
319  if (count_a == 0) return 1; // 1n
320  else if ( (p == 1) && (count_b == 0) ) return 15; // pn
321  }
322 
323  if ( (n == 2) && (count_a == 0) ) return 2; // 2 n
324 
325  if ( (n >= 3) && (count_a == 0) ) return 3; // >= 3 n
326 
327  if (pi == 1) {
328  if (count_c == 0) return 4; // 1 pi
329  else if ( (p == 1) && (count_d == 0) ) return 7; // 1 pi 1 p
330  else if ( (p == 2) && (count_d == 0) ) return 8; // 1 pi 1 p
331  else if ( (n == 1) && (count_e == 0) ) return 7; // 1 pi 1 n
332  else if ( (n == 2) && (count_e == 0) ) return 8; // 1 pi 1 n
333  else if ( (n == 1) && (p == 1) && (count_n == 0) ) return 8;
334  }
335 
336  if (pi == 2) {
337  if (count_c == 0) return 5; // 2pi
338  else if ( (p == 1) && (count_d == 0) ) return 9; // 2pi p
339  else if ( (n == 1) && (count_e == 0) ) return 9; // 2pi n
340  }
341 
342  if ( pi0 == 1 ) {
343  if (count_f == 0) return 6; // 1 pi0
344  else if ( (n == 1) && (count_i == 0) ) return 10; // 1pi0 1 p
345  else if ( (n == 2) && (count_i == 0) ) return 11; // 1pi0 1 p
346  else if ( (p == 1) && (count_j == 0) ) return 10; // 1pi0 1 n
347  else if ( (p == 2) && (count_j == 0) ) return 11;
348  else if ( (n == 1) && (p == 1) && (count_o == 0) ) return 11;
349  else if ( (pi == 1) && ((p == 1) || (n == 1)) && (count_p == 0)) return 12;
350  }
351 
352 
353  if ( (p == 1) && (count_g == 0) ) return 13; // 1 p
354  if ( (p == 2) && (count_g == 0) ) return 14; // 2 p
355 
356  if (k0l == 1) return 16;
357  if (kp == 1) return 17;
358  if (k0s == 1) return 18;
359 
360  if ( (exotic > 0) && (count_h == 0) ) return 19;
361 
362  return 20;
363  }

365  int PhotoNuclearDQM::classifyCompactEvent(const SimParticle* pnGamma, const std::vector< const SimParticle*> daughters, double threshold) {
366 
367  short n{0}, n_t{0}, k0l{0}, kp{0}, k0s{0}, soft{0};
368 
369  // Loop through all of the PN daughters and extract kinematic
370  // information.
371  for (const auto& daughter : daughters ) {
372 
373  // Calculate the kinetic energy
374  auto ke{daughter->getEnergy() - daughter->getMass()};
375 
376  // Get the PDG ID
377  auto pdgID{abs(daughter->getPdgID())};
378 
379  if (ke < 500) {
380  soft++;
381  continue;
382  }
383 
384  if (ke >= 0.8*pnGamma->getEnergy()) {
385  if (pdgID == 2112) n++;
386  else if (pdgID == 130) k0l++;
387  else if (pdgID == 321) kp++;
388  else if (pdgID == 310) k0s++;
389  continue;
390  }
391 
392  if ((pdgID == 2112) && ke > threshold) n_t++;
393  }
394 
395  int neutral_kaons{k0l + k0s};
396 
397  if (n != 0) return 0;
398  if (kp != 0) return 1;
399  if (neutral_kaons != 0) return 2;
400  if (n_t == 2) return 3;
401  if (soft == daughters.size()) return 4;
402 
403  return 5;
404 
405  }

428  std::vector< int > PhotoNuclearDQM::printDaughters(std::map< int, SimParticle > particleMap,
429  const SimParticle particle, int depth) {
430 
431  std::vector< int > printedParticles;
432 
433  // Don't print anything if a particle doesn't have any daughters
434  if (particle.getDaughters().size() == 0) return printedParticles;
435 
436  // Generate the prefix
437  std::string prefix{""};
438  for (auto i{0}; i < depth; ++i) prefix += "\t";
439 
440  // Loop through all of the daughter particles and print them
441  for (const auto& daughter : particle.getDaughters()) {
442 
443  // Print the ith daughter particle
444  std::cout << prefix;
445  particleMap[daughter].Print();
446  printedParticles.push_back(daughter);
447 
448  // Print the Daughters
449  std::vector< int > printedDaughters = printDaughters(particleMap, particleMap[daughter], depth + 1);
450  printedParticles.insert(printedParticles.end(), printedDaughters.begin(), printedDaughters.end());
451  }
452  }
