18void EcalDigiVerifier::analyze(const framework::Event &event) {
19 // get truth information sorted into an ID based map
20 std::vector<ldmx::SimCalorimeterHit> ecalSimHits =
21 event.getCollection<ldmx::SimCalorimeterHit>(ecalSimHitColl_,
22 ecalSimHitPass_);
23
24 // sort sim hits by ID
25 std::sort(ecalSimHits.begin(), ecalSimHits.end(),
26 [](const ldmx::SimCalorimeterHit &lhs,
27 const ldmx::SimCalorimeterHit &rhs) {
28 return lhs.getID() < rhs.getID();
29 });
30
31 std::vector<ldmx::EcalHit> ecalRecHits =
32 event.getCollection<ldmx::EcalHit>(ecalRecHitColl_, ecalRecHitPass_);
33
34 // sort rec hits by ID
35 std::sort(ecalRecHits.begin(), ecalRecHits.end(),
36 [](const ldmx::EcalHit &lhs, const ldmx::EcalHit &rhs) {
37 return lhs.getID() < rhs.getID();
38 });
39
40 double totalRecEnergy = 0.;
41 for (const ldmx::EcalHit &recHit : ecalRecHits) {
42 // skip anything that digi flagged as noise
43 if (recHit.isNoise()) continue;
44
45 int rawID = recHit.getID();
46
47 // get information for this hit
48 int numSimHits = 0;
49 double totalSimEDep = 0.;
50 for (const ldmx::SimCalorimeterHit &simHit : ecalSimHits) {
51 if (rawID == simHit.getID()) {
52 numSimHits += simHit.getNumberOfContribs();
53 totalSimEDep += simHit.getEdep();
54 } else if (rawID < simHit.getID()) {
55 // later sim hits - all done
56 break;
57 }
58 }
59
60 histograms_.fill("num_sim_hits_per_cell", numSimHits);
61 histograms_.fill("sim_edep__rec_amplitude", totalSimEDep,
62 recHit.getAmplitude());
63
64 totalRecEnergy += recHit.getEnergy();
65 }
66
67 histograms_.fill("total_rec_energy", totalRecEnergy);
68
69 if (totalRecEnergy > 6000.) {
70 setStorageHint(framework::hint_shouldKeep);
71 } else {
72 setStorageHint(framework::hint_shouldDrop);
73 }
74
75 return;
76}
