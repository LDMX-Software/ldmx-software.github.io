27 struct PolarfireQueueEntry {
29 int spill;
31 int ticks;
32 ldmx::HgcrocDigiCollection digis;
33 PolarfireQueueEntry(const framework::Event& event,
34 const std::string& input_name,
35 const std::string& input_pass,
36 std::pair<int, int>& spill_counter) {
37 int spilln = event.getObject<int>(input_name + "Spill", input_pass);
38 if (spilln != spill_counter.second) {
39 spill_counter.first++;
40 spill_counter.second = spilln;
41 }
42 spill = spill_counter.first;
43 ticks = event.getObject<int>(input_name + "Ticks", input_pass);
44 digis =
45 event.getObject<ldmx::HgcrocDigiCollection>(input_name, input_pass);
46 }
47 bool same_event(const PolarfireQueueEntry& rhs) {
48 return (spill == rhs.spill and abs(ticks - rhs.ticks) < max_tick_diff_);
49 }
50 bool earlier_event(const PolarfireQueueEntry& rhs) {
51 if (spill == rhs.spill) return ticks < rhs.ticks;
52 return spill < rhs.spill;
53 }
54 };
56 std::queue<PolarfireQueueEntry> pf0_queue, pf1_queue;
58 std::pair<int, int> pf0_spill_counter{0, -1}, pf1_spill_counter{0, -1};

77void HcalAlignPolarfires::produce(framework::Event& event) {
78 // put next package of decoding into the queues
79 pf0_queue.emplace(event, input_names_[0], input_pass_, pf0_spill_counter);
80 pf1_queue.emplace(event, input_names_[1], input_pass_, pf1_spill_counter);
81
82 // remove empty events from front of queues for end-of-file condition
83 while (pf0_queue.size() > 0 and pf0_queue.front().digis.getNumDigis() == 0)
84 pf0_queue.pop();
85 while (pf1_queue.size() > 0 and pf1_queue.front().digis.getNumDigis() == 0)
86 pf1_queue.pop();
87
88 bool aligned{false};
89 ldmx::HgcrocDigiCollection merged;
90 if (pf0_queue.size() > 0 and pf1_queue.size() > 0) {
91 // we have non-empty events in the queues, start alignment procedure
92 if (pf0_queue.front().same_event(pf1_queue.front())) {
93 // same spill and ticks within max_tick_diff_ of each other
94 // add them together and put them into same object
95 merged = pf0_queue.front().digis;
96 const auto& unmerged{pf1_queue.front().digis};
97 for (int i{0}; i < unmerged.getNumDigis(); i++) {
98 auto digi{unmerged.getDigi(i)};
99 std::vector<ldmx::HgcrocDigiCollection::Sample> samples;
100 for (int j{0}; j < unmerged.getNumSamplesPerDigi(); j++) {
101 samples.push_back(digi.at(i));
102 }
103 merged.addDigi(digi.id(), samples);
104 }
105 aligned = true;
106 pf0_queue.pop();
107 pf1_queue.pop();
108 setStorageHint(framework::hint_shouldKeep);
109 } else if (pf0_queue.front().earlier_event(pf1_queue.front())) {
110 // should add pf0 but signal event is unmerged
111 merged = pf0_queue.front().digis;
112 pf0_queue.pop();
113 setStorageHint(framework::hint_shouldDrop);
114 } else {
115 // should add pf1 but signal event is unmerged
116 merged = pf1_queue.front().digis;
117 pf1_queue.pop();
118 setStorageHint(framework::hint_shouldDrop);
119 }
120 } else if (pf0_queue.size() > 0) {
121 // only pf0 has non-empty events left
122 // should add pf0 but signal event is unmerged
123 merged = pf0_queue.front().digis;
124 pf0_queue.pop();
125 setStorageHint(framework::hint_shouldDrop);
126 } else if (pf1_queue.size() > 0) {
127 // only pf1 has non-empty events left
128 // should add pf1 but signal event is unmerged
129 merged = pf1_queue.front().digis;
130 pf1_queue.pop();
131 setStorageHint(framework::hint_shouldDrop);
132 } else {
133 // no more events, both decoders are returning empty events
134 abortEvent();
135 }
136
137 event.add(output_name_, merged);
138 event.add(output_name_ + "Aligned", aligned);
139} // produce
