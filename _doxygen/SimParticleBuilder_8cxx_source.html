27 void SimParticleBuilder::buildSimParticles(framework::Event *outputEvent) {
28  // Get the trajectory container for the event.
29  auto trajectories{
30  (const_cast<G4Event *>(currentEvent_))->GetTrajectoryContainer()};
31 
32  if (!trajectories or !(trajectories->GetVector())) {
33  EXCEPTION_RAISE("PersistFault",
34  "Event's trajectory container does not exist.");
35  }
36 
37  const std::vector<G4VTrajectory *> &trajectory_list =
38  *(trajectories->GetVector());
39 
40  // Create empty SimParticle objects and create the map of track ID to
41  // particles.
42  std::map<int, ldmx::SimParticle> outputParticleMap;
43  for (G4VTrajectory *trajectory : trajectory_list) {
44  if (!trajectory)
45  EXCEPTION_RAISE("PersistFault",
46  "NULL G4VTrajectory ended up in storage.");
47  outputParticleMap[trajectory->GetTrackID()];
48  }
49 
50  // Fill information into the particles.
51  for (G4VTrajectory *trajectory : trajectory_list) {
52  Trajectory *traj = dynamic_cast<Trajectory *>(trajectory);
53  if (!traj)
54  EXCEPTION_RAISE("PersistFault", "NULL Trajectory ended up in storage.");
55 
56  ldmx::SimParticle *simParticle = &outputParticleMap[traj->GetTrackID()];
57 
58  simParticle->setGenStatus(traj->getGenStatus());
59  simParticle->setPdgID(traj->GetPDGEncoding());
60  simParticle->setCharge(traj->GetCharge());
61  simParticle->setMass(traj->getMass());
62  simParticle->setEnergy(traj->getEnergy());
63  simParticle->setTime(traj->getGlobalTime());
64  simParticle->setProcessType(traj->getProcessType());
65  simParticle->setVertexVolume(traj->getVertexVolume());
66 
67  const G4ThreeVector &vertex = traj->getVertexPosition();
68  simParticle->setVertex(vertex[0], vertex[1], vertex[2]);
69 
70  const G4ThreeVector &momentum = traj->GetInitialMomentum();
71  simParticle->setMomentum(momentum[0], momentum[1], momentum[2]);
72 
73  const G4ThreeVector &endpMomentum = traj->getEndPointMomentum();
74  simParticle->setEndPointMomentum(endpMomentum[0], endpMomentum[1],
75  endpMomentum[2]);
76 
77  G4ThreeVector endpoint = traj->getEndPoint();
78  simParticle->setEndPoint(endpoint[0], endpoint[1], endpoint[2]);
79 
80  if (traj->GetParentID() > 0) {
81  simParticle->addParent(traj->GetParentID());
82  // check if parent track is being stored
83  auto parentParticle = outputParticleMap.find(traj->GetParentID());
84  if (parentParticle != outputParticleMap.end()) {
85  // this parent has been found in the particle map
86  outputParticleMap[traj->GetParentID()].addDaughter(
87  trajectory->GetTrackID());
88  } // check if parent exists in output map
89  } // check if particle has a parent
90  }
91 
92  // Add the collection data to the output event.
93  outputEvent->add("SimParticles", outputParticleMap);
94 }
