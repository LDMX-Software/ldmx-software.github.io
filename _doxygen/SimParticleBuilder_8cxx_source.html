32 
33  // Get the trajectory container for the event.

35  (const_cast<G4Event *>(currentEvent_))->GetTrajectoryContainer()};
36 
37  // Create empty SimParticle objects and create the map of track ID to

40  for (auto trajectory : *trajectories->GetVector()) {
41  outputParticleMap[trajectory->GetTrackID()] = SimParticle();
42  }

45  for (auto trajectory : *trajectories->GetVector()) {
46

48

50

53  simParticle->setCharge(traj->GetCharge());

64  simParticle->setMomentum(momentum[0], momentum[1], momentum[2]);
65

70  G4ThreeVector endpoint = traj->getEndPoint();
71  simParticle->setEndPoint(endpoint[0], endpoint[1], endpoint[2]);
72 
73  if (traj->GetParentID() > 0) {

76  auto parentParticle = outputParticleMap.find(traj->GetParentID());
