19void StepPrinter::stepping(const G4Step* step) {
20 // Get the track associated with this step
21 auto track{step->GetTrack()};
22
23 if (auto trackID{track->GetTrackID()};
24 (trackID_ > 0) && (trackID != trackID_))
25 return;
26
27 // Get the particle name.
28 auto particleName{track->GetParticleDefinition()->GetParticleName()};
29
30 // Get the energy of the particle
31 auto energy{step->GetPostStepPoint()->GetTotalEnergy()};
32
33 // Get the volume the particle is in.
34 auto volume{track->GetVolume()->GetName()};
35
36 // Get the next volume
37 auto nextVolume{track->GetNextVolume()->GetName()};
38
39 // Get the region
40 auto region{track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
41
42 std::cout << " Step " << track->GetCurrentStepNumber() << " {"
43 << " Energy: " << energy << " Track ID: " << track->GetTrackID()
44 << " Particle currently in: " << volume << " Region: " << region
45 << " Next volume: " << nextVolume
46 << " Weight: " << track->GetWeight() << " Children:";
47 for (auto const& track : *(step->GetSecondaryInCurrentStep()))
48 std::cout << " " << track->GetParticleDefinition()->GetPDGEncoding();
49
50 std::cout << " }" << std::endl;
51}
