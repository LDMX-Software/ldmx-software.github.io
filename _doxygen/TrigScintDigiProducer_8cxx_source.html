48 void TrigScintDigiProducer::produce(Event &event) {
49 
50  // Need to handle seeding on the first event
51  if (!noiseGenerator_->hasSeed()) {
52  const auto& rseed = getCondition<RandomNumberSeedService>(RandomNumberSeedService::CONDITIONS_OBJECT_NAME);
53  noiseGenerator_->seedGenerator(rseed.getSeed("TrigScintDigiProducer::NoiseGenerator"));
54  }
55 
56 
57  std::map<TrigScintID, int> cellPEs;
58  std::map<TrigScintID, int> cellMinPEs;
59  std::map<TrigScintID, float> Xpos, Ypos, Zpos, Edep, Time, beamFrac;
60  std::set<TrigScintID> noiseHitIDs;
61 
62  auto numRecHits{0};
63 
64  // looper over sim hits and aggregate energy depositions for each detID
65  const auto simHits{
66  event.getCollection<SimCalorimeterHit>(inputCollection_, inputPassName_)};
67  auto particleMap{event.getMap<int, SimParticle>("SimParticles")};
68 
69  int module{-1};
70  for (const auto &simHit : simHits) {
71 
72  TrigScintID id(simHit.getID());
73 
74  // Just set the module ID to use for noise hits here. Given that
75  // we are currently processing a single module at a time, setting
76  // it within the loop shouldn't matter.
77  module = id.module();
78  std::vector<float> position = simHit.getPosition();
79 
80  if (verbose_) {
81  std::cout << id << std::endl;
82  }
83 
84  unsigned int detIDRaw = id.raw();
85 
86  // check if hits is from beam electron and, if so, add to beamFrac
87  for (int i = 0; i < simHit.getNumberOfContribs(); i++) {
88 
89  auto contrib = simHit.getContrib(i);
90  if (verbose_) {
91  std::cout << "contrib " << i << " trackID: " << contrib.trackID
92  << " pdgID: " << contrib.pdgCode << " edep: " << contrib.edep
93  << std::endl;
94  std::cout << "\t particle id: "
95  << particleMap[contrib.trackID].getPdgID()
96  << " particle status: "
97  << particleMap[contrib.trackID].getGenStatus() << std::endl;
98  }
99  if (particleMap[contrib.trackID].getPdgID() == 11 &&
100  particleMap[contrib.trackID].getGenStatus() == 1) {
101  if (beamFrac.find(id) == beamFrac.end())
102  beamFrac[id] = contrib.edep;
103  else
104  beamFrac[id] += contrib.edep;
105  }
106  }
107 
108  // for now, we take am energy weighted average of the hit in each stip to
109  // simulate the hit position. AJW: these should be dropped, they are likely
110  // to lead to a problem since we can't measure them anyway except roughly y
111  // and z, which is encoded in the ids.
112  if (Edep.find(id) == Edep.end()) {
113 
114  // first hit, initialize
115  Edep[id] = simHit.getEdep();
116  Time[id] = simHit.getTime() * simHit.getEdep();
117  Xpos[id] = position[0] * simHit.getEdep();
118  Ypos[id] = position[1] * simHit.getEdep();
119  Zpos[id] = position[2] * simHit.getEdep();
120  numRecHits++;
121 
122  } else {
123 
124  // not first hit, aggregate, and store the largest radius hit
125  Xpos[id] += position[0] * simHit.getEdep();
126  Ypos[id] += position[1] * simHit.getEdep();
127  Zpos[id] += position[2] * simHit.getEdep();
128  Edep[id] += simHit.getEdep();
129  // AJW: need to figure out a better way to model this...
130  Time[id] += simHit.getTime() * simHit.getEdep();
131  }
132  }
133 
134  // Create the container to hold the digitized trigger scintillator hits.
135  std::vector<TrigScintHit> trigScintHits;
136 
137  // loop over detIDs and simulate number of PEs
138  int ihit = 0;
139  for (std::map<TrigScintID, float>::iterator it = Edep.begin();
140  it != Edep.end(); ++it) {
141  TrigScintID id(it->first);
142 
143  double depEnergy = Edep[id];
144  Time[id] = Time[id] / Edep[id];
145  Xpos[id] = Xpos[id] / Edep[id];
146  Ypos[id] = Ypos[id] / Edep[id];
147  Zpos[id] = Zpos[id] / Edep[id];
148  double meanPE = depEnergy / mevPerMip_ * pePerMip_;
149  cellPEs[id] = random_->Poisson(meanPE + meanNoise_);
150 
151  // If a cell has a PE count above threshold, persit the hit.
152  if (cellPEs[id] >= 1) {
153 
154  TrigScintHit hit;
155  hit.setID(id.raw());
156  hit.setPE(cellPEs[id]);
157  hit.setMinPE(cellMinPEs[id]);
158  hit.setAmplitude(cellPEs[id]);
159  hit.setEnergy(depEnergy);
160  hit.setTime(Time[id]);
161  hit.setXPos(Xpos[id]);
162  hit.setYPos(Ypos[id]);
163  hit.setZPos(Zpos[id]);
164  hit.setModuleID(module);
165  hit.setBarID(id.bar()); // getFieldValue("bar"));
166  hit.setNoise(false);
167  hit.setBeamEfrac(beamFrac[id] / depEnergy);
168 
169  trigScintHits.push_back(hit);
170  }
171 
172  if (verbose_) {
173 
174  std::cout << id << std::endl;
175  std::cout << "Edep: " << Edep[id] << std::endl;
176  std::cout << "numPEs: " << cellPEs[id] << std::endl;
177  std::cout << "time: " << Time[id] << std::endl;
178  std::cout << "z: " << Zpos[id] << std::endl;
179  std::cout << "\t X: " << Xpos[id] << "\t Y: " << Ypos[id]
180  << "\t Z: " << Zpos[id] << std::endl;
181  } // end verbose
182  }
183 
184  // ------------------------------- Noise simulation -----------------------//
185  // ------------------------------------------------------------------------//
186  int numEmptyCells =
187  stripsPerArray_ - numRecHits; // only simulating for single array until
188  // all arrays are merged into one collection
189  std::vector<double> noiseHits_PE =
190  noiseGenerator_->generateNoiseHits(numEmptyCells);
191 
192  TrigScintID tempID;
193 
194  for (auto &noiseHitPE : noiseHits_PE) {
195 
196  TrigScintHit hit;
197  // generate random ID from remaining cells
198  do {
199  tempID = generateRandomID(module);
200  } while (Edep.find(tempID) != Edep.end() ||
201  noiseHitIDs.find(tempID) != noiseHitIDs.end());
202 
203  TrigScintID noiseID = tempID;
204 
205  noiseHitIDs.insert(noiseID);
206  hit.setID(noiseID.raw());
207  hit.setPE(noiseHitPE);
208  hit.setMinPE(noiseHitPE);
209  hit.setAmplitude(noiseHitPE);
210  hit.setEnergy(0.);
211  hit.setTime(0.);
212  hit.setXPos(0.);
213  hit.setYPos(0.);
214  hit.setZPos(0.);
215  hit.setModuleID(module);
216  hit.setBarID(noiseID.bar());
217  hit.setNoise(true);
218  hit.setBeamEfrac(0.);
219 
220  trigScintHits.push_back(hit);
221  }
222  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
223  // - -
224 
225  event.add(outputCollection_, trigScintHits);
226 }
