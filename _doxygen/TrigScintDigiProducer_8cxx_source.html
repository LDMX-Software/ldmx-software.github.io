47 void TrigScintDigiProducer::produce(framework::Event &event) {
48  // Need to handle seeding on the first event
49  if (!noiseGenerator_->hasSeed()) {
50  const auto &rseed = getCondition<framework::RandomNumberSeedService>(
51  framework::RandomNumberSeedService::CONDITIONS_OBJECT_NAME);
52  noiseGenerator_->seedGenerator(
53  rseed.getSeed("TrigScintDigiProducer::NoiseGenerator"));
54  }
55 
56  std::map<ldmx::TrigScintID, int> cellPEs;
57  std::map<ldmx::TrigScintID, int> cellMinPEs;
58  std::map<ldmx::TrigScintID, float> Xpos, Ypos, Zpos, Edep, Time, beamFrac;
59  std::set<ldmx::TrigScintID> noiseHitIDs;
60 
61  auto numRecHits{0};
62 
63  // looper over sim hits and aggregate energy depositions for each detID
64  const auto simHits{event.getCollection<ldmx::SimCalorimeterHit>(
65  inputCollection_, inputPassName_)};
66  auto particleMap{event.getMap<int, ldmx::SimParticle>("SimParticles")};
67 
68  int module{-1};
69  for (const auto &simHit : simHits) {
70  ldmx::TrigScintID id(simHit.getID());
71 
72  // Just set the module ID to use for noise hits here. Given that
73  // we are currently processing a single module at a time, setting
74  // it within the loop shouldn't matter.
75  module = id.module();
76  std::vector<float> position = simHit.getPosition();
77 
78  if (verbose_) {
79  std::cout << id << std::endl;
80  }
81 
82  unsigned int detIDRaw = id.raw();
83 
84  // check if hits is from beam electron and, if so, add to beamFrac
85  for (int i = 0; i < simHit.getNumberOfContribs(); i++) {
86  auto contrib = simHit.getContrib(i);
87  if (verbose_) {
88  std::cout << "contrib " << i << " trackID: " << contrib.trackID
89  << " pdgID: " << contrib.pdgCode << " edep: " << contrib.edep
90  << std::endl;
91  std::cout << "\t particle id: "
92  << particleMap[contrib.trackID].getPdgID()
93  << " particle status: "
94  << particleMap[contrib.trackID].getGenStatus() << std::endl;
95  }
96  if (particleMap[contrib.trackID].getPdgID() == 11 &&
97  particleMap[contrib.trackID].getGenStatus() == 1) {
98  if (beamFrac.find(id) == beamFrac.end())
99  beamFrac[id] = contrib.edep;
100  else
101  beamFrac[id] += contrib.edep;
102  }
103  }
104 
105  // for now, we take am energy weighted average of the hit in each stip to
106  // simulate the hit position. AJW: these should be dropped, they are likely
107  // to lead to a problem since we can't measure them anyway except roughly y
108  // and z, which is encoded in the ids.
109  if (Edep.find(id) == Edep.end()) {
110  // first hit, initialize
111  Edep[id] = simHit.getEdep();
112  Time[id] = simHit.getTime() * simHit.getEdep();
113  Xpos[id] = position[0] * simHit.getEdep();
114  Ypos[id] = position[1] * simHit.getEdep();
115  Zpos[id] = position[2] * simHit.getEdep();
116  numRecHits++;
117 
118  } else {
119  // not first hit, aggregate, and store the largest radius hit
120  Xpos[id] += position[0] * simHit.getEdep();
121  Ypos[id] += position[1] * simHit.getEdep();
122  Zpos[id] += position[2] * simHit.getEdep();
123  Edep[id] += simHit.getEdep();
124  // AJW: need to figure out a better way to model this...
125  Time[id] += simHit.getTime() * simHit.getEdep();
126  }
127  }
128 
129  // Create the container to hold the digitized trigger scintillator hits.
130  std::vector<ldmx::TrigScintHit> trigScintHits;
131 
132  // loop over detIDs and simulate number of PEs
133  int ihit = 0;
134  for (std::map<ldmx::TrigScintID, float>::iterator it = Edep.begin();
135  it != Edep.end(); ++it) {
136  ldmx::TrigScintID id(it->first);
137 
138  double depEnergy = Edep[id];
139  Time[id] = Time[id] / Edep[id];
140  Xpos[id] = Xpos[id] / Edep[id];
141  Ypos[id] = Ypos[id] / Edep[id];
142  Zpos[id] = Zpos[id] / Edep[id];
143  double meanPE = depEnergy / mevPerMip_ * pePerMip_;
144  cellPEs[id] = random_->Poisson(meanPE + meanNoise_);
145 
146  // If a cell has a PE count above threshold, persit the hit.
147  if (cellPEs[id] >= 1) {
148  ldmx::TrigScintHit hit;
149  hit.setID(id.raw());
150  hit.setPE(cellPEs[id]);
151  hit.setMinPE(cellMinPEs[id]);
152  hit.setAmplitude(cellPEs[id]);
153  hit.setEnergy(depEnergy);
154  hit.setTime(Time[id]);
155  hit.setXPos(Xpos[id]);
156  hit.setYPos(Ypos[id]);
157  hit.setZPos(Zpos[id]);
158  hit.setModuleID(module);
159  hit.setBarID(id.bar()); // getFieldValue("bar"));
160  hit.setNoise(false);
161  hit.setBeamEfrac(beamFrac[id] / depEnergy);
162 
163  trigScintHits.push_back(hit);
164  }
165 
166  if (verbose_) {
167  std::cout << id << std::endl;
168  std::cout << "Edep: " << Edep[id] << std::endl;
169  std::cout << "numPEs: " << cellPEs[id] << std::endl;
170  std::cout << "time: " << Time[id] << std::endl;
171  std::cout << "z: " << Zpos[id] << std::endl;
172  std::cout << "\t X: " << Xpos[id] << "\t Y: " << Ypos[id]
173  << "\t Z: " << Zpos[id] << std::endl;
174  } // end verbose
175  }
176 
177  // ------------------------------- Noise simulation -----------------------//
178  // ------------------------------------------------------------------------//
179  int numEmptyCells = stripsPerArray_ -
180  numRecHits; // only simulating for single array until
181  // all arrays are merged into one collection
182  std::vector<double> noiseHits_PE =
183  noiseGenerator_->generateNoiseHits(numEmptyCells);
184 
185  ldmx::TrigScintID tempID;
186 
187  for (auto &noiseHitPE : noiseHits_PE) {
188  ldmx::TrigScintHit hit;
189  // generate random ID from remaining cells
190  do {
191  tempID = generateRandomID(module);
192  } while (Edep.find(tempID) != Edep.end() ||
193  noiseHitIDs.find(tempID) != noiseHitIDs.end());
194 
195  ldmx::TrigScintID noiseID = tempID;
196 
197  noiseHitIDs.insert(noiseID);
198  hit.setID(noiseID.raw());
199  hit.setPE(noiseHitPE);
200  hit.setMinPE(noiseHitPE);
201  hit.setAmplitude(noiseHitPE);
202  hit.setEnergy(0.);
203  hit.setTime(0.);
204  hit.setXPos(0.);
205  hit.setYPos(0.);
206  hit.setZPos(0.);
207  hit.setModuleID(module);
208  hit.setBarID(noiseID.bar());
209  hit.setNoise(true);
210  hit.setBeamEfrac(0.);
211 
212  trigScintHits.push_back(hit);
213  }
214  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
215  // - -
216 
217  event.add(outputCollection_, trigScintHits);
218 }
