50  void TrigScintDigiProducer::produce(Event& event) {
51 
52  std::map<unsigned int, int> cellPEs;
53  std::map<unsigned int, int> cellMinPEs;
54  std::map<unsigned int, float> Xpos, Ypos, Zpos, Edep, Time, beamFrac;
55  std::unordered_set<unsigned int> noiseHitIDs;
56 
57  auto numRecHits{0};
58 
59  // looper over sim hits and aggregate energy depositions for each detID
60  const auto simHits{event.getCollection< SimCalorimeterHit >(inputCollection_,inputPassName_)};
61  auto particleMap{event.getMap< int, SimParticle >("SimParticles")};
62 
63  int module{-1};
64  for (const auto& simHit : simHits) {
65 
66  int detIDRaw{simHit.getID()};
67  detID_->setRawValue(detIDRaw);
68  detID_->unpack();
69 
70  // Just set the module ID to use for noise hits here. Given that
71  // we are currently processing a single module at a time, setting
72  // it within the loop shouldn't matter.
73  module = detID_->getFieldValue("module");
74  std::vector<float> position = simHit.getPosition();
75 
76  if (verbose_) {
77  std::cout << "Module: " << module
78  << " Bar: " << detID_->getFieldValue("bar")
79  << std::endl;
80  }
81 
82  // check if hits is from beam electron and, if so, add to beamFrac
83  for( int i = 0 ; i < simHit.getNumberOfContribs() ; i++){
84  auto contrib = simHit.getContrib(i);
85  if( verbose_ ){
86  std::cout << "contrib " << i << " trackID: " << contrib.trackID << " pdgID: " << contrib.pdgCode << " edep: " << contrib.edep << std::endl;
87  std::cout << "\t particle id: " << particleMap[contrib.trackID].getPdgID() << " particle status: " << particleMap[contrib.trackID].getGenStatus() << std::endl;
88  }
89  if( particleMap[contrib.trackID].getPdgID() == 11 && particleMap[contrib.trackID].getGenStatus() == 1 ){
90  if( beamFrac.find(detIDRaw) == beamFrac.end() )
91  beamFrac[detIDRaw]=contrib.edep;
92  else
93  beamFrac[detIDRaw]+=contrib.edep;
94  }
95  }
96 
97  // for now, we take am energy weighted average of the hit in each stip to simulate the hit position.
98  // AJW: these should be dropped, they are likely to lead to a problem since we can't measure them anyway
99  // except roughly y and z, which is encoded in the ids.
100  if (Edep.find(detIDRaw) == Edep.end()) {
101 
102  // first hit, initialize
103  Edep[detIDRaw] = simHit.getEdep();
104  Time[detIDRaw] = simHit.getTime() * simHit.getEdep();
105  Xpos[detIDRaw] = position[0]* simHit.getEdep();
106  Ypos[detIDRaw] = position[1]* simHit.getEdep();
107  Zpos[detIDRaw] = position[2]* simHit.getEdep();
108  numRecHits++;
109 
110  } else {
111 
112  // not first hit, aggregate, and store the largest radius hit
113  Xpos[detIDRaw] += position[0]* simHit.getEdep();
114  Ypos[detIDRaw] += position[1]* simHit.getEdep();
115  Zpos[detIDRaw] += position[2]* simHit.getEdep();
116  Edep[detIDRaw] += simHit.getEdep();
117  // AJW: need to figure out a better way to model this...
118  Time[detIDRaw] += simHit.getTime() * simHit.getEdep();
119  }
120 
121  }
122 
123  // Create the container to hold the digitized trigger scintillator hits.
124  std::vector< TrigScintHit > trigScintHits;
125 
126  // loop over detIDs and simulate number of PEs
127  int ihit = 0;
128  for (std::map<unsigned int, float>::iterator it = Edep.begin(); it != Edep.end(); ++it) {
129  int detIDRaw = it->first;
130  double depEnergy = Edep[detIDRaw];
131  Time[detIDRaw] = Time[detIDRaw] / Edep[detIDRaw];
132  Xpos[detIDRaw] = Xpos[detIDRaw] / Edep[detIDRaw];
133  Ypos[detIDRaw] = Ypos[detIDRaw] / Edep[detIDRaw];
134  Zpos[detIDRaw] = Zpos[detIDRaw] / Edep[detIDRaw];
135  double meanPE = depEnergy / mevPerMip_ * pePerMip_;
136  cellPEs[detIDRaw] = random_->Poisson(meanPE + meanNoise_);
137 
138 
139 
140  // If a cell has a PE count above threshold, persit the hit.
141  if( cellPEs[detIDRaw] >= 1 ){
142 
143  detID_->setRawValue(detIDRaw);
144  detID_->unpack();
145  auto bar{detID_->getFieldValue("bar")};
146 
147  TrigScintHit hit;
148  hit.setID(detIDRaw);
149  hit.setPE(cellPEs[detIDRaw]);
150  hit.setMinPE(cellMinPEs[detIDRaw]);
151  hit.setAmplitude(cellPEs[detIDRaw]);
152  hit.setEnergy(depEnergy);
153  hit.setTime(Time[detIDRaw]);
154  hit.setXpos(Xpos[detIDRaw]);
155  hit.setYpos(Ypos[detIDRaw]);
156  hit.setZpos(Zpos[detIDRaw]);
157  hit.setModuleID(module);
158  hit.setBarID(detID_->getBarID() ); //getFieldValue("bar"));
159  hit.setNoise(false);
160  hit.setBeamEfrac(beamFrac[detIDRaw]/depEnergy);
161 
162  trigScintHits.push_back(hit);
163  }
164 
165  if (verbose_) {
166  detID_->setRawValue(detIDRaw);
167  detID_->unpack();
168  auto bar{detID_->getFieldValue("bar")};
169 
170  std::cout << "Module: " << module << " Bar: " << bar << std::endl;
171  std::cout << "Edep: " << Edep[detIDRaw] << std::endl;
172  std::cout << "numPEs: " << cellPEs[detIDRaw] << std::endl;
173  std::cout << "time: " << Time[detIDRaw] << std::endl;std::cout << "z: " << Zpos[detIDRaw] << std::endl;
174  std::cout << "\t X: " << Xpos[detIDRaw] << "\t Y: " << Ypos[detIDRaw] << "\t Z: " << Zpos[detIDRaw] << std::endl;
175  } // end verbose
176  }
177 
178  // ------------------------------- Noise simulation -------------------------------
179  int numEmptyCells = stripsPerArray_ - numRecHits; // only simulating for single array until all arrays are merged into one collection
180  std::vector<double> noiseHits_PE = noiseGenerator_->generateNoiseHits( numEmptyCells );
181 
182  int tempID;
183 
184  for (auto& noiseHitPE : noiseHits_PE) {
185 
186  TrigScintHit hit;
187  // generate random ID from remaining cells
188  do {
189  tempID = generateRandomID(module);
190  } while( Edep.find(tempID) != Edep.end() ||
191  noiseHitIDs.find(tempID) != noiseHitIDs.end() );
192 
193  TrigScintID noiseID;
194  noiseID.setRawValue(tempID);
195  noiseID.unpack();
196 
197  hit.setID(tempID);
198  hit.setPE(noiseHitPE);
199  hit.setMinPE(noiseHitPE);
200  hit.setAmplitude(noiseHitPE);
201  hit.setEnergy(0.);
202  hit.setTime(0.);
203  hit.setXpos(0.);
204  hit.setYpos(0.);
205  hit.setZpos(0.);
206  hit.setModuleID(module);
207  hit.setBarID(noiseID.getFieldValue("bar"));
208  hit.setNoise(true);
209  hit.setBeamEfrac(0.);
210 
211  trigScintHits.push_back(hit);
212  }
213  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
214 
215  event.add(outputCollection_, trigScintHits);
216  }
217 }
