17 if (track->GetKineticEnergy() < threshold_) return 0;
18
19 std::string currentProcess =
20 callingProcess->GetWrappedProcess()->GetProcessName();

22 G4double interactionLength =
23 callingProcess->GetWrappedProcess()->GetCurrentInteractionLength();
24
25 double neutInXsecUnbiased = 1. / interactionLength;
26

28

30 } else
31 return 0;
32}
