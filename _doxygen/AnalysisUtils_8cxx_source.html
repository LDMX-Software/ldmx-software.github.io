35const ldmx::SimParticle* getPNGamma(
36 const std::map<int, ldmx::SimParticle>& particleMap,
37 const ldmx::SimParticle* recoil, const float& energyThreshold) {
38 // Get all of the daughter track IDs
39 auto daughterTrackIDs{recoil->getDaughters()};
40
41 auto pit = std::find_if(
42 daughterTrackIDs.begin(), daughterTrackIDs.end(),
43 [energyThreshold, particleMap](const int& id) {
44 // Get the SimParticle from the map
45 auto daughter{particleMap.at(id)};
46
47 // If the particle doesn't have any daughters, return false
48 if (daughter.getDaughters().size() == 0
49 or
50 particleMap.find(daughter.getDaughters().front()) != particleMap.end()
51 ) return false;
52
53 // If the particle has daughters that were a result of a
54 // photo-nuclear reaction, and its energy is above threshold,
55 // then tag it as the PN gamma.
56 return (
57 (particleMap.at(daughter.getDaughters().front()).getProcessType() ==
58 ldmx::SimParticle::ProcessType::photonNuclear) &&
59 (daughter.getEnergy() >= energyThreshold));
60 });
61
62 // If a PN daughter was found, return it.
63 if (pit != daughterTrackIDs.end()) return &particleMap.at(*pit);
64
65 return nullptr;
66}
