49Acts::CuboidVolumeBuilder::VolumeConfig
50TrackersTrackingGeometry::buildRecoilVolume() {
51 Acts::CuboidVolumeBuilder::VolumeConfig subDetVolumeConfig;
52 Acts::Transform3 subDet_transform = GetTransform(*Recoil_, true);
53 if (debug_) {
54 std::cout << subDet_transform.translation() << std::endl;
55 std::cout << subDet_transform.rotation() << std::endl;
56 }
57
58 // Add 1mm to not make it sit on the first layer surface - Ask Omar if it's
59 // OK
60 Acts::Vector3 sub_det_position = {
61 subDet_transform.translation()(0) - 1,
62 subDet_transform.translation()(1),
63 subDet_transform.translation()(2),
64 };
65
66 // Get the size of the volume
67 G4Box* subDetBox = (G4Box*)(Recoil_->GetLogicalVolume()->GetSolid());
68
69 // In tracker coordinates. I add 1mm so that it compensates with the 1mm
70 // movement of above
71 double x_length =
72 2 * (subDetBox->GetZHalfLength() + 1) * Acts::UnitConstants::mm;
73
74 // double y_length = 2*subDetBox->GetXHalfLength() * Acts::UnitConstants::mm;
75 // double z_length = 2*subDetBox->GetYHalfLength() * Acts::UnitConstants::mm;
76
77 // The bField is defined only between -70, 70 along ACTS Z Axis (-130,130 in
78 // the extended version)
79 double y_length = TrackerYLength_;
80 double z_length = TrackerZLength_;
81
82 if (debug_) {
83 std::cout << Recoil_->GetName() << std::endl;
84 std::cout << "position" << std::endl;
85 std::cout << sub_det_position << std::endl;
86 std::cout << "x_length " << x_length << " y_length " << y_length
87 << " z_length " << z_length << std::endl;
88 }
89
90 subDetVolumeConfig.position = sub_det_position;
91 subDetVolumeConfig.length = {x_length, y_length, z_length};
92 subDetVolumeConfig.name = "Recoil";
93
94 // Vacuum material
95 Acts::Material subdet_mat = Acts::Material();
96 subDetVolumeConfig.volumeMaterial =
97 std::make_shared<Acts::HomogeneousVolumeMaterial>(subdet_mat);
98
99 std::vector<Acts::CuboidVolumeBuilder::LayerConfig> layerConfig;
100
101 // Prepare the layers
102 for (auto& layer : recoil_layout) {
103 if (debug_) {
104 std::cout << layer.first << " : surfaces==>" << layer.second.size()
105 << std::endl;
106 }
107
108 Acts::CuboidVolumeBuilder::LayerConfig lcfg;
109 lcfg.surfaces = layer.second;
110 // Get the surface thickness
111 double clearance = 0.01;
112 double thickness = layer.second.front()
113 ->surfaceMaterial()
114 ->materialSlab(Acts::Vector2{0., 0.})
115 .thickness();
116
117 // std::cout<<"Sensor Thickness from Material slab "<< thickness<<std::endl;
118
119 lcfg.envelopeX = std::pair<double, double>{thickness / 2. + clearance,
120 thickness / 2. + clearance};
121 lcfg.active = true;
122 layerConfig.push_back(lcfg);
123 }
124
125 subDetVolumeConfig.layerCfg = layerConfig;
126
127 return subDetVolumeConfig;
128}

215 std::string surfacename) {
216 if (debug_)
217 std::cout << "Building layout for the " << pvol->GetName() << " tracker"
218 << std::endl;
219
220 // Get the global transform
221 Acts::Transform3 tracker_transform = GetTransform(*pvol);
222
223 G4LogicalVolume* l_vol = pvol->GetLogicalVolume();
224 for (G4int i = 0; i < l_vol->GetNoDaughters(); i++) {
225 std::string sln = l_vol->GetDaughter(i)->GetName();
226 if (sln.find(surfacename) != std::string::npos) {
227 getAllDaughters(l_vol->GetDaughter(i));
228
229 G4VPhysicalVolume* _Component0Volume{nullptr};
230 G4VPhysicalVolume* _ActiveSensor{nullptr};
231 Acts::Transform3 ref1_transform = GetTransform(*(l_vol->GetDaughter(i)));
232
233 // recoil_l1(4)_axial(stereo)->LDMXRecoilL14ModuleVolume_component0_physvol
234
235 if (sln.find("axial") != std::string::npos ||
236 sln.find("stereo") != std::string::npos) {
237 _Component0Volume =
238 findDaughterByName(l_vol->GetDaughter(i),
239 "LDMXRecoilL14ModuleVolume_component0_physvol");
240 if (!_Component0Volume)
241 throw std::runtime_error(
242 "Could not find component0 volume for L14 Recoil");
243 _ActiveSensor = findDaughterByName(
244 _Component0Volume,
245 "LDMXRecoilL14ModuleVolume_component0Sensor0_physvol");
246
247 }
248
249 // recoil_l5_sensorX->LDMXRecoilL56ModuleVolume_component0_physvol
250
251 else if (sln.find("l5") != std::string::npos ||
252 sln.find("l6") != std::string::npos) {
253 _Component0Volume =
254 findDaughterByName(l_vol->GetDaughter(i),
255 "LDMXRecoilL56ModuleVolume_component0_physvol");
256 if (!_Component0Volume)
257 throw std::runtime_error(
258 "Could not find component0 volume for L56 Recoil");
259 _ActiveSensor = findDaughterByName(
260 _Component0Volume,
261 "LDMXRecoilL56ModuleVolume_component0Sensor0_physvol");
262 }
263
264 else
265 throw std::runtime_error("Could not build recoil layout");
266
267 if (!_ActiveSensor)
268 throw std::runtime_error(
269 "Could not find ActiveSensor for recoil volume");
270
271 Acts::Transform3 ref2_transform = GetTransform(*(_Component0Volume));

273 _ActiveSensor, tracker_transform * ref1_transform * ref2_transform);

275 // Build the layout

285 if (sln == "recoil_l4_axial" || sln == "recoil_l4_stereo")

309void TrackersTrackingGeometry::BuildTaggerLayoutMap(G4VPhysicalVolume* pvol,
310 std::string surfacename) {
311 if (debug_)
312 std::cout << "Building layout for the " << pvol->GetName() << " tracker"
313 << std::endl;
314
315 // Get the global transform
316 Acts::Transform3 tracker_transform = GetTransform(*pvol);
317
318 G4LogicalVolume* l_vol = pvol->GetLogicalVolume();
319 for (G4int i = 0; i < l_vol->GetNoDaughters(); i++) {
320 std::string sln = l_vol->GetDaughter(i)->GetName();
321
322 // To distinguish which layers need to be selected
323 if (sln.find(surfacename) != std::string::npos) {
324 // Box for the module (slightly bigger than the sensor)
325 // LDMXTaggerModuleVolume_physvol -> LDMXTaggerModuleVolume_component0Box,
326 // this is the sensor + inactive region
327 // LDMXTaggerModuleVolume_component0Box->
328 // LDMXTaggerModuleVolume_component0Sensor0Box, this is the sensor itself
329 // (active region)
330
331 // Tagger_ -> LDMXTaggerModuleVolume_physvol1 ->
332 // LDMXTaggerModuleVolume_component0_physvol
333 // ->LDMXTaggerModuleVolume_component0Sensor0_physvol
334 // -> Get Box for the dimension: GetLogical->GetSolid
335 // For the positioning of the sensor:
336 // A position of physvol1 + position of the component0_physvol +
337 // position of the sensor physvol. Thickness I can grab it from the box,
338 // or I just hardcode it. S transform1 * transform2 * transform3
339 // ....
340
341 // Get the sensor volume placement
342 Acts::Transform3 ref1_transform = GetTransform(*(l_vol->GetDaughter(i)));
343
344 G4VPhysicalVolume* _Component0Volume = findDaughterByName(
345 l_vol->GetDaughter(i), "LDMXTaggerModuleVolume_component0_physvol");
346 if (!_Component0Volume)
347 throw std::runtime_error("Missing component0_physvol from physvol");
348 // Get Component0 transform
349 Acts::Transform3 ref2_transform = GetTransform(*(_Component0Volume));
350
351 G4VPhysicalVolume* _ActiveSensor = findDaughterByName(
352 _Component0Volume,
353 "LDMXTaggerModuleVolume_component0Sensor0_physvol");
354 if (!_ActiveSensor) {
355 throw std::runtime_error(
356 "Could not find the ActiveSensor from Component0Volume");
357 }
358
359 // Get the surface
360 std::shared_ptr<Acts::PlaneSurface> sensorSurface = GetSurface(
361 _ActiveSensor, tracker_transform * ref1_transform * ref2_transform);
362 if (sln == "LDMXTaggerModuleVolume_physvol1" ||
363 sln == "LDMXTaggerModuleVolume_physvol2")
364 tagger_layout["tagger_tracker_L1"].push_back(sensorSurface);
365
366 if (sln == "LDMXTaggerModuleVolume_physvol3" ||
367 sln == "LDMXTaggerModuleVolume_physvol4")
368 tagger_layout["tagger_tracker_L2"].push_back(sensorSurface);
369
370 if (sln == "LDMXTaggerModuleVolume_physvol5" ||
371 sln == "LDMXTaggerModuleVolume_physvol6")
372 tagger_layout["tagger_tracker_L3"].push_back(sensorSurface);
373
374 if (sln == "LDMXTaggerModuleVolume_physvol7" ||
375 sln == "LDMXTaggerModuleVolume_physvol8")
376 tagger_layout["tagger_tracker_L4"].push_back(sensorSurface);
377
378 if (sln == "LDMXTaggerModuleVolume_physvol9" ||
379 sln == "LDMXTaggerModuleVolume_physvol10")
380 tagger_layout["tagger_tracker_L5"].push_back(sensorSurface);
381
382 if (sln == "LDMXTaggerModuleVolume_physvol11" ||
383 sln == "LDMXTaggerModuleVolume_physvol12")
384 tagger_layout["tagger_tracker_L6"].push_back(sensorSurface);
385
386 if (sln == "LDMXTaggerModuleVolume_physvol13" ||
387 sln == "LDMXTaggerModuleVolume_physvol14")
388 tagger_layout["tagger_tracker_L7"].push_back(sensorSurface);
389
390 } // found a silicon surface
391 } // loop on daughters
392} // build the layout

394std::shared_ptr<Acts::PlaneSurface> TrackersTrackingGeometry::GetSurface(
395 G4VPhysicalVolume* pvol, Acts::Transform3 ref_trans) {
396 if (!pvol)
397 throw std::runtime_error(
398 "TrackersTrackingGeometry::GetSurface:: pvol is nullptr");
399
400 // Get the surface transform
401 Acts::Transform3 surface_transform = GetTransform(*pvol);
402 // Compose the sensor_transform with the reference transform
403 surface_transform = ref_trans * surface_transform;
404
405 // Now transform to the tracker frame
406 Acts::Transform3 surface_transform_tracker = toTracker(surface_transform);
407
408 if (debug_) {
409 std::cout << "THE SENSOR TRANSFORM - TRANSLATION" << std::endl;
410 std::cout << surface_transform.translation()(0) << std::endl;
411 std::cout << surface_transform.translation()(1) << std::endl;
412 std::cout << surface_transform.translation()(2) << std::endl;
413 std::cout << "THE SENSOR TRANSFORM - ROTATION" << std::endl;
414 std::cout << surface_transform.rotation() << std::endl;
415
416 std::cout << "TO THE TRACKER FRAME" << std::endl;
417 std::cout << surface_transform_tracker.translation()(0) << std::endl;
418 std::cout << surface_transform_tracker.translation()(1) << std::endl;
419 std::cout << surface_transform_tracker.translation()(2) << std::endl;
420 std::cout << "THE SENSOR TRANSFORM - ROTATION" << std::endl;
421 std::cout << surface_transform_tracker.rotation() << std::endl;
422 }
423
424 // This material is defined in different units with respect what acts expects.
425 // I decided to hardcode here. TODO: fix this
426
427 /*

429 if (debug_) {

434 std::cout<<"RL="<<sens_mat->GetRadlen()<<"

437 rho="<<sens_mat->GetDensity()<<std::endl;

447
448 // Define the silicon material
449 Acts::Material silicon = Acts::Material::fromMassDensity(
450 95.7 * Acts::UnitConstants::mm, 465.2 * Acts::UnitConstants::mm, 28.03,
451 14., 2.32 * Acts::UnitConstants::g / Acts::UnitConstants::cm3);
452
453 // Get the active sensor box
454 G4Box* surfaceSolid = (G4Box*)(pvol->GetLogicalVolume()->GetSolid());
455
456 if (debug_) {
457 std::cout << "Sensor Dimensions" << std::endl;
458 std::cout << surfaceSolid->GetXHalfLength() << " "
459 << surfaceSolid->GetYHalfLength() << " "
460 << surfaceSolid->GetZHalfLength() << " " << std::endl;
461 }
462
463 // Form the material slab
464 double thickness =
465 2 * surfaceSolid->GetZHalfLength() * Acts::UnitConstants::mm;
466 Acts::MaterialSlab silicon_slab(silicon, thickness);
467
468 // Get the bounds
469 std::shared_ptr<const Acts::RectangleBounds> rect_bounds =
470 std::make_shared<const Acts::RectangleBounds>(Acts::RectangleBounds(
471 surfaceSolid->GetXHalfLength() * Acts::UnitConstants::mm,
472 surfaceSolid->GetYHalfLength() * Acts::UnitConstants::mm));
473
474 // Form the active sensor surface
475 std::shared_ptr<Acts::PlaneSurface> surface =
476 Acts::Surface::makeShared<Acts::PlaneSurface>(surface_transform_tracker,
477 rect_bounds);
478 surface->assignSurfaceMaterial(
479 std::make_shared<Acts::HomogeneousSurfaceMaterial>(silicon_slab));
480
481 return surface;
