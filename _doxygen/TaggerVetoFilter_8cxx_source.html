20void TaggerVetoFilter::stepping(const G4Step* step) {
21 // Get the track associated with this step
22 auto track{step->GetTrack()};
23
24 // Only process the primary electron track
25 if (track->GetParentID() != 0) return;
26
27 // Get the PDG ID of the track and make sure it's an electron. If
28 // another particle type is found, thrown an exception.
29 if (auto pdgID{track->GetParticleDefinition()->GetPDGEncoding()}; pdgID != 11)
30 return;
31
32 // Get the region the particle is currently in. Continue processing
33 // the particle only if it's in the tagger region.
34 if (auto region{
35 track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
36 region.compareTo("tagger") != 0)
37 return;
38
39 // If the energy of the particle falls below threshold, stop
40 // processing the event.
41 if (auto energy{step->GetPostStepPoint()->GetTotalEnergy()};
42 energy < threshold_) {
43 track->SetTrackStatus(fKillTrackAndSecondaries);
44 G4RunManager::GetRunManager()->AbortEvent();
45 return;
46 }
47}
