20  void TaggerVetoFilter::stepping(const G4Step* step) {
21 
22  // Get the track associated with this step
23  auto track{step->GetTrack()};
24 
25  // Only process the primary electron track
26  if (track->GetParentID() != 0) return;
27 
28  // Get the PDG ID of the track and make sure it's an electron. If
29  // another particle type is found, thrown an exception.
30  if (auto pdgID{track->GetParticleDefinition()->GetPDGEncoding()}; pdgID != 11) return;
31 
32  // Get the region the particle is currently in. Continue processing
33  // the particle only if it's in the tagger region.
34  if (auto region{track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
35  region.compareTo("tagger") != 0) return;
36 
37  // If the energy of the particle falls below threshold, stop
38  // processing the event.
39  if (auto energy{step->GetPostStepPoint()->GetTotalEnergy()}; energy < threshold_) {
40  track->SetTrackStatus(fKillTrackAndSecondaries);
41  G4RunManager::GetRunManager()->AbortEvent();
42  return;
43  }
44  }
