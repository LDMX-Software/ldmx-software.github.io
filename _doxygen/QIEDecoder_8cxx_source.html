9 void QIEDecoder::configure(
10 framework::config::Parameters &ps) {
11 // Configure this instance of the encoder
12 outputCollection_ = ps.getParameter<std::string>("output_collection");
13 inputCollection_ = ps.getParameter<std::string>("input_collection");
14 inputPassName_ = ps.getParameter<std::string>("input_pass_name");
15 channelMapFileName_ = ps.getParameter<std::string>("channel_map_file");
16 nChannels_ = ps.getParameter<int>("number_channels");
17 nSamples_ = ps.getParameter<int>("number_time_samples");
18 isRealData_ = ps.getParameter<bool>("is_real_data");
19 verbose_ = ps.getParameter<bool>("verbose");
20
21 ldmx_log(debug) << "In configure, got parameters:" <<
22 "\noutput_collection = " << outputCollection_ <<
23 "\ninput_collection = " << inputCollection_ <<
24 "\ninput_pass_name = " << inputPassName_ <<
25 "\nchannel_map_file = " << channelMapFileName_ <<
26 "\nnumber_channels = " << nChannels_ <<
27 "\nnumber_time_samples = " << nSamples_ <<
28 "\nis_real_data = " << isRealData_ <<
29 "\nverbose = " << verbose_ ;
30
31 channelMapFile_.open(channelMapFileName_, std::ios::in);
32 if (! channelMapFile_.is_open() ){
33 EXCEPTION_RAISE("BadMapFile", "The channel mapping file cannot be opened."); // <-- appears this needs implementing first
34 ldmx_log(fatal) << "The channel mapping file cannot be opened.";
35 return;
36 }
37 int chID, elID;
38 while (! channelMapFile_.eof() ){
39 channelMapFile_ >> elID >> chID ;
40 // make the map based on electronics ID, to look up channel ID.
41 // the reason is, we will only know the elecID from the position
42 // of the word in the stream. so these need to be strictly ordered.
43 // writing the right bar in the right position is easier if we can just
44 // read this map from beginning to end.
45 // barID can always be set, or looked up, as a property of the digi.
46
47 // here make the elecID the key (other way around when encoding)
48 channelMap_.insert(std::pair<int,int>(elID, chID));
49 ldmx_log(debug) << elID << " chID " << chID ;
50 }
51 channelMapFile_.close();
52 if (elID != nChannels_ - 1 )
53 ldmx_log(fatal) << "The set number of channels " << nChannels_ << " seems not to match the number from the map (+1) :" << elID;
54 return;
55
56 }

58 void QIEDecoder::produce(framework::Event &event) {
59 ldmx_log(debug)
60 << "QIEDecoder: produce() starts! Event number: "
61 << event.getEventHeader().getEventNumber();
62
63 // ldmx::EventHeader *eh = (ldmx::EventHeader*)event.getEventHeaderPtr();
64 //eh->setIsRealData(isRealData_); //doesn't help
65
66 /* // comment for now, requires pushing change ot EventHeader
67 event.getEventHeader().setIsRealData(isRealData_);
68 ldmx_log(debug) << "Decoder bool isRealData = " << isRealData_ << " and after attempt of setting it, event header returns " << event.getEventHeader().isRealData();
69 */
70 //turns out this need to be configurable for now, to read real data
71 int nSamp = nSamples_; //QIEStream::NUM_SAMPLES ;
72 ldmx_log(debug) << "num samples = " << nSamp;
73
74 ldmx_log(debug) << "Looking up input collection " << inputCollection_ << "_" << inputPassName_;
75 const auto eventStream{event.getCollection<uint8_t>( inputCollection_, inputPassName_)};
76 ldmx_log(debug) << "Got input collection" << inputCollection_ << "_" << inputPassName_;
77
78
79 uint32_t timeEpoch=0;
80 //these don't have to be in any particular order, position is anyway looked up from definition in header
81 for (int iW = 0 ; iW < QIEStream::TIMESTAMP_LEN_BYTES; iW++) {
82 int pos = QIEStream::TIMESTAMP_POS + iW;
83 uint8_t timeWord = eventStream.at( pos );
84 ldmx_log(debug) << "time stamp word at position " << pos << " (with iW = " << iW << ") = " << std::bitset<8>(timeWord ) ;
85 timeEpoch |= (timeWord << iW*8); //shift by a byte at a time
86 }
87
88 uint32_t timeClock=0;
89 for (int iW = 0 ; iW < QIEStream::TIMESTAMPCLOCK_LEN_BYTES; iW++) {
90 int pos = QIEStream::TIMESTAMPCLOCK_POS + iW;
91 uint8_t timeWord = eventStream.at( pos );
92 ldmx_log(debug) << "time stamp ns word at position " << pos << " (with iW = " << iW << ") = " << std::bitset<8>(timeWord ) ;
93 timeClock |= (timeWord << iW*8); //shift by a byte at a time
94 }
95
96 uint32_t timeSpill=0;
97 ldmx_log(debug) << "Before starting, timeSpill = " << timeSpill << " (" << std::bitset<64>(timeSpill) << ", or, " << std::hex << timeSpill << std::dec << ") counts since start of spill";
98
99 for (int iW = 0 ; iW < QIEStream::TIMESINCESPILL_LEN_BYTES; iW++) {
100 int pos = QIEStream::TIMESINCESPILL_POS + iW;
101 uint8_t timeWord = eventStream.at( pos );
102 ldmx_log(debug) << "time since spill word at position " << pos << " (with iW = " << iW << ") = " << std::bitset<8>(timeWord ) ;
103 timeSpill |= (timeWord << iW*8); //shift by a byte at a time
104 }
105 ldmx_log(debug) << "time stamp words are : " << timeEpoch << " (" << std::bitset<64>(timeEpoch) << ") and " << timeClock << " (" << std::bitset<64>(timeClock) << ") clock ticks, and " << timeSpill << " (" << std::bitset<64>(timeSpill) << ", or, " << std::hex << timeSpill << std::dec << ") counts since start of spill";
106
107 int sigBitsSkip=6; //the first 6 bits are part of something else.
108 int divisor=TMath::Power(2, 32-sigBitsSkip);
109 timeSpill=timeSpill%divisor; //remove them by taking remainder in division by the values of the last skipped bit
110 //timeSpill=timeSpill/spillTimeConv_;
111 ldmx_log(debug) << "After taking it mod 2^" << 32-sigBitsSkip << " (which is " << divisor << ", spill time is " << timeSpill;
112 event.getEventHeader().setIntParameter("timeSinceSpill", timeSpill);
113 // event.getEventHeader().setTimeSinceSpill(timeSpill); //not working
114
115 TTimeStamp *timeStamp = new TTimeStamp(timeEpoch);
116 // timeStamp->SetNanoSec(timeClock); //not sure about this one...
117 event.getEventHeader().setTimestamp(*timeStamp);
118
119 // trigger ID event number
120 uint32_t triggerID =0;
121
122 for (int iW = 0 ; iW < QIEStream::TRIGID_LEN_BYTES; iW++) {
123 //assume the whole 3B are written as a single 24 bits word
124 int pos = QIEStream::TRIGID_POS+iW;
125 uint8_t tIDword = eventStream.at( pos );
126 ldmx_log(debug) << "trigger word at position " << pos << " (with iW = " << iW << ") = " << std::bitset<8>(tIDword ) ;
127 triggerID |= (tIDword << iW*8); //shift by a byte at a time
128 }
129
130 // ldmx_log(debug) << " got triggerID " << std::bitset<16>(triggerID) ; //eventStream.at(0);
131 ldmx_log(debug) << " got triggerID " << std::bitset<32>(triggerID) ; //eventStream.at(0);
132
133 if ( triggerID != event.getEventHeader().getEventNumber() ) {
134 // this probably only applies to digi emulation,
135 // unless an event number is explicitly set in unpacking
136 ldmx_log(fatal) << "Got event number mismatch: framework reports " <<
137 event.getEventHeader().getEventNumber() <<", stream says " << triggerID ;
138 }
139
140 // error word
141 /* the error word contains
142 - 4 trailing reserved 0's, for now
143 - isCIDunsync : if there is a mismatch between CID reported by channels within the same time sample
144 - isCIDskipped : if there is a gap in the CID increment of a channel beweeen samples
145 - isCRC0malformed : if there was an issue with CRC from fiber0
146 - isCRC1malformed : if there was an issue with CRC from fiber1
147 */
148 uint8_t flags = eventStream.at(QIEStream::ERROR_POS);
149
150 bool isCIDskipped { (flags >> QIEStream::CID_SKIP_POS) & mask8<QIEStream::FLAG_SIZE_BITS>::m};
151 bool isCIDunsync { (flags >> QIEStream::CID_UNSYNC_POS) & mask8<QIEStream::FLAG_SIZE_BITS>::m};
152 bool isCRC1malformed{ (flags >> QIEStream::CRC1_ERR_POS) & mask8<QIEStream::FLAG_SIZE_BITS>::m};
153 bool isCRC0malformed{ (flags >> QIEStream::CRC0_ERR_POS) & mask8<QIEStream::FLAG_SIZE_BITS>::m};
154 //checksum
155 uint8_t referenceChecksum = 0; // really, this is just empty for now. TODO> implement a checksum set/get
156 int checksum{ (flags >> QIEStream::CHECKSUM_POS) & mask8<QIEStream::CHECKSUM_SIZE_BITS>::m}; //eventStream.at(QIEStream::CRC0_ERR_POS) QIEStream::CHECKSUM_POS);
157 if ( checksum != referenceChecksum)
158 ldmx_log(fatal) << "Got checksum mismatch: expected " << (int)referenceChecksum << ", stream says " << checksum ;
159 if (isCIDunsync)
160 ldmx_log(debug) << "Found unsynced CIDs!";
161 if (isCIDskipped)
162 ldmx_log(fatal) << "Found skipped CIDs!" ;
163
164 /* -- TS event header done; read the channel contents -- */
165 std::vector<trigscint::TrigScintQIEDigis> outDigis;
166 std::map < int, std::vector<int>> ADCmap;
167 std::map < int, std::vector<int>> TDCmap;
168
169 // read in words from the stream. line them up per channel and time sample.
170 // channels are in the electronics ordering
171 int iWstart = std::max( std::max(QIEStream::ERROR_POS, QIEStream::CHECKSUM_POS),
172 QIEStream::TRIGID_POS+(QIEStream::TRIGID_LEN_BYTES)) +1; //make sure we're at end of header
173 int nWords = nSamp*nChannels_*2 + iWstart; //1 ADC, 1 TDC per channel per sample, + the words in the header
174 int iWord = iWstart;
175 ldmx_log(debug) << "Event parsing starts at vector idx " << iWstart << " and nWords = " << nWords;
176 // outer loop: over nSamples
177 // inner loop over nChannels to get ADCs, then repeat to get TDCs
178 for (int iS = 0; iS < nSamp; iS++) {
179 for (int iQ = 0; iQ < nChannels_ ; iQ++) {
180 if ( iWord >= nWords ) {
181 ldmx_log(fatal) << "More words than expected! Breaking ADC loop in sample " << iS << " at iQ = " << iQ;
182 break;
183 }
184 uint8_t val = eventStream.at(iWord);
185 if (val > 0) { //add only the digis with non-zero ADC value
186 ldmx_log(debug) << "got ADC value " << (unsigned)val << " at channel (elec) idx " << iQ ;
187 if ( ADCmap.find( iQ ) == ADCmap.end() ) { // we have a new channel
188 std::vector <int > adcs(nSamp, 0);
189 ADCmap.insert(std::pair<int, std::vector <int> >(iQ, adcs));
190 }
191 ADCmap[ iQ ].at(iS) = val;
192 }
193 iWord++;
194 }
195 for (int iQ = 0; iQ < nChannels_ ; iQ++) {
196 if ( iWord >= nWords ) {
197 ldmx_log(debug) << "More words than expected! Breaking TDC loop in sample " << iS << " at iQ = " << iQ;
198 break;
199 }
200 uint8_t val = eventStream.at(iWord);
201 if (val > 0) { // TODO: check if this channel is also present in ADC map? in the end?
202 ldmx_log(debug) << "got TDC value " << (unsigned)val << " at channel (elec) idx " << iQ ;;
203 if ( TDCmap.find( iQ ) == TDCmap.end() ) { // we have a new channel
204 std::vector <int > tdcs(nSamp, 0);
205 TDCmap.insert(std::pair<int, std::vector <int> >(iQ, tdcs));
206 }
207 //this is LETDC; only the two most significant bits included
208 // they are shipped as least significant bits --> shift them
209 TDCmap[ iQ ].at(iS) = (val+1)*16; // want LE TDC = 3 to correspond to 64 > 49 (which is maxTDC in sim)
210 }
211 iWord++;
212 }
213 ldmx_log(debug) << "Done with sample " << iS ;
214
215 }
216
217 ldmx_log(debug) << "Done reading in header, ADC and TDC for event " << (unsigned)triggerID ;
218 for (std::map<int, std::vector<int>>::iterator itr = ADCmap.begin(); itr != ADCmap.end(); ++itr) {
219 TrigScintQIEDigis digi;
220 digi.setADC( itr->second );
221 if (channelMap_.find( itr->first ) == channelMap_.end() ) {
222 ldmx_log(fatal) << "Couldn't find the bar ID corresponding to electronics ID " << itr->first << "!! Skipping." ;
223 continue;
224 }
225 int bar = channelMap_[itr->first];
226 digi.setElecID( itr->first );
227 digi.setChanID( bar );
228 digi.setTDC( TDCmap[itr->first] );
229 digi.setTimeSinceSpill(timeSpill);
230 if (bar == 0)
231 ldmx_log(debug) << "for bar 0, got time since spill "<< digi.getTimeSinceSpill();
232 outDigis.push_back( digi );
233 ldmx_log(debug) << "Iterator points to key " << itr->first
234 << " and mapped channel supposedly is " << bar ;
235 ldmx_log(debug) << "Made digi with elecID = " << digi.getElecID()
236 << ", barID = " << digi.getChanID()
237 << ", third adc value " << digi.getADC().at(2)
238 << " and third tdc " << digi.getTDC().at(2) ;
239 }
240
241 event.add(outputCollection_, outDigis);
242 }
