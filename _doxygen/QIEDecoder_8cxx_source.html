10 void QIEDecoder::configure(
11 framework::config::Parameters &ps) {
12 // Configure this instance of the encoder
13 outputCollection_ = ps.getParameter<std::string>("output_collection");
14 inputCollection_ = ps.getParameter<std::string>("input_collection");
15 inputPassName_ = ps.getParameter<std::string>("input_pass_name");
16 channelMapFileName_ = ps.getParameter<std::string>("channel_map_file");
17 nChannels_ = ps.getParameter<int>("number_channels");
18 verbose_ = ps.getParameter<bool>("verbose");
19
20 ldmx_log(debug) << "In configure, got parameters:" <<
21 "\noutput_collection = " << outputCollection_ <<
22 "\ninput_collection = " << inputCollection_ <<
23 "\ninput_pass_name = " << inputPassName_ <<
24 "\nchannel_map_file = " << channelMapFileName_ <<
25 "\nnumber_channels = " << nChannels_ <<
26 "\nverbose = " << verbose_ ;
27
28 channelMapFile_.open(channelMapFileName_, std::ios::in);
29 if (! channelMapFile_.is_open() ){
30 EXCEPTION_RAISE("BadMapFile", "The channel mapping file cannot be opened."); // <-- appears this needs implementing first
31 ldmx_log(fatal) << "The channel mapping file cannot be opened.";
32 return;
33 }
34 int chID, elID;
35 while (! channelMapFile_.eof() ){
36 channelMapFile_ >> elID >> chID ;
37 // make the map based on electronics ID, to look up channel ID.
38 // the reason is, we will only know the elecID from the position
39 // of the word in the stream. so these need to be strictly ordered.
40 // writing the right bar in the right position is easier if we can just
41 // read this map from beginning to end.
42 // barID can always be set, or looked up, as a property of the digi.
43
44 // here make the elecID the key (other way around when encoding)
45 channelMap_.insert(std::pair<int,int>(elID, chID));
46 // channelMap_[ chID ] = elID;
47 ldmx_log(debug) << elID << " chID " << chID ;
48 }
49 channelMapFile_.close();
50 if (elID != nChannels_ - 1 )
51 ldmx_log(fatal) << "The set number of channels " << nChannels_ << " seems not to match the number from the map (+1) :" << elID;
52 return;
53
54 }

56 void QIEDecoder::produce(framework::Event &event) {
57 ldmx_log(debug)
58 << "QIEDecoder: produce() starts! Event number: "
59 << event.getEventHeader().getEventNumber();
60
61 std::vector <trigscint::QIEStream> qieOuts;
62 int nSamp = QIEStream::NUM_SAMPLES ;
63 std::vector < int > initVec(nSamp, 0);
64
65 ldmx_log(debug) << "num samples = " << nSamp;
66
67 for (int iQ = 0; iQ < nChannels_ ; iQ++) {
68 QIEStream qieOut;
69 qieOut.setADC(initVec);
70 qieOut.setTDC(initVec);
71 qieOut.setCID(initVec);
72 qieOuts.push_back(qieOut);
73 }
74
75 ldmx_log(debug) << "Looking up input collection " << inputCollection_ << "_" << inputPassName_;
76 const auto eventStream{event.getCollection<int>( inputCollection_, inputPassName_)};
77 ldmx_log(debug) << "Got input collection" << inputCollection_ << "_" << inputPassName_;
78
79 // trigger IDevent number
80 uint16_t triggerID =eventStream.at(QIEStream::TRIGID_POS);
81 ldmx_log(debug) << " got triggerID " << eventStream.at(0);
82 if ( triggerID != event.getEventHeader().getEventNumber() ) {
83 // this probably only applies to digi emulation,
84 // unless an event number is explicitly set in unpacking
85 ldmx_log(fatal) << "Got event number mismatch: framework reports " <<
86 event.getEventHeader().getEventNumber() <<", stream says " << triggerID ;
87 }
88 // error word
89 /* the error word contains
90 - 4 trailing reserved 0's, for now
91 - isCIDunsync : if there is a mismatch between CID reported by channels within the same time sample
92 - isCIDskipped : if there is a gap in the CID increment of a channel beweeen samples
93 - isCRC0malformed : if there was an issue with CRC from fiber0
94 - isCRC1malformed : if there was an issue with CRC from fiber1
95 */
96 uint8_t flags = eventStream.at(QIEStream::ERROR_POS);
97 bool isCIDskipped { (flags >> QIEStream::CID_SKIP_POS) & mask8<QIEStream::FLAG_SIZE>::m};
98 bool isCIDunsync { (flags >> QIEStream::CID_UNSYNC_POS) & mask8<QIEStream::FLAG_SIZE>::m};
99 bool isCRC1malformed{ (flags >> QIEStream::CRC1_ERR_POS) & mask8<QIEStream::FLAG_SIZE>::m};
100 bool isCRC0malformed{ (flags >> QIEStream::CRC0_ERR_POS) & mask8<QIEStream::FLAG_SIZE>::m};
101 //checksum
102 uint8_t referenceChecksum = 30; // really, this is just a random number for now. TODO> implement a checksum set/get
103 int checksum = eventStream.at(QIEStream::CHECKSUM_POS);
104 if ( checksum != referenceChecksum)
105 ldmx_log(fatal) << "Got checksum mismatch: expected " << (int)referenceChecksum << ", stream says " << checksum ;
106 if (isCIDunsync)
107 ldmx_log(debug) << "Found unsynced CIDs!";
108 if (isCIDskipped)
109 ldmx_log(fatal) << "Found skipped CIDs!" ;
110
111 /* -- TS event header done; read the channel contents -- */
112 std::vector<trigscint::TrigScintQIEDigis> outDigis;
113 std::map < int, std::vector<int>> ADCmap;
114 std::map < int, std::vector<int>> TDCmap;
115
116 // read in words from the stream. line them up per channel and time sample.
117 // channels are in the electronics ordering
118
119 // outer loop: over nSamples
120 // inner loop over nChannels to get ADCs, then repeat to get TDCs
121 int iWstart = std::max( std::max(QIEStream::ERROR_POS, QIEStream::CHECKSUM_POS), QIEStream::TRIGID_POS) + 1 ; //overkill :D should be 3
122 int nWords = nSamp*nChannels_*2 + iWstart; //1 ADC, 1 TDC per channel per sample, + the words in the header
123 int iWord = iWstart;
124 for (int iS = 0; iS < nSamp; iS++) {
125 for (int iQ = 0; iQ < nChannels_ ; iQ++) {
126 int val = eventStream.at(iWord);
127 if (val > 0) { //add only the digis with non-zero ADC value
128 ldmx_log(debug) << "got ADC value " << val ;
129 if ( ADCmap.find( iQ ) == ADCmap.end() ) { // we have a new channel
130 std::vector <int > adcs(nSamp, 0);
131 ADCmap.insert(std::pair<int, std::vector <int> >(iQ, adcs));
132 }
133 ADCmap[ iQ ].at(iS) = val;
134 }
135 iWord++;
136 }
137 for (int iQ = 0; iQ < nChannels_ ; iQ++) {
138 int val = eventStream.at(iWord);
139 if (val > 0) { // TODO: check if this channel is also present in ADC map? in the end?
140 ldmx_log(debug) << "got TDC value " << val ;
141 if ( TDCmap.find( iQ ) == TDCmap.end() ) { // we have a new channel
142 std::vector <int > tdcs(nSamp, 0);
143 TDCmap.insert(std::pair<int, std::vector <int> >(iQ, tdcs));
144 }
145 //this is LETDC; only the two most significant bits included
146 // they are shipped as least significant bits --> shift them
147 TDCmap[ iQ ].at(iS) = val*16;
148 }
149 iWord++;
150 }
151 if ( iWord > nWords )
152 ldmx_log(fatal) << "More words than expected! ";
153 }
154
155 for (std::map<int, std::vector<int>>::iterator itr = ADCmap.begin(); itr != ADCmap.end(); ++itr) {
156 TrigScintQIEDigis digi;
157 digi.setADC( itr->second );
158 int bar = channelMap_[itr->first]; // same as .find( itr->first )->second;
159 digi.setChanID( bar );
160 digi.setTDC( TDCmap[itr->first] );
161 outDigis.push_back( digi );
162 ldmx_log(debug) << "Iterator points to key " << itr->first
163 << " and mapped channel supposedly is " << bar ;
164 ldmx_log(debug) << "Made digi with barID = " << digi.getChanID()
165 << ", third adc value " << digi.getADC().at(2)
166 << " and third tdc " << digi.getTDC().at(2) ;
167 }
168
169 event.add(outputCollection_, outDigis);
170 }
