66void HcalGeometry::printPositionMap(int section) const {
67 // Note that layer numbering starts at 1 rather than 0
68 for (int layer = 1; layer <= num_layers_[section]; ++layer) {
69 for (int strip = 0; strip < getNumStrips(section, layer); ++strip) {
70 HcalID id(section, layer, strip);
71 auto centerPosition = getStripCenterPosition(id);
72 auto x = centerPosition[0];
73 auto y = centerPosition[1];
74 auto z = centerPosition[2];
75 std::cout << id << ": Center position: (" << x << ", " << y << ", " << z

80void HcalGeometry::buildStripPositionMap() {
81 // We hard-code the number of sections as seen in HcalID
82 for (int section = 0; section < num_sections_; section++) {
83 for (int layer = 1; layer <= num_layers_[section]; layer++) {
84 for (int strip = 0; strip < getNumStrips(section, layer); strip++) {
85 // initialize values
86 double x{-99999}, y{-99999}, z{-99999};
87
88 // get hcal section
89 ldmx::HcalID::HcalSection hcalsection =
90 (ldmx::HcalID::HcalSection)section;
91
92 // the center of a layer: (layer-1) * (layer_thickness) + scint_thickness/2
93 double layercenter =
94 (layer - 1) * layer_thickness_.at(section) + 0.5 * scint_thickness_;
95
96 // the center of a strip: (strip + 0.5) * (strip_dx)
97 double stripcenter = (strip + 0.5) * scint_width_;
98
99 if (hcalsection == ldmx::HcalID::HcalSection::BACK) {
106 // z position: zero-layer(z) + layer_z + scint_thickness
107 z = zero_layer_.at(section) + layercenter;
108
117 if (layerIsHorizontal(layer)) {
118 y = stripcenter - getZeroStrip(section, layer);
119 x = 0;
120 } else {
121 x = stripcenter - getZeroStrip(section, layer);
122 y = 0;
123 }
124 // std::cout << "back (section,layer,strip)" << section << " " << layer << " " << strip;
125 // std::cout << " (x,y,z) " << x << " " << y << " " << z << std::endl;
126 } else {
138 if(side_3d_readout_ && (layer%2==1)){
139 // z position: zero-strip + half-width (center_strip) of strip
140 z = getZeroStrip(section, layer) + getHalfTotalWidth(section, layer);
141 }
142 else {
143 // z position: zero-strip(z) + strip_center(z)
144 z = getZeroStrip(section, layer) + stripcenter;
145 }
146
147 if (hcalsection == ldmx::HcalID::HcalSection::TOP or
148 hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
149 y = zero_layer_.at(section) + layercenter;
150 x = getHalfTotalWidth(section, layer);
151 if(side_3d_readout_ && layer%2==1)
152 x = getZeroStrip(section, layer) + stripcenter;
153 if (hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
154 y *= -1;
155 x *= -1;
156 }
157 } else {
158 x = zero_layer_.at(section) + layercenter;
159 y = getHalfTotalWidth(section, layer);
160 if(side_3d_readout_ && layer%2==1)
161 y = getZeroStrip(section, layer) + stripcenter;
162 if (hcalsection == ldmx::HcalID::HcalSection::RIGHT) {
163 x *= -1;
164 y *= -1;
165 }
166 }
167
168 // std::cout << "side (section,layer,strip)" << section << " " << layer << " " << strip;
169 // std::cout << " (x,y,z) " << x << " " << y << " " << z << std::endl;
170 }
171 TVector3 pos;
172 pos.SetXYZ(x, y, z);
173 strip_position_map_[ldmx::HcalID(section, layer, strip)] = pos;
174 } // loop over strips
175 } // loop over layers
176 } // loop over sections
177} // strip position map
