28void HcalGeometry::buildStripPositionMap() {
29 // We hard-code the number of sections as seen in HcalID
30 for (int section = 0; section < NumSections_; section++) {
31 for (int layer = 1; layer <= NumLayers_[section]; layer++) {
32 for (int strip = 0; strip < NumStrips_[section]; strip++) {
33 // initialize values
34 double x{-99999}, y{-99999}, z{-99999};
35
36 // get hcal section
37 ldmx::HcalID::HcalSection hcalsection =
38 (ldmx::HcalID::HcalSection)section;
39
40 // the center of a layer: layer * (layer_dz) + (layer_dz)/2
41 double layercenter =
42 layer * LayerThickness_.at(section) + 0.5 * ThicknessScint_;
43
44 // the center of a strip: (strip + 0.5) * (strip_dx)
45 double stripcenter = (strip + 0.5) * WidthScint_;
46
57 if (hcalsection == ldmx::HcalID::HcalSection::BACK) {
58 // z position: zero-layer(z) + layer_z + thickness_scint
59 z = ZeroLayer_.at(section) + layercenter;
60
68 if (layer % 2 == 1) {
69 y = stripcenter - ZeroStrip_.at(section);
70 x = 0;
71 } else {
72 x = stripcenter - ZeroStrip_.at(section);
73 y = 0;
74 }
75 } else {
76 // z position: zero-strip(z) + strip_center(z)
77 z = ZeroStrip_.at(section) + stripcenter;
78
83 if (hcalsection == ldmx::HcalID::HcalSection::TOP or
84 hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
85 y = ZeroLayer_.at(section) + layercenter;
86 x = HalfTotalWidth_.at(section);
87 if (hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
88 y *= -1;
89 x *= -1;
90 }
91 } else {
92 x = ZeroLayer_.at(section) + layercenter;
93 y = HalfTotalWidth_.at(section);
94 if (hcalsection == ldmx::HcalID::HcalSection::RIGHT) {
95 x *= -1;
96 y *= -1;
97 }
98 }
99 }
100 TVector3 pos;
101 pos.SetXYZ(x, y, z);
102 stripPositionMap_[ldmx::HcalID(section, layer, strip)] = pos;
103 } // loop over strips
104 } // loop over layers
105 } // loop over sections
106} // strip position map
