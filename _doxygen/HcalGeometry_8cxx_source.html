48std::vector<double> HcalGeometry::rotateGlobalToLocalBarPosition(
49 const std::vector<double> &globalPosition, const ldmx::HcalID &id) const {
50 const auto orientation{getScintillatorOrientation(id)};
51 switch (id.section()) {
52 case ldmx::HcalID::HcalSection::BACK:
53 switch (orientation) {
54 case ScintillatorOrientation::horizontal:
55 return {globalPosition[2], globalPosition[1], globalPosition[0]};
56 case ScintillatorOrientation::vertical:
57 return {globalPosition[2], globalPosition[0], globalPosition[1]};
58 }
59 case ldmx::HcalID::HcalSection::TOP:
60 case ldmx::HcalID::HcalSection::BOTTOM:
61 switch (orientation) {
62 case ScintillatorOrientation::horizontal:
63 return {globalPosition[1], globalPosition[2], globalPosition[0]};
64 case ScintillatorOrientation::depth:
65 return {globalPosition[1], globalPosition[0], globalPosition[2]};
66 }
67 case ldmx::HcalID::HcalSection::LEFT:
68 case ldmx::HcalID::HcalSection::RIGHT:
69 switch (orientation) {
70 case ScintillatorOrientation::vertical:
71 return {globalPosition[0], globalPosition[2], globalPosition[1]};
72 case ScintillatorOrientation::depth:
73 return globalPosition;
74 }
75 }

78HcalGeometry::ScintillatorOrientation
79HcalGeometry::getScintillatorOrientation(const ldmx::HcalID id) const {
80 if (id.section() == ldmx::HcalID::HcalSection::BACK) {
81 // Configurable, same for all geometries
82 return id.layer() % 2 == back_horizontal_parity_
83 ? ScintillatorOrientation::horizontal
84 : ScintillatorOrientation::vertical;
85 }
86 if (hasSide3DReadout()) {
87 // v14 or later detector
88 switch (id.section()) {
89 case ldmx::HcalID::HcalSection::TOP:
90 case ldmx::HcalID::HcalSection::BOTTOM:
91 // Odd layers are in z/depth direction, even are in the x/horizontal
92 // direction
93 return id.layer() % 2 == 0 ? ScintillatorOrientation::horizontal
94 : ScintillatorOrientation::depth;
95
96 case ldmx::HcalID::HcalSection::LEFT:
97 case ldmx::HcalID::HcalSection::RIGHT:
98 // Odd layers are in the z/depth direction, even are in the y/vertical
99 // direction
100 return id.layer() % 2 == 0 ? ScintillatorOrientation::vertical
101 : ScintillatorOrientation::depth;
102 } // V14 or later detector
103 }
104 // v13/v12
105 switch (id.section()) {
106 // For the v13 side hcal, the bars in each section have the same
107 // orientation
108 case ldmx::HcalID::HcalSection::TOP:
109 case ldmx::HcalID::HcalSection::BOTTOM:
110 return ScintillatorOrientation::horizontal;
111 case ldmx::HcalID::HcalSection::LEFT:
112 case ldmx::HcalID::HcalSection::RIGHT:
113 return ScintillatorOrientation::vertical;
114 } // v13/v12 detector
115}
116void HcalGeometry::printPositionMap(int section) const {
117 // Note that layer numbering starts at 1 rather than 0
118 for (int layer = 1; layer <= num_layers_[section]; ++layer) {
119 for (int strip = 0; strip < getNumStrips(section, layer); ++strip) {

121 auto centerPosition = getStripCenterPosition(id);
122 auto x = centerPosition[0];
123 auto y = centerPosition[1];
124 auto z = centerPosition[2];
125 std::cout << id << ": Center position: (" << x << ", " << y << ", " << z

130void HcalGeometry::buildStripPositionMap() {
131 // We hard-code the number of sections as seen in HcalID
132 for (int section = 0; section < num_sections_; section++) {
133 for (int layer = 1; layer <= num_layers_[section]; layer++) {
134 for (int strip = 0; strip < getNumStrips(section, layer); strip++) {
135 // initialize values
136 double x{-99999}, y{-99999}, z{-99999};
137
138 // get hcal section
139 ldmx::HcalID::HcalSection hcalsection =
140 (ldmx::HcalID::HcalSection)section;
141
142 const ldmx::HcalID id{section, layer, strip};
143 const auto orientation{getScintillatorOrientation(id)};
144 // the center of a layer: (layer-1) * (layer_thickness) +
145 // scint_thickness/2
146 double layercenter =
147 (layer - 1) * layer_thickness_.at(section) + 0.5 * scint_thickness_;
148
149 // the center of a strip: (strip + 0.5) * (strip_dx)
150 double stripcenter = (strip + 0.5) * scint_width_;
151
152 if (hcalsection == ldmx::HcalID::HcalSection::BACK) {
159 // z position: zero-layer(z) + layer_z + scint_thickness / 2
160 z = zero_layer_.at(section) + layercenter;
161
170 if (orientation == ScintillatorOrientation::horizontal) {
171 y = stripcenter - getZeroStrip(section, layer);
172 x = 0;
173 } else {
174 x = stripcenter - getZeroStrip(section, layer);
175 y = 0;
176 }
177 } else {
190 if (side_3d_readout_ &&
191 orientation == ScintillatorOrientation::depth) {
192 // z position: zero-strip + half-width (center_strip) of strip
193 z = getZeroStrip(section, layer) +
194 getHalfTotalWidth(section, layer);
195 } else {
196 // z position: zero-strip(z) + strip_center(z)
197 z = getZeroStrip(section, layer) + stripcenter;
198 }
199
200 if (hcalsection == ldmx::HcalID::HcalSection::TOP or
201 hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
202 y = zero_layer_.at(section) + layercenter;
203 x = getHalfTotalWidth(section, layer);
204 if (side_3d_readout_ &&
205 orientation == ScintillatorOrientation::horizontal) {
206 x = getZeroStrip(section, layer) + stripcenter;
207 }
208 if (hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
209 y *= -1;
210 x *= -1;
211 }
212 } else {
213 x = zero_layer_.at(section) + layercenter;
214 y = getHalfTotalWidth(section, layer);
215 if (side_3d_readout_ &&
216 orientation == ScintillatorOrientation::vertical) {
217 y = getZeroStrip(section, layer) + stripcenter;
218 }
219 if (hcalsection == ldmx::HcalID::HcalSection::RIGHT) {
220 x *= -1;
221 y *= -1;
222 }
223 }
224 }
225 TVector3 pos;
226 pos.SetXYZ(x, y, z);
227 strip_position_map_[ldmx::HcalID(section, layer, strip)] = pos;
228 } // loop over strips
229 } // loop over layers
230 } // loop over sections
231} // strip position map
