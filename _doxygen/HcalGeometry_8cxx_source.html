56void HcalGeometry::printPositionMap(int section) const {
57 // Note that layer numbering starts at 1 rather than 0
58 for (int layer = 1; layer <= NumLayers_[section]; ++layer) {
59 for (int strip = 0; strip < getNumStrips(section, layer); ++strip) {
60 HcalID id(section, layer, strip);
61 auto centerPosition = getStripCenterPosition(id);
62 auto x = centerPosition[0];
63 auto y = centerPosition[1];
64 auto z = centerPosition[2];
65 std::cout << id << ": Center position: (" << x << ", " << y << ", " << z
66 << ")\n";
67 }
68 }

70void HcalGeometry::buildStripPositionMap() {
71 // We hard-code the number of sections as seen in HcalID
72 for (int section = 0; section < NumSections_; section++) {
73 for (int layer = 1; layer <= NumLayers_[section]; layer++) {
74 for (int strip = 0; strip < getNumStrips(section, layer); strip++) {
75 // initialize values
76 double x{-99999}, y{-99999}, z{-99999};
77
78 // get hcal section
79 ldmx::HcalID::HcalSection hcalsection =
80 (ldmx::HcalID::HcalSection)section;
81
82 // the center of a layer: layer * (layer_dz) + (layer_dz)/2
83 double layercenter =
84 (layer - 1) * LayerThickness_.at(section) + 0.5 * ThicknessScint_;
85
86 // the center of a strip: (strip + 0.5) * (strip_dx)
87 double stripcenter = (strip + 0.5) * WidthScint_;
88
99 if (hcalsection == ldmx::HcalID::HcalSection::BACK) {
100 // z position: zero-layer(z) + layer_z + thickness_scint
101 z = ZeroLayer_.at(section) + layercenter;
102
110 if (layer % 2 == 1) {
111 y = stripcenter - getZeroStrip(section, layer);
112 x = 0;
113 } else {
114 x = stripcenter - getZeroStrip(section, layer);
115 y = 0;
116 }
117 } else {
118 // z position: zero-strip(z) + strip_center(z)
119 z = getZeroStrip(section, layer) + stripcenter;
120
125 if (hcalsection == ldmx::HcalID::HcalSection::TOP or
126 hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
127 y = ZeroLayer_.at(section) + layercenter;
128 x = getHalfTotalWidth(section, layer);
129 if (hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
130 y *= -1;
131 x *= -1;
132 }
133 } else {
134 x = ZeroLayer_.at(section) + layercenter;
135 y = getHalfTotalWidth(section, layer);
136 if (hcalsection == ldmx::HcalID::HcalSection::RIGHT) {
137 x *= -1;
138 y *= -1;
139 }
140 }
141 }
142 TVector3 pos;
143 pos.SetXYZ(x, y, z);
144 stripPositionMap_[ldmx::HcalID(section, layer, strip)] = pos;
145 } // loop over strips
146 } // loop over layers
147 } // loop over sections
148} // strip position map
