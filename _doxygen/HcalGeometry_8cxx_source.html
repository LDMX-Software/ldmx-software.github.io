49std::vector<double> HcalGeometry::rotateGlobalToLocalBarPosition(
50 const std::vector<double> &globalPosition, const ldmx::HcalID &id) const {
51 const auto orientation{getScintillatorOrientation(id)};
52 switch (id.section()) {
53 case ldmx::HcalID::HcalSection::BACK:
54 switch (orientation) {
55 case ScintillatorOrientation::horizontal:
56 return {globalPosition[2], globalPosition[1], globalPosition[0]};
57 case ScintillatorOrientation::vertical:
58 return {globalPosition[2], globalPosition[0], globalPosition[1]};
59 }
60 case ldmx::HcalID::HcalSection::TOP:
61 case ldmx::HcalID::HcalSection::BOTTOM:
62 switch (orientation) {
63 case ScintillatorOrientation::horizontal:
64 return {globalPosition[1], globalPosition[2], globalPosition[0]};
65 case ScintillatorOrientation::depth:
66 return {globalPosition[1], globalPosition[0], globalPosition[2]};
67 }
68 case ldmx::HcalID::HcalSection::LEFT:
69 case ldmx::HcalID::HcalSection::RIGHT:
70 switch (orientation) {
71 case ScintillatorOrientation::vertical:
72 return {globalPosition[0], globalPosition[2], globalPosition[1]};
73 case ScintillatorOrientation::depth:
74 return globalPosition;
75 }
76 }

79HcalGeometry::ScintillatorOrientation HcalGeometry::getScintillatorOrientation(
80 const ldmx::HcalID id) const {
81 if (hasSide3DReadout()) {
82 // v14 or later detector
83 switch (id.section()) {
84 case ldmx::HcalID::HcalSection::TOP:
85 case ldmx::HcalID::HcalSection::BOTTOM:
86 // Odd layers are in z/depth direction, even are in the x/horizontal
87 // direction
88 return id.layer() % 2 == 0 ? ScintillatorOrientation::horizontal
89 : ScintillatorOrientation::depth;
90
91 case ldmx::HcalID::HcalSection::LEFT:
92 case ldmx::HcalID::HcalSection::RIGHT:
93 // Odd layers are in the z/depth direction, even are in the y/vertical
94 // direction
95 return id.layer() % 2 == 0 ? ScintillatorOrientation::vertical
96 : ScintillatorOrientation::depth;
97 case ldmx::HcalID::HcalSection::BACK:
98 // Configurable
99 return id.layer() % 2 == back_horizontal_parity_
100 ? ScintillatorOrientation::horizontal
101 : ScintillatorOrientation::vertical;
102 } // V14 or later detector
103 }
104 if (isPrototype()) {
105 // The prototype only has the back section. However, the orientation
106 // depends on the configuration so we delegate to the
107 // back_horizontal_parity parameter
108 return id.layer() % 2 == back_horizontal_parity_
109 ? ScintillatorOrientation::horizontal
110 : ScintillatorOrientation::vertical;
111 } // Prototype detector
112 // v13/v12
113 switch (id.section()) {
114 // For the v13 side hcal, the bars in each section have the same
115 // orientation
116 case ldmx::HcalID::HcalSection::TOP:
117 case ldmx::HcalID::HcalSection::BOTTOM:
118 return ScintillatorOrientation::horizontal;
119 case ldmx::HcalID::HcalSection::LEFT:
120 case ldmx::HcalID::HcalSection::RIGHT:
121 return ScintillatorOrientation::vertical;
122 case ldmx::HcalID::HcalSection::BACK:
123 // Configurable
124 return id.layer() % 2 == back_horizontal_parity_
125 ? ScintillatorOrientation::horizontal
126 : ScintillatorOrientation::vertical;
127 } // v13/v12 detector
128}

144void HcalGeometry::buildStripPositionMap() {
145 // We hard-code the number of sections as seen in HcalID
146 for (int section = 0; section < num_sections_; section++) {
147 for (int layer = 1; layer <= num_layers_[section]; layer++) {
148 for (int strip = 0; strip < getNumStrips(section, layer); strip++) {
149 // initialize values
150 double x{-99999}, y{-99999}, z{-99999};
151
152 // get hcal section
153 ldmx::HcalID::HcalSection hcalsection =
154 (ldmx::HcalID::HcalSection)section;
155
156 const ldmx::HcalID id{section, layer, strip};
157 const auto orientation{getScintillatorOrientation(id)};
158 // the center of a layer: (layer-1) * (layer_thickness) +
159 // scint_thickness/2
160 double layercenter =
161 (layer - 1) * layer_thickness_.at(section) + 0.5 * scint_thickness_;
162
163 // the center of a strip: (strip + 0.5) * (strip_dx)
164 double stripcenter = (strip + 0.5) * scint_width_;
165
166 if (hcalsection == ldmx::HcalID::HcalSection::BACK) {
174 // z position: zero-layer(z) + layer_z + scint_thickness / 2
175 z = zero_layer_.at(section) + layercenter;
176
185 if (orientation == ScintillatorOrientation::horizontal) {
186 y = stripcenter - getZeroStrip(section, layer);
187 x = 0;
188 } else {
189 x = stripcenter - getZeroStrip(section, layer);
190 y = 0;
191 }
192 } else {
193 if (side_3d_readout_) {
194 /*
195 *
196 * For 3D readout:
197 * - odd layers have strips in z
198 * - even layers have strips in x(y) for top-bottom (left-right)
199 * sections
200 * - odd layers have strips occupying width of scintillator in x(y)
201 * - even layers have strips occupying width of scintillator in z
202 *
203 */
204 switch (hcalsection) {
205 case ldmx::HcalID::HcalSection::LEFT:
206 case ldmx::HcalID::HcalSection::RIGHT:
207 if (orientation == ScintillatorOrientation::vertical) {
208 x = zero_layer_[section] + 0.5 * scint_thickness_ +
209 (layer - 1) * layer_thickness_[section];
210 y = ecal_dy_ -
211 (getScintillatorLength({id.section(), 2, id.strip()}) -
212 getScintillatorLength(id)) /
213 2;
214 z = getZeroStrip(section, layer) +
215 (strip + 0.5) * getScintillatorWidth();
216 } else if (orientation == ScintillatorOrientation::depth) {
217 x = zero_layer_[section] + 0.5 * scint_thickness_ +
218 layer_thickness_[section] * (layer - 1);
219 y = -ecal_dy_ / 2 + (strip + 0.5) * getScintillatorWidth();
220 z = getZeroStrip(section, layer + 1) +
221 getScintillatorLength(id) / 2;
222 }
223 if (section == ldmx::HcalID::HcalSection::LEFT) {
224 y *= -1;
225 x *= -1;
226 }
227 break;
228
229 case ldmx::HcalID::HcalSection::BOTTOM:
230 case ldmx::HcalID::HcalSection::TOP:
231 if (orientation == ScintillatorOrientation::horizontal) {
232 //
233 // Second half of the expression is the difference between the
234 // longest strips (first module) and the current module.
235 //
236 // 22 mm extra for space for 1 absorber and one air box
237 x = -ecal_dx_ / 2 - 2 - 20 +
238 (getScintillatorLength({id.section(), 2, id.strip()}) -
239 getScintillatorLength(id)) /
240 2;
241 y = zero_layer_[section] + 0.5 * scint_thickness_ +
242 (layer - 1) * layer_thickness_[section];
243 z = getZeroStrip(section, layer) +
244 (strip + 0.5) * getScintillatorWidth();
245 }
246 if (orientation == ScintillatorOrientation::depth) {
247 x = (ecal_dx_ / 2) - (strip + 0.5) * getScintillatorWidth();
248 y = zero_layer_[section] + 0.5 * scint_thickness_ +
249 layer_thickness_[section] * (layer - 1);
250 z = getZeroStrip(section, layer + 1) +
251 getScintillatorLength(id) / 2;
252 }
253 if (section == ldmx::HcalID::HcalSection::BOTTOM) {
254 y *= -1;
255 x *= -1;
256 }
257 break;
258 }
259
260 } else {
270 // z position: zero-strip(z) + strip_center(z)
271 z = getZeroStrip(section, layer) + stripcenter;
272 if (hcalsection == ldmx::HcalID::HcalSection::TOP or
273 hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
274 y = zero_layer_.at(section) + layercenter;
275 x = getHalfTotalWidth(section, layer);
276 if (hcalsection == ldmx::HcalID::HcalSection::BOTTOM) {
277 y *= -1;
278 x *= -1;
279 }
280
281 } else {
282 x = zero_layer_.at(section) + layercenter;
283 y = getHalfTotalWidth(section, layer);
284 if (hcalsection == ldmx::HcalID::HcalSection::RIGHT) {
285 x *= -1;
286 y *= -1;
287 }
288 }
289 }
290 }
291
292 y += y_offset_;
293 TVector3 pos;
294 pos.SetXYZ(x, y, z);
295 strip_position_map_[ldmx::HcalID(section, layer, strip)] = pos;
296 } // loop over strips
297 } // loop over layers
298 } // loop over sections
299} // strip position map
