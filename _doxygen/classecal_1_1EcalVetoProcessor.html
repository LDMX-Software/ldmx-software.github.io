300 GetShowerCentroidIDAndRMS(ecalRecHits, showerRMS_);
301 /* ~~ Fill the hit map ~~ O(n) */
302 fillHitMap(ecalRecHits, cellMap_);
303 bool doTight = true;
304 /* ~~ Fill the isolated hit maps ~~ O(n) */
305 fillIsolatedHitMap(ecalRecHits, globalCentroid, cellMap_, cellMapTightIso_,
306 doTight);

308 // Loop over the hits from the event to calculate the rest of the important
309 // quantities

311 float wavgLayerHit = 0;
312 float xMean = 0;
313 float yMean = 0;
314
315 // Containment variables
316 unsigned int nregions = 5;
317 std::vector<float> electronContainmentEnergy(nregions, 0.0);
318 std::vector<float> photonContainmentEnergy(nregions, 0.0);
319 std::vector<float> outsideContainmentEnergy(nregions, 0.0);
320 std::vector<int> outsideContainmentNHits(nregions, 0);
321 std::vector<float> outsideContainmentXmean(nregions, 0.0);
322 std::vector<float> outsideContainmentYmean(nregions, 0.0);
323 std::vector<float> outsideContainmentXstd(nregions, 0.0);
324 std::vector<float> outsideContainmentYstd(nregions, 0.0);

326 // MIP tracking: vector of hits to be used in the MIP tracking algorithm. All hits inside the
327 // electron ROC (or all hits in the ECal if the event is missing an electron) will be included.
328 std::vector<HitData> trackingHitList;

330 for (const ldmx::EcalHit &hit : ecalRecHits) {
331 // Layer-wise quantities
332 ldmx::EcalID id(hit.getID());
333 ecalLayerEdepRaw_[id.layer()] =
334 ecalLayerEdepRaw_[id.layer()] + hit.getEnergy();
335 if (id.layer() >= 20) ecalBackEnergy_ += hit.getEnergy();
336 if (maxCellDep_ < hit.getEnergy()) maxCellDep_ = hit.getEnergy();
337 if (hit.getEnergy() > 0) {
338 nReadoutHits_++;
339 ecalLayerEdepReadout_[id.layer()] += hit.getEnergy();
340 ecalLayerTime_[id.layer()] += (hit.getEnergy()) * hit.getTime();
341 auto [x,y,z] = geometry_->getPosition(id);
342 xMean += x * hit.getEnergy();
343 yMean += y * hit.getEnergy();
344 avgLayerHit_ += id.layer();
345 wavgLayerHit += id.layer() * hit.getEnergy();
346 if (deepestLayerHit_ < id.layer()) {
347 deepestLayerHit_ = id.layer();
348 }
349 XYCoords xy_pair = std::make_pair(x,y);
350 float distance_ele_trajectory =
351 ele_trajectory.size()
352 ? sqrt(
353 pow((xy_pair.first - ele_trajectory[id.layer()].first), 2) +
354 pow((xy_pair.second - ele_trajectory[id.layer()].second),
355 2))
356 : -1.0;
357 float distance_photon_trajectory =
358 photon_trajectory.size()
359 ? sqrt(
360 pow((xy_pair.first - photon_trajectory[id.layer()].first),
361 2) +
362 pow((xy_pair.second - photon_trajectory[id.layer()].second),
363 2))
364 : -1.0;
365 // Decide which region a hit goes into and add to sums
366 for (unsigned int ireg = 0; ireg < nregions; ireg++) {
367 if (distance_ele_trajectory >= ireg * ele_radii[id.layer()] &&
368 distance_ele_trajectory < (ireg + 1) * ele_radii[id.layer()])
369 electronContainmentEnergy[ireg] += hit.getEnergy();
370 if (distance_photon_trajectory >= ireg * photon_radii[id.layer()] &&
371 distance_photon_trajectory < (ireg + 1) * photon_radii[id.layer()])
372 photonContainmentEnergy[ireg] += hit.getEnergy();
373 if (distance_ele_trajectory > (ireg + 1) * ele_radii[id.layer()] &&
374 distance_photon_trajectory >
375 (ireg + 1) * photon_radii[id.layer()]) {
376 outsideContainmentEnergy[ireg] += hit.getEnergy();
377 outsideContainmentNHits[ireg] += 1;
378 outsideContainmentXmean[ireg] += xy_pair.first * hit.getEnergy();
379 outsideContainmentYmean[ireg] += xy_pair.second * hit.getEnergy();
380 }
381 }
382
383 // MIP tracking: Decide whether hit should be added to trackingHitList
384 // If inside e- RoC or if etraj is missing, use the hit for tracking:
385 if(distance_ele_trajectory >= ele_radii[id.layer()] || distance_ele_trajectory == -1.0) {
386 HitData hd;
387 hd.pos = TVector3(xy_pair.first, xy_pair.second, geometry_->getZPosition(id.layer()));
388 hd.layer = id.layer();
389 trackingHitList.push_back(hd);
390 }
391
392 }

395 for (const auto &[id, energy] : cellMapTightIso_) {
396 if (energy > 0) summedTightIso_ += energy;
397 }

399 for (int iLayer = 0; iLayer < ecalLayerEdepReadout_.size(); iLayer++) {
400 ecalLayerTime_[iLayer] =
401 ecalLayerTime_[iLayer] / ecalLayerEdepReadout_[iLayer];
402 summedDet_ += ecalLayerEdepReadout_[iLayer];
403 }
404
405 if (nReadoutHits_ > 0) {
406 avgLayerHit_ /= nReadoutHits_;
407 wavgLayerHit /= summedDet_;
408 xMean /= summedDet_;
409 yMean /= summedDet_;
410 } else {
411 wavgLayerHit = 0;
412 avgLayerHit_ = 0;
413 xMean = 0;
414 yMean = 0;
415 }
416
417 for (unsigned int ireg = 0; ireg < nregions; ireg++) {
418 if (outsideContainmentEnergy[ireg] > 0) {
419 outsideContainmentXmean[ireg] /= outsideContainmentEnergy[ireg];
420 outsideContainmentYmean[ireg] /= outsideContainmentEnergy[ireg];
421 }
422 }

424 // Loop over hits a second time to find the standard deviations.
425 for (const ldmx::EcalHit &hit : ecalRecHits) {
426 ldmx::EcalID id(hit.getID());
427 auto [x,y,z] = geometry_->getPosition(id);
428 if (hit.getEnergy() > 0) {
429 xStd_ += pow((x - xMean), 2) * hit.getEnergy();
430 yStd_ += pow((y - yMean), 2) * hit.getEnergy();
431 stdLayerHit_ += pow((id.layer() - wavgLayerHit), 2) * hit.getEnergy();
432 }
433 XYCoords xy_pair = std::make_pair(x,y);
434 float distance_ele_trajectory =
435 ele_trajectory.size()
436 ? sqrt(pow((xy_pair.first - ele_trajectory[id.layer()].first), 2) +
437 pow((xy_pair.second - ele_trajectory[id.layer()].second), 2))
438 : -1.0;
439 float distance_photon_trajectory =
440 photon_trajectory.size()
441 ? sqrt(pow((xy_pair.first - photon_trajectory[id.layer()].first),
442 2) +
443 pow((xy_pair.second - photon_trajectory[id.layer()].second),
444 2))
445 : -1.0;
446 for (unsigned int ireg = 0; ireg < nregions; ireg++) {
447 if (distance_ele_trajectory > (ireg + 1) * ele_radii[id.layer()] &&
448 distance_photon_trajectory > (ireg + 1) * photon_radii[id.layer()]) {
449 outsideContainmentXstd[ireg] +=
450 pow((xy_pair.first - outsideContainmentXmean[ireg]), 2) *
451 hit.getEnergy();
452 outsideContainmentYstd[ireg] +=
453 pow((xy_pair.second - outsideContainmentYmean[ireg]), 2) *
454 hit.getEnergy();
455 }
456 }
457 }
458
459 if (nReadoutHits_ > 0) {
460 xStd_ = sqrt(xStd_ / summedDet_);
461 yStd_ = sqrt(yStd_ / summedDet_);
462 stdLayerHit_ = sqrt(stdLayerHit_ / summedDet_);
463 } else {
464 xStd_ = 0;
465 yStd_ = 0;
466 stdLayerHit_ = 0;
467 }
468
469 for (unsigned int ireg = 0; ireg < nregions; ireg++) {
470 if (outsideContainmentEnergy[ireg] > 0) {
471 outsideContainmentXstd[ireg] =
472 sqrt(outsideContainmentXstd[ireg] / outsideContainmentEnergy[ireg]);
473 outsideContainmentYstd[ireg] =
474 sqrt(outsideContainmentYstd[ireg] / outsideContainmentEnergy[ireg]);
475 }
476 }
477
478 // end loop over sim hits

480 /* Code for fiducial region below */
481
482 std::vector<float> faceXY(2);
483
484 if (!recoilP.empty() && recoilP[2] != 0) {
485 faceXY[0] = ((223.8 - 220.0) * (recoilP[0] / recoilP[2])) + recoilPos[0];
486 faceXY[1] = ((223.8 - 220.0) * (recoilP[1] / recoilP[2])) + recoilPos[1];
487 } else {
488 faceXY[0] = -9999.0;
489 faceXY[1] = -9999.0;
490 }
491
492 int inside = 0;
493 int up = 0;
494 int step = 0;
495 int index;
496 float cell_radius = 5.0;
497
498 std::vector<float>::iterator it;
499 it = std::lower_bound(mapsx.begin(), mapsx.end(), faceXY[0]);
500
501 index = std::distance(mapsx.begin(), it);
502
503 if (index == mapsx.size()) {
504 index += -1;
505 }
506
507 if (!recoilP.empty() && faceXY[0] != -9999.0) {
508 while (true) {
509 std::vector<double> dis(2);
510
511 dis[0] = faceXY[0] - mapsx[index + step];
512 dis[1] = faceXY[1] - mapsy[index + step];
513
514 float celldis = sqrt(pow(dis[0], 2) + pow(dis[1], 2));
515
516 if (celldis <= cell_radius) {
517 inside = 1;
518 break;
519 }
520
521 if ((abs(dis[0]) > 5 && up == 0) || index + step == mapsx.size() - 1) {
522 up = 1;
523 step = 0;
524 } else if ((abs(dis[0]) > 5 && up == 1) ||
525 (index + step == 0 && up == 1)) {
526 break;
527 }
528
529 if (up == 0) {
530 step += 1;
531 } else {
532 step += -1;
533 }
534 }
535 }

537 // MIP tracking starts here
538
539 /* Goal: Calculate
540 * nStraightTracks (self-explanatory),
541 * nLinregTracks (tracks found by linreg algorithm),
542 */
543
544 // Find epAng and epSep, and prepare EP trajectory vectors:
545 TVector3 e_traj_start;
546 TVector3 e_traj_end;
547 TVector3 p_traj_start;
548 TVector3 p_traj_end;
549 if(ele_trajectory.size() > 0 && photon_trajectory.size() > 0) {
550 // Create TVector3s marking the start and endpoints of each projected trajectory
551 e_traj_start.SetXYZ(ele_trajectory[0].first, ele_trajectory[0].second, geometry_->getZPosition(0));
552 e_traj_end.SetXYZ( ele_trajectory[33].first, ele_trajectory[33].second, geometry_->getZPosition(33));
553 p_traj_start.SetXYZ(photon_trajectory[0].first, photon_trajectory[0].second, geometry_->getZPosition(0));
554 p_traj_end.SetXYZ( photon_trajectory[33].first, photon_trajectory[33].second, geometry_->getZPosition(33));
555
556 TVector3 evec = e_traj_end - e_traj_start;
557 TVector3 e_norm = evec.Unit();
558 TVector3 pvec = p_traj_end - p_traj_start;
559 TVector3 p_norm = pvec.Unit();
560 // Separation variables are currently unused due to pT bias concerns and low efficiency
561 // May be removed after a more careful MIP tracking study
562 float epDot = e_norm.Dot(p_norm);
563 epAng_ = acos(epDot) * 180.0 / M_PI;
564 epSep_ = sqrt( pow(e_traj_start.X() - p_traj_start.X(), 2) +
565 pow(e_traj_start.Y() - p_traj_start.Y(), 2) );
566 } else {
567 // Electron trajectory is missing, so all hits in the Ecal are fair game.
568 // Pick e/ptraj so that they won't restrict the tracking algorithm (place them far outside the ECal).
569 e_traj_start = TVector3(999,999,geometry_->getZPosition(0)); //0);
570 e_traj_end = TVector3(999,999,geometry_->getZPosition(33)); // 999);
571 p_traj_start = TVector3(1000,1000,geometry_->getZPosition(0)); //0);
572 p_traj_end = TVector3(1000,1000,geometry_->getZPosition(33)); //1000);
573 epAng_ = 3.0 + 1.0; /*ensures event will not be vetoed by angle/separation cut */
574 epSep_ = 10.0 + 1.0;
575 }
576
577 // Near photon step: Find the first layer of the ECal where a hit near the projected
578 // photon trajectory is found
579 // Currently unusued pending further study; performance has dropped between v9 and v12.
580 firstNearPhLayer_ = geometry_->getZPosition(33);
581
582 if(photon_trajectory.size() != 0) { //If no photon trajectory, leave this at the default (ECal back)
583 for(std::vector<HitData>::iterator it = trackingHitList.begin(); it != trackingHitList.end(); ++it) {
584 float ehDist = sqrt( pow((*it).pos.X() - photon_trajectory[(*it).layer].first, 2)
585 + pow((*it).pos.Y() - photon_trajectory[(*it).layer].second, 2));
586 if(ehDist < 8.7 && (*it).layer < firstNearPhLayer_) {
587 firstNearPhLayer_ = (*it).layer;
588 }
589 }
590 }
591
592 // Find straight MIP tracks:
593
594 std::sort(trackingHitList.begin(), trackingHitList.end(), [](HitData ha, HitData hb) {return ha.layer > hb.layer;});
595 // For merging tracks: Need to keep track of existing tracks
596 // Candidate tracks to merge in will always be in front of the current track (lower z), so only store the last hit
597 // 3-layer vector: each track = vector of 3-tuples (xy+layer).
598 std::vector<std::vector<HitData>> track_list;
599
600 float cellWidth = 8.7;
601 for (int iHit = 0; iHit < trackingHitList.size(); iHit++) {
602 int track[34]; // list of hit numbers in track (34 = maximum theoretical length)
603 int currenthit;
604 int trackLen;
605
606 track[0] = iHit;
607 currenthit = iHit;
608 trackLen = 1;
609
610 // Search for hits to add to the track: if hit is in the next two layers behind the current hit,
611 // consider adding.
612 for (int jHit = 0; jHit < trackingHitList.size(); jHit++) {
613 if (trackingHitList[jHit].layer == trackingHitList[currenthit].layer ||
614 trackingHitList[jHit].layer > trackingHitList[currenthit].layer + 3) {
615 continue; // if not in the next two layers, continue
616 }
617 //if it is: add to track if new hit is directly behind the current hit.
618 if (trackingHitList[jHit].pos.X() == trackingHitList[currenthit].pos.X() &&
619 trackingHitList[jHit].pos.Y() == trackingHitList[currenthit].pos.Y()) {
620 track[trackLen] = jHit;
621 trackLen++;
622 }
623 }
624 // Confirm that the track is valid:
625 if (trackLen < 2) continue; // Track must contain at least 2 hits
626 float closest_e = distTwoLines(trackingHitList[track[0]].pos, trackingHitList[track[trackLen-1]].pos,
627 e_traj_start, e_traj_end);
628 float closest_p = distTwoLines(trackingHitList[track[0]].pos, trackingHitList[track[trackLen-1]].pos,
629 p_traj_start, p_traj_end);
630 // Make sure that the track is near the photon trajectory and away from the electron trajectory
631 // Details of these constraints may be revised
632 if (closest_p > cellWidth and closest_e < 2*cellWidth) continue;
633 if (trackLen < 4 and closest_e > closest_p) continue;
634
635 //if track found, increment nStraightTracks and remove all hits in track from future consideration
636 if (trackLen >= 2) {
637 std::vector<HitData> temp_track_list;
638 for (int kHit = 0; kHit < trackLen; kHit++) {
639 temp_track_list.push_back(trackingHitList[track[kHit]]);
640 trackingHitList.erase(trackingHitList.begin() + track[kHit]);
641 }
642 track_list.push_back(temp_track_list);
643 //The *current* hit will have been removed, so iHit is currently pointing to the next hit.
644 iHit--; //Decrement iHit so no hits will get skipped by iHit++
645 //nStraightTracks_++; // moved to post-merging
646 }
647 }
648
649 //Optional addition: Merge nearby straight tracks. Not necessary for veto.
650 // Criteria: consider tail of track. Merge if head of next track is 1/2 layers behind, within 1 cell of xy position.
651 //std::cout << "BEGINNING TRACK MERGING" << std::endl;
652
653 for (int track_i = 0; track_i < track_list.size(); track_i++) {
654 // for each track, check the remainder of the track list for compatible tracks
655 std::vector<HitData> base_track = track_list[track_i];
656 HitData tail_hitdata = base_track.back(); // xylayer of last hit in track
657 //std::cout << " Considering track " << track_i << std::endl;
658 for (int track_j = track_i+1; track_j < track_list.size(); track_j++) {
659 //std::cout << " Checking for compatibility: " << track_j << std::endl;
660 std::vector<HitData> checking_track = track_list[track_j];
661 HitData head_hitdata = checking_track.front();
662 // if 1-2 layers behind, and xy within one cell...
663 if ((head_hitdata.layer == tail_hitdata.layer+1 || head_hitdata.layer == tail_hitdata.layer+2)
664 && pow(pow(head_hitdata.pos.X()-tail_hitdata.pos.X(),2)
665 + pow(head_hitdata.pos.Y()-tail_hitdata.pos.Y(),2), 0.5) <= cellWidth) {
666 // ...then append the second track to the first one and delete it
667 // NOTE: TO ADD: (trackingHitList[iHit].pos - trackingHitList[jHit].pos).Mag()
668 /*
669 std::cout << " **Compatible track found! Adding track, deleting stuff..." << std::endl;
670 std::cout << " Tail xylayer: " << head_hitdata.pos.X() << "," << head_hitdata.pos.Y() << "," << head_hitdata.layer << std::endl;
671 std::cout << " Head xylayer: " << tail_hitdata.pos.X() << "," << tail_hitdata.pos.Y() << "," << tail_hitdata.layer << std::endl;
672 */
673 for (int hit_k = 0; hit_k < checking_track.size(); hit_k++) {
674 base_track.push_back(track_list[track_j][hit_k]);
675 }
676 track_list.erase(track_list.begin() + track_j);
677 break;
678 }
679 }
680 }
681 nStraightTracks_ = track_list.size();
682
683 // Linreg tracking:
684
685 //std::cout << "Finding linreg tracks..." << std::endl;
686
687 for (int iHit = 0; iHit < trackingHitList.size(); iHit++) {
688 //std::cout << "hit " << iHit << std::endl;
689 int track[34];
690 int trackLen;
691 int currenthit;
692 int hitsInRegion[50]; // Hits being considered at one time
693 int nHitsInRegion; // Number of hits under consideration
694 TMatrixD svdMatrix(3,3);
695 TMatrixD Vm(3,3);
696 TMatrixD hdt(3,3);
697 TVector3 slopeVec;
698 TVector3 hmean;
699 TVector3 hpoint;
700 float r_corr_best;
701 int hitNums_best[3]; // Hit numbers of current best track candidate
702 int hitNums[3];
703
704 trackLen = 0;
705 nHitsInRegion = 1;
706 currenthit = iHit;
707 hitsInRegion[0] = iHit;
708 //std::cout << "filling hitsInRegion, tracking list size=" << trackingHitList.size() << std::endl;
709 // Find all hits within 2 cells of the primary hit:
710 for (int jHit = 0; jHit < trackingHitList.size(); jHit++) {
711 //std::cout << "iHit, jHit: " << iHit << ", " << jHit << std::endl;
712 float dstToHit = (trackingHitList[iHit].pos - trackingHitList[jHit].pos).Mag();
713 //std::cout << " Filling hitsInRegion, tracking list len=" << trackingHitList.size() << std::endl;
714 if (dstToHit <= 2*cellWidth) {
715 hitsInRegion[nHitsInRegion] = jHit;
716 nHitsInRegion++;
717 }
718 }
719
720 //std::cout << "...ready" << std::endl;
721
722 // Look at combinations of hits within the region (do not consider the same combination twice):
723 hitNums[0] = iHit;
724 for (int jHit = 1; jHit < nHitsInRegion - 1; jHit++) {
725 if (trackingHitList.size() < 3) break;
726 hitNums[1] = jHit;
727 for (int kHit = jHit + 1; kHit < nHitsInRegion; kHit++) {
728 hitNums[2] = kHit;
729 for (int hInd = 0; hInd < 3; hInd++) {
730 //std::cout << " inner loop" << std::endl;
731 // hmean = geometric mean, subtract off from hits to improve SVD performance
732 hmean(hInd) = (trackingHitList[hitNums[0]].pos(hInd) +
733 trackingHitList[hitNums[1]].pos(hInd) +
734 trackingHitList[hitNums[2]].pos(hInd))/3.0;
735 }
736 for (int hInd = 0; hInd < 3; hInd++) {
737 for (int lInd = 0; lInd < 3; lInd++) {
738 hdt(hInd,lInd) = trackingHitList[hitNums[hInd]].pos(lInd) - hmean(lInd);
739 }
740 }
741
742 // Perform "linreg" on selected points:
743 TDecompSVD svdObj = TDecompSVD(hdt);
744 bool decomposed = svdObj.Decompose();
745 //std::cout << " Decomposed" << std::endl;
746 if (!decomposed) continue;
747
748 Vm = svdObj.GetV(); // First col of V matrix is the slope of the best-fit line
749 for (int hInd = 0; hInd < 3; hInd++) {
750 slopeVec(hInd) = Vm[0][hInd];
751 }
752 hpoint = slopeVec + hmean; // hmean, hpoint are points on the best-fit line
753 //linreg complete: Now have best-fit line for 3 hits under consideration
754 //Check whether the track is valid: r^2 must be high, and the track must plausibly originate from the photon
755 float closest_e = distTwoLines(hmean, hpoint, e_traj_start, e_traj_end);
756 float closest_p = distTwoLines(hmean, hpoint, p_traj_start, p_traj_end);
757 // Projected track must be close to the photon; details may change after future study.
758 if (closest_p > cellWidth or closest_e < 1.5*cellWidth) continue;
759 //find r^2
760 float vrnc = (trackingHitList[hitNums[0]].pos - hmean).Mag() +
761 (trackingHitList[hitNums[1]].pos - hmean).Mag() +
762 (trackingHitList[hitNums[2]].pos - hmean).Mag(); // ~variance
763 float sumerr = distPtToLine(trackingHitList[hitNums[0]].pos, hmean, hpoint) +
764 distPtToLine(trackingHitList[hitNums[1]].pos, hmean, hpoint) +
765 distPtToLine(trackingHitList[hitNums[2]].pos, hmean, hpoint); // sum of |errors|
766 float r_corr = 1 - sumerr/vrnc;
767 // Check whether r^2 exceeds a low minimum r_corr: "Fake" tracks are still much more common in background, so making the algorithm
768 // oversensitive doesn't lower performance significantly
769 if (r_corr > r_corr_best and r_corr > .6) {
770 r_corr_best = r_corr;
771 trackLen = 0;
772 for (int k=0; k<3; k++) { // Only looking for 3-hit tracks currently
773 track[k] = hitNums[k];
774 trackLen++;
775 }
776 }
777 //std::cout << " inner loop done!" << std::endl;
778 }
779 }
780 // Ordinarily, additional hits in line w/ track would be added here. However, this doesn't affect the results of the simple veto.
781 // Exclude all hits in a found track from further consideration:
782 if (trackLen >= 2) {
783 nLinregTracks_++;
784 //std::cout << " Hitlist size = " << trackingHitList.size() << std::endl;
785 for (int kHit = 0; kHit < trackLen; kHit++) {
786 trackingHitList.erase(trackingHitList.begin() + track[kHit]);
787 //std::cout << " Hitlist size = " << trackingHitList.size() << std::endl;
788 }
789 iHit--;
790 }
791 }
792

796 result.setVariables(
797 nReadoutHits_, deepestLayerHit_, summedDet_, summedTightIso_, maxCellDep_,
798 showerRMS_, xStd_, yStd_, avgLayerHit_, stdLayerHit_, ecalBackEnergy_,
799 nStraightTracks_, nLinregTracks_, firstNearPhLayer_, epAng_, epSep_,
800 electronContainmentEnergy, photonContainmentEnergy,
801 outsideContainmentEnergy, outsideContainmentNHits, outsideContainmentXstd,
802 outsideContainmentYstd, ecalLayerEdepReadout_, recoilP, recoilPos);
803
804 if (doBdt_) {
805 buildBDTFeatureVector(result);
806 ldmx::Ort::FloatArrays inputs({bdtFeatures_});
807 float pred = rt_->run({"features"}, inputs, {"probabilities"})[0].at(1);
808 // Removing electron-photon separation step, near photon step due to lower v12 performance; may reconsider
809 bool passesTrackingVeto = (nStraightTracks_ < 3) && (nLinregTracks_ == 0);
810 //&& //Commenting the remainder for now
811 //(firstNearPhLayer_ >= 6); //&& (epAng_ > 3.0 || epSep_ > 10.0);
812 result.setVetoResult(pred > bdtCutVal_ && passesTrackingVeto);
813 result.setDiscValue(pred);
814 // std::cout << " pred > bdtCutVal = " << (pred > bdtCutVal_) << std::endl;
815
816 // If the event passes the veto, keep it. Otherwise,
817 // drop the event.
818 if (result.passesVeto() && inside) {
819 setStorageHint(framework::hint_shouldKeep);
