30void HCalDQM::analyzeSimHits(const std::vector<ldmx::SimCalorimeterHit> &hits) {
31
32 const auto &geometry = getCondition<ldmx::HcalGeometry>(
33 ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
34
35 std::map<ldmx::HcalID, double> simEnergyPerBar;
36 int hitMultiplicity{0};
37
38 for (const auto &hit : hits) {
39
40 ldmx::HcalID id(hit.getID());
41 if (skipHit(id)) {
42 continue;
43 }
44 const auto energy{hit.getEdep()};
45 if (simEnergyPerBar.count(id) == 0) {
46 simEnergyPerBar[id] = energy;
47 } else {
48 simEnergyPerBar[id] += energy;
49 }
50 const auto orientation{geometry.getScintillatorOrientation(id)};
51 const auto section{id.section()};
52 const auto layer{id.layer()};
53 const auto strip{id.strip()};
54 const auto pos{hit.getPosition()};
55 const auto x{pos[0]};
56 const auto y{pos[1]};
57 const auto z{pos[2]};
58 const auto t{hit.getTime()};
59 hitMultiplicity++;
60 histograms_.fill("sim_hit_time", t);
61 histograms_.fill("sim_layer", layer);
62 histograms_.fill("sim_layer:strip", layer, strip);
63 histograms_.fill("sim_energy", energy);
64 switch (orientation) {
65 case ldmx::HcalGeometry::ScintillatorOrientation::horizontal:
66 histograms_.fill("sim_along_x", x);
67 break;
68 case ldmx::HcalGeometry::ScintillatorOrientation::vertical:
69 histograms_.fill("sim_along_y", y);
70 break;
71 case ldmx::HcalGeometry::ScintillatorOrientation::depth:
72 histograms_.fill("sim_along_z", z);
73 break;
74 }
75 }
76
77 histograms_.fill("sim_hit_multiplicity", hitMultiplicity);
78 histograms_.fill("sim_num_bars_hit", simEnergyPerBar.size());
79
80 double total_energy{0};
81 for (const auto [id, energy] : simEnergyPerBar) {
82 histograms_.fill("sim_energy_per_bar", energy);
83 total_energy += energy;
84 }
85 histograms_.fill("sim_total_energy", total_energy);
86}
87void HCalDQM::analyzeRecHits(const std::vector<ldmx::HcalHit> &hits) {
88
89 const auto &geometry = getCondition<ldmx::HcalGeometry>(
90 ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
91
92 float totalPE{0};
93 float maxPE{-1};
94 float maxPETime{-1};
95 float E{0};
96 float totalE{0};
97 int vetoableHitMultiplicity{0};
98 int hitMultiplicity{0};
99
100 for (const ldmx::HcalHit &hit : hits) {
101 ldmx::HcalID id(hit.getID());
102 const auto orientation{geometry.getScintillatorOrientation(id)};
103 const auto section{id.section()};
104 const auto layer{id.layer()};
105 const auto strip{id.strip()};
106 if (skipHit(id)) {
107 continue;
108 }
109
110 if (hit.isNoise()) {
111 histograms_.fill("noise", 1);
112 } else {
113 histograms_.fill("noise", 0);
114 }
115 if (hitPassesVeto(hit, section)) {
116 hitMultiplicity++;
117 } else {
118 hitMultiplicity++;
119 vetoableHitMultiplicity++;
120 }
121 const auto pe{hit.getPE()};
122 const auto t{hit.getTime()};
123 const auto e{hit.getEnergy()};
124 const auto x{hit.getXPos()};
125 const auto y{hit.getYPos()};
126 const auto z{hit.getZPos()};
127 switch (orientation) {
128 case ldmx::HcalGeometry::ScintillatorOrientation::horizontal:
129 histograms_.fill("along_x", x);
130 break;
131 case ldmx::HcalGeometry::ScintillatorOrientation::vertical:
132 histograms_.fill("along_y", y);
133 break;
134 case ldmx::HcalGeometry::ScintillatorOrientation::depth:
135 histograms_.fill("along_z", z);
136 break;
137 }
138
139 totalE += e;
140 totalPE += pe;
141
142 if (pe > maxPE) {
143 maxPE = pe;
144 maxPETime = t;
145 }
146 histograms_.fill("layer:strip", layer, strip);
147 histograms_.fill("pe", pe);
148 histograms_.fill("hit_time", t);
149 histograms_.fill("layer", layer);
150 histograms_.fill("noise", hit.isNoise());
151 histograms_.fill("energy", e);
152 histograms_.fill("hit_z", z);
153 }
154 histograms_.fill("total_energy", totalE);
155 histograms_.fill("total_pe", totalPE);
156 histograms_.fill("max_pe", maxPE);
157 histograms_.fill("max_pe_time", maxPETime);
158 histograms_.fill("hit_multiplicity", hitMultiplicity);
159 histograms_.fill("vetoable_hit_multiplicity", vetoableHitMultiplicity);
160}
