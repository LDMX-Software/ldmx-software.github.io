19void HCalDQM::analyze(const framework::Event& event) {
20 // hard-code PE threshold because I'm lazy
21 static const float pe_veto_threshold{5.};
22
23 // Get the collection of HCalDQM digitized hits if the exists
24 const auto& hcalHits{event.getCollection<ldmx::HcalHit>(rec_coll_name_,rec_pass_name_)};
25
26 float totalPE{0}, total_back_pe{0}, minTime{999}, minTimePE{-1};
27 float maxPE{-1}, maxPETime{-1};
28
29 // Loop through all HCal hits in the event
30 // Get non-noise generated hits into new vector for sorting
31 for (const ldmx::HcalHit& hit : hcalHits) {
32 ldmx::HcalID id(hit.getID());
33
34 histograms_.fill("pe", hit.getPE());
35 histograms_.fill("hit_time", hit.getTime());
36 totalPE += hit.getPE();
37
38 std::string section{"side"};
39 if (id.section() == ldmx::HcalID::HcalSection::BACK) {
40 section = "back";
41 total_back_pe += hit.getPE();
42 }
43
44 histograms_.fill(section+"_pe:layer", hit.getPE(), id.layer());
45 histograms_.fill(section+"_layer:strip", id.layer(), id.strip());
46
51 if (hit.getTime() > -999. and hit.getPE() > pe_veto_threshold and hit.getTime() < minTime) {
52 minTime = hit.getTime();
53 minTimePE = hit.getPE();
54 }
55
60 if (hit.getPE() > maxPE) {
61 maxPE = hit.getPE();
62 maxPETime = hit.getTime();
63 }
64 }
65
66 // let's fill our once-per-event histograms
67 histograms_.fill("n_hits", hcalHits.size());
68 histograms_.fill("total_pe", totalPE);
69 histograms_.fill("back_total_pe", total_back_pe);
70 histograms_.fill("max_pe", maxPE);
71 histograms_.fill("hit_time_max_pe", maxPETime);
72 histograms_.fill("max_pe:time", maxPE, maxPETime);
73 histograms_.fill("min_time_hit_above_thresh", minTime);
74 histograms_.fill("min_time_hit_above_thresh:pe", minTimePE, minTime);
75
76}
