12  void HCalDQM::analyze(const Event & event) {
13 
14  // Check if the collection of digitized HCal hits exist. If it doesn't
15  // don't continue processing.
16  if (!event.exists("hcalDigis")) return;
17 
18  // Get the collection of HCalDQM digitized hits if the exists
19  const std::vector<HcalHit> hcalHits = event.getCollection<HcalHit>("hcalDigis");
20 
21  // Get the total hit count
22  int hitCount = hcalHits.size();
23  histograms_.fill("n_hits",hitCount);
24 
25  double totalPE{0};
26 
27  // Loop through all HCal hits in the event
28  // Get non-noise generated hits into new vector for sorting
29  std::vector<const HcalHit *> filteredHits;
30  for (const HcalHit &hit : hcalHits ) {
31 
32  histograms_.fill("pe",hit.getPE());
33  histograms_.fill("hit_time",hit.getTime());
34 
35  totalPE += hit.getPE();
36 
37  if ( hit.getTime() > -999. ) { filteredHits.push_back( &hit ); }
38  }
39 
40  histograms_.fill("total_pe",totalPE);
41 
42  // Sort the array by hit time
43  std::sort (filteredHits.begin(), filteredHits.end(), [ ](const auto& lhs, const auto& rhs)
44  {
45  return lhs->getTime() < rhs->getTime();
46  });
47 
48  //get first time and PE of hit over threshold
49  double minTime{-1};
50  double minTimePE{-1};
51  for (const auto& hit : filteredHits) {
52  if (hit->getPE() < maxPEThreshold_) continue;
53  minTime = hit->getTime();
54  minTimePE = hit->getPE();
55  break;
56  }
57 
58  histograms_.fill("min_time_hit_above_thresh",minTime);
59  histograms_.fill("min_time_hit_above_thresh:pe",minTimePE, minTime);
60 
61  float maxPE{-1};
62  float maxPETime{-1};
63  bool passesHcalVeto{false};
64  // Check if the HcalVeto result exists
65  if (event.exists("HcalVeto")) {
66 
67  // Get the collection of HCalDQM digitized hits if the exists
68  const HcalVetoResult hcalVeto = event.getObject<HcalVetoResult>("HcalVeto");
69 
70  HcalHit maxPEHit = hcalVeto.getMaxPEHit();
71 
72  // Get the max PE and it's time
73  maxPE = maxPEHit.getPE();
74  maxPETime = maxPEHit.getTime();
75 
76  histograms_.fill("max_pe",maxPE);
77  histograms_.fill("hit_time_max_pe",maxPETime);
78  histograms_.fill("max_pe:time",maxPE, maxPETime);
79  histograms_.fill("veto",hcalVeto.passesVeto());
80 
81  if (hcalVeto.passesVeto()) {
82  histograms_.fill("max_pe_hcal_veto",maxPE);
83  histograms_.fill("hit_time_max_pe_hcal_veto",maxPETime);
84  histograms_.fill("max_pe:time_hcal_veto",maxPE, maxPETime);
85  histograms_.fill("total_pe_hcal_veto",totalPE);
86  histograms_.fill("n_hits_hcal_veto",hitCount);
87  passesHcalVeto = true;
88  }
89  }
90  }
