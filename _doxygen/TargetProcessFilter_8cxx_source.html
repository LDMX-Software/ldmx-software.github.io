51  void TargetProcessFilter::stepping(const G4Step* step) {
52 
53  // Get the track associated with this step.
54  auto track{step->GetTrack()};
55 
56  // Get the track info and check if this track is a brem candidate
57  auto trackInfo{static_cast< UserTrackInformation* >(track->GetUserInformation())};
58  if ((trackInfo != nullptr) && !trackInfo->isBremCandidate()) return;
59 
60  // Get the event info to keep track of the number of brem candidates
61  auto eventInfo{
62  static_cast< UserEventInformation* >(
63  G4EventManager::GetEventManager()->GetUserInformation())};
64  if (eventInfo == nullptr) {
65  // thrown an exception
66  }
67 
68  // Get the region the particle is currently in. Continue processing
69  // the particle only if it's in the calorimeter region.
70  if (auto region{track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
71  region.compareTo("target") != 0) return;
72 
73  // Get the particles daughters.
74  auto secondaries{step->GetSecondary()};
75 
76  // If the brem photon doesn't undergo any reaction in the target, stop
77  // processing the rest of the event if the particle is exiting the
78  // target region.
79  if (secondaries->size() == 0) {
80 
81  // Check if the electron will be exiting the target
82  if (auto volume{track->GetNextVolume()->GetName()}; volume.compareTo("recoil_PV") == 0) {
83  if (secondaries->size() != 0) {
84 
85  if (eventInfo->bremCandidateCount() == 1) {
86  track->SetTrackStatus(fKillTrackAndSecondaries);
87  G4RunManager::GetRunManager()->AbortEvent();
88  currentTrack_ = nullptr;
89  } else {
90  currentTrack_ = track;
91  track->SetTrackStatus(fSuspend);
92  eventInfo->decBremCandidateCount();
93  trackInfo->tagBremCandidate(false);
94  }
95  }
96  return;
97  }
98  } else {
99 
100  G4String processName = secondaries->at(0)->GetCreatorProcess()->GetProcessName();
101 
102  // Only record the process that is being biased
103  if (!processName.contains(process_)) {
104 
105  if (eventInfo->bremCandidateCount() == 1) {
106  track->SetTrackStatus(fKillTrackAndSecondaries);
107  G4RunManager::GetRunManager()->AbortEvent();
108  currentTrack_ = nullptr;
109  } else {
110  currentTrack_ = track;
111  track->SetTrackStatus(fSuspend);
112  eventInfo->decBremCandidateCount();
113  trackInfo->tagBremCandidate(false);
114  }
115  }
116 
117  std::cout << "[ TargetProcessFilter ]: "
118  << "Brem photon produced " << secondaries->size()
119  << " particle via " << processName << " process."
120  << std::endl;
121  trackInfo->tagBremCandidate(false);
122  eventInfo->decBremCandidateCount();
123  eventInfo->setWeight(track->GetWeight());
124 
125  }
