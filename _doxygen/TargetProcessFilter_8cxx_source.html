49void TargetProcessFilter::stepping(const G4Step* step) {
50 // Get the track associated with this step.
51 auto track{step->GetTrack()};
52
53 if (G4EventManager::GetEventManager()->GetConstCurrentEvent()->IsAborted())
54 return;
55
56 // Get the track info and check if this track is a brem candidate
57 auto trackInfo{simcore::UserTrackInformation::get(track)};
58 if ((trackInfo != nullptr) && !trackInfo->isBremCandidate()) return;
59
60 // Get the particles daughters.
61 auto secondaries{step->GetSecondary()};
62
63 // Get the region the particle is currently in. Continue processing
64 // the particle only if it's in the target region.
65 if (auto region{
66 track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
67 region.compareTo("target") != 0) {
68 // If secondaries were produced outside of the volume of interest,
69 // and there aren't additional brems to process, abort the event.
70 // Otherwise, suspend the track and move on to the next brem.
71 if (secondaries->size() != 0) {
72 if (getEventInfo()->bremCandidateCount() == 1) {
73 track->SetTrackStatus(fKillTrackAndSecondaries);
74 G4RunManager::GetRunManager()->AbortEvent();
75 currentTrack_ = nullptr;
76 } else {
77 currentTrack_ = track;
78 track->SetTrackStatus(fSuspend);
79 getEventInfo()->decBremCandidateCount();
80 trackInfo->tagBremCandidate(false);
81 }
82 }
83 return;
84 }
85
86 // If the brem photon doesn't undergo any reaction in the target, stop
87 // processing the rest of the event if the particle is exiting the
88 // target region.
89 if (secondaries->size() == 0) {
102 if (auto volume{track->GetNextVolume()->GetName()};
103 volume.compareTo("recoil_PV") == 0 or
104 volume.compareTo("World_PV") == 0) {
105 if (getEventInfo()->bremCandidateCount() == 1) {
106 track->SetTrackStatus(fKillTrackAndSecondaries);
107 G4RunManager::GetRunManager()->AbortEvent();
108 currentTrack_ = nullptr;
109 } else {
110 currentTrack_ = track;
111 track->SetTrackStatus(fSuspend);
112 getEventInfo()->decBremCandidateCount();
113 trackInfo->tagBremCandidate(false);
114 }
115 }
116 return;
117 } else {
118 // If the brem gamma interacts and produced secondaries, get the
119 // process used to create them.
120 G4String processName =
121 secondaries->at(0)->GetCreatorProcess()->GetProcessName();
122
123 // Only record the process that is being biased
124 if (!processName.contains(process_)) {
125 if (getEventInfo()->bremCandidateCount() == 1) {
126 track->SetTrackStatus(fKillTrackAndSecondaries);
127 G4RunManager::GetRunManager()->AbortEvent();
128 currentTrack_ = nullptr;
129 } else {
130 currentTrack_ = track;
131 track->SetTrackStatus(fSuspend);
132 getEventInfo()->decBremCandidateCount();
133 trackInfo->tagBremCandidate(false);
134 }
135 return;
136 }
137
138 if (G4RunManager::GetRunManager()->GetVerboseLevel() > 1) {
139 std::cout << "[ TargetProcessFilter ]: "
140 << G4EventManager::GetEventManager()
141 ->GetConstCurrentEvent()
142 ->GetEventID()
143 << " Brem photon produced " << secondaries->size()
144 << " particle via " << processName << " process." << std::endl;
145 }
146 trackInfo->tagBremCandidate(false);
147 trackInfo->setSaveFlag(true);
148 trackInfo->tagPNGamma();
149 getEventInfo()->decBremCandidateCount();
150 }
151}
