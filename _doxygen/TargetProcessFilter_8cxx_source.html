49 void TargetProcessFilter::stepping(const G4Step* step) {
50  // Get the track associated with this step.
51  auto track{step->GetTrack()};
52 
53  // Get the track info and check if this track is a brem candidate
54  auto trackInfo{simcore::UserTrackInformation::get(track)};
55  if ((trackInfo != nullptr) && !trackInfo->isBremCandidate()) return;
56 
57  // Get the region the particle is currently in. Continue processing
58  // the particle only if it's in the calorimeter region.
59  if (auto region{
60  track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
61  region.compareTo("target") != 0)
62  return;
63 
64  // Get the particles daughters.
65  auto secondaries{step->GetSecondary()};
66 
67  // If the brem photon doesn't undergo any reaction in the target, stop
68  // processing the rest of the event if the particle is exiting the
69  // target region.
70  if (secondaries->size() == 0) {
71  // Check if the electron will be exiting the target
72  if (auto volume{track->GetNextVolume()->GetName()};
73  volume.compareTo("recoil_PV") == 0) {
74  if (secondaries->size() != 0) {
75  if (getEventInfo()->bremCandidateCount() == 1) {
76  track->SetTrackStatus(fKillTrackAndSecondaries);
77  G4RunManager::GetRunManager()->AbortEvent();
78  currentTrack_ = nullptr;
79  } else {
80  currentTrack_ = track;
81  track->SetTrackStatus(fSuspend);
82  getEventInfo()->decBremCandidateCount();
83  trackInfo->tagBremCandidate(false);
84  }
85  }
86  return;
87  }
88  } else {
89  G4String processName =
90  secondaries->at(0)->GetCreatorProcess()->GetProcessName();
91 
92  // Only record the process that is being biased
93  if (!processName.contains(process_)) {
94  if (getEventInfo()->bremCandidateCount() == 1) {
95  track->SetTrackStatus(fKillTrackAndSecondaries);
96  G4RunManager::GetRunManager()->AbortEvent();
97  currentTrack_ = nullptr;
98  } else {
99  currentTrack_ = track;
100  track->SetTrackStatus(fSuspend);
101  getEventInfo()->decBremCandidateCount();
102  trackInfo->tagBremCandidate(false);
103  }
104  }
105 
106  std::cout << "[ TargetProcessFilter ]: "
107  << "Brem photon produced " << secondaries->size()
108  << " particle via " << processName << " process." << std::endl;
109  trackInfo->tagBremCandidate(false);
110  trackInfo->tagPNGamma();
111  getEventInfo()->decBremCandidateCount();
112  }
113 }
