184 
185  try {
186  the_file_ = std::make_unique<framework::EventFile>(file.Data());
187  the_file_->setupEvent(&the_event_);
188  if (verbose_) {
189  std::cout << "[ Display ] : Input root file opened successfully."
190  << std::endl;
191  }
192  } catch (const framework::exception::Exception& e) {
193  std::cerr << "[ Display ] : Input root file cannot be opened." << std::endl;

195  << " at " << e.module() << ":" << e.line() << " in "
196  << e.function() << "\nStack trace: " << std::endl
197  << e.stackTrace();
198  return false;
199  }
200 
201  return true;
202 }

257 bool Display::SetSimThresh() {
258  double thresh = atof(textBoxSimThresh_->GetText());
259  if (thresh == 0 && std::string(textBoxSimThresh_->GetText()) != "0") {
260  std::cout << "[ Display ] : Invalid sim energy threshold entered: \""

262  return false;
