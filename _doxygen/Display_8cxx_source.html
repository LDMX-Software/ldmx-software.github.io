183bool Display::SetFile(const TString file) {
184
185 try {
186 framework::config::Parameters file_config;
187 file_config.addParameter("tree_name",std::string("LDMX_Events"));
188 the_file_ = std::make_unique<framework::EventFile>(file_config,file.Data());
189 the_file_->setupEvent(&the_event_);
190 if (verbose_) {
191 std::cout << "[ Display ] : Input root file opened successfully."
192 << std::endl;
193 }
194 } catch (const framework::exception::Exception& e) {
195 std::cerr << "[ Display ] : Input root file cannot be opened." << std::endl;
196 std::cerr << "[" << e.name() << "] : " << e.message() << "\n"
197 << " at " << e.module() << ":" << e.line() << " in "
198 << e.function() << "\nStack trace: " << std::endl
199 << e.stackTrace();
200 return false;
201 }
202
203 return true;
204}

259bool Display::SetSimThresh() {
260 double thresh = atof(textBoxSimThresh_->GetText());
261 if (thresh == 0 && std::string(textBoxSimThresh_->GetText()) != "0") {
262 std::cout << "[ Display ] : Invalid sim energy threshold entered: \""
263 << textBoxSimThresh_->GetText() << "\"" << std::endl;
264 return false;
265 } else if (verbose_) {
266 std::cout << "[ Display ] : Setting SimParticle energy threshold to "
267 << thresh << std::endl;
268 }
269
270 objects_.SetSimThresh(thresh);
271
272 manager_->RegisterRedraw3D();
273 manager_->FullRedraw3D(kFALSE, kTRUE);
274
275 return true;
276}
