53 void RootSimFromEcalSP::GeneratePrimaryVertex(G4Event* anEvent) {
54  // go to next event
55  // if there isn't another event ==> EventFile::nextEvent returns false
56  if (not ifile_->nextEvent()) {
57  std::cout << "[ RootSimFromEcalSP ]: End of file reached." << std::endl;
58  G4RunManager::GetRunManager()->AbortRun(true);
59  anEvent->SetEventAborted();
60  }
61 
62  // In this mode, we need to loop through all ECal scoring plane hits
63  // and find the subset of unique hits created by particles exiting
64  // the ECal. These particles will be stored in a container and
65  // re-fired into the HCal.
66  std::unordered_map<int, const ldmx::SimTrackerHit*> spHits;
67 
68  auto ecalSPParticles{ievent_.getCollection<ldmx::SimTrackerHit>(
69  ecalSPHitsCollName_, ecalSPHitsPassName_)};
70  // Loop through all of the ECal scoring plane hits.
71  for (const ldmx::SimTrackerHit& spHit : ecalSPParticles) {
72  // First, start by skipping all hits that were created by
73  // particles entering the ECal volume.
74  if (spHit.getLayerID() == 1 and spHit.getMomentum()[2] > 0) continue;
75  if (spHit.getLayerID() == 2 and spHit.getMomentum()[2] < 0) continue;
76  if (spHit.getLayerID() == 3 and spHit.getMomentum()[1] < 0) continue;
77  if (spHit.getLayerID() == 4 and spHit.getMomentum()[1] > 0) continue;
78  if (spHit.getLayerID() == 5 and spHit.getMomentum()[0] > 0) continue;
79  if (spHit.getLayerID() == 6 and spHit.getMomentum()[0] < 0) continue;
80 
81  // Don't consider particles created outside of the HCal readout
82  // window. Currently, this is estimated to be 50 ns.
83  if (spHit.getTime() > timeCutoff_) continue;
84 
85  if (spHits.find(spHit.getTrackID()) == spHits.end()) {
86  spHits[spHit.getTrackID()] = &spHit;
87  } else {
88  float currentPMag =
89  sqrt(pow(spHit.getMomentum()[0], 2) + pow(spHit.getMomentum()[1], 2) +
90  pow(spHit.getMomentum()[2], 2));
91  float pMag = sqrt(pow(spHits[spHit.getTrackID()]->getMomentum()[0], 2) +
92  pow(spHits[spHit.getTrackID()]->getMomentum()[1], 2) +
93  pow(spHits[spHit.getTrackID()]->getMomentum()[2], 2));
94 
95  if (pMag < currentPMag) spHits[spHit.getTrackID()] = &spHit;
96  }
97  }
98 
99  for (auto const& spHit : spHits) {
100  auto cVertex{new G4PrimaryVertex()};
101  cVertex->SetPosition(spHit.second->getPosition()[0] * mm,
102  spHit.second->getPosition()[1] * mm,
103  spHit.second->getPosition()[2] * mm);
104  cVertex->SetWeight(1.);
105 
106  auto primary{new G4PrimaryParticle()};
107  primary->SetPDGcode(spHit.second->getPdgID());
108  primary->SetMomentum(spHit.second->getMomentum()[0] * MeV,
109  spHit.second->getMomentum()[1] * MeV,
110  spHit.second->getMomentum()[2] * MeV);
111 
112  cVertex->SetPrimary(primary);
113  anEvent->AddPrimaryVertex(cVertex);
114  }

117  // The input stream is then passed to the random engine to restore
118  // the state.
119  std::istringstream iss(
120  ievent_.getEventHeader().getStringParameter("eventSeed"));
121  G4Random::restoreFullState(iss);
122 }
