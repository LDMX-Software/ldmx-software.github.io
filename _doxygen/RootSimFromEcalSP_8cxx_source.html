55void RootSimFromEcalSP::GeneratePrimaryVertex(G4Event* anEvent) {
56 // go to next event
57 // if there isn't another event ==> EventFile::nextEvent returns false
58 if (not ifile_->nextEvent()) {
59 std::cout << "[ RootSimFromEcalSP ]: End of file reached." << std::endl;
60 G4RunManager::GetRunManager()->AbortRun(true);
61 anEvent->SetEventAborted();
62 }
63
64 // In this mode, we need to loop through all ECal scoring plane hits
65 // and find the subset of unique hits created by particles exiting
66 // the ECal. These particles will be stored in a container and
67 // re-fired into the HCal.
68 std::unordered_map<int, const ldmx::SimTrackerHit*> spHits;
69
70 auto ecalSPParticles{ievent_.getCollection<ldmx::SimTrackerHit>(
71 ecalSPHitsCollName_, ecalSPHitsPassName_)};
72 // Loop through all of the ECal scoring plane hits.
73 for (const ldmx::SimTrackerHit& spHit : ecalSPParticles) {
74 // First, start by skipping all hits that were created by
75 // particles entering the ECal volume.
76 if (spHit.getLayerID() == 1 and spHit.getMomentum()[2] > 0) continue;
77 if (spHit.getLayerID() == 2 and spHit.getMomentum()[2] < 0) continue;
78 if (spHit.getLayerID() == 3 and spHit.getMomentum()[1] < 0) continue;
79 if (spHit.getLayerID() == 4 and spHit.getMomentum()[1] > 0) continue;
80 if (spHit.getLayerID() == 5 and spHit.getMomentum()[0] > 0) continue;
81 if (spHit.getLayerID() == 6 and spHit.getMomentum()[0] < 0) continue;
82
83 // Don't consider particles created outside of the HCal readout
84 // window. Currently, this is estimated to be 50 ns.
85 if (spHit.getTime() > timeCutoff_) continue;
86
87 if (spHits.find(spHit.getTrackID()) == spHits.end()) {
88 spHits[spHit.getTrackID()] = &spHit;
89 } else {
90 float currentPMag =
91 sqrt(pow(spHit.getMomentum()[0], 2) + pow(spHit.getMomentum()[1], 2) +
92 pow(spHit.getMomentum()[2], 2));
93 float pMag = sqrt(pow(spHits[spHit.getTrackID()]->getMomentum()[0], 2) +
94 pow(spHits[spHit.getTrackID()]->getMomentum()[1], 2) +
95 pow(spHits[spHit.getTrackID()]->getMomentum()[2], 2));
96
97 if (pMag < currentPMag) spHits[spHit.getTrackID()] = &spHit;
98 }
99 }
100
101 for (auto const& spHit : spHits) {
102 auto cVertex{new G4PrimaryVertex()};
103 cVertex->SetPosition(spHit.second->getPosition()[0] * mm,
104 spHit.second->getPosition()[1] * mm,
105 spHit.second->getPosition()[2] * mm);
106 cVertex->SetWeight(1.);
107
108 auto primary{new G4PrimaryParticle()};
109 primary->SetPDGcode(spHit.second->getPdgID());
110 primary->SetMomentum(spHit.second->getMomentum()[0] * MeV,
111 spHit.second->getMomentum()[1] * MeV,
112 spHit.second->getMomentum()[2] * MeV);
113
114 cVertex->SetPrimary(primary);
115 anEvent->AddPrimaryVertex(cVertex);
116 }
117
118 // Create an input stream with a copy of the event seed as content.
119 // The input stream is then passed to the random engine to restore
120 // the state.
121 std::istringstream iss(
122 ievent_.getEventHeader().getStringParameter("eventSeed"));
123 G4Random::restoreFullState(iss);
124}
