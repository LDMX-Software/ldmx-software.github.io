22G4double ElementXsecCache::get(G4double energy, G4double A, G4double Z) {
23 key_t key = computeKey(energy, A, Z);
24 if (the_cache_.find(key) == the_cache_.end()) {
25 if (model_.get() == nullptr) {
26 EXCEPTION_RAISE("BadCache",
27 "ElementXsecCache not given a model to calculate cross "
28 "sections with.");
29 }
30 the_cache_[key] = model_->ComputeCrossSectionPerAtom(energy, A, Z);
31 }
32 return the_cache_.at(key);
33}

35void ElementXsecCache::stream(std::ostream& o) const {
36 o << "A [au],Z [protons],Energy [MeV],Xsec [pb]\n"
37 << std::setprecision(std::numeric_limits<double>::digits10 +
38 1); // maximum precision
39 for (auto const& [key, xsec] : the_cache_) {
40 key_t E = key % MAX_E;
41 key_t A = ((key - E) / MAX_E) % MAX_A;
42 key_t Z = ((key - E) / MAX_E - A) / MAX_A;
43 o << A << "," << Z << "," << E << "," << xsec / CLHEP::picobarn << "\n";
44 }
45 o << std::endl;
46}

48ElementXsecCache::key_t ElementXsecCache::computeKey(G4double energy,
49 G4double A,
50 G4double Z) const {
51 key_t energyKey = energy;
52 key_t AKey = A;
53 key_t ZKey = Z;
54 return (ZKey * MAX_A + AKey) * MAX_E + energyKey;
55}

59G4eDarkBremsstrahlung::G4eDarkBremsstrahlung(
60 const framework::config::Parameters& params)
61 : G4VDiscreteProcess(G4eDarkBremsstrahlung::PROCESS_NAME,
62 fElectromagnetic) {
63 // we need to pretend to be an EM process so the biasing framework recognizes
64 // us
65 SetProcessSubType(63); // needs to be different from the other Em Subtypes
66
67 only_one_per_event_ = params.getParameter<bool>("only_one_per_event");
68 cache_xsec_ = params.getParameter<bool>("cache_xsec");
69 ap_mass_ = params.getParameter<double>("ap_mass");
70
71 auto model{params.getParameter<framework::config::Parameters>("model")};
72 auto model_name{model.getParameter<std::string>("name")};
73 if (model_name == "vertex_library") {
74 model_ = std::make_shared<DarkBremVertexLibraryModel>(model);
75 } else {
76 EXCEPTION_RAISE("DarkBremModel",
77 "Model named '" + model_name + "' is not known.");
78 }
79
80 // now that the model is set, calculate common xsec and put them into the
81 // cache
82 if (cache_xsec_) {
83 element_xsec_cache_ =
84 ElementXsecCache(model_); // remake cache with model attached
85 CalculateCommonXsec(); // calculate common cross sections
86 }
87}

106G4VParticleChange* G4eDarkBremsstrahlung::PostStepDoIt(const G4Track& track,
107 const G4Step& step) {
108 // Debugging Purposes: Check if track we get is an electron
109 if (not IsApplicable(*track.GetParticleDefinition()))
110 EXCEPTION_RAISE(
111 "DBBadTrack",
112 "Dark brem process receieved a track that isn't applicable.");
113
114 /*
115 * Geant4 has decided that it is our time to interact,
116 * so we are going to change the particle
117 */
118 ldmx_log(debug) << "A dark brem occurred!";
119
120 if (only_one_per_event_) {
121 // Deactivate the process after one dark brem if we restrict ourselves to
122 // only one per event. If this is in the stepping action instead, more than
123 // one brem can occur within each step. Reactivated in
124 // RunManager::TerminateOneEvent Both biased and unbiased process could be
125 // in the run (but not at the same time),
126 // so we turn off both while silencing the warnings from the process table.
127 std::vector<G4String> db_process_name_options = {
128 "biasWrapper(" + PROCESS_NAME + ")", PROCESS_NAME};
129 G4ProcessTable* ptable = G4ProcessTable::GetProcessTable();
130 G4int verbosity = ptable->GetVerboseLevel();
131 ptable->SetVerboseLevel(0);
132 for (auto const& name : db_process_name_options)
133 ptable->SetProcessActivation(name, false);
134 ptable->SetVerboseLevel(verbosity);
135 }
136
137 aParticleChange.Initialize(track);
138
139 model_->GenerateChange(aParticleChange, track, step);
140
141 /*
142 * Parent class has some internal counters that need to be reset,
143 * so we call it before returning. It will return our shared
144 * protected member variable aParticleChange that we have been modifying
145 */
146 return G4VDiscreteProcess::PostStepDoIt(track, step);
147}

149void G4eDarkBremsstrahlung::CalculateCommonXsec() {
150 // first in pair is A, second is Z
151 std::vector<std::pair<G4double, G4double>> elements = {
152 std::make_pair(183.84, 74), // tungsten
153 std::make_pair(28.085, 14) // silicon
154 };
155
156 G4double current_energy = 2.0 * GeV;
157 G4double maximum_energy = 4.0 * GeV;
158 G4double energy_step = 1.0 * MeV;
159 while (current_energy <= maximum_energy) {
160 for (auto const& [A, Z] : elements)
161 element_xsec_cache_.get(current_energy, A, Z);
162 current_energy += energy_step;
163 }
164}

166G4double G4eDarkBremsstrahlung::GetMeanFreePath(const G4Track& track, G4double,
167 G4ForceCondition*) {
168 // won't happen if it isn't applicable
169 if (not IsApplicable(*track.GetParticleDefinition())) return DBL_MAX;
170
171 G4Material* materialWeAreIn = track.GetMaterial();
172 const G4ElementVector* theElementVector = materialWeAreIn->GetElementVector();
173 const G4double* NbOfAtomsPerVolume =
174 materialWeAreIn->GetVecNbOfAtomsPerVolume();
175
176 G4double SIGMA = 0;
177 for (size_t i = 0; i < materialWeAreIn->GetNumberOfElements(); i++) {
178 G4double AtomicZ = (*theElementVector)[i]->GetZ();
179 G4double AtomicA = (*theElementVector)[i]->GetA() / (g / mole);
180 G4double energy = track.GetDynamicParticle()->GetKineticEnergy();
181
182 G4double element_xsec;
183
184 if (cache_xsec_)
185 element_xsec = element_xsec_cache_.get(energy, AtomicA, AtomicZ);
186 else
187 element_xsec =
188 model_->ComputeCrossSectionPerAtom(energy, AtomicA, AtomicZ);
189
190 SIGMA += NbOfAtomsPerVolume[i] * element_xsec;
191 }
192
193 return SIGMA > DBL_MIN ? 1. / SIGMA : DBL_MAX;
194}
