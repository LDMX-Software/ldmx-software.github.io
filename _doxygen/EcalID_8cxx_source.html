41EcalID::EcalID(unsigned int layer, unsigned int module, unsigned int u,
42 unsigned int v)
43 : EcalAbstractID(EcalAbstractID::PrecisionGlobal, 0) {
44 id_ |= (layer & LAYER_MASK) << LAYER_SHIFT;
45 id_ |= (module & MODULE_MASK) << MODULE_SHIFT;
46
47 unsigned int cell = 0;
48 if (v > max_v) {
49 EXCEPTION_RAISE("InvalidIdException",
50 "Attempted to create EcalID with invalid (u,v)=(" +
51 std::to_string(u) + "," + std::to_string(v) + ")");
52 }
53
54 if (v <= v_middle) { // simple case...
55 if (u > (base_row_w - 1 + v)) {
56 EXCEPTION_RAISE("InvalidIdException",
57 "Attempted to create EcalID with invalid (u,v)=(" +
58 std::to_string(u) + "," + std::to_string(v) + ")");
59 }
60 cell = u + v * base_row_w + (v - 1) * v / 2;
61 } else {
62 unsigned int umin = v - v_middle;
63 static unsigned int umax = 23; // constant
64 unsigned int vrel = v - v_middle - 1;
65 if (u < umin || u > umax) {
66 EXCEPTION_RAISE("InvalidIdException",
67 "Attempted to create EcalID with invalid (u,v)=(" +
68 std::to_string(u) + "," + std::to_string(v) + ")");
69 }
70 cell = 222 + (u - umin) + vrel * base_row_w + 55 -
71 (v_middle - vrel - 1) * (v_middle - vrel) / 2;
72 }
73 id_ |= (cell & CELL_MASK) << CELL_SHIFT;
74}
