61  static std::vector<std::string> splitCSV(const std::string& s) {
62  std::vector<std::string> rv;
63  std::string field;
64  bool inquote=false;
65  for (auto chr : s ) {
66  if (chr=='"') {
67  inquote=!inquote;
68  } else if ((chr==',' || chr=='\t') && !inquote) {
69  if (!field.empty()) {
70  // std::cout << "Field: '" << field << "'\n";
71  rv.push_back(field);
72  }
73  field.clear();
74  } else if (isspace(chr) && !inquote) { // do not add spaces
75  } else if (chr=='#' && !inquote) {
76  break; // comment
77  } else {
78  field+=chr;
79  }
80  }
81  if (!field.empty()) {
82  rv.push_back(field);
83  // std::cout << "Field: '" << field << "'\n";
84  }
85  return rv;

95  void loadT(T& table, std::istream& is) {
96  table.clear();
97  // first work with the header line
98  std::string line;
99  std::vector<std::string> split;
100  int iDetID=-1;
101  int iline=0;
102  size_t ncolin;
103 
104  while (!is.eof()) {
105  iline++;
106  std::getline(is,line);
107  split=splitCSV(line);
108  if (split.size()<1) {
109  split.clear();
110  continue; // need at least an id column and as many columns as requested
111  }
112  if (find(split,"DetID")==split.end() && find(split,"subdetector")==split.end()) {
113  EXCEPTION_RAISE("ConditionsException","Malformed CSV file with no DetId or subdetector column");
114  } else {
115  break;
116  }
117  }
118  if (is.eof()) {
119  EXCEPTION_RAISE("ConditionsException","CSV file has no valid header");
120  }
121  // ok, we have a header line. Do we have a DetID column?
122  std::vector<std::string>::const_iterator id=find(split,"DetID");
123  if (id!=split.end()) {
124  // good this is simpler...
125  iDetID=int(id-split.begin());
126  } else {
127  EXCEPTION_RAISE("MissingFeatureException","Cannot actually load CSV file without valid DetID column at this point");
128  }
129  ncolin=split.size();
130  // check for columns which match all those requested in the table
131  std::map<unsigned int, int> table_to_csv;
132  for (unsigned int ic=0; ic!=table.getColumnCount(); ic++) {
133  auto fc = find(split,table.getColumnName(ic));
134  if (fc==split.end()) {
135  EXCEPTION_RAISE("ConditionsException","Missing column '"+table.getColumnName(ic)+"' in CSV table load");
136  }
137  table_to_csv[ic]=int(fc-split.begin());
138  }
139 
140  // processing additional lines
141  while (!is.eof()) {
142  iline++;
143  std::getline(is,line);
144  split=splitCSV(line);
145  if (split.size()==0) continue; // ignore comment lines
146  if (split.size()!=ncolin) {
147  EXCEPTION_RAISE("ConditionsException","Mismatched number of columns (" +
148  std::to_string(split.size()) + "!=" + std::to_string(ncolin) + ") on line " + std::to_string(iline));
149  }
150  unsigned int id(0);
151  std::vector<V> values(table_to_csv.size(),0);
152  if (iDetID>=0) id=strtoul(split[iDetID].c_str(),0,0);
153  V dummy(0);
154  for (auto icopy : table_to_csv) {
155  values[icopy.first]=convert(split[icopy.second],dummy);
156  }
157  if (id!=0) {
158  table.add(id,values);
159  }
160  }
161  }
