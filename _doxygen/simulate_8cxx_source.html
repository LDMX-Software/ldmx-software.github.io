112class Hunk : public G4VUserDetectorConstruction {

116 std::string material_;
117 public:
121 Hunk(double d, const std::string& m)
122 : G4VUserDetectorConstruction(), depth_{d}, material_{m} {}
123
132 virtual G4VPhysicalVolume* Construct() final override {
133 // Get nist material manager
134 G4NistManager* nist = G4NistManager::Instance();
135 using CLHEP::mm;
136
137 G4double box_half_x{500*mm},
138 box_half_y{500*mm},
139 box_half_z{depth_/2*mm};
140 G4Material* box_mat = nist->FindOrBuildMaterial(material_);
141 if (not box_mat) {
142 throw std::runtime_error("Material '"+material_+"' unknown to G4NistManager.");
143 }
144
145 G4Material* world_mat = nist->FindOrBuildMaterial("G4_AIR");
146 if (not world_mat) {
147 throw std::runtime_error("Material 'G4_AIR' unknown to G4NistManager.");
148 }
149
150 G4double world_half_z = 2*box_half_z+2;
151 G4Box* solidWorld =
152 new G4Box("World", 1.1*box_half_x, 1.1*box_half_y, world_half_z);
153
154 G4LogicalVolume* logicWorld =
155 new G4LogicalVolume(solidWorld,
156 world_mat,
157 "World");
158
159 G4VPhysicalVolume* physWorld =
160 new G4PVPlacement(0, //no rotation
161 G4ThreeVector(), // center nudged upstream a bit
162 logicWorld, //its logical volume
163 "World", //its name
164 0, //its mother volume
165 false, //no boolean operation
166 0, //copy number
167 false); //overlaps checking
168
169 G4Box* solidBox = new G4Box("Box",
170 box_half_x, box_half_y, box_half_z);
171
172 G4LogicalVolume* logicBox = new G4LogicalVolume(solidBox,
173 box_mat, "Box");
174
175 // providing mother volume attaches us to the world volume
176 new G4PVPlacement(0, //no rotation
177 G4ThreeVector(0.,0.,box_half_z+1), //at (0,0,box_half_z+1)
178 logicBox, //its logical volume
179 "Envelope", //its name
180 logicWorld, //its mother volume
181 false, //no boolean operation
182 0, //copy number
183 false); //overlaps checking
184
185 //always return the physical World
186 return physWorld;
187 }

385void usage() {
386 std::cout <<
387 "\n"
388 "USAGE\n"
389 " g4db-simulate [options] DB-LIB NUM-EVENTS\n"
390 "\n"
391 "ARGUMENTS\n"
392 " DB-LIB : dark brem library to scale from\n"
393 " the user is expected to make sure that this argument aligns with\n"
394 " the other options (lepton, incident beam energy, A' mass, etc...)\n"
395 " NUM-EVENTS : number of events to **request**\n"
396 " since Geant4 decides when a dark brem will occurr, it is important\n"
397 " to allow some beam leptons to /not/ dark brem in the target so a realistic\n"
398 " distribution of dark brem vertices is sampled."
399 "\n"
400 "OPTIONS\n"
401 " -h, --help : print this usage and exit"
402 " --muons : run using muons (without this flag, assumes electrons)\n"
403 " -m, --ap-mass : mass of the dark photon (A') in GeV (defaults to 0.1 for electrons and 1. for muons)\n"
404 " -d, --depth : thickness of target in mm (defaults to 18 for electrons, 2000 for muons)\n"
405 " -t, --target : target material, must be findable by G4NistManager\n"
406 " (defaults to G4_W for electrons and G4_Cu for muons)\n"
407 " -o, --output : output file to write CSV data to (defaults to 'events.csv')\n"
408 " -b, --bias : biasing factor to use to encourage dark brem\n"
409 " a good starting point is generally the A' mass squared, so that is the default\n"
410 " -e, --beam : Beam energy in GeV (defaults to 4 for electrons and 100 for muons)\n"
411 " --mat-list : print the full list from G4NistManager and exit\n"
412 "\n"
413 << std::flush;
414}
