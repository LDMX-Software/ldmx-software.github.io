256 isGoodEventFile(const std::string& pass, const int& entries, const int& runs,
257 bool existColl = true, bool existObj = true)
258 : pass_(pass),
259 entries_(entries),
260 runs_(runs),
261 existCollection_(existColl),
262 existObject_(existObj) {}
263
272 bool match(const std::string& filename) const override {
273 TFile* f = TFile::Open(filename.c_str());
274 if (!f) return false;
275
276 TTreeReader events("LDMX_Events", f);
277
278 if (events.GetEntries(true) != entries_) {
279 f->Close();
280 return false;
281 }
282
283 // Event tree should _always_ have the ldmx::EventHeader
284 TTreeReaderValue<ldmx::EventHeader> header(events, "EventHeader");
285
286 if (existCollection_) {
287 // make sure collection matches pattern
288 TTreeReaderValue<std::vector<ldmx::CalorimeterHit>> collection(
289 events, ("TestCollection_" + pass_).c_str());
290 while (events.Next()) {
291 if (collection->size() != header->getEventNumber()) {
292 f->Close();
293 return false;
294 }
295 for (unsigned int i = 0; i < collection->size(); i++)
296 if (collection->at(i).getID() != header->getEventNumber() * 10 + i) {
297 f->Close();
298 return false;
299 }
300 }
301 // restart in case checking object as well
302 events.Restart();
303 } else {
304 // check to make sure collection is NOT there
305 auto t{(TTree*)f->Get("LDMX_Events")};
306 if (t and t->GetBranch(("TestCollection_" + pass_).c_str())) {
307 f->Close();
308 return false;
309 }
310 }
311
312 if (existObject_) {
313 // make sure object matches pattern
314 TTreeReaderValue<ldmx::HcalVetoResult> object(
315 events, ("TestObject_" + pass_).c_str());
316 while (events.Next()) {
317 if (object->getMaxPEHit().getID() != header->getEventNumber()) {
318 f->Close();
319 return false;
320 }
321 }
322 } else {
323 // check to make sure object is NOT there
324 auto t{(TTree*)f->Get("LDMX_Events")};
325 if (t and t->GetBranch(("TestObject_" + pass_).c_str())) {
326 f->Close();
327 return false;
328 }
329 }
330
331 TTreeReader runs("LDMX_Run", f);
332
333 if (runs.GetEntries(true) != runs_) {
334 f->Close();
335 return false;
336 }
337
338 TTreeReaderValue<ldmx::RunHeader> runHeader(runs, "RunHeader");
339
340 while (runs.Next()) {
341 if (runHeader->getRunNumber() !=
342 runHeader->getIntParameter("Should Be Run Number")) {
343 f->Close();
344 return false;
345 }
346 }
347
348 f->Close();
349
350 return true;
351 }
