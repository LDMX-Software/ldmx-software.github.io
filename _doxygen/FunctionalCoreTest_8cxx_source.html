256  bool match(const std::string& filename) const override {
257  TFile* f = TFile::Open(filename.c_str());
258  if (!f) return false;
259 
260  TTreeReader events("LDMX_Events", f);
261 
262  if (events.GetEntries(true) != entries_) {
263  f->Close();
264  return false;
265  }
266 
267  // Event tree should _always_ have the ldmx::EventHeader
268  TTreeReaderValue<ldmx::EventHeader> header(events, "EventHeader");
269 
270  if (existCollection_) {
271  // make sure collection matches pattern
272  TTreeReaderValue<std::vector<ldmx::CalorimeterHit>> collection(
273  events, ("TestCollection_" + pass_).c_str());
274  while (events.Next()) {
275  if (collection->size() != header->getEventNumber()) {
276  f->Close();
277  return false;
278  }
279  for (unsigned int i = 0; i < collection->size(); i++)
280  if (collection->at(i).getID() != header->getEventNumber() * 10 + i) {
281  f->Close();
282  return false;
283  }
284  }
285  // restart in case checking object as well
286  events.Restart();
287  } else {
288  // check to make sure collection is NOT there
289  auto t{(TTree*)f->Get("LDMX_Events")};
290  if (t and t->GetBranch(("TestCollection_" + pass_).c_str())) {
291  f->Close();
292  return false;
293  }
294  }
295 
296  if (existObject_) {
297  // make sure object matches pattern
298  TTreeReaderValue<ldmx::HcalVetoResult> object(
299  events, ("TestObject_" + pass_).c_str());
300  while (events.Next()) {
301  if (object->getMaxPEHit().getID() != header->getEventNumber()) {
302  f->Close();
303  return false;
304  }
305  }
306  } else {
307  // check to make sure object is NOT there
308  auto t{(TTree*)f->Get("LDMX_Events")};
309  if (t and t->GetBranch(("TestObject_" + pass_).c_str())) {
310  f->Close();
311  return false;
312  }
313  }
314 
315  TTreeReader runs("LDMX_Run", f);
316 
317  if (runs.GetEntries(true) != runs_) {
318  f->Close();
319  return false;
320  }
321 
322  TTreeReaderValue<ldmx::RunHeader> runHeader(runs, "RunHeader");
323 
324  while (runs.Next()) {
325  if (runHeader->getRunNumber() !=
326  runHeader->getIntParameter("Should Be Run Number")) {
327  f->Close();
328  return false;
329  }
330  }
331 
332  f->Close();
333 
334  return true;
335  }
