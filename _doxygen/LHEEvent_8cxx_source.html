13LHEEvent::LHEEvent(std::string& line) {
14 std::istringstream iss(line);
15 std::vector<std::string> tokens;
16 do {
17 std::string elem;
18 iss >> elem;
19 if (elem.size() != 0) {
20 tokens.push_back(elem);
21 }
22 } while (iss);
23
24 if (tokens.size() != 6) {
25 EXCEPTION_RAISE("TokenNum",
26 "Wrong number of tokens in LHE event information record.");
27 }
28
29 nup_ = atoi(tokens[0].c_str());
30 idprup_ = atoi(tokens[1].c_str());
31 xwgtup_ = atof(tokens[2].c_str());
32 scalup_ = atof(tokens[3].c_str());
33 aqedup_ = atof(tokens[4].c_str());
34 aqcdup_ = atof(tokens[5].c_str());
35
36 vtx_[0] = 0;
37 vtx_[1] = 0;
38 vtx_[2] = 0;
39}

81void LHEEvent::setVertex(const std::string& line) {
82 std::istringstream iss(line);
83 std::vector<std::string> tokens;
84 do {
85 std::string elem;
86 iss >> elem;
87 if (elem.size() != 0) {
88 tokens.push_back(elem);
89 }
90 } while (iss);
91
92 if (tokens.size() != 4 && tokens.size() != 5) {
93 EXCEPTION_RAISE("TokenNum",
94 "Wrong number of tokens or format in LHE event vertex "
95 "information record.");
96 }
97 vtx_[0] = atof(tokens[1].c_str());
98 vtx_[1] = atof(tokens[2].c_str());
99 vtx_[2] = atof(tokens[3].c_str());
100 if (tokens.size() > 4) {
101 vtxt_ = atof(tokens[4].c_str());
102 }
103}
