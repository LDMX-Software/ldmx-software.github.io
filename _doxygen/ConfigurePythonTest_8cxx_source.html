131TEST_CASE("Configure Python Test", "[Framework][functionality]") {
132 const std::string config_file_name{"config_python_test_config.py"};
133
134 // Arguments to pass to ConfigurePython constructor
135 char *args[1];
136
137 // Process handle
138 framework::ProcessHandle p;
139
140 // Run a check of the python configuration class without arguments.
141 SECTION("No arguments to python script") {
142 framework::ConfigurePython cfg(config_file_name, args, 0);
143 p = cfg.makeProcess();
144
145 CHECK(p->getPassName() == "test");
146 }
147
148 // Update the python config so we can pass the log frequency as a parameter.
149 std::ifstream in_file;
150 std::ofstream out_file;
151
152 in_file.open(config_file_name.c_str(), std::ios::in | std::ios::binary);
153
154 const std::string config_file_name_arg{"/tmp/config_python_test_config_arg.py"};
155 out_file.open(config_file_name_arg, std::ios::out | std::ios::binary);
156 out_file << in_file.rdbuf();
157 out_file << "import sys" << std::endl;
158 out_file << "p.logFrequency = int(sys.argv[1])" << std::endl;
159
160 in_file.close();
161 out_file.close();
162
163 // Pass the log frequency as a parameter to the process and check that it
164 // was set correctly.
165 auto correct_log_freq{9000};
166 SECTION("Single argument to python script") {
167 args[0] = "9000";
168 framework::ConfigurePython cfg(config_file_name_arg, args, 1);
169 p = cfg.makeProcess();
170
171 CHECK(p->getLogFrequency() == correct_log_freq);
172 }
173
174 // add a malformed parameter to test failing
175 in_file.open(config_file_name.c_str(), std::ios::in | std::ios::binary);
176
177 out_file.open(config_file_name_arg, std::ios::out | std::ios::binary);
178 out_file << in_file.rdbuf();
179 out_file << "p.sequence[0].bad_param = ('tuples','are','not','supported')" << std::endl;
180
181 in_file.close();
182 out_file.close();
183
184 // warning: this test will fail if the repr of a tuple changes format
185 SECTION("Bad parameter exception test") {
186 REQUIRE_THROWS_WITH(
187 std::make_unique<framework::ConfigurePython>(config_file_name_arg, args, 0),
188 ContainsSubstring("('tuples', 'are', 'not', 'supported')")
189 );
190 // we need to manually close up our python interpreter
191 // because we left during an exception without closing it above
192 Py_FinalizeEx();
193 }
