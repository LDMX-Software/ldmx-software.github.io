80void SimObjects::analyze(const framework::Event &event) {
81 static std::vector<framework::ProductTag> sp_maps, calo_colls, track_colls;
82 if (sp_maps.empty()) {
83 sp_maps = event.searchProducts("SimParticles","","");
84 if (sp_maps.size() != 1) {
85 ldmx_log(warn) << sp_maps.size() << " SimParticle maps which is not one!";
86 }
87 }
88 if (calo_colls.empty()) {
89 calo_colls = event.searchProducts("",sim_pass_,".*SimCalorimeterHit.*");
90 for (auto pt : calo_colls)
91 createCalorimeterHists(pt.name());
92 }
93 if (track_colls.empty()) {
94 track_colls = event.searchProducts("",sim_pass_,".*SimTrackerHit.*");
95 for (auto pt : track_colls)
96 createTrackerHists(pt.name());
97 }
98
99 auto const& particle_map{event.getMap<int,ldmx::SimParticle>("SimParticles")};
100 for (auto const& [track_id, particle] : particle_map) {
101 auto const& momentum{particle.getMomentum()};
102 auto const& vertex{particle.getVertex()};
103 histograms_.fill("SimParticles.E", particle.getEnergy());
104 histograms_.fill("SimParticles.px",momentum.at(0));
105 histograms_.fill("SimParticles.py",momentum.at(1));
106 histograms_.fill("SimParticles.pz",momentum.at(2));
107 histograms_.fill("SimParticles.time",particle.getTime());
108 histograms_.fill("SimParticles.pdg",particle.getPdgID());
109 histograms_.fill("SimParticles.x", vertex.at(0));
110 histograms_.fill("SimParticles.y", vertex.at(1));
111 histograms_.fill("SimParticles.z", vertex.at(2));
112 histograms_.fill("SimParticles.process", particle.getProcessType());
113 histograms_.fill("SimParticles.track_id", track_id);
114 for (auto const& parent : particle.getParents())
115 histograms_.fill("SimParticles.parent", parent);
116 for (auto const& child : particle.getDaughters())
117 histograms_.fill("SimParticles.children", child);
118
119 // PN particles are special
120 if (particle.getProcessType() == ldmx::SimParticle::ProcessType::photonNuclear) {
121 histograms_.fill("pn_child.E", particle.getEnergy());
122 histograms_.fill("pn_child.px",momentum.at(0));
123 histograms_.fill("pn_child.py",momentum.at(1));
124 histograms_.fill("pn_child.pz",momentum.at(2));
125 histograms_.fill("pn_child.time",particle.getTime());
126 histograms_.fill("pn_child.pdg",particle.getPdgID());
127 histograms_.fill("pn_child.x", vertex.at(0));
128 histograms_.fill("pn_child.y", vertex.at(1));
129 histograms_.fill("pn_child.z", vertex.at(2));
130 histograms_.fill("pn_child.track_id", track_id);
131 for (auto const& parent : particle.getParents())
132 histograms_.fill("pn_child.parent", parent);
133 for (auto const& child : particle.getDaughters())
134 histograms_.fill("pn_child.children", child);
135 }
136 } // loop over sim particle map
137
138 for (auto const& pt : calo_colls) {
139 auto const& coll{event.getCollection<ldmx::SimCalorimeterHit>(pt.name(),sim_pass_)};
140 for (auto const& hit : coll) {
141 unsigned int n_contribs{hit.getNumberOfContribs()};
142 histograms_.fill(pt.name()+".n_contribs", n_contribs);
143 for (unsigned int i_contrib{0}; i_contrib<n_contribs; i_contrib++) {
144 ldmx::SimCalorimeterHit::Contrib contrib{hit.getContrib(i_contrib)};
145 histograms_.fill(pt.name()+".incidents", contrib.incidentID);
146 histograms_.fill(pt.name()+".tracks", contrib.trackID);
147 histograms_.fill(pt.name()+".pdg", contrib.pdgCode);
148 }
149
150 histograms_.fill(pt.name()+".edep", hit.getEdep());
151 auto pos{hit.getPosition()};
152 histograms_.fill(pt.name()+".x",pos.at(0));
153 histograms_.fill(pt.name()+".y",pos.at(1));
154 histograms_.fill(pt.name()+".z",pos.at(2));
155 histograms_.fill(pt.name()+".time", hit.getTime());
156 } // loop over hits in the calorimeter collection
157 } // loop over different calorimeter hit collections
158
159 for (auto const& pt : track_colls) {
160 auto const& coll{event.getCollection<ldmx::SimTrackerHit>(pt.name(),sim_pass_)};
161 for (auto const& hit : coll) {
162 histograms_.fill(pt.name()+".particle_E", hit.getEnergy());
163 auto momentum{hit.getMomentum()};
164 histograms_.fill(pt.name()+".particle_px", momentum.at(0));
165 histograms_.fill(pt.name()+".particle_py", momentum.at(1));
166 histograms_.fill(pt.name()+".particle_pz", momentum.at(2));
167 histograms_.fill(pt.name()+".edep", hit.getEdep());
168 histograms_.fill(pt.name()+".time", hit.getTime());
169 auto pos{hit.getPosition()};
170 histograms_.fill(pt.name()+".x", pos.at(0));
171 histograms_.fill(pt.name()+".y", pos.at(1));
172 histograms_.fill(pt.name()+".z", pos.at(2));
173 histograms_.fill(pt.name()+".track", hit.getTrackID());
174 histograms_.fill(pt.name()+".pdg", hit.getPdgID());
175 } // loop over hits in the tracker collection
176 } // loop over different tracker hit collections
177
178 return;
179}
