225  void addOrMerge(const std::pair<double,double>& hit, double hit_merge_ns) {
226  auto imerge{hits_.begin()};
227  for (; imerge!=hits_.end(); imerge++)
228  if (fabs(imerge->second-hit.second)<hit_merge_ns) break;
229  if (imerge == hits_.end()) { // didn't find a match, add to the list
230  hits_.push_back(hit);
231  } else { // merge hits, shifting time to average
232  imerge->second=(imerge->second*imerge->first+hit.first*hit.second);
233  imerge->first+=hit.first;
234  imerge->second/=imerge->first;
235  }
236  }

250  double findCrossing(double low, double high, double level, double prec=0.01) {
251  // use midpoint algorithm, assumes low is below and high is above
252  double step=high-low;
253  double pt=(high+low)/2;
254  while (step>prec) {
255  double vmid=at(pt);
256  if (vmid<level) {
257  low=pt;
258  } else {
259  high=pt;
260  }
261  step=high-low;
262  pt=(high+low)/2;
263  }
264  return pt;
265  }
