258 void addOrMerge(const std::pair<double,double>& hit, double hit_merge_ns) {
259 auto imerge{hits_.begin()};
260 for (; imerge!=hits_.end(); imerge++)
261 if (fabs(imerge->second-hit.second)<hit_merge_ns) break;
262 if (imerge == hits_.end()) { // didn't find a match, add to the list
263 hits_.push_back(hit);
264 } else { // merge hits, shifting time to average
265 imerge->second=(imerge->second*imerge->first+hit.first*hit.second);
266 imerge->first+=hit.first;
267 imerge->second/=imerge->first;
268 }
269 }

283 double findCrossing(double low, double high, double level, double prec=0.01) {
284 // use midpoint algorithm, assumes low is below and high is above
285 double step=high-low;
286 double pt=(high+low)/2;
287 while (step>prec) {
288 double vmid=at(pt);
289 if (vmid<level) {
290 low=pt;
291 } else {
292 high=pt;
293 }
294 step=high-low;
295 pt=(high+low)/2;
296 }
297 return pt;
298 }
