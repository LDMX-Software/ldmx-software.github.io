78 static void httpstream(const std::string& url, std::string& strbuf,
79  int depth = 0) {
80  using boost::asio::ip::tcp;
81  boost::asio::io_service io_service;
82 
83  int locslash = url.find("//");
84  std::string hostname =
85  url.substr(locslash + 2, url.find("/", locslash + 2) - locslash - 2);
86 
87  // Get a list of endpoints corresponding to the server name.
88  tcp::socket socket(io_service);
89  try {
90  tcp::resolver resolver(io_service);
91  tcp::resolver::query query(hostname, "http");
92  tcp::resolver::iterator endpoint_iterator = resolver.resolve(query);
93  // Try each endpoint until we successfully establish a connection.
94  boost::asio::connect(socket, endpoint_iterator);
95  } catch (...) {
96  EXCEPTION_RAISE("ConditionsException",
97  "Unable to access " + url + " to load conditions table.");
98  }
99 
100  // Form the request. We specify the "Connection: close" header so that the
101  // server will close the socket after transmitting the response. This will
102  // allow us to treat all data up until the EOF as the content.
103  boost::asio::streambuf request;
104  std::ostream request_stream(&request);
105  request_stream << "GET " << url << " HTTP/1.0\r\n";
106  request_stream << "Host: " << hostname << "\r\n";
107  request_stream << "Accept: */*\r\n";
108  request_stream << "Connection: close\r\n\r\n";
109 
110  // Send the request.
111  boost::asio::write(socket, request);
112 
113  // Read the response status line. The response streambuf will automatically
114  // grow to accommodate the entire line. The growth may be limited by passing
115  // a maximum size to the streambuf constructor.
116  boost::asio::streambuf response;
117  boost::asio::read_until(socket, response, "\r\n");
118 
119  // Check that response is OK.
120  std::istream response_stream(&response);
121  std::string http_version;
122  response_stream >> http_version;
123  unsigned int status_code;
124  response_stream >> status_code;
125  std::string status_message;
126  std::getline(response_stream, status_message);
127  if (!response_stream || http_version.substr(0, 5) != "HTTP/") {
128  EXCEPTION_RAISE("ConditionsException",
129  "HTTP invalid response loading CSV file " + url +
130  " from host " + hostname);
131  }
132 
133  if (status_code >= 400) {
134  EXCEPTION_RAISE("ConditionsException",
135  "HTTP status code " + std::to_string(status_code) +
136  " loading CSV file " + url + " from host " + hostname);
137  }
138 
139  if (status_code >= 300) { // redirect...
140  // Read the response headers, which are terminated by a blank line.
141  boost::asio::read_until(socket, response, "\r\n\r\n");
142 
143  // Process the response headers.
144  std::string header;
145  std::string newurl;
146  while (std::getline(response_stream, header) && header != "\r")
147  if (header.find("Location: ") != std::string::npos)
148  newurl = header.substr(10);
149  if (newurl[newurl.size() - 1] == '\r') newurl.erase(newurl.size() - 1);
150 
151  // std::cout << "'" << newurl << "'\n";
152 
153  // std::cout << newurl << "," << newhost <<"\n";
154 
155  // read to the end for good behavior
156  boost::system::error_code error;
157  bool storing = false;
158  while (boost::asio::read(socket, response,
159  boost::asio::transfer_at_least(1), error))
160  ;
161 
162  if (depth > 10) {
163  EXCEPTION_RAISE("ConditionsException",
164  "Too much redirection in attempt to load HTTP CSV file");
165  }
166  httpstream(newurl, strbuf, depth + 1); // recursion...
167  }
168 
169  // Read until EOF, writing data to output as we go.
170  boost::system::error_code error;
171  bool storing = false;
172  while (boost::asio::read(socket, response, boost::asio::transfer_at_least(1),
173  error))
174  ;
175 
176  while (response.size() > 0) {
177  std::string s;
178  std::getline(response_stream, s);
179  if (s.empty() || s == "\r") {
180  storing = true;
181  continue;
182  }
183  if (storing) {
184  strbuf += s;
185  strbuf += '\n';
186  }
187  }

219 std::string SimpleCSVTableProvider::expandEnv(const std::string& s) const {
220  std::string retval;
221  std::string::size_type j = 0;
222  for (std::string::size_type i = s.find("${", j); i < s.size();
223  i = s.find("${", j)) {
224  if (i != j) retval.append(s, j, i - j); // append
225  j = s.find("}", i); // look for the end of request
226  std::string key = std::string(s, i + 2, j - i - 2);
227  if (key == "LDMX_CONDITION_TAG")
228  retval += getTagName();
229  else {
230  const char* cenv = getenv(key.c_str());
231  if (cenv != 0) {
232  retval += cenv;
233  } else {
234  // ERROR!
235  }
236  }
237  j++;
238  }
239  if (j < s.size()) retval.append(s, j);
240  // std::cout << s << "=>" << retval << std::endl;
