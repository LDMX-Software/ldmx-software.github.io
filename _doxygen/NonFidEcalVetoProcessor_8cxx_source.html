177 
178  clearProcessor();
179 
180  // Get the collection of digitized Ecal hits from the event.
181  const std::vector<EcalHit> ecalRecHits = event.getCollection<EcalHit>("EcalRecHits");
182  int nEcalHits = ecalRecHits.size();
183 
184  std::cout << "[ NonFidEcalVetoProcessor ] : Got " << nEcalHits << " ECal digis in event "
185  << event.getEventHeader().getEventNumber() << std::endl;
186 
187  int globalCentroid = GetShowerCentroidIDAndRMS(ecalRecHits, showerRMS_);
188  /* ~~ Fill the hit map ~~ O(n) */
189  fillHitMap(ecalRecHits, cellMap_);
190  bool doTight = true;
191  /* ~~ Fill the isolated hit maps ~~ O(n) */
192  fillIsolatedHitMap(ecalRecHits, globalCentroid, cellMap_, cellMapTightIso_, doTight);
193 
194  //Loop over the hits from the event to calculate the rest of the important quantities
195 
196  float wavgLayerHit = 0;
197  float xMean = 0;
198  float yMean = 0;
199 
200  for (const EcalHit &hit : ecalRecHits ) {
201  //Layer-wise quantities
202  LayerCellPair hit_pair = hitToPair(hit);
203  ecalLayerEdepRaw_[hit_pair.first] = ecalLayerEdepRaw_[hit_pair.first] + hit.getEnergy();
204  if (maxCellDep_ < hit.getEnergy())
205  maxCellDep_ = hit.getEnergy();
206  if (hit.getEnergy() > 0) {
207  nReadoutHits_++;
208  ecalLayerEdepReadout_[hit_pair.first] += hit.getEnergy();
209  ecalLayerTime_[hit_pair.first] += (hit.getEnergy()) * hit.getTime();
210  xMean += getCellCentroidXYPair(hit_pair.second).first * hit.getEnergy();
211  yMean += getCellCentroidXYPair(hit_pair.second).second * hit.getEnergy();
212  avgLayerHit_ += hit.getLayer();
213  wavgLayerHit += hit.getLayer() * hit.getEnergy();
214  if (deepestLayerHit_ < hit.getLayer()) {
215  deepestLayerHit_ = hit.getLayer();
216  }
217  }
218  }
219 
220  for (int iLayer = 0; iLayer < ecalLayerEdepReadout_.size(); iLayer++) {
221  for (auto cell : cellMapTightIso_[iLayer]) {
222  if (cell.second > 0) {
223  summedTightIso_ += cell.second;
224  }
225  }
226  ecalLayerTime_[iLayer] = ecalLayerTime_[iLayer] / ecalLayerEdepReadout_[iLayer];
227  summedDet_ += ecalLayerEdepReadout_[iLayer];
228  }
229 
230  if (nReadoutHits_ > 0) {
231  avgLayerHit_ /= nReadoutHits_;
232  wavgLayerHit /= summedDet_;
233  xMean /= summedDet_;
234  yMean /= summedDet_;
235  } else {
236  wavgLayerHit = 0;
237  avgLayerHit_ = 0;
238  xMean = 0;
239  yMean = 0;
240  }
241 
242  // Loop over hits a second time to find the standard deviations.
243  for (const EcalHit &hit : ecalRecHits ) {
244  LayerCellPair hit_pair = hitToPair(hit);
245  if (hit.getEnergy() > 0) {
246  xStd_ += pow((getCellCentroidXYPair(hit_pair.second).first - xMean), 2) * hit.getEnergy();
247  yStd_ += pow((getCellCentroidXYPair(hit_pair.second).second - yMean), 2) * hit.getEnergy();
248  stdLayerHit_ += pow((hit.getLayer() - wavgLayerHit), 2) * hit.getEnergy();
249  }
250  }
251 
252  if (nReadoutHits_ > 0) {
253  xStd_ = sqrt (xStd_ / summedDet_);
254  yStd_ = sqrt (yStd_ / summedDet_);
255  stdLayerHit_ = sqrt (stdLayerHit_ / summedDet_);
256  } else {
257  xStd_ = 0;
258  yStd_ = 0;
259  stdLayerHit_ = 0;
260  }
261 
262  // end loop over sim hits
263 
264  // Get the collection of Ecal scoring plane hits. If it doesn't exist,
265  // don't bother adding any truth tracking information.
266 
267  std::vector<double> recoilP;
268  std::vector<float> recoilPos;
269 
270  if (event.exists("EcalScoringPlaneHits")) {
271 
272  // Get the collection of simulated particles from the event
273  auto particleMap{event.getMap< int, SimParticle >("SimParticles")};
274 
275  // Get the ECal scoring plane hits from the event
276  auto ecalSpHits{event.getCollection< SimTrackerHit >( "EcalScoringPlaneHits" )};
277 
278  // Search for the recoil electron
279  auto [recoilTrackID, recoilElectron] = Analysis::getRecoil(particleMap);
280 
281  for ( const SimTrackerHit &spHit : ecalSpHits ) {
282 
283  if (spHit.getLayerID() != 1) continue;
284 
285  if (spHit.getTrackID() == recoilTrackID) {
286  recoilP = spHit.getMomentum();
287  recoilPos = spHit.getPosition();
288  if (recoilP[2] <= 0) continue;
289  break;
290  }
291  }
292  }
293 
294  /* Code for fiducial region below */
295 
296  std::vector<float> faceXY(2);
297 
298  if (!recoilP.empty() && recoilP[2] != 0) {
299  faceXY[0] = ((223.8 - 220.0) * (recoilP[0] / recoilP[2])) + recoilPos[0];
300  faceXY[1] = ((223.8 - 220.0) * (recoilP[1] / recoilP[2])) + recoilPos[1];
301  } else {
302  faceXY[0] = -9999.0;
303  faceXY[1] = -9999.0;
304  }
305 
306  int inside = 0;
307  int up = 0;
308  int step = 0;
309  int index;
310  float cell_radius = 5.0;
311 
312  std::vector<float>::iterator it;
313  it = std::lower_bound(mapsx.begin(), mapsx.end(), faceXY[0]);
314 
315  index = std::distance( mapsx.begin(), it);
316 
317  if (index == mapsx.size()) {
318  index += -1;
319  }
320 
321  if (!recoilP.empty() && faceXY[0] != -9999.0) {
322  while (true) {
323  std::vector<double> dis(2);
324 
325  dis[0] = faceXY[0] - mapsx[index + step];
326  dis[1] = faceXY[1] - mapsy[index + step];
327 
328  float celldis = sqrt (pow(dis[0],2) + pow(dis[1],2));
329 
330  if (celldis <= cell_radius) {
331  inside = 1;
332  break;
333  }
334 
335  if ((abs(dis[0]) > 5 && up == 0) || index + step == mapsx.size()-1) {
336  up = 1;
337  step = 0;
338  } else if ((abs(dis[0]) > 5 && up == 1) || (index + step == 0 && up == 1)) {
339  break;
340  }
341 
342  if (up == 0) {
343  step += 1;
344  } else {
345  step += -1;
346  }
347  }
348  }
349 
350  NonFidEcalVetoResult result;
351  result.setVariables(nReadoutHits_, deepestLayerHit_, inside, summedDet_, summedTightIso_, maxCellDep_,
352  showerRMS_, xStd_, yStd_, avgLayerHit_, stdLayerHit_, ecalLayerEdepReadout_, recoilP, recoilPos, faceXY);
353 
354  if (doBdt_) {
355  std::vector<float> preds(4, -1);
356  std::vector<int> res(4, -1);
357 
358  p001BDTHelper_->buildFeatureVector(bdtFeatures_, result);
359  preds[0] = p001BDTHelper_->getSinglePred(bdtFeatures_, "p001");
360  bdtFeatures_.clear();
361 
362  p01BDTHelper_->buildFeatureVector(bdtFeatures_, result);
363  preds[1] = p01BDTHelper_->getSinglePred(bdtFeatures_, "p01");
364  bdtFeatures_.clear();
365 
366  p1BDTHelper_->buildFeatureVector(bdtFeatures_, result);
367  preds[2] = p1BDTHelper_->getSinglePred(bdtFeatures_, "p1");
368  bdtFeatures_.clear();
369 
370  p0BDTHelper_->buildFeatureVector(bdtFeatures_, result);
371  preds[3] = p0BDTHelper_->getSinglePred(bdtFeatures_, "p0");
372 
373  for (int i = 0; i<nfbdtFileNames_.size(); i++) {
374  res[i] = preds[i] > bdtCutVal_[i];
375  }
376 
377  result.setVetoResult(res);
378  result.setDiscValue(preds);
379 
380  if ((res[0] || res[1] || res[2] || res[3]) && !inside) {
381  setStorageHint(hint_shouldKeep);
382  } else {
383  setStorageHint(hint_shouldDrop);
384  }
385  }
386 
387  if (!inside) {
388  setStorageHint(hint_shouldKeep);
389  } else {
390  setStorageHint(hint_shouldDrop);
391  }
392 
393  event.add("NonFidEcalVeto", result);
394  }

407  /* Function to calculate the energy weighted shower centroid */
408  int NonFidEcalVetoProcessor::GetShowerCentroidIDAndRMS(const std::vector<EcalHit> &ecalRecHits, double& showerRMS) {
409  XYCoords wgtCentroidCoords = std::make_pair<float, float>(0., 0.);
410  float sumEdep = 0;
411  int returnCellId = 1e6;
412  //Calculate Energy Weighted Centroid
413  for ( const EcalHit &hit : ecalRecHits ) {
414  LayerCellPair hit_pair = hitToPair(hit);
415  CellEnergyPair cell_energy_pair = std::make_pair(hit_pair.second, hit.getEnergy());
416  XYCoords centroidCoords = getCellCentroidXYPair(hit_pair.second);
417  wgtCentroidCoords.first = wgtCentroidCoords.first + centroidCoords.first * cell_energy_pair.second;
418  wgtCentroidCoords.second = wgtCentroidCoords.second + centroidCoords.second * cell_energy_pair.second;
419  sumEdep += cell_energy_pair.second;
420  }
421  wgtCentroidCoords.first = (sumEdep > 1E-6) ? wgtCentroidCoords.first / sumEdep : wgtCentroidCoords.first;
422  wgtCentroidCoords.second = (sumEdep > 1E-6) ? wgtCentroidCoords.second / sumEdep : wgtCentroidCoords.second;
423  //Find Nearest Cell to Centroid
424  float maxDist = 1e6;
425  for ( const EcalHit &hit : ecalRecHits ) {
426  LayerCellPair hit_pair = hitToPair(hit);
427  XYCoords centroidCoords = getCellCentroidXYPair(hit_pair.second);
428 
429  float deltaR = pow(pow((centroidCoords.first - wgtCentroidCoords.first), 2) + pow((centroidCoords.second - wgtCentroidCoords.second), 2), .5);
430  showerRMS += deltaR * hit.getEnergy();
431  if (deltaR < maxDist) {
432  maxDist = deltaR;
433  returnCellId = hit_pair.second;
434  }
435  }
436  if (sumEdep > 0)
437  showerRMS = showerRMS / sumEdep;
438  return returnCellId;
439  }

453  void NonFidEcalVetoProcessor::fillIsolatedHitMap(const std::vector<EcalHit> &ecalRecHits, float globalCentroid,
454  std::vector<std::map<int, float>>& cellMap_, std::vector<std::map<int, float>>& cellMapIso_, bool doTight) {
455  for ( const EcalHit &hit : ecalRecHits ) {
456  std::pair<bool, int> isolatedHit = std::make_pair(true, 0);
457  LayerCellPair hit_pair = hitToPair(hit);
458  if (doTight) {
459  //Disregard hits that are on the centroid.
460  if (hit_pair.second == globalCentroid)
461  continue;
462 
463  //Skip hits that are on centroid inner ring
464  if (isInShowerInnerRing(globalCentroid, hit_pair.second)) {
465  continue;
466  }
467  }
468 
469  //Skip hits that have a readout neighbor
470  std::vector<int> cellNbrIds = getInnerRingCellIds(hit_pair.second);
471 
472  //Get neighboring cell id's and try to look them up in the full cell map (constant speed algo.)
473  for (int k = 0; k < 6; k++) {
474  std::map<int, float>::iterator it = cellMap_[hit_pair.first].find(cellNbrIds[k]);
475  if (it != cellMap_[hit_pair.first].end()) {
476  isolatedHit = std::make_pair(false, cellNbrIds[k]);
477  break;
478  }
479  }
480  if (!isolatedHit.first) {
481  continue;
482  }
483  //Insert isolated hit
484  CellEnergyPair cell_energy_pair = std::make_pair(hit_pair.second, hit.getEnergy());
485  cellMapIso_[hit_pair.first].insert(cell_energy_pair);
486  }
487  }
