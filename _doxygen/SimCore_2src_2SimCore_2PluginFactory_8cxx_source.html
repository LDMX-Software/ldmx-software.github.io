81 void PluginFactory::createAction(const std::string& className,
82  const std::string& instanceName,
83  framework::config::Parameters& parameters) {
84  auto it{registeredActions_.find(className)};
85  if (it == registeredActions_.end()) {
86  EXCEPTION_RAISE("PluginFactory", "Failed to create " + className);
87  }
88 
89  auto act{it->second(instanceName, parameters)};
90 
91  std::vector<TYPE> types = act->getTypes();
92  for (auto& type : types) {
93  if (type == TYPE::RUN)
94  std::get<UserRunAction*>(actions_[TYPE::RUN])->registerAction(act);
95  else if (type == TYPE::EVENT)
96  std::get<UserEventAction*>(actions_[TYPE::EVENT])->registerAction(act);
97  else if (type == TYPE::TRACKING)
98  std::get<UserTrackingAction*>(actions_[TYPE::TRACKING])
99  ->registerAction(act);
100  else if (type == TYPE::STEPPING)
101  std::get<USteppingAction*>(actions_[TYPE::STEPPING])->registerAction(act);
102  else if (type == TYPE::STACKING)
103  std::get<UserStackingAction*>(actions_[TYPE::STACKING])
104  ->registerAction(act);
105  else
106  EXCEPTION_RAISE("PluginFactory", "User action type doesn't exist.");
107  }
108 
109  // now that the biasing is built --> put it on active list
110  ldmx_log(info) << "User Action '" << instanceName << "' of class '"
111  << className << "' has been created.";
112 }
