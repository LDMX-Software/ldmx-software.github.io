130inline void flatCov(Acts::BoundSymMatrix cov, std::vector<double>& v_cov) {
131
132 v_cov.clear();
133 v_cov.reserve(cov.rows() * (cov.rows() + 1) / 2);
134 for (int i = 0 ; i<cov.rows(); i++)
135 for (int j = i ; j<cov.cols(); j++)
136 v_cov.push_back(cov(i,j));
137}

139inline Acts::BoundSymMatrix unpackCov(const std::vector<double>& v_cov) {
140
141 Acts::BoundSymMatrix cov;
142 int e{0};
143 for (int i = 0; i < cov.rows(); i++)
144 for (int j = i; j < cov.cols(); j++) {
145 cov(i,j)=v_cov.at(e);
146 cov(j,i)=cov(i,j);
147 e++;
148 }
149
150 return cov;
151}

173inline Acts::FreeVector toFreeParameters(Acts::Vector3 pos, Acts::Vector3 mom, Acts::ActsScalar q) {
174
175 Acts::FreeVector free_params;
176 Acts::ActsScalar p = mom.norm() * Acts::UnitConstants::MeV;
177
178 free_params[Acts::eFreePos0] = pos(Acts::ePos0) * Acts::UnitConstants::mm;
179 free_params[Acts::eFreePos1] = pos(Acts::ePos1) * Acts::UnitConstants::mm;
180 free_params[Acts::eFreePos2] = pos(Acts::ePos2) * Acts::UnitConstants::mm;
181 free_params[Acts::eFreeTime] = 0.;
182 free_params[Acts::eFreeDir0] = mom(0) / mom.norm();
183 free_params[Acts::eFreeDir1] = mom(1) / mom.norm();
184 free_params[Acts::eFreeDir2] = mom(2) / mom.norm();
185 free_params[Acts::eFreeQOverP] = (q != Acts::ActsScalar(0)) ? (q / p) : 0.; //1. / p instead?
186
187 return free_params;
188}

224//Return an unbound surface along the beam axis
225inline const std::shared_ptr<Acts::Surface> unboundSurface(double surf_location) {
226
227 //Define the target surface - be careful:
228 // x - downstream
229 // y - left (when looking along x)
230 // z - up
231 // Passing identity here means that your target surface is oriented in the same way
232 Acts::RotationMatrix3 surf_rotation = Acts::RotationMatrix3::Zero();
233 //u direction along +Y
234 surf_rotation(1,0) = 1;
235 //v direction along +Z
236 surf_rotation(2,1) = 1;
237 //w direction along +X
238 surf_rotation(0,2) = 1;
239
240 Acts::Vector3 pos(surf_location, 0., 0.);
241 Acts::Translation3 surf_translation(pos);
242 Acts::Transform3 surf_transform(surf_translation * surf_rotation);
243
244 //Unbounded surface
245 const std::shared_ptr<Acts::PlaneSurface> target_surface =
246 Acts::Surface::makeShared<Acts::PlaneSurface>(surf_transform);
247
248 return Acts::Surface::makeShared<Acts::PlaneSurface>(surf_transform);
