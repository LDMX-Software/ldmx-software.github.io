127inline void flatCov(Acts::BoundSymMatrix cov, std::vector<double>& v_cov) {
128
129 v_cov.clear();
130 v_cov.reserve(cov.rows() * (cov.rows() + 1) / 2);
131 for (int i = 0 ; i<cov.rows(); i++)
132 for (int j = i ; j<cov.cols(); j++)
133 v_cov.push_back(cov(i,j));
134}

136inline Acts::BoundSymMatrix unpackCov(const std::vector<double>& v_cov) {
137
138 Acts::BoundSymMatrix cov;
139 int e{0};
140 for (int i = 0; i < cov.rows(); i++)
141 for (int j = i; j < cov.cols(); j++) {
142 cov(i,j)=v_cov.at(e);
143 cov(j,i)=cov(i,j);
144 e++;
145 }
146
147 return cov;
148}

168//Transform position, momentum and charge to free parameters
169
170inline Acts::FreeVector toFreeParameters(Acts::Vector3 pos, Acts::Vector3 mom, Acts::ActsScalar q) {
171
172 Acts::FreeVector free_params;
173 Acts::ActsScalar p = mom.norm() * Acts::UnitConstants::MeV;
174
175 free_params[Acts::eFreePos0] = pos(Acts::ePos0) * Acts::UnitConstants::mm;
176 free_params[Acts::eFreePos1] = pos(Acts::ePos1) * Acts::UnitConstants::mm;
177 free_params[Acts::eFreePos2] = pos(Acts::ePos2) * Acts::UnitConstants::mm;
178 free_params[Acts::eFreeTime] = 0.;
179 free_params[Acts::eFreeDir0] = mom(0) / mom.norm();
180 free_params[Acts::eFreeDir1] = mom(1) / mom.norm();
181 free_params[Acts::eFreeDir2] = mom(2) / mom.norm();
182 free_params[Acts::eFreeQOverP] = (q != Acts::ActsScalar(0)) ? (q / p) : 0.; //1. / p instead?
183
184 return free_params;
185}
