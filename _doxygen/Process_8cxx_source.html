116void Process::run() {
117 // set up the logging for this run
118 logging::open(logging::convertLevel(termLevelInt_),
119 logging::convertLevel(fileLevelInt_),
120 logFileName_ // if this is empty string, no file is logged to
121 );
122
123 // Counter to keep track of the number of events that have been
124 // procesed
125 auto n_events_processed{0};
126
127 // make sure the ntuple manager is in a blank state
128 NtupleManager::getInstance().reset();
129
130 // event bus for this process
131 Event theEvent(passname_);
132
133 // Start by notifying everyone that modules processing is beginning
134 conditions_.onProcessStart();
135 for (auto module : sequence_) module->onProcessStart();
136
137 // If we have no input files, but do have an event number, run for
138 // that number of events and generate an output file.
139 if (inputFiles_.empty() && eventLimit_ > 0) {
140 if (outputFiles_.empty()) {
141 EXCEPTION_RAISE("InvalidConfig",
142 "No input files or output files were given.");
143 } else if (outputFiles_.size() > 1) {
144 ldmx_log(warn) << "Several output files given with no input files. "
145 << "Only the first output file '" << outputFiles_.at(0)
146 << "' will be used.";
147 }
148 std::string outputFileName = outputFiles_.at(0);
149
150 // Configure the event file to create an output file with no parent. This
151 // requires setting the parameters isOutputFile and isSingleOutput to true.
152 EventFile outFile(config_, outputFileName, nullptr, true, true, false);
153
154 for (auto module : sequence_) module->onFileOpen(outFile);
155
156 outFile.setupEvent(&theEvent);
157
158 for (auto rule : dropKeepRules_) outFile.addDrop(rule);
159
160 ldmx::RunHeader runHeader(runForGeneration_);
161 runHeader.setRunStart(std::time(nullptr)); // set run starting
162 runHeader_ = &runHeader; // give handle to run header to process
163 outFile.writeRunHeader(runHeader); // add run header to file
164
165 newRun(runHeader);
166
167 int numTries = 0; // number of tries for the current event number
168 while (n_events_processed < eventLimit_) {
169 ldmx::EventHeader &eh = theEvent.getEventHeader();
170 eh.setRun(runForGeneration_);
171 eh.setEventNumber(n_events_processed + 1);
172 eh.setTimestamp(TTimeStamp());
173
174 // event header pointer grab
175 eventHeader_ = theEvent.getEventHeaderPtr();
176
177 numTries++;
178
179 // reset the storage controller state
180 storageController_.resetEventState();
181
182 bool completed = process(n_events_processed,theEvent);
183
184 outFile.nextEvent(storageController_.keepEvent(completed));
185
186 if (completed or numTries >= maxTries_) {
187 n_events_processed++; // increment events made
188 NtupleManager::getInstance().fill(); // fill ntuples
189 numTries = 0; // reset try counter
190 }
191
192 NtupleManager::getInstance().clear();
193 }
194
195 for (auto module : sequence_) module->onFileClose(outFile);
196
197 runHeader.setRunEnd(std::time(nullptr));
198 ldmx_log(info) << runHeader;
199 outFile.close();
200
201 } else {
202 // there are input files
203
204 EventFile *outFile(0);
205
206 bool singleOutput = false;
207 if (outputFiles_.size() == 1) {
208 singleOutput = true;
209 } else if (!outputFiles_.empty() and
210 outputFiles_.size() != inputFiles_.size()) {
211 EXCEPTION_RAISE("Process",
212 "Unable to handle case of different number of input and "
213 "output files (other than zero/one ouput file).");
214 }
215
216 // next, loop through the files
217 int ifile = 0;
218 int wasRun = -1;
219 for (auto infilename : inputFiles_) {
220 EventFile inFile(config_, infilename);
221
222 ldmx_log(info) << "Opening file " << infilename;
223
224 for (auto module : sequence_) module->onFileOpen(inFile);
225
226 // configure event file that will be iterated over
227 EventFile *masterFile;
228 if (!outputFiles_.empty()) {
229 // setup new output file if either
230 // 1) we are not in single output mode
231 // 2) this is the first input file
232 if (!singleOutput or ifile == 0) {
233 // setup new output file
234 outFile = new EventFile(config_, outputFiles_[ifile], &inFile, singleOutput);
235 ifile++;
236
237 // setup theEvent we will iterate over
238 if (outFile) {
239 outFile->setupEvent(&theEvent);
240 masterFile = outFile;
241 } else {
242 EXCEPTION_RAISE("Process", "Unable to construct output file for " +
243 outputFiles_[ifile]);
244 }
245
246 for (auto rule : dropKeepRules_) outFile->addDrop(rule);
247
248 } else {
249 // all other input files
250 outFile->updateParent(&inFile);
251 masterFile = outFile;
252
253 } // check if in singleOutput mode
254
255 } else {
256 // empty output file list, use inputFile as master file
257 inFile.setupEvent(&theEvent);
258 masterFile = &inFile;
259 }
260
261 bool event_completed = true;
262 while (masterFile->nextEvent(storageController_.keepEvent(event_completed))
263 && (eventLimit_ < 0 || (n_events_processed) < eventLimit_)) {
264 // clean up for storage control calculation
265 storageController_.resetEventState();
266
267 // event header pointer grab
268 eventHeader_ = theEvent.getEventHeaderPtr();
269
270 // notify for new run if necessary
271 if (theEvent.getEventHeader().getRun() != wasRun) {
272 wasRun = theEvent.getEventHeader().getRun();
273 try {
274 auto runHeader = masterFile->getRunHeader(wasRun);
275 runHeader_ = &runHeader; // save current run header for later
276 ldmx_log(info) << "Got new run header from '"
277 << masterFile->getFileName() << "' ...\n"
278 << runHeader;
279 newRun(runHeader);
280 } catch (const framework::exception::Exception &) {
281 ldmx_log(warn) << "Run header for run " << wasRun
282 << " was not found!";
283 }
284 }
285
286 event_completed = process(n_events_processed,theEvent);
287
288 if (event_completed) NtupleManager::getInstance().fill();
289 NtupleManager::getInstance().clear();
290
291 n_events_processed++;
292 } // loop through events
293
294 if (eventLimit_ > 0 && n_events_processed == eventLimit_) {
295 ldmx_log(info) << "Reached event limit of " << eventLimit_ << " events";
296 }
297
298 if (eventLimit_ == 0 && n_events_processed > eventLimit_) {
299 ldmx_log(warn) << "Processing interrupted";
300 }
301
302 ldmx_log(info) << "Closing file " << infilename;
303
304 for (auto module : sequence_) module->onFileClose(inFile);
305
306 inFile.close();
307
308 // Reset the event in case of multiple input files
309 theEvent.onEndOfFile();
310
311 if (outFile and !singleOutput) {
312 outFile->close();
313 delete outFile;
314 outFile = nullptr;
315 }
316
317 } // loop through input files
318
319 if (outFile) {
320 // close outFile
321 // outFile would survive to here in single output mode
322 outFile->close();
323 delete outFile;
324 outFile = nullptr;
325 }
326
327 } // are there input files? if-else tree
328
329 // close up histogram file if anything was put into it
330 if (histoTFile_) {
331 histoTFile_->Write();
332 delete histoTFile_;
333 histoTFile_ = 0;
334 }
335
336 // finally, notify everyone that we are stopping
337 for (auto module : sequence_) {
338 module->onProcessEnd();
339 }
340
341 // we're done so let's close up the logging
342 logging::close();
343}

357 TDirectory *owner{nullptr};
358
359 if (histoFilename_.empty()) {

361 EXCEPTION_RAISE("NoHistFileName",

369 } else

371 owner->cd();

376void Process::newRun(ldmx::RunHeader& header) {
377 // Producers are allowed to put parameters into
378 // the run header through 'beforeNewRun' method
379 for (auto module : sequence_)
380 if (dynamic_cast<Producer *>(module))
381 dynamic_cast<Producer *>(module)->beforeNewRun(header);
382 // now run header has been modified by Producers,
383 // it is valid to read from for everyone else in 'onNewRun'
384 conditions_.onNewRun(header);
385 for (auto module : sequence_)
386 module->onNewRun(header);
387}

389bool Process::process(int n, Event& event) const {
390 if ((logFrequency_ != -1) &&
391 ((n+1) % logFrequency_ == 0)) {
392 TTimeStamp t;
393 ldmx_log(info) << "Processing " << n+1 << " Run "
394 << event.getEventHeader().getRun() << " Event "
395 << event.getEventHeader().getEventNumber() << " ("
396 << t.AsString("lc") << ")";
397 }
398
399 try {
400 for (auto module : sequence_) {
401 if (dynamic_cast<Producer *>(module)) {
402 (dynamic_cast<Producer *>(module))->produce(event);
403 } else if (dynamic_cast<Analyzer *>(module)) {
404 (dynamic_cast<Analyzer *>(module))->analyze(event);
405 }
406 }
407 } catch (AbortEventException &) {
408 return false;
409 }
410 return true;
411}
