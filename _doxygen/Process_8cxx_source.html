124  // other objects will have their own channels
125  // the ldmx_log macro uses a variable called theLog_,
126  // so we are going to name it that for now.
127  auto theLog_{logging::makeLogger("Process")};

140  // If we have no input files, but do have an event number, run for
141  // that number of events and generate an output file.
142  if (inputFiles_.empty() && eventLimit_ > 0) {
143  if (outputFiles_.empty()) {
144  EXCEPTION_RAISE("InvalidConfig",
145  "No input files or output files were given.");
146  } else if (outputFiles_.size() > 1) {
147  ldmx_log(warn) << "Several output files given with no input files. "
148  << "Only the first output file '" << outputFiles_.at(0)
149  << "' will be used.";
150  }
151  std::string outputFileName = outputFiles_.at(0);
152 
153  // Configure the event file to create an output file with no parent. This
154  // requires setting the parameters isOutputFile and isSingleOutput to true.
155  EventFile outFile(config_, outputFileName, nullptr, true, true, false);
156

158

160 
161  for (auto rule : dropKeepRules_) outFile.addDrop(rule);
162 
163  ldmx::RunHeader runHeader(runForGeneration_);

165  runHeader_ = &runHeader; // give handle to run header to process
166  outFile.writeRunHeader(runHeader); // add run header to file
167 
168  for (auto module : sequence_)
169  if (dynamic_cast<Producer *>(module))
170  dynamic_cast<Producer *>(module)->beforeNewRun(runHeader);
171 
172  // now run header has been modified by Producers, so it is valid to read
173  // from

175  for (auto module : sequence_) module->onNewRun(runHeader);
176 
177  int numTries = 0; // number of tries for the current event number
178  while (n_events_processed < eventLimit_) {
179  ldmx::EventHeader &eh = theEvent.getEventHeader();
180  eh.setRun(runForGeneration_);
181  eh.setEventNumber(n_events_processed + 1);
182  eh.setTimestamp(TTimeStamp());
183 
184  // event header pointer grab
185  eventHeader_ = theEvent.getEventHeaderPtr();
186 
187  numTries++;
188 
189  // reset the storage controller state
190  m_storageController.resetEventState();
191 
192  if (numTries <= 1 and getLogFrequency() > 0 and
193  (eh.getEventNumber() % getLogFrequency() == 0)) {
194  TTimeStamp t;
195  ldmx_log(info) << "Processing " << n_events_processed + 1 << " Run "
196  << theEvent.getEventHeader().getRun() << " Event "
197  << theEvent.getEventHeader().getEventNumber() << " ("
198  << t.AsString("lc") << ")";
199  }
200 
201  bool eventAborted = false;

205  (dynamic_cast<Producer *>(module))->produce(theEvent);
206  } else if (dynamic_cast<Analyzer *>(module)) {
207  (dynamic_cast<Analyzer *>(module))->analyze(theEvent);
208  }
209  } catch (AbortEventException &) {

211  break;

213  }

323  for (auto module : sequence_) module->onNewRun(runHeader);

329

332  TTimeStamp t;

334  << theEvent.getEventHeader().getRun() << " Event "
335  << theEvent.getEventHeader().getEventNumber() << " ("
336  << t.AsString("lc") << ")";
337  }
338

341  try {

343  (dynamic_cast<Producer *>(module))->produce(theEvent);
344  } else if (dynamic_cast<Analyzer *>(module)) {
345  (dynamic_cast<Analyzer *>(module))->analyze(theEvent);
346  }
347  } catch (AbortEventException &) {

351  }
352

354  NtupleManager::getInstance().clear();
355 
356  n_events_processed++;
357  } // loop through events
358 
359  if (eventLimit_ > 0 && n_events_processed == eventLimit_) {
360  ldmx_log(info) << "Reached event limit of " << eventLimit_ << " events";
361  }
362 
363  if (eventLimit_ == 0 && n_events_processed > eventLimit_) {
364  ldmx_log(warn) << "Processing interrupted";
365  }
366 
367  ldmx_log(info) << "Closing file " << infilename;
368 
369  for (auto module : sequence_) module->onFileClose(inFile);
370 
371  inFile.close();
372 
373  // Reset the event in case of multiple input files
374  theEvent.onEndOfFile();
375 
376  if (outFile and !singleOutput) {
377  outFile->close();
378  delete outFile;
379  outFile = nullptr;
380  }
381 
382  } // loop through input files
383 
384  if (outFile) {
385  // close outFile
386  // outFile would survive to here in single output mode
387  outFile->close();
388  delete outFile;
389  outFile = nullptr;
390  }
391 
392  } // are there input files? if-else tree
393 
394  // close up histogram file if anything was put into it
395  if (histoTFile_) {

398  histoTFile_ = 0;
399  }
400

403  module->onProcessEnd();
404  }
405 
406  // we're done so let's close up the logging
