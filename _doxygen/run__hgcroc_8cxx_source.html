14 framework::config::Parameters parameters;
15 parameters.addParameter("clockCycle", 25.);
16 parameters.addParameter("timingJitter",0.25);
17 parameters.addParameter("nADCs", 10 );
18 parameters.addParameter("iSOI", 0 );
19 parameters.addParameter("rateUpSlope", -0.345);
20 parameters.addParameter("timeUpSlope", 70.6547);
21 parameters.addParameter("rateDnSlope", 0.140068);
22 parameters.addParameter("timeDnSlope", 87.7649);
23 parameters.addParameter("timePeak", 77.732);
24 parameters.addParameter("noiseRMS", (700.+25.*20.)*(0.162/1000.)/20.);
25 parameters.addParameter("noise",true);
26
27 conditions::DoubleTableCondition chip_conditions("RUN_HGCROC_TABLE", {
28 "PEDESTAL",
29 "MEAS_TIME",
30 "PAD_CAPACITANCE",
31 "TOT_MAX",
32 "DRAIN_RATE",
33 "GAIN",
34 "READOUT_THRESHOLD",
35 "TOA_THRESHOLD",
36 "TOT_THRESHOLD"
37 });
38 chip_conditions.setIdMask(0); // all ids are the same
39 chip_conditions.add(0, {
40 50. , //PEDESTAL
41 0.0, //MEAS_TIME - ns
42 20., //PAD_CAPACITANCE - pF
43 200., //TOT_MAX - ns - maximum time chip would be in TOT mode
44 10240. / 200., //DRAIN_RATE - fC/ns
45 gain, //GAIN - 320. fC / 1024. counts / 20 pF - conversion from ADC to mV
46 50. + 3., //READOUT_THRESHOLD - 3 ADC counts above pedestal
47 50.*gain + 5 *37*0.162/20., //TOA_THRESHOLD - mV - ~5 MIPs above pedestal
48 50.*gain + 50*37*0.162/20., //TOT_THRESHOLD - mV - ~50 MIPs above pedestal
49 });
50
51 ldmx::HgcrocEmulator hgcroc(parameters);
52 hgcroc.seedGenerator(420);
53 hgcroc.condition(chip_conditions);
54
55 double readout_threshold = gain*53.;
56 double tot_threshold = 15.76225;
57
58 float min_voltage_test{0.5};
59 float max_voltage_test{20.};
60 float voltage_step{0.1};
61 int num_voltages{int((max_voltage_test-min_voltage_test)/voltage_step)};
62
63 std::vector<float> time_tests {
64 0., // nominal in-time with light speed particle
65 0.2, // off-peak but within normal shower time
66 1.0, // very late in shower
67 20., // totally separate, probably cosmic
68 -8.0, // pre-hit (e.g. cosmic)
69 };
70
71 // arbitrary cell id
72 unsigned int cell_id = 0x14002020;
73
74 TFile f("hgcroc_emulation.root","RECREATE");
75 TTree t("hgcroc","hgcroc");
76
77 float voltage, time;
78 int adc{0}, tdc{0};
79 bool readout, sim_totmode,
80 digitized, digi_totmode;
81
82 t.Branch("sim_voltage",&voltage);
83 t.Branch("sim_time",&time);
84 t.Branch("sim_readout",&readout);
85 t.Branch("sim_totmode",&sim_totmode);
86
87 t.Branch("digitized",&digitized);
88 t.Branch("adc",&adc);
89 t.Branch("digi_totmode",&digi_totmode);
90 t.Branch("tdc",&tdc);
91
92 ldmx::HgcrocDigiCollection all_digis;
93 all_digis.setNumSamplesPerDigi(10);
94 all_digis.setSampleOfInterestIndex(0);
95
96 unsigned int last_digi=0;
97 std::vector<std::pair<double,double>> the_pulse(1,{0.,0.});
98 for (const float ti : time_tests) {
99 for (int iv{0}; iv < num_voltages; iv++) {
100 voltage = min_voltage_test + iv*voltage_step;
101 time = ti;
102 readout = (voltage > readout_threshold);
103 sim_totmode = (voltage > tot_threshold);
104
105 the_pulse[0].first = voltage;
106 the_pulse[0].second = time;
107
108 std::vector<ldmx::HgcrocDigiCollection::Sample> digi_to_add;
109 digitized = hgcroc.digitize(cell_id,the_pulse,digi_to_add);
110 if (digitized) {
111 all_digis.addDigi(cell_id, digi_to_add);
112
113 auto digi = all_digis.getDigi(last_digi++);
114
115 digi_totmode = digi.isTOT();
116 if (digi_totmode) {
117 adc = 0;
118 tdc = digi.tot();
119 } else {
120 adc = digi.soi().adc_t();
121 tdc = 0;
122 }
123 } else {
124 adc = -1;
125 tdc = -1;
126 }
127 t.Fill();
128 }
129 }
130
131 t.Write();
132 f.Close();
133}
