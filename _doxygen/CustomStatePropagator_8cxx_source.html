213void CustomStatePropagator::fillTree(int state,
214 int q,
215 const Acts::Vector3 gen_pos,
216 const Acts::Vector3 gen_mom,
217 const Acts::BoundTrackParameters* endParams) {
218
219 state_nr = state;
220 charge = q;
221 gen_x = gen_pos(0);
222 gen_y = gen_pos(1);
223 gen_z = gen_pos(2);
224
225 gen_px = gen_mom(0);
226 gen_py = gen_mom(1);
227 gen_pz = gen_mom(2);
228
229 Acts::Vector3 end_pos = endParams->position(gctx_);
230 end_x = end_pos(0);
231 end_y = end_pos(1);
232 end_z = end_pos(2);
233
234 Acts::BoundVector bound_parameters = endParams->parameters();
235 end_loc0 = bound_parameters[Acts::eBoundLoc0];
236 end_loc1 = bound_parameters[Acts::eBoundLoc1];
237
238 Acts::Vector3 end_mom = endParams->momentum();
239 end_px = end_mom(0);
240 end_py = end_mom(1);
241 end_pz = end_mom(2);
242
243 outTree_->Fill();
244}
