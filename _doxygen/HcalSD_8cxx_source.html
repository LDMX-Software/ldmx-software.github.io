30G4bool HcalSD::ProcessHits(G4Step* aStep, G4TouchableHistory* ROhist) {
31 // Determine if current particle of this step is a Geantino.
32 G4ParticleDefinition* pdef = aStep->GetTrack()->GetDefinition();
33 bool isGeantino = false;
34 if (pdef == G4Geantino::Definition() ||
35 pdef == G4ChargedGeantino::Definition()) {
36 isGeantino = true;
37 }
38
39 // Get the edep from the step.
40 G4double edep = aStep->GetTotalEnergyDeposit();
41
42 // Skip steps with no energy dep which come from non-Geantino particles.
43 if (edep == 0.0 && !isGeantino) {
44 if (verboseLevel > 2) {
45 std::cout << "CalorimeterSD skipping step with zero edep." << std::endl
46 << std::endl;
47 }
48 return false;
49 }
50
51 //---------------------------------------------------------------------------------------------------
52 // Birks' Law
53 // ===========
54 //
55 // In the case of Scintillator as active medium, we can
56 // describe the quenching effects with the Birks' law,
57 // using the expression and the coefficients taken from
58 // the paper NIM 80 (1970) 239-244 for the organic
59 // scintillator NE-102:
60 // S*dE/dr
61 // dL/dr = -----------------------------------
62 // 1 + C1*(dE/dr)
63 // with:
64 // S=1
65 // C1 = 1.29 x 10^-2 g*cm^-2*MeV^-1
66 // C2 = 9.59 x 10^-6 g^2*cm^-4*MeV^-2
67 // These are the same values used by ATLAS TileCal
68 // and CMS HCAL (and also the default in Geant3).
69 // You can try different values for these parameters,
70 // to have an idea on the uncertainties due to them,
71 // by uncommenting one of the lines below.
72 // To get the "dE/dr" that appears in the formula,
73 // which has the dimensions
74 // [ dE/dr ] = MeV * cm^2 / g
75 // we have to divide the energy deposit in MeV by the
76 // product of the step length (in cm) and the density
77 // of the scintillator:
78
79 G4double birksFactor(1.0);
80 G4double stepLength = aStep->GetStepLength() / CLHEP::cm;
81 // Do not apply Birks for gamma deposits!
82 if (stepLength > 1.0e-6) // Check, cut if necessary.
83 {
84 G4double rho = aStep->GetPreStepPoint()->GetMaterial()->GetDensity() /
85 (CLHEP::g / CLHEP::cm3);
86 G4double dedx = edep / (rho * stepLength); //[MeV*cm^2/g]
87 birksFactor = 1.0 / (1.0 + birksc1_ * dedx + birksc2_ * dedx * dedx);
88 if (aStep->GetTrack()->GetDefinition() == G4Gamma::GammaDefinition())
89 birksFactor = 1.0;
90 if (aStep->GetTrack()->GetDefinition() == G4Neutron::NeutronDefinition())
91 birksFactor = 1.0;
92 }
93
94 // Create a new cal hit.
95 G4CalorimeterHit* hit = new G4CalorimeterHit();
96
97 // Set the edep.
98 hit->setEdep(edep * birksFactor);
99
100 // Get the scintillator solid box
101 G4Box* scint = static_cast<G4Box*>(aStep->GetPreStepPoint()
102 ->GetTouchableHandle()
103 ->GetVolume()
104 ->GetLogicalVolume()
105 ->GetSolid());
106
107 // Set the step mid-point as the hit position.
108 G4StepPoint* prePoint = aStep->GetPreStepPoint();
109 G4StepPoint* postPoint = aStep->GetPostStepPoint();
110 G4ThreeVector position =
111 0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
112 G4ThreeVector localPosition = aStep->GetPreStepPoint()
113 ->GetTouchableHandle()
114 ->GetHistory()
115 ->GetTopTransform()
116 .TransformPoint(position);
117 hit->setPosition(position[0], position[1], position[2]);
118
119 // Set the global time.
120 hit->setTime(aStep->GetTrack()->GetGlobalTime());
121
122 // Create the ID for the hit.
123 int copyNum = aStep->GetPreStepPoint()
124 ->GetTouchableHandle()
125 ->GetHistory()
126 ->GetVolume(layerDepth_)
127 ->GetCopyNo();
128 int section = copyNum / 1000;
129 int layer = copyNum % 1000;
130
131 // stripID: back Hcal, segmented along y direction for now every 10 cm --
132 // alternate x-y in the future?
133 // left/right side hcal: segmented along x direction every 10 cm
134 // top/bottom side hcal: segmented along y direction every 10 cm
135
136 int stripID = -1;
137 // Odd layers have bars horizontal
138 // Even layers have bars vertical
139 // 5cm wide bars are HARD-CODED
140 if (section == ldmx::HcalID::BACK && layer % 2 == 1)
141 stripID = int((localPosition.y() + scint->GetYHalfLength()) / 50.0);
142 else if (section == ldmx::HcalID::BACK && layer % 2 == 0)
143 stripID = int((localPosition.x() + scint->GetXHalfLength()) / 50.0);
144 else
145 stripID = int((localPosition.z() + scint->GetZHalfLength()) / 50.0);
146
147 // std::cout << "---" << std::endl;
148 // std::cout << "GetXHalfLength = " << scint->GetXHalfLength() << "\t
149 // GetYHalfLength = " << scint->GetYHalfLength() << "\t GetZHalfLength = " <<
150 // scint->GetZHalfLength() << std::endl; std::cout << "xpos = " <<
151 // localPosition.x() << "\t ypos = " << localPosition.y() << "\t zpos = " <<
152 // localPosition.z() << std::endl; std::cout << "xpos_g = " << position.x() <<
153 // "\t ypos_g = " << position.y() << "\t zpos_g = " << position.z() <<
154 // std::endl; std::cout << "Layer = " << layer << "\t section = " << section
155 // << "\t strip = " << stripID << std::endl;
156
157 ldmx::HcalID id(section, layer, stripID);
158 hit->setID(id.raw());
159
160 // Set the track ID on the hit.
161 hit->setTrackID(aStep->GetTrack()->GetTrackID());
162
163 // Set the PDG code from the track.
164 hit->setPdgCode(aStep->GetTrack()->GetParticleDefinition()->GetPDGEncoding());
165
166 // do we want to set the hit coordinate in the middle of the absorber?
167 // G4ThreeVector volumePosition =
168 // aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()->GetTopTransform().Inverse().TransformPoint(G4ThreeVector());
169 // if (section==ldmx::HcalID::BACK) hit->setPosition(position[0], position[1],
170 // volumePosition.z()); elseif (section==ldmx::HcalID::TOP ||
171 // section==ldmx::HcalID::BOTTOM) hit->setPosition(position[0],
172 // volumePosition.y(), position[2]); elseif (section==ldmx::HcalID::LEFT ||
173 // section==ldmx::HcalID::RIGHT)
174 // hit->setPosition(volumePosition.x(),position[1] , position[2]);
175
176 if (this->verboseLevel > 2) {
177 std::cout << "Created new SimCalorimeterHit in detector " << this->GetName()
178 << " subdet ID <" << subdet_ << ">, layer <" << layer
179 << "> and section <" << section << ">, copynum <" << copyNum
180 << ">" << std::endl;
181 hit->Print();
182 std::cout << std::endl;
183 }
184
185 // Insert the hit into the hits collection.
186 hitsCollection_->insert(hit);
187
188 return true;
189}
