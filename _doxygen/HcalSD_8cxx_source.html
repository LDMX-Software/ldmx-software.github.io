29ldmx::HcalID HcalSD::decodeCopyNumber(const std::uint32_t copyNumber,
30 const G4ThreeVector& localPosition,
31 const G4Box* scint) {
32 const unsigned int version{copyNumber / 0x01000000};
33 if (version != 0) {
34 using Index = ldmx::PackedIndex<256, 256, 256>;
35 return ldmx::HcalID{Index(copyNumber).field2(), Index(copyNumber).field1(),
36 Index(copyNumber).field0()};
37 }
38 const auto& geometry = getCondition<ldmx::HcalGeometry>(
39 ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
40 unsigned int stripID = 0;
41 const unsigned int section = copyNumber / 1000;
42 const unsigned int layer = copyNumber % 1000;
43
44 // 5cm wide bars are HARD-CODED
45 if (section == ldmx::HcalID::BACK) {
46 if (geometry.backLayerIsHorizontal(layer)) {
47 stripID = int((localPosition.y() + scint->GetYHalfLength()) / 50.0);
48 } else {
49 stripID = int((localPosition.x() + scint->GetXHalfLength()) / 50.0);
50 }
51 } else {
52 stripID = int((localPosition.z() + scint->GetZHalfLength()) / 50.0);
53 }
54 return ldmx::HcalID{section, layer, stripID};
55}

57G4bool HcalSD::ProcessHits(G4Step* aStep, G4TouchableHistory* ROhist) {
58 // Get the edep from the step.
59 G4double edep = aStep->GetTotalEnergyDeposit();
60
61 // Skip steps with no energy dep which come from non-Geantino particles.
62 if (edep == 0.0 and not isGeantino(aStep)) {
63 if (verboseLevel > 2) {
64 std::cout << "CalorimeterSD skipping step with zero edep." << std::endl
65 << std::endl;
66 }
67 return false;
68 }
69
70 //---------------------------------------------------------------------------------------------------
71 // Birks' Law
72 // ===========
73 //
74 // In the case of Scintillator as active medium, we can
75 // describe the quenching effects with the Birks' law,
76 // using the expression and the coefficients taken from
77 // the paper NIM 80 (1970) 239-244 for the organic
78 // scintillator NE-102:
79 // S*dE/dr
80 // dL/dr = -----------------------------------
81 // 1 + C1*(dE/dr)
82 // with:
83 // S=1
84 // C1 = 1.29 x 10^-2 g*cm^-2*MeV^-1
85 // C2 = 9.59 x 10^-6 g^2*cm^-4*MeV^-2
86 // These are the same values used by ATLAS TileCal
87 // and CMS HCAL (and also the default in Geant3).
88 // You can try different values for these parameters,
89 // to have an idea on the uncertainties due to them,
90 // by uncommenting one of the lines below.
91 // To get the "dE/dr" that appears in the formula,
92 // which has the dimensions
93 // [ dE/dr ] = MeV * cm^2 / g
94 // we have to divide the energy deposit in MeV by the
95 // product of the step length (in cm) and the density
96 // of the scintillator:
97
98 G4double birksFactor(1.0);
99 G4double stepLength = aStep->GetStepLength() / CLHEP::cm;
100 // Do not apply Birks for gamma deposits!
101 if (stepLength > 1.0e-6) // Check, cut if necessary.
102 {
103 G4double rho = aStep->GetPreStepPoint()->GetMaterial()->GetDensity() /
104 (CLHEP::g / CLHEP::cm3);
105 G4double dedx = edep / (rho * stepLength); //[MeV*cm^2/g]
106 birksFactor = 1.0 / (1.0 + birksc1_ * dedx + birksc2_ * dedx * dedx);
107 if (aStep->GetTrack()->GetDefinition() == G4Gamma::GammaDefinition())
108 birksFactor = 1.0;
109 if (aStep->GetTrack()->GetDefinition() == G4Neutron::NeutronDefinition())
110 birksFactor = 1.0;
111 }
112
113 // update edep to include birksFactor
114 edep *= birksFactor;
115
116 // Create a new cal hit.
117 ldmx::SimCalorimeterHit& hit{hits_.emplace_back()};
118
119 // Get the scintillator solid box
120 G4Box* scint = static_cast<G4Box*>(aStep->GetPreStepPoint()
121 ->GetTouchableHandle()
122 ->GetVolume()
123 ->GetLogicalVolume()
124 ->GetSolid());
125
126 // Set the step mid-point as the hit position.
127 G4StepPoint* prePoint = aStep->GetPreStepPoint();
128 G4StepPoint* postPoint = aStep->GetPostStepPoint();
129 // A Geant4 "touchable" is a way to uniquely identify a particular volume,
130 // short for touchable detector element. See the detector definition and
131 // response section of the Geant4 application developers manual for details.
132 //
133 // The TouchableHandle is just a reference counted pointer to a
134 // G4TouchableHistory object, which is a concrete implementation of a
135 // G4Touchable interface.
136 //
137 auto touchableHistory{prePoint->GetTouchableHandle()->GetHistory()};
138 // Affine transform for converting between local and global coordinates
139 auto topTransform{touchableHistory->GetTopTransform()};
140 G4ThreeVector position =
141 0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
142 G4ThreeVector localPosition = topTransform.TransformPoint(position);
143 hit.setPosition(position[0], position[1], position[2]);
144
145 // Create the ID for the hit. Note 2 here corresponds to the "depth" of the
146 // geometry tree. If this changes in the GDML, this would have to be updated
147 // here. Currently, 0 corresponds to the world volume, 1 corresponds to the
148 // Hcal, and 2 to the bars/absorbers
149 int copyNum = touchableHistory->GetVolume(2)->GetCopyNo();
150 ldmx::HcalID id = decodeCopyNumber(copyNum, localPosition, scint);
151 hit.setID(id.raw());
152
153 // add one contributor for this hit with
154 // ID of ancestor incident on Cal-Region
155 // ID of this track
156 // PDG of this track
157 // EDEP (including birks factor)
158 // time of this hit
159 const G4Track* track = aStep->GetTrack();
160 int track_id = track->GetTrackID();
161 hit.addContrib(getTrackMap().findIncident(track_id), track_id,
162 track->GetParticleDefinition()->GetPDGEncoding(), edep,
163 track->GetGlobalTime());
164 //
165 // Pre/post step details for scintillator response simulation
166
167 // Convert back to mm
168 hit.setPathLength(stepLength * CLHEP::cm / CLHEP::mm);
169 hit.setVelocity(track->GetVelocity());
170 const auto& geometry = getCondition<ldmx::HcalGeometry>(
171 ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
172 // Convert pre/post step position from global coordinates to coordinates
173 // within the scintillator bar
174 const auto localPreStepPoint{
175 topTransform.TransformPoint(prePoint->GetPosition())};
176 const auto localPostStepPoint{
177 topTransform.TransformPoint(postPoint->GetPosition())};
178
179 // And rotate them to a local coordinate system for the bar that always has
180 // the same x/y/z definitions (see HcalGeometry for details)
181 auto localPrePositionRotated{geometry.rotateGlobalToLocalBarPosition(
182 {localPreStepPoint[0], localPreStepPoint[1], localPreStepPoint[2]}, id)};
183
184 auto localPostPositionRotated{geometry.rotateGlobalToLocalBarPosition(
185 {localPostStepPoint[0], localPostStepPoint[1], localPostStepPoint[2]},
186 id)};
187 hit.setPreStepPosition(localPrePositionRotated[0], localPrePositionRotated[1],
188 localPrePositionRotated[2]);
189 hit.setPostStepPosition(localPostPositionRotated[0],
190 localPostPositionRotated[1],
191 localPostPositionRotated[2]);
192 hit.setPreStepTime(prePoint->GetGlobalTime());
193 hit.setPostStepTime(postPoint->GetGlobalTime());
194
195 if (this->verboseLevel > 2) {
196 hit.Print();
197 }
198
199 return true;
200}
