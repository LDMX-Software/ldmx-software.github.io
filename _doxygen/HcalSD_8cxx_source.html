28G4bool HcalSD::ProcessHits(G4Step* aStep, G4TouchableHistory* ROhist) {
29
30 // Get the edep from the step.
31 G4double edep = aStep->GetTotalEnergyDeposit();
32
33 // Skip steps with no energy dep which come from non-Geantino particles.
34 if (edep == 0.0 and not isGeantino(aStep)) {
35 if (verboseLevel > 2) {
36 std::cout << "CalorimeterSD skipping step with zero edep." << std::endl
37 << std::endl;
38 }
39 return false;
40 }
41
42 //---------------------------------------------------------------------------------------------------
43 // Birks' Law
44 // ===========
45 //
46 // In the case of Scintillator as active medium, we can
47 // describe the quenching effects with the Birks' law,
48 // using the expression and the coefficients taken from
49 // the paper NIM 80 (1970) 239-244 for the organic
50 // scintillator NE-102:
51 // S*dE/dr
52 // dL/dr = -----------------------------------
53 // 1 + C1*(dE/dr)
54 // with:
55 // S=1
56 // C1 = 1.29 x 10^-2 g*cm^-2*MeV^-1
57 // C2 = 9.59 x 10^-6 g^2*cm^-4*MeV^-2
58 // These are the same values used by ATLAS TileCal
59 // and CMS HCAL (and also the default in Geant3).
60 // You can try different values for these parameters,
61 // to have an idea on the uncertainties due to them,
62 // by uncommenting one of the lines below.
63 // To get the "dE/dr" that appears in the formula,
64 // which has the dimensions
65 // [ dE/dr ] = MeV * cm^2 / g
66 // we have to divide the energy deposit in MeV by the
67 // product of the step length (in cm) and the density
68 // of the scintillator:
69
70
71 G4double birksFactor(1.0);
72 G4double stepLength = aStep->GetStepLength() / CLHEP::cm;
73 // Do not apply Birks for gamma deposits!
74 if (stepLength > 1.0e-6) // Check, cut if necessary.
75 {
76 G4double rho = aStep->GetPreStepPoint()->GetMaterial()->GetDensity() /
77 (CLHEP::g / CLHEP::cm3);
78 G4double dedx = edep / (rho * stepLength); //[MeV*cm^2/g]
79 birksFactor = 1.0 / (1.0 + birksc1_ * dedx + birksc2_ * dedx * dedx);
80 if (aStep->GetTrack()->GetDefinition() == G4Gamma::GammaDefinition())
81 birksFactor = 1.0;
82 if (aStep->GetTrack()->GetDefinition() == G4Neutron::NeutronDefinition())
83 birksFactor = 1.0;
84 }
85
86 // update edep to include birksFactor
87 edep *= birksFactor;
88
89 // Create a new cal hit.
90 ldmx::SimCalorimeterHit& hit{hits_.emplace_back()};
91
92 // Get the scintillator solid box
93 G4Box* scint = static_cast<G4Box*>(aStep->GetPreStepPoint()
94 ->GetTouchableHandle()
95 ->GetVolume()
96 ->GetLogicalVolume()
97 ->GetSolid());
98
99 // Set the step mid-point as the hit position.
100 G4StepPoint* prePoint = aStep->GetPreStepPoint();
101 G4StepPoint* postPoint = aStep->GetPostStepPoint();
102 G4ThreeVector position =
103 0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
104 G4ThreeVector localPosition = aStep->GetPreStepPoint()
105 ->GetTouchableHandle()
106 ->GetHistory()
107 ->GetTopTransform()
108 .TransformPoint(position);
109 hit.setPosition(position[0], position[1], position[2]);
110
111 // Create the ID for the hit.
112 int copyNum = aStep->GetPreStepPoint()
113 ->GetTouchableHandle()
114 ->GetHistory()
115 ->GetVolume(2)
116 ->GetCopyNo();
117 int section = copyNum / 1000;
118 int layer = copyNum % 1000;
119
120 // stripID: back Hcal, segmented along y direction for now every 10 cm --
121 // alternate x-y in the future?
122 // left/right side hcal: segmented along x direction every 10 cm
123 // top/bottom side hcal: segmented along y direction every 10 cm
124
125 int stripID = -1;
126 // 5cm wide bars are HARD-CODED
127 const auto& hcalGeometry{getCondition<ldmx::HcalGeometry>(ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME)};
128 if (section == ldmx::HcalID::BACK) {
129 if (hcalGeometry.layerIsHorizontal(layer)) {
130 stripID = int((localPosition.y() + scint->GetYHalfLength()) / 50.0);
131 } else {
132 stripID = int((localPosition.x() + scint->GetXHalfLength()) / 50.0);
133 }
134 } else {
135 stripID = int((localPosition.z() + scint->GetZHalfLength()) / 50.0);
136 }
137
138 // std::cout << "---" << std::endl;
139 // std::cout << "GetXHalfLength = " << scint->GetXHalfLength() << "\t
140 // GetYHalfLength = " << scint->GetYHalfLength() << "\t GetZHalfLength = " <<
141 // scint->GetZHalfLength() << std::endl; std::cout << "xpos = " <<
142 // localPosition.x() << "\t ypos = " << localPosition.y() << "\t zpos = " <<
143 // localPosition.z() << std::endl; std::cout << "xpos_g = " << position.x() <<
144 // "\t ypos_g = " << position.y() << "\t zpos_g = " << position.z() <<
145 // std::endl; std::cout << "Layer = " << layer << "\t section = " << section
146 // << "\t strip = " << stripID << std::endl;
147
148 ldmx::HcalID id(section, layer, stripID);
149 hit.setID(id.raw());
150
151 // add one contributor for this hit with
152 // ID of ancestor incident on Cal-Region
153 // ID of this track
154 // PDG of this track
155 // EDEP (including birks factor)
156 // time of this hit
157 const G4Track* track = aStep->GetTrack();
158 int track_id = track->GetTrackID();
159 hit.addContrib(getTrackMap().findIncident(track_id), track_id,
160 track->GetParticleDefinition()->GetPDGEncoding(),
161 edep, track->GetGlobalTime());
162
163 // do we want to set the hit coordinate in the middle of the absorber?
164 // G4ThreeVector volumePosition =
165 // aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()->GetTopTransform().Inverse().TransformPoint(G4ThreeVector());
166 // if (section==ldmx::HcalID::BACK) hit->setPosition(position[0], position[1],
167 // volumePosition.z()); elseif (section==ldmx::HcalID::TOP ||
168 // section==ldmx::HcalID::BOTTOM) hit->setPosition(position[0], volumePosition.y(),
169 // position[2]); elseif (section==ldmx::HcalID::LEFT || section==ldmx::HcalID::RIGHT)
170 // hit->setPosition(volumePosition.x(),position[1] , position[2]);
171
172 if (this->verboseLevel > 2) { hit.Print(); }
173
174 return true;
175}
