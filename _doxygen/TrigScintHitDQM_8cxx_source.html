98 
99 
100  // Check if the collection of digitized TrigScint hits exist. If it doesn't
101  // don't continue processing.
102  if ( !event.exists(hitCollectionName_.c_str()) ) {
103  std::cout << "No collection called " << hitCollectionName_ << std::endl;
104  return;
105  }
106 
107  // Get the collection of TrigScintHit digitized hits if the exists
108  const std::vector<TrigScintHit> TrigScintHits = event.getCollection<TrigScintHit>( hitCollectionName_);
109 
110  // Get the total hit count
111  int hitCount = TrigScintHits.size();
112  histograms_->get("n_hits_"+padName_)->Fill(hitCount);
113 
114  double totalPE{0};
115  int noiseHitCount = 0;
116 
117  // Loop through all TrigScint hits in the event
118 
119  for (const TrigScintHit &hit : TrigScintHits ) {
120 
121  histograms_->get("pe_"+padName_)->Fill(hit.getPE());
122  histograms_->get("hit_time_"+padName_)->Fill(hit.getTime());
123  histograms_->get("id_"+padName_)->Fill(hit.getBarID() );
124 
125  totalPE += hit.getPE();
126  if ( hit.getNoise()>0 ) {
127  noiseHitCount++;
128  histograms_->get("pe_noise_"+padName_)->Fill(hit.getPE());
129  histograms_->get("id_noise_"+padName_)->Fill(hit.getBarID() );
130  }
131  else { //x, y, z not set for noise hits
132  histograms_->get("x_"+padName_)->Fill( hit.getX() );
133  histograms_->get("y_"+padName_)->Fill( hit.getY() );
134  histograms_->get("z_"+padName_)->Fill( hit.getZ() );
135  }
136 
137  }
138 
139  histograms_->get("total_pe_"+padName_)->Fill(totalPE);
140  histograms_->get("n_hits_noise_"+padName_)->Fill(noiseHitCount);
