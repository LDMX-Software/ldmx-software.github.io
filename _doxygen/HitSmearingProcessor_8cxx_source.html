60// This method smears the SimHit according to two independent gaussian
61// distributions in u and v direction. Tagger and Recoil hits can be smeared
62// with different sigma factors. The hits generated in the two detectors are
63// distinguished by checking the location along the beam.
64
65ldmx::SimTrackerHit
66HitSmearingProcessor::smearSimHit(const ldmx::SimTrackerHit &hit) {
67
68 auto sim_hit_pos{hit.getPosition()};
69 ldmx::SimTrackerHit smeared_hit;
70
71 auto sigma_u{tagger_sigma_u_};
72 auto sigma_v{tagger_sigma_v_};
73
74 // Check if the sim hit is in the tagger or in the recoil to choose the
75 // smearing factor.
76 if (sim_hit_pos[2] > 0) {
77 sigma_u = recoil_sigma_u_;
78 sigma_v = recoil_sigma_v_;
79 }
80
81 // LDMX Global X, along the less sensitive direction
82 float smear_factor{(*normal_)(generator_)};
83 sim_hit_pos[0] += smear_factor * sigma_v;
84
85 // LDMX Global Y, along the sensitive direction
86 smear_factor = (*normal_)(generator_);
87 sim_hit_pos[1] += smear_factor * sigma_u;
88
89 // Fill the smeared hit
90
91 // The ID will be the same
92 smeared_hit.setID(hit.getID());
93
94 smeared_hit.setLayerID(hit.getLayerID());
95 smeared_hit.setModuleID(hit.getModuleID());
96 smeared_hit.setPosition(sim_hit_pos[0], sim_hit_pos[1], sim_hit_pos[2]);
97 smeared_hit.setEdep(hit.getEdep());
98 smeared_hit.setEnergy(hit.getEnergy());
99 smeared_hit.setTime(hit.getTime());
100
101 // Change path-length will be the same
102 smeared_hit.setPathLength(hit.getPathLength());
103
104 smeared_hit.setMomentum(hit.getMomentum()[0], hit.getMomentum()[1],
105 hit.getMomentum()[2]);
106 smeared_hit.setTrackID(hit.getTrackID());
107 smeared_hit.setPdgID(hit.getPdgID());
108}
