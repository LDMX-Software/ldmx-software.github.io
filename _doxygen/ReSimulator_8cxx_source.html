21void ReSimulator::produce(framework::Event& event) {
22 /* numEventsBegan_++; */
23 auto& eventHeader{event.getEventHeader()};
24 const auto eventNumber{eventHeader.getEventNumber()};
25 if (!resimulate_all_events) {
26 auto found = std::find(std::begin(events_to_resimulate_),
27 std::end(events_to_resimulate_), eventNumber);
28 if (found == std::end(events_to_resimulate_)) {
29 if (verbosity_ > 1) {
30 std::cout << "Skipping event: " << eventNumber
31 << " since it wasn't part of the requested events..."
32 << std::endl;
33 }
34 this->abortEvent(); // get out of processors loop
35 return;
36 }
37 }
38 if (verbosity_ > 0) {
39 std::cout << "Resimulating " << eventNumber << std::endl;
40 }
41
42 std::istringstream iss(eventHeader.getStringParameter("eventSeed"));
43 G4Random::restoreFullState(iss);
44 runManager_->ProcessOneEvent(eventNumber);
45 if (verbosity_ > 1) {
46 std::cout << "Finished with event number " << eventNumber << std::endl;
47 }
48 if (runManager_->GetCurrentEvent()->IsAborted()) {
49 runManager_->TerminateOneEvent();
50 SensitiveDetector::Factory::get().apply(
51 [](auto sd) { sd->OnFinishedEvent(); });
52 EXCEPTION_RAISE(
53 "ReSimAbortedEvent",
54 "Resimulation resulted in an aborted event, something is wrong with "
55 "the seed from event " +
56 std::to_string(eventNumber));
57 }
58
59 eventHeader.setEventNumber(++events_resimulated);
60 updateEventHeader(eventHeader);
61 saveTracks(event);
62
63 saveSDHits(event);
64
65 runManager_->TerminateOneEvent();
66}
