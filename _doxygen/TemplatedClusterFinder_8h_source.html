30  void cluster(double seed_threshold, double cutoff) {
31  int ncluster = clusters_.size();
32  double minwgt = cutoff;
33 
34  std::sort(clusters_.begin(), clusters_.end(), compClusters);
35  do {
36  bool any = false;
37  size_t mi(0),mj(0);
38 
39  int nseeds = 0;
40 
41  for (size_t i = 0; i < clusters_.size(); i++) {
42 
43  if (clusters_[i].empty()) continue;
44 
45  bool iseed = (clusters_[i].centroid().E() >= seed_threshold);
46  if (iseed) {
47  nseeds++;
48  } else {
49  // Since we sorted initially if we find a hit below seed threshold
50  // it is guaranteed that there will be no seeds after.
51  break;
52  }
53 
54  for (size_t j = i + 1; j < clusters_.size(); j++) {
55 
56  if (clusters_[j].empty() || (!iseed && clusters_[j].centroid().E() < seed_threshold)) continue;
57  double wgt = wgt_(clusters_[i],clusters_[j]);
58  if (!any || wgt < minwgt) {
59  any = true;
60  minwgt = wgt;
61  mi = i;
62  mj = j;
63  }
64  }
65  }
66 
67  nseeds_ = nseeds;
68  transitionWeights_.insert(std::pair<int, double>(ncluster, minwgt));
69 
70  if (any && minwgt < cutoff) {
71  // put the bigger one in mi
72  if (clusters_[mi].centroid().E() < clusters_[mj].centroid().E()) { std::swap(mi,mj); }
73  // now we have the smallest, merge
74  clusters_[mi].add(clusters_[mj]);
75  clusters_[mj].clear();
76  // decrement cluster count
77  ncluster--;
78  }
79 
80  } while (minwgt < cutoff && ncluster > 1);
81  finalwgt_ = minwgt;
82  }
