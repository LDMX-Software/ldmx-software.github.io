64 void PrimaryGeneratorAction::GeneratePrimaries(G4Event* event) {
65  /*
66  * Create our Event information first so that it
67  * can be accessed by everyone from now on.
68  */
69  // Make sure we aren't overwriting a different information container
70  if (event->GetUserInformation()) {
71  EXCEPTION_RAISE(
72  "Misconfig",
73  "There was a UserEventInformation attached before beginning event."
74  "\nI don't know how this happend!!");
75  }
76 
77  // Make our information container and give it to geant4
78  // G4Event owns the event information and will delete it
79  auto event_info = new UserEventInformation;
80  event->SetUserInformation(event_info);
81 
83  auto generators{manager_.getGenerators()};
84 
85  // Generate the primary vertices using the generators
86  std::for_each(generators.begin(), generators.end(),
87  [event](const auto& generator) {
88  generator->GeneratePrimaryVertex(event);
89  });
90 
91  // smear all primary vertices (if activated)
92  int nPV = event->GetNumberOfPrimaryVertex();
93  if (nPV > 0) {
94  // loop over all vertices generated
95  for (int iPV = 0; iPV < nPV; ++iPV) {
96  G4PrimaryVertex* primary_vertex = event->GetPrimaryVertex(iPV);
97 
98  // Loop over all particle associated with the primary vertex and
99  // set the generator status to 1.
100  for (int iparticle = 0; iparticle < primary_vertex->GetNumberOfParticle();
101  ++iparticle) {
102  G4PrimaryParticle* primary = primary_vertex->GetPrimary(iparticle);
103 
104  auto primary_info{dynamic_cast<UserPrimaryParticleInformation*>(
105  primary->GetUserInformation())};
106  if (not primary_info) {
107  // no user info defined
108  // ==> make a new one
109  primary_info = new UserPrimaryParticleInformation;
110  primary->SetUserInformation(primary_info);
111  } // check if primaryinfo is defined
112 
113  int hepStatus = primary_info->getHepEvtStatus();
114  if (hepStatus <= 0) {
115  // undefined hepStatus ==> set to 1
116  primary_info->setHepEvtStatus(1);
117  } // check if hepStatus defined
118 
119  } // iparticle - loop over primary particles from this vertex
120 
121  // include the weight of this primary vertex in the event weight
122  event_info->incWeight(primary_vertex->GetWeight());
123 
124  // smear beamspot if it is turned on
125  if (useBeamspot_) {
126  double x0_i = primary_vertex->GetX0();
127  double y0_i = primary_vertex->GetY0();
128  double z0_i = primary_vertex->GetZ0();
129  /*
130  * G4UniformRand returns a number in [0,1]
131  * - we shift this range so that it is [-0.5,0.5]
132  * - multiply by the width to get [-0.5*size,0.5*size]
133  * - add the initial point (in case its off center) to get
134  * [init-0.5*size, init+0.5*size]
135  */
136  double x0_f = beamspotXSize_ * (G4UniformRand() - 0.5) + x0_i;
137  double y0_f = beamspotYSize_ * (G4UniformRand() - 0.5) + y0_i;
138  double z0_f = beamspotZSize_ * (G4UniformRand() - 0.5) + z0_i;
139  primary_vertex->SetPosition(x0_f, y0_f, z0_f);
140  }
141 
142  // shift so that t=0 coincides with primaries arriving at (or coming from)
143  // the target
144  if (time_shift_primaries_) {
145  primary_vertex->SetT0(primary_vertex->GetT0() +
146  primary_vertex->GetZ0() / 299.702547);
147  }
148 
149  } // iPV - loop over primary vertices
150  } else {
151  EXCEPTION_RAISE(
152  "NoPrimaries",
153  "No primary vertices were produced by any of the generators.");
154  }
155 }
