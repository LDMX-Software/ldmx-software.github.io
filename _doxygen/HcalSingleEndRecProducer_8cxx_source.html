68std::tuple<double, double, int> HcalSingleEndRecProducer::extract_measurements(
69 const ldmx::HgcrocDigiCollection::HgcrocDigi& digi, double pedestal, double bx_shift) {
70 // sum_adc = total of all but first in-time adc measurements
71 double sum_adc{0};
72 // sum_tot = total of all tot measurements
73 int sum_tot{0};
74 // first, get time of arrival w.r.t to start BX
75 int toa_sample{0}, toa_startbx{0};
76 // get the correction for the wrong BX assignment
77 int bx_to_time{0};
78 // and figure out sample of maximum amplitude
79 int max_sample{0};
80 double max_meas{0};
81 for (std::size_t i_sample{0}; i_sample < digi.size(); i_sample++) {
82 // adc logic
83 if (i_sample > 0) sum_adc += (digi.at(i_sample).adc_t() - pedestal);
84
85 // tot logic
86 sum_tot += digi.at(i_sample).tot();
87
88 // toa logic
89 if (digi.at(i_sample).toa() > 0) {
90 if (digi.at(i_sample).toa() >= bx_shift) {
91 toa_sample = i_sample;
92 }
93 else {
94 toa_sample = i_sample + 1;
95 }
96
97 // sum toa in given bx - given that multiple bx may be associated with the TOA measurement
98 toa_startbx += digi.at(i_sample).toa() * (clock_cycle_ / 1024) + (clock_cycle_ * toa_sample);
99 }
100
101 if (digi.at(i_sample).adc_t() - pedestal > max_meas) {
102 max_meas = digi.at(i_sample).adc_t() - pedestal;
103 max_sample = i_sample;
104 }
105 }
106 // get toa
107 double toa = toa_startbx;
108
109 // get toa w.r.t the peak
110 //double toa = (max_sample - toa_sample) * clock_cycle_ - toa_startbx;
111 // get toa w.r.t the SOI
112 //toa += ((int)isoi_ - max_sample) * clock_cycle_;
113
114 return std::make_tuple(toa, sum_adc, sum_tot);
115}

129void HcalSingleEndRecProducer::produce(framework::Event& event) {
130 const auto& hcalGeometry = getCondition<ldmx::HcalGeometry>(
131 ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
132
133 const auto& conditions{
134 getCondition<HcalReconConditions>(HcalReconConditions::CONDITIONS_NAME)};
135
136 auto hcalDigis =
137 event.getObject<ldmx::HgcrocDigiCollection>(coll_name_, pass_name_);
138
139 std::vector<ldmx::HcalHit> hcalRecHits;
140
141 isoi_ = hcalDigis.getSampleOfInterestIndex();
142
143 for (auto const& digi : hcalDigis) {
144 ldmx::HcalDigiID id_digi(digi.id());
145 ldmx::HcalID id(id_digi.section(), id_digi.layer(), id_digi.strip());
146
147 // amplitude/TOT reconstruction
148 double num_mips_equivalent{0};
149 auto [toa, sum_adc, sum_tot] =
150 extract_measurements(digi, conditions.adcPedestal(digi.id()), conditions.toaCalib(digi.id(), 0));
151 auto is_adc = conditions.is_adc(digi.id(), sum_tot);
152 if (is_adc) {
153 double adc_calib = sum_adc / conditions.adcGain(digi.id(), 0);
154 num_mips_equivalent = adc_calib;
155 } else {
156 double tot_calib = conditions.linearize(digi.id(), sum_tot);
157 num_mips_equivalent = tot_calib / conditions.adcGain(digi.id(), 0);
158 }
159 int PEs = num_mips_equivalent * pe_per_mip_;
160 double reconstructed_energy =
161 num_mips_equivalent * pe_per_mip_ * mip_energy_;
162
163 // time reconstruction
164 double hitTime = toa;
165
166 // position single ended (taken directly from geometry)
167 auto position = hcalGeometry.getStripCenterPosition(id);
168
169 // reconstructed Hit
170 ldmx::HcalHit recHit;
171 recHit.setID(id.raw());
172 recHit.setXPos(position.X());
173 recHit.setYPos(position.Y());
174 recHit.setZPos(position.Z());
175 recHit.setSection(id.section());
176 recHit.setStrip(id.strip());
177 recHit.setLayer(id.layer());
178 recHit.setEnd(id_digi.end());
179 recHit.setPE(PEs);
180 recHit.setMinPE(PEs);
181 recHit.setAmplitude(num_mips_equivalent);
182 recHit.setEnergy(reconstructed_energy);
183 recHit.setTime(hitTime);
184 recHit.setIsADC(is_adc);
185 hcalRecHits.push_back(recHit);
186 }
187
188 // add collection to event bus
189 event.add(rec_coll_name_, hcalRecHits);
190}
