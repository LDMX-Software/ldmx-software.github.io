76  void EventFile::addDrop(const std::string& rule) {
77 
78  int offset;
79  bool isKeep=false,isDrop=false,isIgnore=false;
80  size_t i = rule.find("keep");
81  if( i != std::string::npos ){
82  offset = i+4 ;
83  isKeep = true;
84  }
85  i = rule.find("drop");
86  if( i != std::string::npos ){
87  offset = i+4 ;
88  isDrop = true;
89  }
90  i = rule.find("ignore");
91  if( i != std::string::npos ){
92  offset = i+6 ;
93  isIgnore = true;
94  }
95 
96  //more than one of (keep,drop,ignore) was provided => not valid rule
97  if( int(isKeep) + int(isDrop) + int(isIgnore) != 1 ) return;
98 
99  std::string srule = rule.substr(offset);
100  for (i = srule.find_first_of(" \t\n\r"); i != std::string::npos; i = srule.find_first_of(" \t\n\r"))
101  srule.erase(i, 1);
102 
103  //name of branch is not given
104  if (srule.length() == 0) return;
105 
106  //add wild card at end for matching purposes
107  if (srule.back() != '*') srule += ".*"; //add wildcard to back
108 
109  if( isKeep ) {
110  //turn both the input and output tree's on
111  //root needs . removed otherwise it gets cranky
112  srule.erase( std::remove( srule.begin(), srule.end(), '.' ) , srule.end() );
113  preCloneRules_.emplace_back( srule , true );
114  //this branch will then be copied over into output tree and be active
115  } else if( isIgnore ) {
116  //don't even read it from the input file
117  //root needs . removed otherwise it gets cranky
118  srule.erase( std::remove( srule.begin(), srule.end(), '.' ) , srule.end() );
119  preCloneRules_.emplace_back( srule , false );
120  //these branches won't be copied over into output tree
121  } else if ( isDrop ) {
122  //drop means allowing it on reading but not writing
123  // pass these regex to event bus so Event::add knows
124  event_->addDrop( srule ); //requires event_ to be set
125 
126  //root needs . removed otherwise it gets cranky
127  srule.erase( std::remove( srule.begin(), srule.end(), '.' ) , srule.end() );
128  preCloneRules_.emplace_back( srule , false );
129  //these branches won't be copied over into output tree
130  //reactivate input branch after clone
131  reactivateRules_.push_back( srule );
132  }
133  }

135  bool EventFile::nextEvent(bool storeCurrentEvent) {
136 
137  if (ientry_ < 0 && parent_) {
138  if (!parent_->tree_) {
139  EXCEPTION_RAISE("EventFile", "No event tree in the file");
140  }
141 
142  //Only clone parent tree if either
143  // 1) There is no tree setup yet (first input file)
144  // 2) This is not single output (new input file --> new output file)
145  if ( !tree_ or !isSingleOutput_ ) {
146  //clones parent_->tree_ to our tree_ keeping drop/keep rules in mind
147  //clone tree (only copies over branches that are active on input tree)
148 
149  file_->cd(); //go into output file
150 
151  for ( auto const& rulePair : preCloneRules_ )
152  parent_->tree_->SetBranchStatus( rulePair.first.c_str() , rulePair.second );
153 
154  tree_ = parent_->tree_->CloneTree(0);
155 
156  //reactivate any drop branches (drop) on input tree
157  for ( auto const& rule : reactivateRules_ )
158  parent_->tree_->SetBranchStatus( rule.c_str(), 1 );
159 
160  }
161  event_->setInputTree( parent_->tree_ );
162  event_->setOutputTree( tree_ );
163  }
164 
165  // close up the last event
166  if (ientry_ >= 0) {
167  if (isOutputFile_) {
168  event_->beforeFill();
169  if (storeCurrentEvent) tree_->Fill(); // fill the clones...
170  }
171  if (event_) {
172  event_->Clear();
173  event_->onEndOfEvent();
174  }
175  }
176 
177  if (parent_) {
178  if (!parent_->nextEvent()) {
179  return false;
180  }
181  parent_->tree_->GetEntry(parent_->ientry_);
182  ientry_ = parent_->ientry_;
183  event_->nextEvent();
184  entries_++;
185  return true;
186 
187  } else {
188 
189  // if we are reading, move the pointer
190  if (!isOutputFile_) {
191 
192  if (ientry_ + 1 >= entries_) {
193  return false;
194  }
195 
196  ientry_++;
197  tree_->LoadTree(ientry_);
198 
199  if (event_) {
200  event_->nextEvent();
201  }
202  return true;
203 
204  } else {
205  ientry_++;
206  entries_++;
207  return true;
208  }
209  }
210  return false;
211  }
