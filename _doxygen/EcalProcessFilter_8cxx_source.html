49 void EcalProcessFilter::stepping(const G4Step* step) {
50  // Get the track associated with this step.
51  auto track{step->GetTrack()};
52 
53  if (G4EventManager::GetEventManager()->GetConstCurrentEvent()->IsAborted())
54  return;
55 
56  // Get the track info and check if this track is a brem candidate
57  auto trackInfo{simcore::UserTrackInformation::get(track)};
58  if ((trackInfo != nullptr) && !trackInfo->isBremCandidate()) return;
59 
60  // Get the particles daughters.
61  auto secondaries{step->GetSecondary()};
62 
63  // Get the region the particle is currently in. Continue processing
64  // the particle only if it's in the calorimeter region.
65  if (auto region{
66  track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
67  region.compareTo("CalorimeterRegion") != 0) {
68  // If secondaries were produced outside of the volume of interest,
69  // and there aren't additional brems to process, abort the
70  // event. Otherwise, suspend the track and move on to the next
71  // brem.
72  if (secondaries->size() != 0) {
73  /*
74  std::cout << "[ EcalProcessFilter ]: "
75  <<
76  G4EventManager::GetEventManager()->GetConstCurrentEvent()->GetEventID()
77  << " secondaries outside ecal...";
78  */
79  if (getEventInfo()->bremCandidateCount() == 1) {
80  // std::cout << "aborting the event." << std::endl;
81  track->SetTrackStatus(fKillTrackAndSecondaries);
82  G4RunManager::GetRunManager()->AbortEvent();
83  currentTrack_ = nullptr;
84  } else {
85  /*
86  std::cout << "suspending the track " << track->GetTrackID()
87  << " , " << getEventInfo()->bremCandidateCount() << " brems left."
88  << std::endl;
89  */
90  currentTrack_ = track;
91  track->SetTrackStatus(fSuspend);
92  getEventInfo()->decBremCandidateCount();
93  trackInfo->tagBremCandidate(false);
94  }
95  }
96  return;
97  }
98 
99  // If the particle doesn't interact, then move on to the next step.
100  if (secondaries->size() == 0) {
101  // Check if the photon will be exiting the ecal
102  if (auto volume{track->GetNextVolume()->GetName()};
103  volume.compareTo("hcal_PV") == 0) {
104  /*
105  std::cout << "[ EcalProcessFilter ]: "
106  <<
107  G4EventManager::GetEventManager()->GetConstCurrentEvent()->GetEventID()
108  << " no secondaries when leaving ecal...";
109  */
110  if (getEventInfo()->bremCandidateCount() == 1) {
111  // std::cout << "aborting the event." << std::endl;
112  track->SetTrackStatus(fKillTrackAndSecondaries);
113  G4RunManager::GetRunManager()->AbortEvent();
114  currentTrack_ = nullptr;
115  } else {
116  /*
117  std::cout << "suspending the track " << track->GetTrackID()
118  << " , " << getEventInfo()->bremCandidateCount() << " brems left."
119  << std::endl;
120  */
121  currentTrack_ = track;
122  track->SetTrackStatus(fSuspend);
123  getEventInfo()->decBremCandidateCount();
124  trackInfo->tagBremCandidate(false);
125  }
126  }
127 
128  return;
129  } else {
130  // If the brem gamma interacts and produces secondaries, get the
131  // process used to create them.
132  auto processName{secondaries->at(0)->GetCreatorProcess()->GetProcessName()};
133 
134  // Only record the process that is being biased
135  if (!processName.contains(process_)) {
136  /*
137  std::cout << "[ EcalProcessFilter ]: "
138  <<
139  G4EventManager::GetEventManager()->GetConstCurrentEvent()->GetEventID()
140  << " not PN products...";
141  */
142  if (getEventInfo()->bremCandidateCount() == 1) {
143  // std::cout << "aborting the event." << std::endl;
144  track->SetTrackStatus(fKillTrackAndSecondaries);
145  G4RunManager::GetRunManager()->AbortEvent();
146  currentTrack_ = nullptr;
147  } else {
148  /*
149  std::cout << "suspending the track " << track->GetTrackID()
150  << " , " << getEventInfo()->bremCandidateCount() << " brems left."
151  << std::endl;
152  */
153  currentTrack_ = track;
154  track->SetTrackStatus(fSuspend);
155  getEventInfo()->decBremCandidateCount();
156  trackInfo->tagBremCandidate(false);
157  }
158  return;
159  }
160 
161  std::cout << "[ EcalProcessFilter ]: "
162  << G4EventManager::GetEventManager()
163  ->GetConstCurrentEvent()
164  ->GetEventID()
165  << " Brem photon produced " << secondaries->size()
166  << " particle via " << processName << " process." << std::endl;
167  trackInfo->tagBremCandidate(false);
168  trackInfo->setSaveFlag(true);
169  trackInfo->tagPNGamma();
170  getEventInfo()->decBremCandidateCount();
171  }
172 }
