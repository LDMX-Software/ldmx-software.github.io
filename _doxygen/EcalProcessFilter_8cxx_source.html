48  void EcalProcessFilter::stepping(const G4Step* step) {
49 
50  // Get the track associated with this step.
51  auto track{step->GetTrack()};
52 
53  // Get the track info and check if this track is a brem candidate
54  auto trackInfo{static_cast< UserTrackInformation* >(track->GetUserInformation())};
55  if ((trackInfo != nullptr) && !trackInfo->isBremCandidate()) return;
56 
57  // Get the event info to keep track of the number of brem candidates
58  auto eventInfo{
59  static_cast< UserEventInformation* >(
60  G4EventManager::GetEventManager()->GetUserInformation())};
61  if (eventInfo == nullptr) {
62  // thrown an exception
63  }
64 
65  // Get the particles daughters.
66  auto secondaries{step->GetSecondary()};
67 
68  // Get the region the particle is currently in. Continue processing
69  // the particle only if it's in the calorimeter region.
70  if (auto region{track->GetVolume()->GetLogicalVolume()->GetRegion()->GetName()};
71  region.compareTo("CalorimeterRegion") != 0) {
72 
73  // If secondaries were produced outside of the volume of interest,
74  // and there aren't additional brems to process, abort the
75  // event. Otherwise, suspend the track and move on to the next
76  // brem.
77  if (secondaries->size() != 0) {
78 
79  if (eventInfo->bremCandidateCount() == 1) {
80  track->SetTrackStatus(fKillTrackAndSecondaries);
81  G4RunManager::GetRunManager()->AbortEvent();
82  currentTrack_ = nullptr;
83  } else {
84  currentTrack_ = track;
85  track->SetTrackStatus(fSuspend);
86  eventInfo->decBremCandidateCount();
87  trackInfo->tagBremCandidate(false);
88  }
89  }
90  return;
91  }
92 
93  // If the particle doesn't interact, then move on to the next step.
94  if (secondaries->size() == 0) {
95 
96 
97  // Check if the electron will be exiting the target
98  if (auto volume{track->GetNextVolume()->GetName()}; volume.compareTo("hcal_PV") == 0) {
99  if (eventInfo->bremCandidateCount() == 1) {
100  track->SetTrackStatus(fKillTrackAndSecondaries);
101  G4RunManager::GetRunManager()->AbortEvent();
102  currentTrack_ = nullptr;
103  } else {
104  currentTrack_ = track;
105  track->SetTrackStatus(fSuspend);
106  eventInfo->decBremCandidateCount();
107  trackInfo->tagBremCandidate(false);
108  }
109  }
110 
111  return;
112  } else {
113 
114  // If the brem gamma interacts and produces secondaries, get the
115  // process used to create them.
116  auto processName{secondaries->at(0)->GetCreatorProcess()->GetProcessName()};
117 
118  // Only record the process that is being biased
119  if (!processName.contains(process_)) {
120 
121  if (eventInfo->bremCandidateCount() == 1) {
122  track->SetTrackStatus(fKillTrackAndSecondaries);
123  G4RunManager::GetRunManager()->AbortEvent();
124  currentTrack_ = nullptr;
125  } else {
126  currentTrack_ = track;
127  track->SetTrackStatus(fSuspend);
128  eventInfo->decBremCandidateCount();
129  trackInfo->tagBremCandidate(false);
130  }
131  return;
132  }
133 
134  std::cout << "[ EcalProcessFilter ]: "
135  << "Brem photon produced " << secondaries->size()
136  << " particle via " << processName << " process."
137  << std::endl;
138  trackInfo->tagBremCandidate(false);
139  trackInfo->tagPNGamma();
140  eventInfo->decBremCandidateCount();
141  eventInfo->setWeight(track->GetWeight());
142  }
143 
144  }
145 }
