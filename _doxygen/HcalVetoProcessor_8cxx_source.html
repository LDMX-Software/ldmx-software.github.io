29void HcalVetoProcessor::produce(framework::Event &event) {
30 // Get the collection of sim particles from the event
31 const std::vector<ldmx::HcalHit> hcalRecHits =
32 event.getCollection<ldmx::HcalHit>("HcalRecHits");
33
34 // Loop over all of the Hcal hits and calculate to total photoelectrons
35 // in the event.
36 float totalPe{0};
37 float maxPE{-1000};
38 const ldmx::HcalHit *maxPEHit;
39 for (const ldmx::HcalHit &hcalHit : hcalRecHits) {
40 // If the hit time is outside the readout window, don't consider it.
41 if (hcalHit.getTime() >= maxTime_) continue;
42
43 // If the hit z position is beyond the maximum HCal depth, skip it.
44 if (hcalHit.getZPos() > maxDepth_) continue;
45
46 // Get the total PE in the bar
47 float pe = hcalHit.getPE();
48
49 // Keep track of the total PE
50 totalPe += pe;
51
52 // Check that both sides of the bar have a PE value above threshold.
53 // If not, don't consider the hit. Double sided readout is only
54 // being used for the back HCal bars. For the side HCal, just
55 // use the maximum PE as before.
56 ldmx::HcalID id(hcalHit.getID());
57 if ((id.section() == ldmx::HcalID::BACK) && (hcalHit.getMinPE() < minPE_))
58 continue;
59
60 // Find the maximum PE in the list
61 if (maxPE < pe) {
62 maxPE = pe;
63 maxPEHit = &hcalHit;
64 }
65 }
66
67 // If the maximum PE found is below threshold, it passes the veto.
68 bool passesVeto = (maxPE < totalPEThreshold_);
69
70 ldmx::HcalVetoResult result;
71 result.setVetoResult(passesVeto);
72 result.setMaxPEHit(*maxPEHit);
73
74 if (passesVeto) {
75 setStorageHint(framework::hint_shouldKeep);
76 } else {
77 setStorageHint(framework::hint_shouldDrop);
78 }
79
80 event.add("HcalVeto", result);
81}
