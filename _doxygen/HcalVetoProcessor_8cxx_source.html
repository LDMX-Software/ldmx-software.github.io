30  void HcalVetoProcessor::produce(Event& event) {
31 
32  // Get the collection of sim particles from the event
33  const std::vector<HcalHit> hcalRecHits = event.getCollection<HcalHit>("HcalRecHits");
34 
35  // Loop over all of the Hcal hits and calculate to total photoelectrons
36  // in the event.
37  float totalPe{0};
38  float maxPE{-1000};
39  const HcalHit *maxPEHit;
40  for ( const HcalHit &hcalHit : hcalRecHits ) {
41 
42  // If the hit time is outside the readout window, don't consider it.
43  if (hcalHit.getTime() >= maxTime_) continue;
44 
45  // If the hit z position is beyond the maximum HCal depth, skip it.
46  if (hcalHit.getZ() > maxDepth_) continue;
47 
48  // Get the total PE in the bar
49  float pe = hcalHit.getPE();
50 
51  // Keep track of the total PE
52  totalPe += pe;
53 
54  // Check that both sides of the bar have a PE value above threshold.
55  // If not, don't consider the hit. Double sided readout is only
56  // being used for the back HCal bars. For the side HCal, just
57  // use the maximum PE as before.
58  if ( (hcalHit.getSection() == HcalSection::BACK) && (hcalHit.getMinPE() < minPE_) ) continue;
59 
60  // Find the maximum PE in the list
61  if (maxPE < pe) {
62  maxPE = pe;
63  maxPEHit = &hcalHit;
64  }
65  }
66 
67  // If the maximum PE found is below threshold, it passes the veto.
68  bool passesVeto = (maxPE < totalPEThreshold_);
69 
70  HcalVetoResult result;
71  result.setVetoResult(passesVeto);
72  result.setMaxPEHit(*maxPEHit);
73 
74  if (passesVeto) {
75  setStorageHint(hint_shouldKeep);
76  } else {
77  setStorageHint(hint_shouldDrop);
78  }
79 
80  event.add("HcalVeto", result);
81  }
82 }
