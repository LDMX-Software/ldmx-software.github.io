203  void RootPersistencyManager::writeHeader(const G4Event* anEvent) {
204 
205  // Retrieve a mutable version of the event header
206  EventHeader& eventHeader = event_->getEventHeader();
207 
208  // Set the event weight
209  double weight{1};
210  if (anEvent->GetUserInformation() != nullptr) {
211  weight = static_cast<UserEventInformation*>(anEvent->GetUserInformation())->getWeight();
212  } else if (anEvent->GetPrimaryVertex(0)) {
213  weight = anEvent->GetPrimaryVertex(0)->GetWeight();
214  }
215  eventHeader.setWeight(weight);
216 
217  // Save the state of the random engine to an output stream. A string

222  eventHeader.setStringParameter("eventSeed", stream.str());

225  void RootPersistencyManager::writeHitsCollections(const G4Event* anEvent, Event* outputEvent) {
226 
227  // Get the HC of this event.
228  G4HCofThisEvent* hce = anEvent->GetHCofThisEvent();
229  int nColl = hce->GetNumberOfCollections();
230 
231  // Loop over all hits collections.
232  for (int iColl = 0; iColl < nColl; iColl++) {
233 
234  // Get a hits collection and its name.
235  G4VHitsCollection* hc = hce->GetHC(iColl);
236  if ( ! hc ) {
237  EXCEPTION_RAISE(
238  "G4HitColl",
239  "G4VHitsCollection indexed " + std::to_string(iColl) + " returned a nullptr."
240  );
241  }
242 
243  std::string collName = hc->GetName();
244 
245  if (dynamic_cast<G4TrackerHitsCollection*>(hc) != nullptr) {
246 
247  // Write G4TrackerHit collection to output SimTrackerHit collection.
248  G4TrackerHitsCollection* trackerHitsColl = dynamic_cast<G4TrackerHitsCollection*>(hc);
249  std::vector<SimTrackerHit> outputColl;
250  writeTrackerHitsCollection( trackerHitsColl, outputColl );
251 
252  // Add hits collection to output event.
253  outputEvent->add( collName, outputColl );
254 
255  } else if (dynamic_cast<G4CalorimeterHitsCollection*>(hc) != nullptr) {
256 
257  G4CalorimeterHitsCollection* calHitsColl = dynamic_cast<G4CalorimeterHitsCollection*>(hc);
258  std::vector<SimCalorimeterHit> outputColl;
259  if (collName == EventConstants::ECAL_SIM_HITS) {
260  // Write ECal G4CalorimeterHit collection to output SimCalorimeterHit collection using helper class.
261  ecalHitIO_.writeHitsCollection(calHitsColl, outputColl );
262  } else {
263  // Write generic G4CalorimeterHit collection to output SimCalorimeterHit collection.
264  writeCalorimeterHitsCollection(calHitsColl, outputColl );
265  }
266 
267  // Add hits collection to output event.
268  outputEvent->add( collName, outputColl );
269  } //switch on type of hit collection
270 
271  } //loop through geant4 hit collections
272 
273  return;
274  }

276  void RootPersistencyManager::writeTrackerHitsCollection(G4TrackerHitsCollection* hc, std::vector<SimTrackerHit> &outputColl) {
277 
278  outputColl.clear();
279  int nHits = hc->GetSize();
280  for (int iHit = 0; iHit < nHits; iHit++) {
281  G4TrackerHit* g4hit = (G4TrackerHit*) hc->GetHit(iHit);
282  const G4ThreeVector& momentum = g4hit->getMomentum();
283  const G4ThreeVector& position = g4hit->getPosition();
284 
285  SimTrackerHit simTrackerHit;
286  simTrackerHit.setID( g4hit->getID() );
287  simTrackerHit.setTime( g4hit->getTime() );
288  simTrackerHit.setLayerID( g4hit->getLayerID() );
289  simTrackerHit.setModuleID( g4hit->getModuleID() );
290  simTrackerHit.setEdep( g4hit->getEdep() );
291  simTrackerHit.setEnergy( g4hit->getEnergy() );
292  simTrackerHit.setPathLength( g4hit->getPathLength() );
293  simTrackerHit.setTrackID( g4hit->getTrackID() );
294  simTrackerHit.setPdgID( g4hit->getPdgID() );
295  simTrackerHit.setPosition(position.x(), position.y(), position.z());
296  simTrackerHit.setMomentum(momentum.x(), momentum.y(), momentum.z());
297 
298  outputColl.push_back( simTrackerHit );
299  }
300 
301  return;
302  }

304  void RootPersistencyManager::writeCalorimeterHitsCollection(G4CalorimeterHitsCollection* hc,
305  std::vector<SimCalorimeterHit> &outputColl) {
306 
307  //get ancestral tracking information
308  auto trackMap{UserTrackingAction::getUserTrackingAction()->getTrackMap()};
309 
310  int nHits = hc->GetSize();
311  for (int iHit = 0; iHit < nHits; iHit++) {
312  G4CalorimeterHit* g4hit = (G4CalorimeterHit*) hc->GetHit(iHit);
313  const G4ThreeVector& pos = g4hit->getPosition();
314 
315  SimCalorimeterHit simHit;
316  simHit.setID( g4hit->getID() );
317  simHit.addContrib( trackMap->findIncident(g4hit->getTrackID()), g4hit->getTrackID(), g4hit->getPdgCode(), g4hit->getEdep(), g4hit->getTime() );
318  simHit.setPosition( pos.x(), pos.y(), pos.z() );
319 
320  outputColl.push_back( simHit );
321  }
322 
323  return;
324  }
