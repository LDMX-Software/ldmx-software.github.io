110 void RootPersistencyManager::writeHeader(const G4Event *anEvent) {
111 
112  // Retrieve a mutable version of the event header
113  EventHeader &eventHeader = event_->getEventHeader();
114 
115  // Set the event weight
116  double weight{1};
117  if (anEvent->GetUserInformation() != nullptr) {
118  weight = static_cast<UserEventInformation *>(anEvent->GetUserInformation())
119  ->getWeight();
120  } else if (anEvent->GetPrimaryVertex(0)) {
121  weight = anEvent->GetPrimaryVertex(0)->GetWeight();
122  }
123  eventHeader.setWeight(weight);
124 
125  // Save the state of the random engine to an output stream. A string
126  // is then extracted and saved to the event header.
127  std::ostringstream stream;
128  G4Random::saveFullState(stream);
129  // std::cout << stream.str() << std::endl;
130  eventHeader.setStringParameter("eventSeed", stream.str());
131 }

133 void RootPersistencyManager::writeHitsCollections(const G4Event *anEvent,
134  Event *outputEvent) {
135 
136  // Get the HC of this event.
137  G4HCofThisEvent *hce = anEvent->GetHCofThisEvent();
138  int nColl = hce->GetNumberOfCollections();
139 
140  // Loop over all hits collections.
141  for (int iColl = 0; iColl < nColl; iColl++) {
142 
143  // Get a hits collection and its name.
144  G4VHitsCollection *hc = hce->GetHC(iColl);
145  if (!hc) {
146  EXCEPTION_RAISE("G4HitColl", "G4VHitsCollection indexed " +
147  std::to_string(iColl) +
148  " returned a nullptr.");
149  }
150 
151  std::string collName = hc->GetName();
152 
153  if (dynamic_cast<G4TrackerHitsCollection *>(hc) != nullptr) {
154 
155  // Write G4TrackerHit collection to output SimTrackerHit collection.
156  G4TrackerHitsCollection *trackerHitsColl =
157  dynamic_cast<G4TrackerHitsCollection *>(hc);
158  std::vector<SimTrackerHit> outputColl;
159  writeTrackerHitsCollection(trackerHitsColl, outputColl);
160 
161  // Add hits collection to output event.
162  outputEvent->add(collName, outputColl);
163 
164  } else if (dynamic_cast<G4CalorimeterHitsCollection *>(hc) != nullptr) {
165 
166  G4CalorimeterHitsCollection *calHitsColl =
167  dynamic_cast<G4CalorimeterHitsCollection *>(hc);
168  std::vector<SimCalorimeterHit> outputColl;
169  if (collName == EventConstants::ECAL_SIM_HITS) {
170  // Write ECal G4CalorimeterHit collection to output SimCalorimeterHit
171  // collection using helper class.
172  ecalHitIO_.writeHitsCollection(calHitsColl, outputColl);
173  } else {
174  // Write generic G4CalorimeterHit collection to output SimCalorimeterHit
175  // collection.
176  writeCalorimeterHitsCollection(calHitsColl, outputColl);
177  }
178 
179  // Add hits collection to output event.
180  outputEvent->add(collName, outputColl);
181  } // switch on type of hit collection
182 
183  } // loop through geant4 hit collections
184 
185  return;
186 }

188 void RootPersistencyManager::writeTrackerHitsCollection(
189  G4TrackerHitsCollection *hc, std::vector<SimTrackerHit> &outputColl) {
190 
191  outputColl.clear();
192  int nHits = hc->GetSize();
193  for (int iHit = 0; iHit < nHits; iHit++) {
194  G4TrackerHit *g4hit = (G4TrackerHit *)hc->GetHit(iHit);
195  const G4ThreeVector &momentum = g4hit->getMomentum();
196  const G4ThreeVector &position = g4hit->getPosition();
197 
198  SimTrackerHit simTrackerHit;
199  simTrackerHit.setID(g4hit->getID());
200  simTrackerHit.setTime(g4hit->getTime());
201  simTrackerHit.setLayerID(g4hit->getLayerID());
202  simTrackerHit.setModuleID(g4hit->getModuleID());
203  simTrackerHit.setEdep(g4hit->getEdep());
204  simTrackerHit.setEnergy(g4hit->getEnergy());
205  simTrackerHit.setPathLength(g4hit->getPathLength());
206  simTrackerHit.setTrackID(g4hit->getTrackID());
207  simTrackerHit.setPdgID(g4hit->getPdgID());
208  simTrackerHit.setPosition(position.x(), position.y(), position.z());
209  simTrackerHit.setMomentum(momentum.x(), momentum.y(), momentum.z());
210 
211  outputColl.push_back(simTrackerHit);
212  }
213 
214  return;
215 }

217 void RootPersistencyManager::writeCalorimeterHitsCollection(
218  G4CalorimeterHitsCollection *hc,
219  std::vector<SimCalorimeterHit> &outputColl) {
220 
221  // get ancestral tracking information
222  auto trackMap{UserTrackingAction::getUserTrackingAction()->getTrackMap()};
223 
224  int nHits = hc->GetSize();
225  for (int iHit = 0; iHit < nHits; iHit++) {
226  G4CalorimeterHit *g4hit = (G4CalorimeterHit *)hc->GetHit(iHit);
227  const G4ThreeVector &pos = g4hit->getPosition();
228 
229  SimCalorimeterHit simHit;
230  simHit.setID(g4hit->getID());
231  simHit.addContrib(trackMap->findIncident(g4hit->getTrackID()),
232  g4hit->getTrackID(), g4hit->getPdgCode(),
233  g4hit->getEdep(), g4hit->getTime());
234  simHit.setPosition(pos.x(), pos.y(), pos.z());
235 
236  outputColl.push_back(simHit);
237  }
238 
239  return;
240 }
