103 void RootPersistencyManager::writeHeader(const G4Event *anEvent) {
104  // Retrieve a mutable version of the event header
105  ldmx::EventHeader &eventHeader = event_->getEventHeader();
106 
107  auto event_info{
108  static_cast<UserEventInformation *>(anEvent->GetUserInformation())};
109 
110  eventHeader.setWeight(event_info->getWeight());
111  eventHeader.setFloatParameter("total_photonuclear_energy",
112  event_info->getPNEnergy());
113  eventHeader.setFloatParameter("total_electronuclear_energy",
114  event_info->getENEnergy());
115 
116  // Save the state of the random engine to an output stream. A string
117  // is then extracted and saved to the event header.
118  std::ostringstream stream;
119  G4Random::saveFullState(stream);
120  // std::cout << stream.str() << std::endl;
121  eventHeader.setStringParameter("eventSeed", stream.str());
122 }

124 void RootPersistencyManager::writeHitsCollections(
125  const G4Event *anEvent, framework::Event *outputEvent) {
126  // Get the HC of this event.
127  G4HCofThisEvent *hce = anEvent->GetHCofThisEvent();
128  int nColl = hce->GetNumberOfCollections();
129 
130  // Loop over all hits collections.
131  for (int iColl = 0; iColl < nColl; iColl++) {
132  // Get a hits collection and its name.
133  G4VHitsCollection *hc = hce->GetHC(iColl);
134  if (!hc) {
135  EXCEPTION_RAISE("G4HitColl", "G4VHitsCollection indexed " +
136  std::to_string(iColl) +
137  " returned a nullptr.");
138  }
139 
140  std::string collName = hc->GetName();
141 
142  if (dynamic_cast<G4TrackerHitsCollection *>(hc) != nullptr) {
143  // Write G4TrackerHit collection to output SimTrackerHit collection.
144  G4TrackerHitsCollection *trackerHitsColl =
145  dynamic_cast<G4TrackerHitsCollection *>(hc);
146  std::vector<ldmx::SimTrackerHit> outputColl;
147  writeTrackerHitsCollection(trackerHitsColl, outputColl);
148 
149  // Add hits collection to output event.
150  outputEvent->add(collName, outputColl);
151 
152  } else if (dynamic_cast<G4CalorimeterHitsCollection *>(hc) != nullptr) {
153  G4CalorimeterHitsCollection *calHitsColl =
154  dynamic_cast<G4CalorimeterHitsCollection *>(hc);
155  std::vector<ldmx::SimCalorimeterHit> outputColl;
156  if (collName == ldmx::EventConstants::ECAL_SIM_HITS) {
157  // Write ECal G4CalorimeterHit collection to output SimCalorimeterHit
158  // collection using helper class.
159  ecalHitIO_.writeHitsCollection(calHitsColl, outputColl);
160  } else {
161  // Write generic G4CalorimeterHit collection to output SimCalorimeterHit
162  // collection.
163  writeCalorimeterHitsCollection(calHitsColl, outputColl);
164  }
165 
166  // Add hits collection to output event.
167  outputEvent->add(collName, outputColl);
168  } // switch on type of hit collection
169 
170  } // loop through geant4 hit collections
171 
172  return;
173 }

175 void RootPersistencyManager::writeTrackerHitsCollection(
176  G4TrackerHitsCollection *hc, std::vector<ldmx::SimTrackerHit> &outputColl) {
177  outputColl.clear();
178  int nHits = hc->GetSize();
179  for (int iHit = 0; iHit < nHits; iHit++) {
180  G4TrackerHit *g4hit = (G4TrackerHit *)hc->GetHit(iHit);
181  const G4ThreeVector &momentum = g4hit->getMomentum();
182  const G4ThreeVector &position = g4hit->getPosition();
183 
184  ldmx::SimTrackerHit simTrackerHit;
185  simTrackerHit.setID(g4hit->getID());
186  simTrackerHit.setTime(g4hit->getTime());
187  simTrackerHit.setLayerID(g4hit->getLayerID());
188  simTrackerHit.setModuleID(g4hit->getModuleID());
189  simTrackerHit.setEdep(g4hit->getEdep());
190  simTrackerHit.setEnergy(g4hit->getEnergy());
191  simTrackerHit.setPathLength(g4hit->getPathLength());
192  simTrackerHit.setTrackID(g4hit->getTrackID());
193  simTrackerHit.setPdgID(g4hit->getPdgID());
194  simTrackerHit.setPosition(position.x(), position.y(), position.z());
195  simTrackerHit.setMomentum(momentum.x(), momentum.y(), momentum.z());
196 
197  outputColl.push_back(simTrackerHit);
198  }
199 
200  return;
201 }

203 void RootPersistencyManager::writeCalorimeterHitsCollection(
204  G4CalorimeterHitsCollection *hc,
205  std::vector<ldmx::SimCalorimeterHit> &outputColl) {
206  // get ancestral tracking information
207  auto trackMap{UserTrackingAction::getUserTrackingAction()->getTrackMap()};
208 
209  int nHits = hc->GetSize();
210  for (int iHit = 0; iHit < nHits; iHit++) {
211  G4CalorimeterHit *g4hit = (G4CalorimeterHit *)hc->GetHit(iHit);
212  const G4ThreeVector &pos = g4hit->getPosition();
213 
214  ldmx::SimCalorimeterHit simHit;
215  simHit.setID(g4hit->getID());
216  simHit.addContrib(trackMap->findIncident(g4hit->getTrackID()),
217  g4hit->getTrackID(), g4hit->getPdgCode(),
218  g4hit->getEdep(), g4hit->getTime());
219  simHit.setPosition(pos.x(), pos.y(), pos.z());
220 
221  outputColl.push_back(simHit);
222  }
223 
224  return;
225 }
