25class Reader {
26 const std::vector<uint8_t> &buffer_;
27 std::size_t i_word_;
28 uint32_t next() {
29 uint32_t w = buffer_.at(i_word_) |
30 (buffer_.at(i_word_ + 1) << 8) |
31 (buffer_.at(i_word_ + 2) << 16) |
32 (buffer_.at(i_word_ + 3) << 24);
33 i_word_ += 4;
34 return w;
35 }
36
37 public:
38 Reader(const std::vector<uint8_t>& b) : buffer_{b}, i_word_{0} {}
39 operator bool() {
40 return (i_word_ < buffer_.size());
41 }
42 Reader& operator>>(uint32_t& w) {
43 if (*this)
44 w = next();
45 return *this;
46 }
47}; // Reader

135 template<typename ReaderType>
136 std::map<ldmx::HcalElectronicsID, std::vector<ldmx::HgcrocDigiCollection::Sample>>
137 read(ReaderType& reader, PolarfireEventHeader& eh) {
141 static const unsigned int common_mode_channel = roc_version_ == 2 ? 19 : 1;
142 static const unsigned int calib_channel = 20;
144 static uint32_t head1, head2, w;
145
146 // special header words not counted in event length
147 do {
148 reader >> head1;
149#ifdef DEBUG
150 if (head1 == 0xbeef2021) {
151 std::cout << "Signal words imply version 1" << std::endl;
152 } else if (head1 == 0xbeef2022) {
153 std::cout << "Signal words imply version 2" << std::endl;
154 } else {
155 std::cout << "Extra header (inserted by rogue): " << debug::hex(head1) << std::endl;
156 }
157#endif
158 } while (head1 != 0xbeef2021 and head1 != 0xbeef2022);
159
163 long int eventlen;
164 long int i_event{0};
165 /* whole event header word looks like
166 *
167 * VERSION (4) | FPGA ID (8) | NSAMPLES (4) | LEN (16)
168 */
169 reader >> head1; i_event++;
170
171 eh.version = (head1 >> 28) & packing::utility::mask<4>;
172 eh.fpga = (head1 >> 20) & packing::utility::mask<8>;
173 eh.nsamples = (head1 >> 16) & packing::utility::mask<4>;
174 eventlen = head1 & packing::utility::mask<16>;
175 if (eh.version == 1u) {
176 // eventlen is 32-bit words in event
177 // do nothing here
178 } else if (eh.version == 2u) {
179 // eventlen is 64-bit words in event,
180 // need to multiply by 2 to get actual 32-bit event length
181 eventlen *= 2;
182 // and subtract off the special header word above
183 eventlen -= 1;
184 } else {
185 EXCEPTION_RAISE("VersMis",
186 "HcalRawDecoder only knows version 1 and 2 of DAQ format.");
187 }
188#ifdef DEBUG
189 std::cout << debug::hex(head1)
190 << " EventHeader(version = " << eh.version
191 << ", fpga = " << eh.fpga
192 << ", nsamples = " << eh.nsamples
193 << ", eventlen = " << eventlen
194 << ")" << std::endl;
195 std::cout << "Sample Lenghts: ";
196#endif
197 // sample counters
198 int n_words{0};
199 std::vector<uint32_t> length_per_sample(eh.nsamples, 0);
200 for (uint32_t i_sample{0}; i_sample < eh.nsamples; i_sample++) {
201 if (i_sample%2 == 0) {
202 n_words++;
203 reader >> w; i_event++;
204 }
205 uint32_t shift_in_word = 16*(i_sample%2);
206 length_per_sample[i_sample] = (w >> shift_in_word) & packing::utility::mask<12>;
207#ifdef DEBUG
208 std::cout << "len(" << i_sample << ") = " << length_per_sample[i_sample] << " ";
209#endif
210 }
211#ifdef DEBUG
212 std::cout << std::endl;
213#endif
214
215 if (eh.version == 2) {
221#ifdef DEBUG
222 std::cout << "Padding words to reach 8 total sample length words." << std::endl;
223#endif
224 for (int i_word{n_words}; i_word < 8; i_word++) {
225 reader >> head1; i_event++;
226#ifdef DEBUG
227 std::cout << " " << debug::hex(head1);
228#endif
229 }
230#ifdef DEBUG
231 std::cout << std::endl;
232#endif
233
237 reader >> head1; i_event++;
238 eh.spill = ((head1 >> 12) & 0xfff);
239 eh.bunch = (head1 & 0xfff);
240#ifdef DEBUG
241 std::cout << " " << debug::hex(head1)
242 << " Spill: " << eh.spill
243 << " Bunch: " << eh.bunch << std::endl;
244#endif
245 reader >> head1; i_event++;
246 eh.ticks = head1;
247#ifdef DEBUG
248 std::cout << " " << debug::hex(head1)
249 << " 5 MHz Ticks since Spill: " << head1
250 << " Time: " << head1/5e6 << "s" << std::endl;
251#endif
252 reader >> head1; i_event++;
253 eh.number = head1;
254#ifdef DEBUG
255 std::cout << " " << debug::hex(head1)
256 << " Event Number: " << head1 << std::endl;
257#endif
258 reader >> head1; i_event++;
259 eh.run = (head1 & 0xFFF);
260 eh.DD = (head1>>23)&0x1F;
261 eh.MM = (head1>>28)&0xF;
262 eh.hh = (head1>>18)&0x1F;
263 eh.mm = (head1>>12)&0x3F;
264#ifdef DEBUG
265 std::cout << " " << debug::hex(head1)
266 << " Run: " << eh.run << " DD-MM hh:mm "
267 << eh.DD << "-" << eh.MM << " " << eh.hh << ":" << eh.mm
268 << std::endl;
269#endif
270 }
271
279 // fill map of **electronic** IDs to the digis that were read out
280 std::map<ldmx::HcalElectronicsID,
281 std::vector<ldmx::HgcrocDigiCollection::Sample>>
282 eid_to_samples;
283 std::size_t i_sample{0};
284 while (i_event < eventlen) {
285#ifdef DEBUG
286 std::cout << "Decoding sample " << i_sample << " on word " << i_event << std::endl;
287#endif
288 reader >> head1 >> head2; i_event += 2;
304 packing::utility::CRC fpga_crc;
305 fpga_crc << head1;
306#ifdef DEBUG
307 std::cout << debug::hex(head1) << " : ";
308#endif
309 uint32_t hgcroc_version = (head1 >> 28) & packing::utility::mask<4>;
310#ifdef DEBUG
311 std::cout << "hgcroc_version " << hgcroc_version << std::flush;
312#endif
313 uint32_t fpga = (head1 >> 20) & packing::utility::mask<8>;
314 uint32_t nlinks = (head1 >> 14) & packing::utility::mask<6>;
315 uint32_t len = head1 & packing::utility::mask<12>;
316
317#ifdef DEBUG
318 std::cout << ", fpga: " << fpga << ", nlinks: " << nlinks << ", len: " << len << std::endl;
319#endif
320 fpga_crc << head2;
321#ifdef DEBUG
322 std::cout << debug::hex(head2) << " : ";
323#endif
324
325 uint32_t bx_id = (head2 >> 20) & packing::utility::mask<12>;
326 uint32_t rreq = (head2 >> 10) & packing::utility::mask<10>;
327 uint32_t orbit = head2 & packing::utility::mask<10>;
328
329#ifdef DEBUG
330 std::cout << "bx_id: " << bx_id << ", rreq: " << rreq << ", orbit: " << orbit << std::endl;
331#endif
332
333 std::vector<uint32_t> length_per_link(nlinks, 0);
334 for (uint32_t i_link{0}; i_link < nlinks; i_link++) {
335 if (i_link % 4 == 0) {
336 i_event++; reader >> w;
337 fpga_crc << w;
338#ifdef DEBUG
339 std::cout << debug::hex(w) << " : Four Link Pack " << std::endl;
340#endif
341 }
342 uint32_t shift_in_word = 8 * (i_link % 4);
343 bool rid_ok = (w >> (shift_in_word + 7)) & packing::utility::mask<1> == 1;
344 bool cdc_ok = (w >> (shift_in_word + 6)) & packing::utility::mask<1> == 1;
345 length_per_link[i_link] =
346 (w >> shift_in_word) & packing::utility::mask<6>;
347#ifdef DEBUG
348 std::cout << " Link " << i_link << " readout " << length_per_link.at(i_link) << " channels" << std::endl;
349#endif
350 }
351
361 for (uint32_t i_link{0}; i_link < nlinks; i_link++) {
362#ifdef DEBUG
363 std::cout << "RO Link " << i_link << std::endl;
364#endif
369 if (length_per_link.at(i_link) < 2) {
370#ifdef DEBUG
371 std::cout << "DOWN" << std::endl;
372#endif
373 continue;
374 }
375 // move on from last word counting links or previous link
376 packing::utility::CRC link_crc;
377 i_event++; reader >> w;
378 fpga_crc << w;
379 link_crc << w;
380 uint32_t roc_id = (w >> 16) & packing::utility::mask<16>;
381 bool crc_ok = (w >> 15) & packing::utility::mask<1> == 1;
382#ifdef DEBUG
383 std::cout << debug::hex(w) << " : roc_id " << roc_id << ", crc_ok (v2 always false) " << std::boolalpha << crc_ok << std::endl;
384#endif
385
386 // get readout map from the last 8 bits of this word
387 // and the entire next word
388 std::bitset<40> ro_map = w & packing::utility::mask<8>;
389 ro_map <<= 32;
390 i_event++; reader >> w;
391 fpga_crc << w;
392 link_crc << w;
393 ro_map |= w;
394#ifdef DEBUG
395 std::cout << debug::hex(w) << " : lower 32 bits of RO map" << std::endl;
396 std::cout << "Start looping through " << length_per_link.at(i_link)
397 << " words for this link" << std::endl;
398#endif
399 // loop through channels on this link,
400 // since some channels may have been suppressed because of low
401 // amplitude the channel ID is not the same as the index it
402 // is listed in.
403 int j{-1};
404 for (uint32_t i_word{2}; i_word < length_per_link.at(i_link); i_word++) {
405 // skip zero-suppressed channel IDs
406 do {
407 j++;
408 } while (j < 40 and not ro_map.test(j));
409
410 // next word is this channel
411 i_event++; reader >> w;
412 fpga_crc << w;
413#ifdef DEBUG
414 std::cout << debug::hex(w) << " " << j;
415#endif
416
417 if (j == 0) {
426#ifdef DEBUG
427 std::cout << " : ROC Header";
428#endif
429 link_crc << w;
430 uint32_t bx_id = (w >> 16) & packing::utility::mask<12>;
431 uint32_t short_event = (w >> 10) & packing::utility::mask<6>;
432 uint32_t short_orbit = (w >> 7) & packing::utility::mask<3>;
433 uint32_t hamming_errs = (w >> 4) & packing::utility::mask<3>;
434 } else if (j == common_mode_channel) {
438 link_crc << w;
439#ifdef DEBUG
440 std::cout << " : Common Mode";
441#endif
442 } else if (j == calib_channel) {
443 // calib channel
444 link_crc << w;
445#ifdef DEBUG
446 std::cout << " : Calib";
447#endif
448 } else if (j == 39) {
449 // trailer on each link added by ROC
450 // ROC v2 - IDLE word
451 // ROC v3 - CRC checksum
452#ifdef DEBUG
453 if (roc_version_ == 2) {
454 std::cout << " : Idle";
455 } else {
456 std::cout << " : CRC checksum : " << debug::hex(link_crc.get()) << " =? " << debug::hex(w);
457 }
458#endif
459 /*
460 if (roc_version_ > 2 and link_crc.get() != w) {
461 EXCEPTION_RAISE("BadCRC",
462 "Our calculated link checksum doesn't match the "
463 "one from raw data.");
464 }
465 */
466 } else {
468
469 link_crc << w;
482 ldmx::HcalElectronicsID eid(fpga, i_link,
483 j - 1 - 1*(j > common_mode_channel) - 1*(j > calib_channel));
484#ifdef DEBUG
485 std::cout << " : DAQ Channel ";
486 std::cout << "EID(" << eid.fiber() << "," << eid.elink() << "," << eid.channel() << ") ";
487#endif
488 // copy data into EID->sample map
489 eid_to_samples[eid].emplace_back(w);
490 } // type of channel
491#ifdef DEBUG
492 std::cout << std::endl;
493#endif
494 } // loop over channels (j in Table 4)
495#ifdef DEBUG
496 std::cout << "done looping through channels" << std::endl;
497#endif
498 } // loop over links
499
500
501 // another CRC checksum from FPGA
502 i_event++; reader >> w;
503 uint32_t crc = w;
504#ifdef DEBUG
505 std::cout << "Done with sample " << i_sample << std::endl;
506 std::cout << "FPGA Checksum : " << debug::hex(fpga_crc.get()) << " =? " << debug::hex(crc) << std::endl;
507 std::cout << " N Sample Words : " << length_per_sample.at(i_sample) << std::endl;
508#endif
509 /* TODO
510 * fix calculation of FPGA checksum
511 * I can't figure out why it isn't matching, but there
512 * is definitely a word here where the FPGA checksum would be.
513 if (fpga_crc.get() != crc) {
514 EXCEPTION_RAISE(
515 "BadCRC",
516 "Our calculated FPGA checksum doesn't match the one read in.");
517 }
518 */
519 // padding to reach 64-bit boundary in version 2
520 if (eh.version == 2u and length_per_sample.at(i_sample) % 2 == 1) {
521 i_event++; reader >> head1;
522#ifdef DEBUG
523 std::cout << "Padding to reach 64-bit boundary: " << debug::hex(head1) << std::endl;
524#endif
525 }
526 i_sample++;
527 }
528
529 if (eh.version == 1u) {
530 // special footer words
531 reader >> head1 >> head2;
532 }
533
534 return eid_to_samples;
535 }

578void HcalRawDecoder::produce(framework::Event& event) {
579
580 std::map<ldmx::HcalElectronicsID,
581 std::vector<ldmx::HgcrocDigiCollection::Sample>>
582 eid_to_samples;
583 PolarfireEventHeader eh;
584 if (read_from_file_) {
585 if (!file_reader_ or file_reader_.eof()) return;
586 eid_to_samples = this->read(file_reader_,eh);
587 } else {
588 for (const auto& name : input_names_) {
589 hcal::utility::Reader bus_reader(event.getCollection<uint8_t>(name, input_pass_));
590 auto single_pf_samples = this->read(bus_reader,eh);
591 for (const auto& [id, samples] : single_pf_samples) {
592 eid_to_samples[id] = samples;
593 }
594 }
595 }
596
597 eh.board(event, output_name_);
598
599 ldmx::HgcrocDigiCollection digis;
600 // assume all channels have same number of samples
601 digis.setNumSamplesPerDigi(eid_to_samples.begin()->second.size());
602 digis.setSampleOfInterestIndex(0); // TODO configurable
603 digis.setVersion(roc_version_);
604 if (translate_eid_) {
612#ifdef DEBUG
613 std::cout << "Translating EIDs into DetIDs. Printing skipped EIDs..." << std::endl;
614#endif
615 auto detmap{
616 getCondition<HcalDetectorMap>(HcalDetectorMap::CONDITIONS_OBJECT_NAME)};
617 for (auto const& [eid, digi] : eid_to_samples) {
618 // The electronics map returns an empty ID of the correct
619 // type when the electronics ID is not found.
620 // need to check if the electronics ID exists
621 // TODO: do we want to end processing if this happens?
622 if (detmap.exists(eid)) {
623 uint32_t did_raw = detmap.get(eid).raw();
624 digis.addDigi(did_raw, digi);
625 } else {
632#ifdef DEBUG
633 std::cout << "EID(" << eid.fiber() << "," << eid.elink() << "," << eid.channel() << ") ";
634 for (auto& s : digi) std::cout << debug::hex(s.raw()) << " ";
635 std::cout << std::endl;
636#endif
637 }
638 }
639 } else {
647 for (auto const& [eid, digi] : eid_to_samples) {
648 digis.addDigi(eid.raw(), digi);
649 }
650 }
651
652#ifdef DEBUG
653 std::cout << "adding " << digis.getNumDigis() << " digis each with " << digis.getNumSamplesPerDigi() << " samples to event bus" << std::endl;
654#endif
655 event.add(output_name_, digis);
656 return;
657} // produce
