88void HcalRawDecoder::produce(framework::Event& event) {
92 static const unsigned int common_mode_channel = roc_version_ == 2 ? 19 : 1;
94 static uint32_t head1, head2, w;
95
96 // if we aren't reading from a file, open
97 // the buffer from the event bus
98 if (not reader_.isFile()) {
99 reader_.open(event.getCollection<uint32_t>(input_name_, input_pass_));
100 } else if (!reader_) {
101 // abort event if reader is a file and in a "bad" state
102 // (most likely EOF)
103 abortEvent();
104 }
105
119 // fill map of **electronic** IDs to the digis that were read out
120 std::map<ldmx::HcalElectronicsID,
121 std::vector<ldmx::HgcrocDigiCollection::Sample>>
122 eid_to_samples;
123 for (unsigned int i_packet{0}; i_packet < num_packets_per_event_;
124 i_packet++) {
125 if (!(reader_ >> head1 >> head2)) {
126 // error reading two header words,
127 // this could be due to a misalignment between the number
128 // of requested packets per event and the number in the file.
129 // this can also happen on the first event after the end
130 // of file has been reached.
131 break;
132 }
133
149 packing::utility::CRC fpga_crc;
150 fpga_crc << head1;
151 std::cout << debug::hex(head1) << " : ";
152 uint32_t version = (head1 >> 28) & packing::utility::mask<4>;
153 std::cout << "version " << version << std::flush;
154 uint32_t one{1};
155 if (version != one)
156 EXCEPTION_RAISE("VersMis",
157 "HcalRawDecoder only knows version 1 of DAQ format.");
158
159 uint32_t fpga = (head1 >> 20) & packing::utility::mask<8>;
160 uint32_t nlinks = (head1 >> 14) & packing::utility::mask<6>;
161 uint32_t len = head1 & packing::utility::mask<12>;
162
163 std::cout << ", fpga: " << fpga << ", nlinks: " << nlinks
164 << ", len: " << len << std::endl;
165 fpga_crc << head2;
166 std::cout << debug::hex(head2) << " : ";
167
168 uint32_t bx_id = (head2 >> 20) & packing::utility::mask<12>;
169 uint32_t rreq = (head2 >> 10) & packing::utility::mask<10>;
170 uint32_t orbit = head2 & packing::utility::mask<10>;
171
172 std::cout << "bx_id: " << bx_id << ", rreq: " << rreq
173 << ", orbit: " << orbit << std::endl;
174
175 std::vector<uint32_t> length_per_link(nlinks, 0);
176 for (uint32_t i_link{0}; i_link < nlinks; i_link++) {
177 if (i_link % 4 == 0) {
178 reader_ >> w;
179 fpga_crc << w;
180 std::cout << debug::hex(w) << " : Four Link Pack " << std::endl;
181 }
182 uint32_t shift_in_word = 8 * (i_link % 4);
183 bool rid_ok = (w >> (shift_in_word + 7)) & packing::utility::mask<1> == 1;
184 bool cdc_ok = (w >> (shift_in_word + 6)) & packing::utility::mask<1> == 1;
185 length_per_link[i_link] =
186 (w >> shift_in_word) & packing::utility::mask<6>;
187 std::cout << " Link " << i_link << " readout "
188 << length_per_link.at(i_link) << " channels" << std::endl;
189 }
190
200 for (uint32_t i_link{0}; i_link < nlinks; i_link++) {
201 // move on from last word counting links or previous link
202 std::cout << "RO Link " << i_link << std::endl;
203 packing::utility::CRC link_crc;
204 reader_ >> w;
205 fpga_crc << w;
206 link_crc << w;
207 uint32_t roc_id = (w >> 16) & packing::utility::mask<16>;
208 bool crc_ok = (w >> 15) & packing::utility::mask<1> == 1;
209 std::cout << debug::hex(w) << " : roc_id " << roc_id << ", crc_ok (v2 always false) "
210 << std::boolalpha << crc_ok << std::endl;
211
212 // get readout map from the last 8 bits of this word
213 // and the entire next word
214 std::bitset<40> ro_map = w & packing::utility::mask<8>;
215 ro_map <<= 32;
216 reader_ >> w;
217 fpga_crc << w;
218 link_crc << w;
219 ro_map |= w;
220
221 std::cout << "Start looping through channels..." << std::endl;
222 // loop through channels on this link,
223 // since some channels may have been suppressed because of low
224 // amplitude the channel ID is not the same as the index it
225 // is listed in.
226 int channel_id{-1};
227 for (uint32_t j{0}; j < length_per_link.at(i_link) - 2; j++) {
228 // skip zero-suppressed channel IDs
229 do {
230 channel_id++;
231 } while (channel_id < 40 and not ro_map.test(channel_id));
232
233 // next word is this channel
234 reader_ >> w;
235 fpga_crc << w;
236 std::cout << debug::hex(w) << " " << channel_id;
237
238 if (channel_id == 0) {
247 std::cout << " : ROC Header";
248 link_crc << w;
249 uint32_t bx_id = (w >> 16) & packing::utility::mask<12>;
250 uint32_t short_event = (w >> 10) & packing::utility::mask<6>;
251 uint32_t short_orbit = (w >> 7) & packing::utility::mask<3>;
252 uint32_t hamming_errs = (w >> 4) & packing::utility::mask<3>;
253 } else if (channel_id == common_mode_channel) {
257 link_crc << w;
258 std::cout << " : Common Mode";
259 } else if (channel_id == 39) {
260 // CRC checksum from ROC
261 uint32_t crc = w;
262 std::cout << " : CRC checksum : " << debug::hex(link_crc.get())
263 << " =? " << debug::hex(crc);
264 /*
265 if (link_crc.get() != crc) {
266 EXCEPTION_RAISE("BadCRC",
267 "Our calculated link checksum doesn't match the "
268 "one from raw data.");
269 }
270 */
271 } else {
273
274 link_crc << w;
285 std::cout << " : DAQ Channel ";
286
287 std::cout << fpga << " " << roc_id << " " << channel_id << " ";
288 // TODO fix hardcoded starting value
289 ldmx::HcalElectronicsID eid(fpga, roc_id-256, channel_id);
290 std::cout << eid.index();
291
292 // copy data into EID->sample map
293 eid_to_samples[eid].emplace_back(w);
294 } // type of channel
295 std::cout << std::endl;
296 } // loop over channels (j in Table 4)
297 std::cout << "done looping through channels" << std::endl;
298 } // loop over links
299
300 // another CRC checksum from FPGA
301 reader_ >> w;
302 uint32_t crc = w;
303 std::cout << "FPGA Checksum : " << debug::hex(fpga_crc.get()) << " =? "
304 << debug::hex(crc) << std::endl;
305 /* TODO
306 * fix calculation of FPGA checksum
307 * I can't figure out why it isn't matching, but there
308 * is definitely a word here where the FPGA checksum would be.
309 if (fpga_crc.get() != crc) {
310 EXCEPTION_RAISE(
311 "BadCRC",
312 "Our calculated FPGA checksum doesn't match the one read in.");
313 }
314 */
315 }
316
317 // check if there was any data to decode
318 // helpful for when reading directly from a raw data file
319 // so we can simply overestimate the number of events
320 if (eid_to_samples.size() == 0) {
321 abortEvent();
322 }
323
324 ldmx::HgcrocDigiCollection digis;
325 // assume all channels have same number of samples
326 digis.setNumSamplesPerDigi(eid_to_samples.begin()->second.size());
327 digis.setSampleOfInterestIndex(0); // TODO configurable
328 digis.setVersion(roc_version_);
329 if (translate_eid_) {
337 auto detmap{
338 getCondition<HcalDetectorMap>(HcalDetectorMap::CONDITIONS_OBJECT_NAME)};
339 for (auto const& [eid, digi] : eid_to_samples) {
340 // The electronics map returns an empty ID of the correct
341 // type when the electronics ID is not found.
342 // need to check if the electronics ID exists
343 // TODO: do we want to end processing if this happens?
344 if (detmap.exists(eid)) {
345 uint32_t did_raw = detmap.get(eid).raw();
346 digis.addDigi(did_raw, digi);
347 } else {
348 // DO NOTHING
349 // skip hits where the EID aren't in the detector mapping
350 // DO WE ACTUALLY WANT TO DO THIS?
351 }
352 }
353 } else {
354 // no EID translation, just add the digis to the digi collection
355 // with their raw electronic ID
356 for (auto const& [eid, digi] : eid_to_samples) {
357 digis.addDigi(eid.raw(), digi);
358 }
359 }
360
361 std::cout << "adding " << digis.getNumDigis() << " digis to event bus" << std::endl;
362 event.add(output_name_, digis);
363 return;
364} // produce
