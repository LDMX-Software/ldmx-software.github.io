22class Reader {
23 const std::vector<uint8_t> &buffer_;
24 std::size_t i_word_;
25 uint32_t next() {
26 uint32_t w = buffer_.at(i_word_) |
27 (buffer_.at(i_word_ + 1) << 8) |
28 (buffer_.at(i_word_ + 2) << 16) |
29 (buffer_.at(i_word_ + 3) << 24);
30 i_word_ += 4;
31 return w;
32 }
33
34 public:
35 Reader(const std::vector<uint8_t>& b) : buffer_{b}, i_word_{0} {}
36 operator bool() {
37 return (i_word_ < buffer_.size());
38 }
39 Reader& operator>>(uint32_t& w) {
40 if (*this)
41 w = next();
42 return *this;
43 }
44}; // Reader

75 static const unsigned int common_mode_channel = roc_version_ == 2 ? 19 : 1;
77 static uint32_t head1, head2, w;

79 utility::Reader reader_{event.getCollection<uint8_t>(input_name_, input_pass_)};
80
87 // fill map of **electronic** IDs to the digis that were read out

91 while (reader_ >> head1 >> head2) {
93 if (head1 == 0x11111111 and head2 == 0xbeef2021) {
94 /* whole event header word looks like
95 *
96 * VERSION (4) | FPGA ID (8) | NSAMPLES (4) | LEN (16)
97 */
98 uint32_t whole_event_header;
99 reader_ >> whole_event_header;
100 uint32_t version = (whole_event_header >> 28) & packing::utility::mask<4>;
101 uint32_t fpga = (whole_event_header >> 20) & packing::utility::mask<8>;
102 uint32_t nsamples = (whole_event_header >> 16) & packing::utility::mask<4>;

104
105 // sample counters
106 std::vector<uint32_t> length_per_sample(nsamples, 0);
107 for (uint32_t i_sample{0}; i_sample < nsamples; i_sample++) {
108 if (i_sample%2 == 0) {
109 reader_ >> w;
110 }
111 uint32_t shift_in_word = 16*(i_sample%2);
112 length_per_sample[i_sample] = (w >> shift_in_word) & packing::utility::mask<12>;
113 }
114
115 // read first sample headers
116 reader_ >> head1 >> head2;
117 } else if (head1 == 0xd07e2021 and head2 == 0x12345678) {
118 // these are the special footer words at the end,
119 // done with event
120 break;
121 }

138 packing::utility::CRC fpga_crc;

140 //std::cout << debug::hex(head1) << " : ";

152 //std::cout << ", fpga: " << fpga << ", nlinks: " << nlinks << ", len: " << len << std::endl;
153 fpga_crc << head2;

157 uint32_t rreq = (head2 >> 10) & packing::utility::mask<10>;
158 uint32_t orbit = head2 & packing::utility::mask<10>;
159

163 for (uint32_t i_link{0}; i_link < nlinks; i_link++) {

165 reader_ >> w;
166 fpga_crc << w;
167 //std::cout << debug::hex(w) << " : Four Link Pack " << std::endl;
168 }
169 uint32_t shift_in_word = 8 * (i_link % 4);

171 bool cdc_ok = (w >> (shift_in_word + 6)) & packing::utility::mask<1> == 1;
172 length_per_link[i_link] =
173 (w >> shift_in_word) & packing::utility::mask<6>;
174 //std::cout << " Link " << i_link << " readout " << length_per_link.at(i_link) << " channels" << std::endl;
175 }
176
186 for (uint32_t i_link{0}; i_link < nlinks; i_link++) {
187 // move on from last word counting links or previous link
188 //std::cout << "RO Link " << i_link << std::endl;
189 packing::utility::CRC link_crc;
190 reader_ >> w;
191 fpga_crc << w;
192 link_crc << w;
193 uint32_t roc_id = (w >> 16) & packing::utility::mask<16>;
194 bool crc_ok = (w >> 15) & packing::utility::mask<1> == 1;
195 //std::cout << debug::hex(w) << " : roc_id " << roc_id << ", crc_ok (v2 always false) " << std::boolalpha << crc_ok << std::endl;
196
197 // get readout map from the last 8 bits of this word
198 // and the entire next word
199 std::bitset<40> ro_map = w & packing::utility::mask<8>;
200 ro_map <<= 32;
201 reader_ >> w;
202 fpga_crc << w;
203 link_crc << w;
204 ro_map |= w;
205
206 //std::cout << "Start looping through channels..." << std::endl;
207 // loop through channels on this link,
208 // since some channels may have been suppressed because of low
209 // amplitude the channel ID is not the same as the index it
210 // is listed in.
211 int channel_id{-1};
212 for (uint32_t j{0}; j < length_per_link.at(i_link) - 2; j++) {
213 // skip zero-suppressed channel IDs
214 do {
215 channel_id++;

279 eid_to_samples[eid].emplace_back(w);
280 } // type of channel
281 //std::cout << std::endl;
282 } // loop over channels (j in Table 4)
283 //std::cout << "done looping through channels" << std::endl;
284 } // loop over links

286 // another CRC checksum from FPGA
287 reader_ >> w;
288 uint32_t crc = w;
289 //std::cout << "FPGA Checksum : " << debug::hex(fpga_crc.get()) << " =? " << debug::hex(crc) << std::endl;
290 /* TODO

302 ldmx::HgcrocDigiCollection digis;
303 // assume all channels have same number of samples
304 digis.setNumSamplesPerDigi(eid_to_samples.begin()->second.size());
305 digis.setSampleOfInterestIndex(0); // TODO configurable
306 digis.setVersion(roc_version_);
307 if (translate_eid_) {

316 getCondition<HcalDetectorMap>(HcalDetectorMap::CONDITIONS_OBJECT_NAME)};
317 for (auto const& [eid, digi] : eid_to_samples) {
318 // The electronics map returns an empty ID of the correct

322 if (detmap.exists(eid)) {

324 digis.addDigi(did_raw, digi);
325 } else {
326 // DO NOTHING

328 // DO WE ACTUALLY WANT TO DO THIS?
329 }
330 }
331 } else {
332 // no EID translation, just add the digis to the digi collection
333 // with their raw electronic ID
334 // TODO: remove this, we shouldn't be able to get past
335 // the decoding stage without translating the EID
336 // into a detector ID to avoid confusion in recon

338 digis.addDigi(eid.raw(), digi);

340 }

342 //std::cout << "adding " << digis.getNumDigis() << " digis each with " << digis.getNumSamplesPerDigi() << " samples to event bus" << std::endl;
343 event.add(output_name_, digis);
