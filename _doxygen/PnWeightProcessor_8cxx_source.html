51  void PnWeightProcessor::produce(Event& event) {
52 
53  // Get the particle map from the event. If the particle map is empty,
54  // don't process the event.
55  auto particleMap{event.getMap<int,SimParticle>("SimParticles")};
56  if (particleMap.size() == 0) return;
57 
58  // Get the recoil electron
59  auto [ trackID, recoil ] = Analysis::getRecoil(particleMap);
60 
61  // Use the recoil electron to retrieve the gamma that underwent a
62  // photo-nuclear reaction.
63  auto pnGamma{Analysis::getPNGamma(particleMap, recoil, 2500.)};
64 
65  // For PN biased events, there should always be a gamma that
66  // underwent a PN reaction.
67  if (pnGamma == nullptr) {
68  EXCEPTION_RAISE( "PnWeightProc" , "Event doesn't contain a PN Gamma that is the daughter of the recoil electron." );
69  }
70 
71  double hardestNucleonKe = -9999;
72  double hardestNucleonTheta = -9999;
73  double hardestNucleonW = -9999;
74  double highestWNucleonKe = -9999;
75  double highestWNucleonTheta = -9999;
76  double highestWNucleonW = -9999;
77  PnWeightResult result;
78  for (const int &daughterTrackID : pnGamma->getDaughters() ) {
79 
80  // Get a daughter of the PN gamma
81  const SimParticle *pnDaughter;
82  if ( particleMap.count( daughterTrackID ) > 0 ) {
83  pnDaughter = &( particleMap.at( daughterTrackID ) );
84  } else {
85  //pnDaughter not stored in particle map
86  EXCEPTION_RAISE(
87  "MissingPNDaughter",
88  "PN Daughter with track ID " + std::to_string(daughterTrackID) +
89  " has not been stored. Have you stored all of the daughters of PN interactions?"
90  );
91  }
92 
93  // Calculate the kinetic energy
94  double ke = (pnDaughter->getEnergy() - pnDaughter->getMass());
95 
96  // Calculate the momentum
97  double px = pnDaughter->getMomentum()[0];
98  double py = pnDaughter->getMomentum()[1];
99  double pz = pnDaughter->getMomentum()[2];
100  double p = sqrt(px*px + py*py + pz*pz);
101 
102  // Calculate the polar angle
103  double theta = acos(pz/p)*180.0/3.14159;
104 
105  // Get the PDG ID of the daughter
106  long int pdgID = std::abs(pnDaughter->getPdgID());
107 
108  double w = this->calculateW(pnDaughter);
109 
110  // Check if the daughter particle is a proton or neutron
111  if ((pdgID == PROTON_PDGID) || (pdgID == NEUTRON_PDGID)) {
112 
113  // Add the W of the current nucleon to the inclusive collection.
114  result.addW(w);
115 
116  // Add the theta of the current nucleon to the inclusive collection.
117  result.addTheta(theta);
118 
119  // Find the nucleon with the greatest kinetic energy
120  if (ke > hardestNucleonKe) {
121  hardestNucleonKe = ke;
122  hardestNucleonTheta = theta;
123  hardestNucleonW = w;
124  }
125 
126  // Find the nucleon with the highest W
127  if (w > highestWNucleonW) {
128  highestWNucleonKe = ke;
129  highestWNucleonTheta = theta;
130  highestWNucleonW = w;
131  }
132  }//pnDaughter is nucleon
133  }//loop through pnDaughters
134 
135  // If the W of the highest W nucleon is above the threshold and the
136  // polar angle is above the angle threshold, reweight the event.
137  // Otherwise, set the event weight to 1.
138  double eventWeight = 1;
139  if ((highestWNucleonW > wThreshold_) && (highestWNucleonTheta > thetaThreshold_)) {
140  eventWeight = calculateWeight(highestWNucleonW);
141  }
142 
143  result.setHardestNucleonKe(hardestNucleonKe);
144  result.setHardestNucleonTheta(hardestNucleonTheta);
145  result.setHardestNucleonW(hardestNucleonW);
146  result.setHighestWNucleonKe(highestWNucleonKe);
147  result.setHighestWNucleonTheta(highestWNucleonTheta);
148  result.setHighestWNucleonW(highestWNucleonW);
149  result.setWeight(eventWeight);
150 
151  // Add the result to the collection
152  event.add("PNweight", result);
153  }

159  double PnWeightProcessor::calculateW(const SimParticle* particle, double delta) {
160  double px = particle->getMomentum()[0];
161  double py = particle->getMomentum()[1];
162  double pz = particle->getMomentum()[2];
163  double p = sqrt(px*px + py*py + pz*pz);
164  double ke = particle->getEnergy() - particle->getMass();
165 
166  return 0.5*(p + ke)*(sqrt(1 + (delta*delta)) - delta*(pz/p));
167  }
168 }
