63void EcalDetectorMap::loadCellMap(conditions::GeneralCSVLoader& loader) {
64 cells_.clear();
65 while (loader.nextRow()) {
66 CellInformation ci;
67 ci.module_cellid = loader.getInteger("CELLID");
68 ci.rocid = loader.getInteger("ROCID");
69 ci.roc_elink_number = loader.getInteger("ROC_ELINK_NUMBER");
70 ci.roc_elink_channel = loader.getInteger("ROC_ELINK_CHANNEL");
71 cells_.push_back(ci);
72 }

99void EcalDetectorMap::buildElectronicsMap() {
100 this->clear(); // empty the electronics map
101 // loop over optical links
102 for (auto olink : layers_) {
103 for (auto elink : elinks_) {
104 // select only matching motherboard types
105 if (elink.motherboard_type != olink.motherboard_type) continue;
106
107 for (auto cell : cells_) {
108 // select only cells which are associated with the appropriate elink
109 if (elink.rocid != cell.rocid ||
110 elink.roc_elink_number != cell.roc_elink_number)
111 continue;
112
113 // now, we have only cells which are relevant
114 ldmx::EcalID precisionId(olink.layer, elink.module, cell.module_cellid);
115 ldmx::EcalElectronicsID elecId(olink.daq_opticallink,
116 elink.polarfire_elink,
117 cell.roc_elink_channel);
118
119 if (this->exists(elecId)) {
120 std::stringstream ss;
121 ss << "Two different mappings for electronics channel " << elecId;
122 EXCEPTION_RAISE("DuplicateMapping", ss.str());
123 }
124 this->addEntry(elecId, precisionId);
125 }
126 }
127 }
