28 void TruthHitProducer::produce(Event &event) {
29 
30  // Check if the collection exists. If not, don't bother processing the event.
31  if (!event.exists(inputCollection_)) {
32  ldmx_log(error) << "No input collection called " << inputCollection_
33  << " found; skipping!";
34  return;
35  }
36  // looper over sim hits and aggregate energy depositions for each detID
37  const auto simHits{
38  event.getCollection<SimCalorimeterHit>(inputCollection_, inputPassName_)};
39  auto particleMap{event.getMap<int, SimParticle>("SimParticles")};
40 
41  std::vector<SimCalorimeterHit> truthBeamElectrons;
42 
43  // TODO: Convert this to using a for_each and lambda
44  for (const auto &simHit : simHits) {
45  bool keep{false};
46  // check if hit is from beam electron and, if so, add to output collection
47  for (int i = 0; i < simHit.getNumberOfContribs(); i++) {
48 
49  auto contrib = simHit.getContrib(i);
50  if (verbose_) {
51  ldmx_log(debug) << "contrib " << i << " trackID: " << contrib.trackID
52  << " pdgID: " << contrib.pdgCode
53  << " edep: " << contrib.edep;
54  ldmx_log(debug) << "\t particle id: "
55  << particleMap[contrib.trackID].getPdgID()
56  << " particle status: "
57  << particleMap[contrib.trackID].getGenStatus();
58  }
59  // if the trackID is in the map
60  if (particleMap.find(contrib.trackID) != particleMap.end()) {
61  // beam electron (PDGID = 11, genStatus == 1)
62  if (particleMap[contrib.trackID].getPdgID() == 11 &&
63  particleMap[contrib.trackID].getGenStatus() == 1) {
64  keep = true;
65  }
66  }
67  if (keep)
68  truthBeamElectrons.push_back(simHit);
69  }
70  }
71  event.add(outputCollection_, truthBeamElectrons);
72 }
73 } // namespace ldmx
