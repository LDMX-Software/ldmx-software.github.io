28 void TruthHitProducer::produce(framework::Event &event) {
29  // Check if the collection exists. If not, don't bother processing the event.
30  if (!event.exists(inputCollection_)) {
31  ldmx_log(error) << "No input collection called " << inputCollection_
32  << " found; skipping!";
33  return;
34  }
35  // looper over sim hits and aggregate energy depositions for each detID
36  const auto simHits{event.getCollection<ldmx::SimCalorimeterHit>(
37  inputCollection_, inputPassName_)};
38  auto particleMap{event.getMap<int, ldmx::SimParticle>("SimParticles")};
39 
40  std::vector<ldmx::SimCalorimeterHit> truthBeamElectrons;
41 
42  // TODO: Convert this to using a for_each and lambda
43  for (const auto &simHit : simHits) {
44  bool keep{false};
45  // check if hit is from beam electron and, if so, add to output collection
46  for (int i = 0; i < simHit.getNumberOfContribs(); i++) {
47  auto contrib = simHit.getContrib(i);
48  if (verbose_) {
49  ldmx_log(debug) << "contrib " << i << " trackID: " << contrib.trackID
50  << " pdgID: " << contrib.pdgCode
51  << " edep: " << contrib.edep;
52  ldmx_log(debug) << "\t particle id: "
53  << particleMap[contrib.trackID].getPdgID()
54  << " particle status: "
55  << particleMap[contrib.trackID].getGenStatus();
56  }
57  // if the trackID is in the map
58  if (particleMap.find(contrib.trackID) != particleMap.end()) {
59  // beam electron (PDGID = 11, genStatus == 1)
60  if (particleMap[contrib.trackID].getPdgID() == 11 &&
61  particleMap[contrib.trackID].getGenStatus() == 1) {
62  keep = true;
63  }
64  }
65  if (keep) truthBeamElectrons.push_back(simHit);
66  }
67  }
68  event.add(outputCollection_, truthBeamElectrons);
69 }
