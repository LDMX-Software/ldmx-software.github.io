18void HcalRecProducer::configure(framework::config::Parameters& ps) {
19 // collection names
20 digiCollName_ = ps.getParameter<std::string>("digiCollName");
21 digiPassName_ = ps.getParameter<std::string>("digiPassName");
22 simHitCollName_ = ps.getParameter<std::string>("simHitCollName");
23 simHitPassName_ = ps.getParameter<std::string>("simHitPassName");
24 recHitCollName_ = ps.getParameter<std::string>("recHitCollName");
25
26 // parameters
27 mip_energy_ = ps.getParameter<double>("mip_energy");
28 pe_per_mip_ = ps.getParameter<double>("pe_per_mip");
29 clock_cycle_ = ps.getParameter<double>("clock_cycle");
30 voltage_per_mip_ = ps.getParameter<double>("voltage_per_mip");
31 attlength_ = ps.getParameter<double>("attenuationLength");
32 nADCs_ = ps.getParameter<int>("nADCs");
33
34 // configuring corrections graphs derived on the fly
35 // TODO: maybe we should save these as a graph instead?
36 rateUpSlope_ = ps.getParameter<double>("rateUpSlope");
37 timeUpSlope_ = ps.getParameter<double>("timeUpSlope");
38 rateDnSlope_ = ps.getParameter<double>("rateDnSlope");
39 timeDnSlope_ = ps.getParameter<double>("timeDnSlope");
40 timePeak_ = ps.getParameter<double>("timePeak");
41 pulseFunc_ =
42 TF1("pulseFunc",
43 "[0]*((1.0+exp([1]*(-[2]+[3])))*(1.0+exp([5]*(-[6]+[3]))))/"
44 "((1.0+exp([1]*(x-[2]+[3]-[4])))*(1.0+exp([5]*(x-[6]+[3]-[4]))))",
45 (double)nADCs_ * clock_cycle_ * -1, (double)nADCs_ * clock_cycle_);
46 pulseFunc_.FixParameter(1, rateUpSlope_);
47 pulseFunc_.FixParameter(2, timeUpSlope_);
48 pulseFunc_.FixParameter(3, timePeak_);
49 pulseFunc_.FixParameter(5, rateDnSlope_);
50 pulseFunc_.FixParameter(6, timeDnSlope_);
51 pulseFunc_.FixParameter(4, 0);
52 pulseFunc_.FixParameter(0, 1);
53
54 // build amplitude correction (Ampl[t-1]/Ampl[t]) with pulse-shape
55 int n = 0;
56 for (double t = -clock_cycle_; t < clock_cycle_; t += 0.01) {
57 double ampl_t = pulseFunc_.Eval(t);
58 double ampl_tm1 = pulseFunc_.Eval(t - clock_cycle_);
59 if (ampl_tm1 > ampl_t) continue;
60 correctionAmpl_.SetPoint(n, ampl_tm1 / ampl_t, ampl_t);
61 if (n == 0) minAmplFraction_ = ampl_tm1 / ampl_t;
62 n++;
63 }
64
65 // build TOA timewalk correction with pulse-shape
66 double toaThreshold = ps.getParameter<double>("avgToaThreshold");
67 double gain = ps.getParameter<double>("avgGain");
68 double pedestal = ps.getParameter<double>("avgPedestal");
69 n = 0;
70 for (double ampl = toaThreshold + 0.1; ampl < 10000; ampl += 0.01) {
71 pulseFunc_.FixParameter(0, ampl);
72 double ampl_t = gain * pedestal + pulseFunc_.Eval(0);
73 double toa =
74 fabs(pulseFunc_.GetX(toaThreshold, (double)nADCs_ * clock_cycle_ * -1,
75 (double)nADCs_ * clock_cycle_));
76 correctionTOA_.SetPoint(n, ampl_t, toa);
77 if (n == 0) minAmpl_ = ampl_t;
78 n++;
79 }
80}

82double HcalRecProducer::getTOA(
83 const ldmx::HgcrocDigiCollection::HgcrocDigi digi, double pedestal,
84 unsigned int iSOI) {
85 // get toa relative to the startBX
86 double toaRelStartBX(0.), maxMeas{0.};
87 int toaSample(0), maxSample(0), iADC(0);
88 for (int i_sample{0}; i_sample < digi.size(); i_sample++) {
89 auto sample{digi.at(i_sample)};
90 if (sample.toa() > 0) {
91 toaRelStartBX = sample.toa() * (clock_cycle_ / 1024); // ns
92 // find in which ADC sample the TOA was taken
93 toaSample = iADC;
94 }
95 if ((sample.adc_t() - pedestal) > maxMeas) {
96 maxMeas = (sample.adc_t() - pedestal);
97 maxSample = iADC;
98 }
99 iADC++;
100 }
101
102 // time w.r.t to the peak
103 double toa = (maxSample - toaSample) * clock_cycle_ - toaRelStartBX;
104
105 // time w.r.t to the SOI
106 toa += ((int)iSOI - maxSample) * clock_cycle_;
107
108 return toa;
109}

111void HcalRecProducer::produce(framework::Event& event) {
112 // get the Hcal Geometry
113 const auto& hcalGeometry = getCondition<ldmx::HcalGeometry>(
114 ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
115
116 // get the reconstruction parameters
117 HcalReconConditions the_conditions(
118 getCondition<conditions::DoubleTableCondition>(
119 HcalReconConditions::CONDITIONS_NAME));
120
121 std::vector<ldmx::HcalHit> hcalRecHits;
122 auto hcalDigis =
123 event.getObject<ldmx::HgcrocDigiCollection>(digiCollName_, digiPassName_);
124 int numDigiHits = hcalDigis.getNumDigis();
125
126 // get sample of interest index
127 unsigned int iSOI = hcalDigis.getSampleOfInterestIndex();
128
129 // loop through digis
130 int iDigi = 0;
131 while (iDigi < numDigiHits) {
132 auto digi_posend = hcalDigis.getDigi(iDigi);
133
134 // ID from first digi sample (which should be in positive end)
135 ldmx::HcalDigiID id_posend(digi_posend.id());
136 ldmx::HcalID id(id_posend.section(), id_posend.layer(), id_posend.strip());
137
138 // position from ID
139 auto position = hcalGeometry.getStripCenterPosition(id);
140 double half_total_width = hcalGeometry.getHalfTotalWidth(id.section(),id.layer());
141 double ecal_dx = hcalGeometry.getEcalDx();
142 double ecal_dy = hcalGeometry.getEcalDy();
143
144 // compute distance to the end of the bar
145 // for back Hcal, we take the half of the bar
146 // for side Hcal, we take the length of the bar (2*half-width)-Ecal_dxy as
147 // an approximation
148 float distance_posend, distance_negend, distance_ecal;
149 if (id.section() == ldmx::HcalID::HcalSection::BACK) {
150 distance_posend = half_total_width;
151 distance_negend = half_total_width;
152 } else {
153 if ((id.section() == ldmx::HcalID::HcalSection::TOP) ||
154 (id.section() == ldmx::HcalID::HcalSection::BOTTOM))
155 distance_ecal = ecal_dx;
156 else
157 distance_ecal = ecal_dy;
158 distance_posend = 2 * half_total_width - distance_ecal / 2.;
159 distance_negend = distance_ecal / 2.;
160 }
161
162 // get the estimated voltage and time from digi samples
163 double voltage(0.);
164 double voltage_min(0.);
165 double hitTime(0.);
166
167 double amplT(0.);
168 double amplT_posend(0.), amplTm1_posend(0.);
169 double amplT_negend(0.), amplTm1_negend(0.);
170
171 // double readout
172 if (id.section() == ldmx::HcalID::HcalSection::BACK) {
173 auto digi_negend = hcalDigis.getDigi(iDigi + 1);
174 ldmx::HcalDigiID id_negend(digi_negend.id());
175
176 double voltage_posend, voltage_negend;
177 double amplT_posend_copy, amplT_negend_copy;
178 if (digi_posend.isTOT()) {
179 voltage_posend =
180 (digi_posend.tot() - the_conditions.totPedestal(id_posend)) *
181 the_conditions.totGain(id_posend);
182 voltage_negend =
183 (digi_negend.tot() - the_conditions.totPedestal(id_negend)) *
184 the_conditions.totGain(id_negend);
185 } else {
186 amplT_posend =
187 digi_posend.soi().adc_t() - the_conditions.adcPedestal(id_posend);
188 amplTm1_posend =
189 digi_posend.soi().adc_tm1() - the_conditions.adcPedestal(id_posend);
190 amplT_negend =
191 digi_negend.soi().adc_t() - the_conditions.adcPedestal(id_negend);
192 amplTm1_negend =
193 digi_negend.soi().adc_tm1() - the_conditions.adcPedestal(id_negend);
194
195 amplT_posend_copy = amplT_posend;
196 amplT_negend_copy = amplT_negend;
197
198 // correct amplitude (amplitude fractions from both ends need to be
199 // above the boundary of the correction)
200 if (amplTm1_posend / amplT_posend > minAmplFraction_ &&
201 amplTm1_negend / amplT_negend > minAmplFraction_) {
202 amplT_posend *= correctionAmpl_.Eval(amplTm1_posend / amplT_posend);
203 amplT_negend *= correctionAmpl_.Eval(amplTm1_negend / amplT_negend);
204 }
205
206 // set voltage
207 voltage_posend = amplT_posend * the_conditions.adcGain(id_posend);
208 voltage_negend = amplT_negend * the_conditions.adcGain(id_negend);
209 }
210
211 // get TOA
212 double TOA_posend =
213 getTOA(digi_posend, the_conditions.adcPedestal(id_posend), iSOI);
214 double TOA_negend =
215 getTOA(digi_negend, the_conditions.adcPedestal(id_negend), iSOI);
216
217 // get sign of position along the bar
218 int position_bar_sign = (TOA_posend - TOA_negend) > 0 ? 1 : -1;
219
220 double TOA_posend_copy = TOA_posend;
221 double TOA_negend_copy = TOA_negend;
222
223 // correct TOA
224 // amplitudes from both ends need to be above the boundary of the
225 // correction otherwise, one TOA gets corrected and the other does not,
226 // which results in a large TOA difference and an out-of-bounds position
227 if (amplT_posend > minAmpl_ && amplT_negend > minAmpl_) {
228 TOA_posend = correctionTOA_.Eval(amplT_posend) - TOA_posend;
229 TOA_negend = correctionTOA_.Eval(amplT_negend) - TOA_negend;
230 }
231
232 // get x(y) coordinate from TOA measurement = (dt*v/2)
233 // if time_posend < time_negend: position is positive
234 double v =
235 299.792 / 1.6; // velocity of light in polystyrene, n = 1.6 = c/v
236 double position_bar =
237 position_bar_sign * fabs(TOA_posend - TOA_negend) * v / 2;
238
239 // reverse voltage attenuation
240 // if position along the bar is positive, then the positive end will have
241 // less attenuation than the negative end
242 // NOTE: For now, reverse attenuation is not applied to the energy
243 // deposited since both ends of the bar are summed.
244 double att_posend =
245 exp(-1. * ((distance_posend - position_bar) / 1000.) / attlength_);
246 double att_negend =
247 exp(-1. * ((distance_negend + position_bar) / 1000.) / attlength_);
248
249 // set voltage as the sum of both bars
250 voltage = (voltage_posend + voltage_negend);
251 voltage_min = std::min(voltage_posend, voltage_negend);
252
253 // set amplitude as the average of both bars (reverse attenuated)
254 amplT = (amplT_posend / att_posend + amplT_negend / att_negend) / 2;
255
256 // set position along the bar
257 if ((id_posend.layer() % 2) == 1) {
258 position.SetX(position_bar);
259 } else {
260 position.SetY(position_bar);
261 }
262
263 // set hit time
264 // TODO: does this need to revert shift because of propagation of light in
265 // polysterene?
266 hitTime = fabs(TOA_posend + TOA_negend) / 2; // ns
267
268 iDigi += 2;
269 } // end double readout loop
270 else { // single readout
271
272 double voltage_i;
273 if (digi_posend.isTOT()) {
274 // TOT - number of clock ticks that pulse was over threshold
275 // this is related to the amplitude of the pulse approximately through a
276 // linear drain rate the amplitude of the pulse is related to the energy
277 // deposited
278
279 // convert the time over threshold into a total energy deposited in the
280 // bar (time over threshold [ns] - pedestal) * gain
281
282 voltage_i =
283 (digi_posend.tot() - the_conditions.totPedestal(id_posend)) *
284 the_conditions.totGain(id_posend);
285
286 } else {
287 // ADC mode of readout
288 // ADC - voltage measurement at a specific time of the pulse
289 amplT_posend =
290 digi_posend.soi().adc_t() - the_conditions.adcPedestal(id_posend);
291 amplTm1_posend =
292 digi_posend.soi().adc_tm1() - the_conditions.adcPedestal(id_posend);
293 voltage_i = amplT_posend * the_conditions.adcGain(id_posend);
294 }
295
296 // reverse voltage attenuation
297 // for now, assume that position along the bar is the half_total_width
298 double distance_end =
299 id_posend.isNegativeEnd() ? distance_negend : distance_posend;
300 double att = exp(-1. * ((distance_end - fabs(half_total_width)) / 1000.) /
301 attlength_);
302
303 // set voltage
304 voltage = voltage_i;
305 voltage_min = voltage_i;
306
307 // set amplitude (reverse attenuated)
308 amplT = amplT_posend / att;
309
310 // get TOA
311 double TOA =
312 getTOA(digi_posend, the_conditions.adcPedestal(id_posend), iSOI);
313
314 // correct TOA
315 TOA = correctionTOA_.Eval(amplT) - TOA;
316
317 // set hit time
318 hitTime = TOA; // ns
319
320 iDigi++;
321 } // end single readout loop
322
323 double num_mips_equivalent = voltage / voltage_per_mip_;
324 double energy_deposited = num_mips_equivalent * mip_energy_;
325
326 // reconstructed energy in the layer (approximate)
327 // TODO: need to incorporate corrections if necessary
344 double reconstructed_energy = energy_deposited;
345
346 int PEs = num_mips_equivalent * pe_per_mip_;
347 int minPEs = (voltage_min / voltage_per_mip_) * pe_per_mip_;
348
349 // copy over information to rec hit structure in new collection
350 ldmx::HcalHit recHit;
351 recHit.setID(id.raw());
352 recHit.setXPos(position.X());
353 recHit.setYPos(position.Y());
354 recHit.setZPos(position.Z());
355 recHit.setPE(PEs);
356 recHit.setMinPE(minPEs);
357 recHit.setAmplitude((amplT / voltage_per_mip_) * mip_energy_);
358 recHit.setEnergy(reconstructed_energy);
359 recHit.setTime(hitTime);
360 hcalRecHits.push_back(recHit);
361 }
362
363 if (event.exists(simHitCollName_, simHitPassName_)) {
364 // hcal sim hits exist ==> label which hits are real and which are pure
365 // noise
366 auto hcalSimHits{event.getCollection<ldmx::SimCalorimeterHit>(
367 simHitCollName_, simHitPassName_)};
368 std::set<int> real_hits;
369 for (auto const& sim_hit : hcalSimHits) real_hits.insert(sim_hit.getID());
370 for (auto& hit : hcalRecHits)
371 hit.setNoise(real_hits.find(hit.getID()) == real_hits.end());
372 }
373
374 // add collection to event bus
375 event.add(recHitCollName_, hcalRecHits);
376}
