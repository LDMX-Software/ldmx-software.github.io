18 void HcalRecProducer::configure(framework::config::Parameters& ps) {
19  // collection names
20  digiCollName_ = ps.getParameter<std::string>("digiCollName");
21  digiPassName_ = ps.getParameter<std::string>("digiPassName");
22  simHitCollName_ = ps.getParameter<std::string>("simHitCollName");
23  simHitPassName_ = ps.getParameter<std::string>("simHitPassName");
24  recHitCollName_ = ps.getParameter<std::string>("recHitCollName");
25 
26  // parameters
27  mip_energy_ = ps.getParameter<double>("mip_energy");
28  pe_per_mip_ = ps.getParameter<double>("pe_per_mip");
29  clock_cycle_ = ps.getParameter<double>("clock_cycle");
30  voltage_per_mip_ = ps.getParameter<double>("voltage_per_mip");
31  attlength_ = ps.getParameter<double>("attenuationLength");
32  nADCs_ = ps.getParameter<int>("nADCs");
33 
34  // configuring corrections graphs derived on the fly
35  // TODO: maybe we should save these as a graph instead?
36  rateUpSlope_ = ps.getParameter<double>("rateUpSlope");
37  timeUpSlope_ = ps.getParameter<double>("timeUpSlope");
38  rateDnSlope_ = ps.getParameter<double>("rateDnSlope");
39  timeDnSlope_ = ps.getParameter<double>("timeDnSlope");
40  timePeak_ = ps.getParameter<double>("timePeak");
41  pulseFunc_ =
42  TF1("pulseFunc",
43  "[0]*((1.0+exp([1]*(-[2]+[3])))*(1.0+exp([5]*(-[6]+[3]))))/"
44  "((1.0+exp([1]*(x-[2]+[3]-[4])))*(1.0+exp([5]*(x-[6]+[3]-[4]))))",
45  (double)nADCs_ * clock_cycle_ * -1, (double)nADCs_ * clock_cycle_);
46  pulseFunc_.FixParameter(1, rateUpSlope_);
47  pulseFunc_.FixParameter(2, timeUpSlope_);
48  pulseFunc_.FixParameter(3, timePeak_);
49  pulseFunc_.FixParameter(5, rateDnSlope_);
50  pulseFunc_.FixParameter(6, timeDnSlope_);
51  pulseFunc_.FixParameter(4, 0);
52  pulseFunc_.FixParameter(0, 1);
53 
54  // build amplitude correction (Ampl[t-1]/Ampl[t]) with pulse-shape
55  int n = 0;
56  for (double t = -clock_cycle_; t < clock_cycle_; t += 0.01) {
57  double ampl_t = pulseFunc_.Eval(t);
58  double ampl_tm1 = pulseFunc_.Eval(t - clock_cycle_);
59  if (ampl_tm1 > ampl_t) continue;
60  correctionAmpl_.SetPoint(n, ampl_tm1 / ampl_t, ampl_t);
61  if (n == 0) minAmplFraction_ = ampl_tm1 / ampl_t;
62  n++;
63  }
64 
65  // build TOA timewalk correction with pulse-shape
66  double toaThreshold = ps.getParameter<double>("avgToaThreshold");
67  double gain = ps.getParameter<double>("avgGain");
68  double pedestal = ps.getParameter<double>("avgPedestal");
69  n = 0;
70  for (double ampl = toaThreshold + 0.1; ampl < 10000; ampl += 0.01) {
71  pulseFunc_.FixParameter(0, ampl);
72  double ampl_t = gain * pedestal + pulseFunc_.Eval(0);
73  double toa =
74  fabs(pulseFunc_.GetX(toaThreshold, (double)nADCs_ * clock_cycle_ * -1,
75  (double)nADCs_ * clock_cycle_));
76  correctionTOA_.SetPoint(n, ampl_t, toa);
77  if (n == 0) minAmpl_ = ampl_t;
78  n++;
79  }

83 double HcalRecProducer::getTOA(
84  const ldmx::HgcrocDigiCollection::HgcrocDigi digi, double pedestal,
85  unsigned int iSOI) {
86  // get toa relative to the startBX
87  double toaRelStartBX(0.), maxMeas{0.};
88  int toaSample(0), maxSample(0), iADC(0);
89  for (auto it = digi.begin(); it < digi.end(); it++) {
90  if (it->toa() > 0) {
91  toaRelStartBX = it->toa() * (clock_cycle_ / 1024); // ns
92  // find in which ADC sample the TOA was taken
93  toaSample = iADC;
94  }
95  if ((it->adc_t() - pedestal) > maxMeas) {
96  maxMeas = (it->adc_t() - pedestal);
97  maxSample = iADC;
98  }
99  iADC++;
100  }
101 
102  // time w.r.t to the peak
103  double toa = (maxSample - toaSample) * clock_cycle_ - toaRelStartBX;
104 
105  // time w.r.t to the SOI
106  toa += ((int)iSOI - maxSample) * clock_cycle_;
107 
108  return toa;
109 }

111 void HcalRecProducer::produce(framework::Event& event) {
112  // get the Hcal Geometry
113  const auto& hcalGeometry = getCondition<ldmx::HcalGeometry>(
114  ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
115 
116  // get the reconstruction parameters
117  HcalReconConditions the_conditions(
118  getCondition<conditions::DoubleTableCondition>(
119  HcalReconConditions::CONDITIONS_NAME));
120 
121  std::vector<ldmx::HcalHit> hcalRecHits;
122  auto hcalDigis =
123  event.getObject<ldmx::HgcrocDigiCollection>(digiCollName_, digiPassName_);
124  int numDigiHits = hcalDigis.getNumDigis();
125 
126  // get sample of interest index
127  unsigned int iSOI = hcalDigis.getSampleOfInterestIndex();
128 
129  // loop through digis
130  int iDigi = 0;
131  while (iDigi < numDigiHits) {
132  auto digi_posend = hcalDigis.getDigi(iDigi);
133 
134  // ID from first digi sample (which should be in positive end)
135  ldmx::HcalDigiID id_posend(digi_posend.id());
136  ldmx::HcalID id(id_posend.section(), id_posend.layer(), id_posend.strip());
137 
138  // position from ID
139  auto position = hcalGeometry.getStripCenterPosition(id);
140  double half_total_width = hcalGeometry.getHalfTotalWidth(id.section());
141  double ecal_dx = hcalGeometry.getEcalDx();
142  double ecal_dy = hcalGeometry.getEcalDy();
143 
144  // compute distance to the end of the bar
145  // for back Hcal, we take the half of the bar
146  // for side Hcal, we take the length of the bar (2*half-width)-Ecal_dxy as
147  // an approximation
148  float distance_posend, distance_negend, distance_ecal;
149  if (id.section() == ldmx::HcalID::HcalSection::BACK) {
150  distance_posend = half_total_width;
151  distance_negend = half_total_width;
152  } else {
153  if ((id.section() == ldmx::HcalID::HcalSection::TOP) ||
154  (id.section() == ldmx::HcalID::HcalSection::BOTTOM))
155  distance_ecal = ecal_dx;
156  else
157  distance_ecal = ecal_dy;
158  distance_posend = 2 * half_total_width - distance_ecal / 2.;
159  distance_negend = distance_ecal / 2.;
160  }
161 
162  // get the estimated voltage and time from digi samples
163  double voltage(0.);
164  double voltage_min(0.);
165  double hitTime(0.);
166 
167  double amplT_posend(0.), amplTm1_posend(0.);
168  double amplT_negend(0.), amplTm1_negend(0.);
169 
170  // double readout
171  if (id.section() == ldmx::HcalID::HcalSection::BACK) {
172  auto digi_negend = hcalDigis.getDigi(iDigi + 1);
173  ldmx::HcalDigiID id_negend(digi_negend.id());
174 
175  double voltage_posend, voltage_negend;
176  double amplT_posend_copy, amplT_negend_copy;
177  if (digi_posend.isTOT()) {
178  voltage_posend =
179  (digi_posend.tot() - the_conditions.totPedestal(id_posend)) *
180  the_conditions.totGain(id_posend);
181  voltage_negend =
182  (digi_negend.tot() - the_conditions.totPedestal(id_negend)) *
183  the_conditions.totGain(id_negend);
184  } else {
185  amplT_posend =
186  digi_posend.soi().adc_t() - the_conditions.adcPedestal(id_posend);
187  amplTm1_posend =
188  digi_posend.soi().adc_tm1() - the_conditions.adcPedestal(id_posend);
189  amplT_negend =
190  digi_negend.soi().adc_t() - the_conditions.adcPedestal(id_negend);
191  amplTm1_negend =
192  digi_negend.soi().adc_tm1() - the_conditions.adcPedestal(id_negend);
193 
194  amplT_posend_copy = amplT_posend;
195  amplT_negend_copy = amplT_negend;
196 
197  // correct amplitude (amplitude fractions from both ends need to be
198  // above the boundary of the correction)
199  if (amplTm1_posend / amplT_posend > minAmplFraction_ &&
200  amplTm1_negend / amplT_negend > minAmplFraction_) {
201  amplT_posend *= correctionAmpl_.Eval(amplTm1_posend / amplT_posend);
202  amplT_negend *= correctionAmpl_.Eval(amplTm1_negend / amplT_negend);
203  }
204 
205  // set voltage
206  voltage_posend = amplT_posend * the_conditions.adcGain(id_posend);
207  voltage_negend = amplT_negend * the_conditions.adcGain(id_negend);
208  }
209 
210  // get TOA
211  double TOA_posend =
212  getTOA(digi_posend, the_conditions.adcPedestal(id_posend), iSOI);
213  double TOA_negend =
214  getTOA(digi_negend, the_conditions.adcPedestal(id_negend), iSOI);
215 
216  // get sign of position along the bar
217  int position_bar_sign = (TOA_posend - TOA_negend) > 0 ? 1 : -1;
218 
219  double TOA_posend_copy = TOA_posend;
220  double TOA_negend_copy = TOA_negend;
221 
222  // correct TOA
223  // amplitudes from both ends need to be above the boundary of the
224  // correction otherwise, one TOA gets corrected and the other does not,
225  // which results in a large TOA difference and an out-of-bounds position
226  if (amplT_posend > minAmpl_ && amplT_negend > minAmpl_) {
227  TOA_posend = correctionTOA_.Eval(amplT_posend) - TOA_posend;
228  TOA_negend = correctionTOA_.Eval(amplT_negend) - TOA_negend;
229  }
230 
231  // get x(y) coordinate from TOA measurement = (dt*v/2)
232  // if time_posend < time_negend: position is positive
233  double v =
234  299.792 / 1.6; // velocity of light in polystyrene, n = 1.6 = c/v
235  double position_bar =
236  position_bar_sign * fabs(TOA_posend - TOA_negend) * v / 2;
237 
238  // reverse voltage attenuation
239  // if position along the bar is positive, then the positive end will have
240  // less attenuation than the negative end
241  double att_posend =
242  exp(-1. * ((distance_posend - position_bar) / 1000.) / attlength_);
243  double att_negend =
244  exp(-1. * ((distance_negend + position_bar) / 1000.) / attlength_);
245 
246  // set voltage
247  voltage = (voltage_posend / att_posend + voltage_negend / att_negend) /
248  2; // mV
249  voltage_min =
250  std::min(voltage_posend / att_posend, voltage_negend / att_negend);
251 
252  // set position along the bar
253  if ((id_posend.layer() % 2) == 1) {
254  position.SetX(position_bar);
255  } else {
256  position.SetY(position_bar);
257  }
258 
259  // set hit time
260  // TODO: does this need to revert shift because of propagation of light in
261  // polysterene?
262  hitTime = fabs(TOA_posend + TOA_negend) / 2; // ns
263 
264  iDigi += 2;
265  } // end double readout loop
266  else { // single readout
267 
268  double voltage_i;
269  if (digi_posend.isTOT()) {
270  // TOT - number of clock ticks that pulse was over threshold
271  // this is related to the amplitude of the pulse approximately through a
272  // linear drain rate the amplitude of the pulse is related to the energy
273  // deposited
274 
275  // convert the time over threshold into a total energy deposited in the
276  // bar (time over threshold [ns] - pedestal) * gain
277 
278  voltage_i =
279  (digi_posend.tot() - the_conditions.totPedestal(id_posend)) *
280  the_conditions.totGain(id_posend);
281 
282  } else {
283  // ADC mode of readout
284  // ADC - voltage measurement at a specific time of the pulse
285  amplT_posend =
286  digi_posend.soi().adc_t() - the_conditions.adcPedestal(id_posend);
287  amplTm1_posend =
288  digi_posend.soi().adc_tm1() - the_conditions.adcPedestal(id_posend);
289  voltage_i = amplT_posend * the_conditions.adcGain(id_posend);
290  }
291 
292  // reverse voltage attenuation
293  // for now, assume that position along the bar is the half_total_width
294  double distance_end =
295  id_posend.isNegativeEnd() ? distance_negend : distance_posend;
296  double att = exp(-1. * ((distance_end - fabs(half_total_width)) / 1000.) /
297  attlength_);
298 
299  // set voltage
300  voltage = voltage_i / att;
301  voltage_min = voltage_i / att;
302 
303  // get TOA
304  double TOA =
305  getTOA(digi_posend, the_conditions.adcPedestal(id_posend), iSOI);
306 
307  // correct TOA
308  TOA = correctionTOA_.Eval(amplT_posend) - TOA;
309 
310  // set hit time
311  hitTime = TOA; // ns
312 
313  iDigi++;
314  } // end single readout loop
315 
316  double num_mips_equivalent = voltage / voltage_per_mip_;
317  double energy_deposited = num_mips_equivalent * mip_energy_;
318 
319  // TODO: need to incorporate corrections if necessary
320  double reconstructed_energy = energy_deposited;
321 
322  int PEs = num_mips_equivalent * pe_per_mip_;
323  int minPEs = (voltage_min / voltage_per_mip_) * pe_per_mip_;
324 
325  // copy over information to rec hit structure in new collection
326  ldmx::HcalHit recHit;
327  recHit.setID(id.raw());
328  recHit.setXPos(position.X());
329  recHit.setYPos(position.Y());
330  recHit.setZPos(position.Z());
331  recHit.setPE(PEs);
332  recHit.setMinPE(minPEs);
333  recHit.setAmplitude(amplT_posend);
334  recHit.setEnergy(energy_deposited);
335  recHit.setTime(hitTime);
336  hcalRecHits.push_back(recHit);
337  }
338 
339  if (event.exists(simHitCollName_, simHitPassName_)) {
340  // hcal sim hits exist ==> label which hits are real and which are pure
341  // noise
342  auto hcalSimHits{event.getCollection<ldmx::SimCalorimeterHit>(
343  simHitCollName_, simHitPassName_)};
344  std::set<int> real_hits;
345  for (auto const& sim_hit : hcalSimHits) real_hits.insert(sim_hit.getID());
346  for (auto& hit : hcalRecHits)
347  hit.setNoise(real_hits.find(hit.getID()) == real_hits.end());
348  }
349 
350  // add collection to event bus
351  event.add(recHitCollName_, hcalRecHits);
352 }
