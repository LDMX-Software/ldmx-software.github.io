33  std::fill(fieldValues_.begin(), fieldValues_.end(), 0);

36  unsigned result = (field->getBitMask() & id_.raw()) >> field->getStartBit();

41 void DetectorIDInterpreter::pack() {
42  DetectorID::RawValue rawValue = 0;

45  rawValue =

120  fields->fieldList_ = fieldList;

122  g_rosettaStone[sig] = fields;
