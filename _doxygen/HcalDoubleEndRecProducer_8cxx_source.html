50void HcalDoubleEndRecProducer::produce(framework::Event& event) {
51 const auto& hcalGeometry = getCondition<ldmx::HcalGeometry>(
52 ldmx::HcalGeometry::CONDITIONS_OBJECT_NAME);
53
54 const auto& conditions{
55 getCondition<HcalReconConditions>(HcalReconConditions::CONDITIONS_NAME)};
56
57 auto hcalRecHits =
58 event.getCollection<ldmx::HcalHit>(coll_name_, pass_name_);
59
60 std::vector<ldmx::HcalHit> doubleHcalRecHits;
61
62 // group hcal rechits by the same HcalID
63 std::map<ldmx::HcalID, std::vector<ldmx::HcalHit>> hitsByID;
64 for (auto const& hit : hcalRecHits) {
65 ldmx::HcalID id(hit.getSection(), hit.getLayer(), hit.getStrip());
66
67 auto idh = hitsByID.find(id);
68 if (idh == hitsByID.end()) {
69 hitsByID[id] = std::vector<ldmx::HcalHit>(1, hit);
70 } else {
71 idh->second.push_back(hit);
72 }
73 }
74
75 // make pairs of hcal rechits indices that belong to the same pulse
76 // @TODO: for now we just take the first two indices that have opposite-ends
77 // we do not cover the case where two hits come separated in time
78 std::map<ldmx::HcalID, std::pair<int,int>> indicesByID;
79 for (auto const& hcalBar : hitsByID) {
80 auto id = hcalBar.first;
81
82 std::pair<int,int> indices(-1,-1);
83 int iHit = 0;
84 while ( iHit < hcalBar.second.size() ) {
85 auto hit = hcalBar.second.at(iHit);
86
87 ldmx::HcalDigiID digi_id(hit.getSection(), hit.getLayer(), hit.getStrip(), hit.getEnd());
88 if(digi_id.isNegativeEnd() && indices.second==-1) {
89 indices.second = iHit;
90 }
91 if(!digi_id.isNegativeEnd() && indices.first==-1) {
92 indices.first = iHit;
93 }
94 }
95 indicesByID[id] = indices;
96 }
97
98 // reconstruct double-ended hits
99 for (auto const& hcalBar : hitsByID) {
100 auto id = hcalBar.first;
101
102 // get bar position from geometry
103 auto position = hcalGeometry.getStripCenterPosition(id);
104
105 // skip non-double-ended layers
106 if (id.layer() != ldmx::HcalID::HcalSection::BACK)
107 continue;
108
109 // get two hits to reconstruct
110 auto hitPosEnd = hcalBar.second.at(indicesByID[id].first);
111 auto hitNegEnd = hcalBar.second.at(indicesByID[id].second);
112
113 // update position in strip according to time measurement
114 double v =
115 299.792 / 1.6; // velocity of light in polystyrene, n = 1.6 = c/v
116 double hitTimeDiff = hitPosEnd.getTime() - hitNegEnd.getTime();
117 int position_bar_sign = hitTimeDiff > 0 ? 1 : -1;
118 double position_bar =
119 position_bar_sign * fabs(hitTimeDiff) * v / 2;
120 if (hcalGeometry.layerIsHorizontal(hitPosEnd.getLayer())) {
121 position.SetX(position_bar);
122 } else {
123 position.SetY(position_bar);
124 }
125
126 // TODO: switch unique hit time for this pulse
127 double hitTime = (hitPosEnd.getTime() + hitNegEnd.getTime());
128
129 // amplitude and PEs
130 double num_mips_equivalent = (hitPosEnd.getAmplitude() + hitNegEnd.getAmplitude());
131 double PEs = (hitPosEnd.getPE() + hitNegEnd.getPE());
132 double reconstructed_energy =
133 num_mips_equivalent * pe_per_mip_ * mip_energy_;
134
135 // reconstructed Hit
136 ldmx::HcalHit recHit;
137 recHit.setID(id.raw());
138 recHit.setXPos(position.X());
139 recHit.setYPos(position.Y());
140 recHit.setZPos(position.Z());
141 recHit.setSection(id.section());
142 recHit.setStrip(id.strip());
143 recHit.setLayer(id.layer());
144 recHit.setPE(PEs);
145 recHit.setMinPE(std::min(hitPosEnd.getPE(),hitNegEnd.getPE()));
146 recHit.setAmplitude(num_mips_equivalent);
147 recHit.setEnergy(reconstructed_energy);
148 recHit.setTime(hitTime);
149 doubleHcalRecHits.push_back(recHit);
150 }
151
152 // add collection to event bus
153 event.add(rec_coll_name_, doubleHcalRecHits);
154}
