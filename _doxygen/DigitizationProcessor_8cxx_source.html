83void DigitizationProcessor::produce(framework::Event& event) {
84 // Get the tracking geometry
85
86 const auto tGeometry = ldmx_tg->getTG();
87
88 ldmx_log(debug) << " Getting the tracking geometry:" << tGeometry;
89
90 // Mode 0: Load simulated hits and produce smeared 1d measurements
91 // Mode 1: Load simulated hits and produce digitized 1d measurements
92
93 std::vector<ldmx::LdmxSpacePoint*> digitized_hits;
94
95 const std::vector<ldmx::SimTrackerHit> sim_hits =
96 event.getCollection<ldmx::SimTrackerHit>(hit_collection_);
97
98 std::vector<ldmx::SimTrackerHit> merged_hits;
99
100 if (merge_hits_) {
101 mergeSimHits(sim_hits, merged_hits);
102 digitizeHits(merged_hits, digitized_hits);
103 }
104
105 else {
106 digitizeHits(sim_hits, digitized_hits);
107 }
108
109 // Get the measurements and run clustering
110
111 std::vector<ldmx::Measurement> output_measurements;
112
113 for (auto dh : digitized_hits) {
114 ldmx::Measurement m;
115 m.setGlobalPosition(dh->global_pos_(0), dh->global_pos_(1),
116 dh->global_pos_(2));
117 m.setLocalPosition(dh->local_pos_(0), dh->local_pos_(1));
118 m.setTime(-999);
119 m.setLayer(dh->layer());
120
121 output_measurements.push_back(m);
122 }
123 ldmx_log(debug)<< "Output measurements " << output_measurements.size();
124
125
126 event.add(out_collection_, output_measurements);

129void DigitizationProcessor::onProcessEnd() {}
130// This method merges hits that have the same track_id on the same layer.
131// The energy of the merged hit is the sum of the energy of the single sub-hits
132// The position/momentum of the merged hit is the energy-weighted average
133// sihits = vector of hits to merge
134// mergedHits = total merged collection
135
136bool DigitizationProcessor::mergeHits(
137 const std::vector<ldmx::SimTrackerHit>& sihits,
138 std::vector<ldmx::SimTrackerHit>& mergedHits) {
139 if (sihits.size() < 1) return false;
140
141 if (sihits.size() == 1) {
142 mergedHits.push_back(sihits[0]);
143 return true;
144 }
145
146 ldmx::SimTrackerHit mergedHit;
147 // Since all the hits will be on the same sensor, just use the ID of the first
148 mergedHit.setLayerID(sihits[0].getLayerID());
149 mergedHit.setModuleID(sihits[0].getModuleID());
150 mergedHit.setID(sihits[0].getID());
151 mergedHit.setTrackID(sihits[0].getTrackID());
152
153 double X{0}, Y{0}, Z{0}, PX{0}, PY{0}, PZ{0};
154 double T{0}, E{0}, EDEP{0}, path{0};
155 int pdgID{0};
156
157 pdgID = sihits[0].getPdgID();
158
159 for (auto hit : sihits) {
160 double edep_hit = hit.getEdep();
161 EDEP += edep_hit;
162 E += hit.getEnergy();
163 T += edep_hit * hit.getTime();
164 X += edep_hit * hit.getPosition()[0];
165 Y += edep_hit * hit.getPosition()[1];
166 Z += edep_hit * hit.getPosition()[2];
167 PX += edep_hit * hit.getMomentum()[0];
168 PY += edep_hit * hit.getMomentum()[1];
169 PZ += edep_hit * hit.getMomentum()[2];
170 path += edep_hit * hit.getPathLength();
171
172 if (hit.getPdgID() != pdgID) {
173 std::cout << "ERROR:: Found hits with compatible sensorID and track_id "
174 "but different PDGID"
175 << std::endl;
176 std::cout << "TRACKID ==" << hit.getTrackID() << " vs "
177 << sihits[0].getTrackID() << std::endl;
178 std::cout << "PDGID== " << hit.getPdgID() << " vs " << pdgID << std::endl;
179 return false;
180 }
181 }
182
183 mergedHit.setTime(T / EDEP);
184 mergedHit.setPosition(X / EDEP, Y / EDEP, Z / EDEP);
185 mergedHit.setMomentum(PX / EDEP, PY / EDEP, PZ / EDEP);
186 mergedHit.setPathLength(path / EDEP);
187 mergedHit.setEnergy(E);
188 mergedHit.setEdep(EDEP);
189 mergedHit.setPdgID(pdgID);
190
191 mergedHits.push_back(mergedHit);
192
193 return true;
194}

233void DigitizationProcessor::digitizeHits(
234 const std::vector<ldmx::SimTrackerHit>& sim_hits,
235 std::vector<ldmx::LdmxSpacePoint*>& ldmxsps) {
236
237 ldmx_log(debug) << "Found:" << sim_hits.size() << " sim hits in the "
238 << hit_collection_;
239
240 // Convert to ldmxsps
241
242 for (auto& simHit : sim_hits) {
243 // Remove low energy deposit hits
244 if (simHit.getEdep() > min_e_dep_) {
245 if (track_id_ > 0 && simHit.getTrackID() != track_id_) continue;
246
247 ldmx::LdmxSpacePoint* ldmxsp =
248 tracking::sim::utils::convertSimHitToLdmxSpacePoint(simHit);
249
250 // Get the layer from the ldmxsp
251 unsigned int layerid = ldmxsp->layer();
252
253 // Get the surface
254 const Acts::Surface* hit_surface = ldmx_tg->getSurface(layerid);
255
256 if (hit_surface) {
257 // Transform the ldmx space point from global to local and store the
258 // information
259
260 ldmx_log(debug) << "Global hit position on layer::" << ldmxsp->layer()
261 << ldmxsp->global_pos_;
262
263 //hit_surface->toStream(gctx_, std::cout);
264 ldmx_log(debug) << "Local to global"
265 << std::endl<< hit_surface->transform(gctx_).rotation()
266 << std::endl<< hit_surface->transform(gctx_).translation();
267
268
269 Acts::Vector3 dummy_momentum;
270 Acts::Vector2 local_pos;
271 double surface_thickness = 0.320 * Acts::UnitConstants::mm;
272
273 try {
274 local_pos = hit_surface
275 ->globalToLocal(gctx_, ldmxsp->global_pos_,
276 dummy_momentum, surface_thickness)
277 .value();
278 } catch (const std::exception& e) {
279 std::cout << "WARNING:: hit not on surface.. Skipping." << std::endl;
280 std::cout << ldmxsp->global_pos_ << std::endl;
281 continue;
282 }
283
284 // Smear the local position
285
286
287
288 if (do_smearing_) {
289 float smear_factor{(*normal_)(generator_)};
290
291 local_pos[0] += smear_factor * sigma_u_;
292 smear_factor = (*normal_)(generator_);
293 local_pos[1] += smear_factor * sigma_v_;
294
295
296 // update covariance
297 ldmxsp->setLocalCovariance(sigma_u_ * sigma_u_, sigma_v_ * sigma_v_);
298
299 // cache the acts x coordinate
300 double original_x = ldmxsp->global_pos_(0);
301
302 // transform to global
303 ldmxsp->global_pos_ =
304 hit_surface->localToGlobal(gctx_, local_pos, dummy_momentum);
305 // update the acts x location
306 ldmxsp->global_pos_(0) = original_x;
307
308 } // do smearing
309
310 ldmxsp->local_pos_ = local_pos;
311
312 ldmxsps.push_back(ldmxsp);
313 } // hit_surface exists
314 } // energy cut
315
316 } // loop on sim-hits
317} // digitizeHits
