81 // Get the tracking geometry

87 // Mode 0: Load simulated hits and produce smeared 1d measurements

91 event.getCollection<ldmx::SimTrackerHit>(hit_collection_);

96 if (merge_hits_) {

108// This method merges hits that have the same track_id on the same layer.
109// The energy of the merged hit is the sum of the energy of the single sub-hits
110// The position/momentum of the merged hit is the energy-weighted average
111// sihits = vector of hits to merge
112// mergedHits = total merged collection
113
114bool DigitizationProcessor::mergeHits(
115 const std::vector<ldmx::SimTrackerHit>& sihits,
116 std::vector<ldmx::SimTrackerHit>& mergedHits) {
117 if (sihits.size() < 1) return false;
118
119 if (sihits.size() == 1) {
120 mergedHits.push_back(sihits[0]);
121 return true;
122 }
123
124 ldmx::SimTrackerHit mergedHit;
125 // Since all the hits will be on the same sensor, just use the ID of the first
126 mergedHit.setLayerID(sihits[0].getLayerID());
127 mergedHit.setModuleID(sihits[0].getModuleID());
128 mergedHit.setID(sihits[0].getID());
129 mergedHit.setTrackID(sihits[0].getTrackID());
130
131 double X{0}, Y{0}, Z{0}, PX{0}, PY{0}, PZ{0};
132 double T{0}, E{0}, EDEP{0}, path{0};
133 int pdgID{0};
134
135 pdgID = sihits[0].getPdgID();
136
137 for (auto hit : sihits) {
138 double edep_hit = hit.getEdep();
139 EDEP += edep_hit;
140 E += hit.getEnergy();
141 T += edep_hit * hit.getTime();
142 X += edep_hit * hit.getPosition()[0];
143 Y += edep_hit * hit.getPosition()[1];
144 Z += edep_hit * hit.getPosition()[2];
145 PX += edep_hit * hit.getMomentum()[0];
146 PY += edep_hit * hit.getMomentum()[1];
147 PZ += edep_hit * hit.getMomentum()[2];
148 path += edep_hit * hit.getPathLength();
149
150 if (hit.getPdgID() != pdgID) {
151 std::cout << "ERROR:: Found hits with compatible sensorID and track_id "
152 "but different PDGID"
153 << std::endl;
154 std::cout << "TRACKID ==" << hit.getTrackID() << " vs "
155 << sihits[0].getTrackID() << std::endl;
156 std::cout << "PDGID== " << hit.getPdgID() << " vs " << pdgID << std::endl;
157 return false;
158 }
159 }
160
161 mergedHit.setTime(T / EDEP);
162 mergedHit.setPosition(X / EDEP, Y / EDEP, Z / EDEP);
163 mergedHit.setMomentum(PX / EDEP, PY / EDEP, PZ / EDEP);
164 mergedHit.setPathLength(path / EDEP);
165 mergedHit.setEnergy(E);
166 mergedHit.setEdep(EDEP);
167 mergedHit.setPdgID(pdgID);
168
169 mergedHits.push_back(mergedHit);
170
171 return true;
172}

187 ldmx_log(debug) << "hitmap being filled, size::[" << index << "]["

189 }
190
191 typedef std::map<int,

193 hitmap_it1;
194 typedef std::map<int, std::vector<ldmx::SimTrackerHit>>::iterator hitmap_it2;

210std::vector<ldmx::Measurement> DigitizationProcessor::digitizeHits(
211 const std::vector<ldmx::SimTrackerHit>& sim_hits) {
212 ldmx_log(debug) << "Found:" << sim_hits.size() << " sim hits in the "
213 << hit_collection_;
214
215 std::vector<ldmx::Measurement> measurements;
216
217 // Loop over all SimTrackerHits and
218 // * Use the position of the SimTrackerHit (global position) and the surface
219 // the hit was created on to extract the local coordinates.
220 // * If specified, smear the local coordinates and update the global
221 // coordinates.
222 // * Create a Measurement object.
223 for (auto& sim_hit : sim_hits) {
224 // Remove low energy deposit hits
225 if (sim_hit.getEdep() > min_e_dep_) {
226 if (track_id_ > 0 && sim_hit.getTrackID() != track_id_) continue;
227
228 ldmx::Measurement measurement(sim_hit);
229
230 // Get the layer ID.
231 auto layer_id = tracking::sim::utils::getSensorID(sim_hit);
232 measurement.setLayerID(layer_id);
233
234
235
236 // Get the surface
237 auto hit_surface{ldmx_tg->getSurface(layer_id)};
238
239 if (hit_surface) {
240 // Transform from global to local coordinates.
241 // hit_surface->toStream(gctx_, std::cout);
242 ldmx_log(debug) << "Local to global" << std::endl
243 << hit_surface->transform(gctx_).rotation() << std::endl
244 << hit_surface->transform(gctx_).translation();
245
246 Acts::Vector3 dummy_momentum;
247 Acts::Vector2 local_pos;
248 double surface_thickness = 0.320 * Acts::UnitConstants::mm;
249 Acts::Vector3 global_pos(measurement.getGlobalPosition()[0],
250 measurement.getGlobalPosition()[1],
251 measurement.getGlobalPosition()[2]);
252
253 try {
254 local_pos = hit_surface
255 ->globalToLocal(gctx_, global_pos, dummy_momentum,
256 surface_thickness)
257 .value();
258 } catch (const std::exception& e) {
259 std::cout << "WARNING:: hit not on surface.. Skipping." << std::endl;
260 continue;
261 }
262
263 // Smear the local position
264 if (do_smearing_) {
265 float smear_factor{(*normal_)(generator_)};
266
267 local_pos[0] += smear_factor * sigma_u_;
268 smear_factor = (*normal_)(generator_);
269 local_pos[1] += smear_factor * sigma_v_;
270
271 // update covariance
272 measurement.setLocalCovariance(sigma_u_ * sigma_u_,
273 sigma_v_ * sigma_v_);
274
275 // transform to global
276 auto global_pos{
277 hit_surface->localToGlobal(gctx_, local_pos, dummy_momentum)};
278 measurement.setGlobalPosition(measurement.getGlobalPosition()[0],
279 global_pos(1), global_pos(2));
280
281 } // do smearing
282 measurement.setLocalPosition(local_pos(0), local_pos(1));
283 measurements.push_back(measurement);
284 } // hit_surface exists
285
286 } // energy cut
287 } // loop on sim-hits
288
289 return measurements;
290
