51  EcalHexReadout::EcalHexReadout(double moduleMinR, double gap, unsigned nCellsWide,
52  const std::vector<double> &layerZPositions, double ecalFrontZ)
53  : layerZPositions_(layerZPositions), ecalFrontZ_(ecalFrontZ) {
54 
55  // ORIENTATION ASSUMPTIONS:
56  // modules are oriented flat side down. cells are oriented corner side down.
57  // module corners are one cell -- the 'single mousebite' cell layout.
58  // cells along center horizontal line form a neat row and share vertical edges.
59  // SOME GEOMETRY:
60  // hexagons have two radii:
61  // r (half of flat-to-flat width) and R (half of corner-to-corner width).
62  // r = (sqrt(3)/2)R and s = R, where s is the length of an edge.
63  // for seven ecal modules oriented flat-side-down, maximum x and y extents are:
64  // deltaY = 6r' + 2g = 3sqrt(3)R' + 2g
65  // deltaX = 4R' + s' + 2g/cos(30 deg) = 5R' + 4g/sqrt(3)
66  // where g is uniform gap width between modules, and primed variables correspond to modules.
67  // THIS GRID:
68  // column-to-column distance in a grid such as ours is 2r = sqrt(3)R.
69  // row-to-row distance is 1.5R (easy to observe that twice that distance = 3R)
70  // a cell will be centered at (0,0).
71 
72  assert(nCellsWide % 2 == 1); // calculations rely on centering a cell at (0,0)
73 
74  // calculate module radius reduction due to module edges cutting off outer cell corners.
75  // r = cell center-to-flat (i.e. minimum) radius
76  // R = cell center-to-corner (i.e. maximum) radius
77  // s = R = cell edge length
78  // nCellsWide = count of cells along horizontal center row
79  // moduleMaxR = module center-to-corner radius
80  // moduleMinR = module center-to-flat radius
81  // lengthWide = total length of CELLS along horizontal center row
82  // x = distance from module's corner to outside flat edge of cell it resides in
83  // THEN
84  // nCellsWide*(2*r) = lengthWide = 2*moduleMaxR + 2*x
85  // AND tan(30 deg) = x/(s/2), or x = s/(2*sqrt(3)),
86  // SO nCellsWide*r = moduleMaxR + R/(2*sqrt(3)) = moduleMaxR + r/3
87  // SO r = moduleMaxR/(nCellsWide - 1/3) = (1/2)*(2*moduleMaxR/(nCellsWide - 1/3))
88  // AND lengthWide = nCellsWide*2*r = nCellsWide*(2*moduleMaxR/(nCellsWide - 1/3))
89  //
90  // Test: moduleMaxR = 1., nCellsWide = 1
91  // r = 1/(1-1/3) = 1.5, lengthWide = 3
92  // Checked with a ruler - correct.
93 
94  gap_ = gap;
95  moduler_ = moduleMinR;
96  moduleR_ = moduler_*(2./sqrt(3.));
97  nCellsWide_ = nCellsWide;
98  cellr_ = moduleR_/(nCellsWide - 1./3.);
99  cellR_ = (2./sqrt(3.))*cellr_;
100  lengthWide_ = nCellsWide*2.*cellr_;
101  if(verbose_>0){
102  std::cout << std::endl << "[EcalHexReadout] Verbosity set in header to " << verbose_ << std::endl;
103  std::cout << TString::Format("Building module map with gap %.2f, lengthWide %.2f, nCellsWide %d ",gap_,lengthWide_,nCellsWide_) << std::endl;
104  std::cout << TString::Format(" min/max radii of cell %.2f %.2f and module %.2f %.2f",cellr_,cellR_,moduler_,moduleR_) << std::endl;
105  }

110  buildNeighborMaps();
111  if(verbose_>0){ std::cout << std::endl; }

115  if(verbose_>0) std::cout << std::endl << "[buildModuleMap] Building module position map for module min r of " << moduler_ << std::endl;
116  // module IDs are 0 for ecal center, 1 at 12 o'clock, and clockwise till 6 at 11 o'clock.
117  double C_PI = 3.14159265358979323846; // or TMath::Pi(), #define, atan(), ...
118  modulePositionMap_[0] = std::pair<double,double>(0.,0.);
119  for(unsigned id = 1 ; id < 7 ; id++){
120  double x = (2.*moduler_+gap_)*sin( (id-1)*(C_PI/3.) );
121  double y = (2.*moduler_+gap_)*cos( (id-1)*(C_PI/3.) );
122  modulePositionMap_[id] = std::pair<double,double>(x,y);
123  if(verbose_>2) std::cout << TString::Format(" id %d is at (%.2f, %.2f)",id,x,y) << std::endl;
124  }

271  double EcalHexReadout::distanceToEdge(int cellModuleID) const {
272  // https://math.stackexchange.com/questions/1210572/find-the-distance-to-the-edge-of-a-hexagon
273  int cellID = separateID(cellModuleID).first;
274  XYCoords cellLocation = getCellCenterRelative(cellID);
275  double x = fabs(cellLocation.first); // bring to first quadrant
276  double y = fabs(cellLocation.second);
277  double r = sqrt(x*x+y*y);
278  double theta = (r > 1E-3) ? fabs(std::atan(y/x)) : 0;
279  if(x < moduleR_/2.) return (moduler_ - y); // closest line is straight vertical to top edge
280  double dist = sqrt(3.)*moduleR_/(std::sin(theta) + sqrt(3.)*std::cos(theta));
281  return dist;
282  }

284  bool EcalHexReadout::isInside(double normX, double normY) const {
285  if(verbose_>2) std::cout << TString::Format("[isInside] Checking if normXY=(%.2f,%.2f) is inside.",normX,normY) << std::endl;
286  normX = fabs(normX), normY = fabs(normY);
287  double xvec = -1, yvec = -1./sqrt(3);
288  double xref = 0.5, yref = sqrt(3)/2.;
289  if( (normX > 1.) || (normY > yref) ){
290  if(verbose_>2) std::cout << "[isInside] they are outside quadrant." << std::endl;
291  return false;
292  }
293  double dotProd = (xvec*(normX-xref) + yvec*(normY-yref));
294  if(verbose_>2) std::cout << TString::Format("[isInside] they are inside quadrant. Dot product (>0 is inside): %.2f ", dotProd) << std::endl;
295  return (dotProd > 0.);
296  }
