12 def makePath( detName , fileName ) :
13  """Return a path to the installed data directory for the input detector and file names.
14 
15  Assumes the detectors are installed in the 'data/detectors' directory in the installation directory.
16  Errors out the python script if the created full path does exist.
17  This could happen because
18  - The inputs were spelled wrong
19  - The detector gdml files were not installed
20  - The detector or file you want doesn't exist
21 
22  Parameters
23  ----------
24  detName : str
25  Name of detector to get a path for (should match the name of one of the detector directories)
26  fileName : str
27  Name of the detector gdml file to get a path for (should match the name -- no extension -- of one of the gdml files)
28 
29  Returns
30  -------
31  str
32  full path to installed detector gdml description
33  """
34 
35  fullPath = '@CMAKE_INSTALL_PREFIX@/data/detectors/' + detName + '/' + fileName + '.gdml'
36  if not os.path.isfile( fullPath ) :
37  print('GDML file \'%s\' does not exist.' % ( fullPath ))
38  sys.exit(1)
39 
40  return fullPath
